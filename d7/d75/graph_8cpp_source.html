765  std::vector<uint_fast64_t> stack(psiStates.begin(), psiStates.end());

768  std::vector<uint_fast64_t> stepStack;

770  if (useStepBound) {
771  stepStack.reserve(numberOfStates);
772  stepStack.insert(stepStack.begin(), psiStates.getNumberOfSetBits(), maximalSteps);
773  remainingSteps.resize(numberOfStates);
774  for (auto state : psiStates) {
775  remainingSteps[state] = maximalSteps;
776  }

781  while(!stack.empty()) {
782  currentState = stack.back();
783  stack.pop_back();

786  currentStepBound = stepStack.back();
787  stepStack.pop_back();
788  if (currentStepBound == 0) {
789  continue;

795  if (!statesWithProbabilityGreater0.get(predecessorEntryIt->getColumn())) {

812  break;

814  }
815

817  addToStatesWithProbabilityGreater0 = false;
818  break;
819  }
820  }
821  }
822

824  if (addToStatesWithProbabilityGreater0) {
825  // If we don't have a bound on the number of steps to take, just add the state to the stack.

827  // If there is at least one more step to go, we need to push the state and the new number of steps.

829  stepStack.push_back(currentStepBound - 1);

848  break;

850  }
851  }
852  }

856  stack.push_back(predecessorEntryIt->getColumn());
857  }
858  }
859  }
860  }
861  }

1444  break;

1447  addPlayer1State = false;
1448  break;

1450  }

1453  player1Solution.set(player1Predecessor);

1472  // We only do this if at least one strategy will be produced.
1473  produceStrategiesInIteration = !produceStrategiesInIteration && strategyPair;

1475  result.player1States = player1Solution;
1476  result.player2States = player2Solution;

1484  template <storm::dd::DdType Type, typename ValueType>
1485  SymbolicGameProb01Result<Type> performProb1(storm::models::symbolic::StochasticTwoPlayerGame<Type, ValueType> const& model, storm::dd::Bdd<Type> const& transitionMatrix, storm::dd::Bdd<Type> const& phiStates, storm::dd::Bdd<Type> const& psiStates, storm::OptimizationDirection const& player1Strategy, storm::OptimizationDirection const& player2Strategy, bool producePlayer1Strategy, bool producePlayer2Strategy, boost::optional<storm::dd::Bdd<Type>> const& player1Candidates) {
1486 
1487  // Create the potential prob1 states of player 1.
1488  storm::dd::Bdd<Type> maybePlayer1States = model.getReachableStates();
1489  if (player1Candidates) {
1490  maybePlayer1States &= player1Candidates.get();
1491  }
1492 
1493  // Initialize potential prob1 states of player 2.
1494  storm::dd::Bdd<Type> maybePlayer2States = model.getManager().getBddZero();
1495 
1496  // A flag that governs whether strategies are produced in the current iteration.
1497  bool produceStrategiesInIteration = false;
1498  boost::optional<storm::dd::Bdd<Type>> player1StrategyBdd;
1499  boost::optional<storm::dd::Bdd<Type>> consideredPlayer1States;
1500  boost::optional<storm::dd::Bdd<Type>> player2StrategyBdd;
1501  boost::optional<storm::dd::Bdd<Type>> consideredPlayer2States;
1502 
1503  bool maybeStatesDone = false;
1504  uint_fast64_t maybeStateIterations = 0;
1505  while (!maybeStatesDone || produceStrategiesInIteration) {
1506  bool solutionStatesDone = false;
1507  uint_fast64_t solutionStateIterations = 0;
1508 
1509  // If we are to produce strategies in this iteration, we prepare some storage.
1510  if (produceStrategiesInIteration) {
1511  if (player1Strategy == storm::OptimizationDirection::Maximize) {
1512  player1StrategyBdd = model.getManager().getBddZero();
1513  consideredPlayer1States = model.getManager().getBddZero();
1514  }
1515  if (player2Strategy == storm::OptimizationDirection::Maximize) {
1516  player2StrategyBdd = model.getManager().getBddZero();
1517  consideredPlayer2States = model.getManager().getBddZero();
1518  }
1519  }
1520 
1521  storm::dd::Bdd<Type> player1Solution = psiStates;
1522  storm::dd::Bdd<Type> player2Solution = model.getManager().getBddZero();
1523  while (!solutionStatesDone) {
1524  // Start by computing the transitions that have only maybe states as successors. Note that at
1525  // this point, there may be illegal transitions.
1526  storm::dd::Bdd<Type> distributionsStayingInMaybe = (!transitionMatrix || maybePlayer1States.swapVariables(model.getRowColumnMetaVariablePairs())).universalAbstract(model.getColumnVariables());
1527 
1528  // Then, determine all distributions that have at least one successor in the states that have
1529  // probability 1.
1530  storm::dd::Bdd<Type> distributionsWithProb1Successor = (transitionMatrix && player1Solution.swapVariables(model.getRowColumnMetaVariablePairs())).existsAbstract(model.getColumnVariables());
1531 
1532  // The valid distributions are then those that emanate from phi states, stay completely in the
1533  // maybe states and have at least one successor with probability 1.
1534  storm::dd::Bdd<Type> valid = phiStates && distributionsStayingInMaybe && distributionsWithProb1Successor;
1535 
1536  // Depending on the strategy of player 2, we need to check whether all choices are valid or
1537  // there exists a valid choice.
1538  if (player2Strategy == OptimizationDirection::Minimize) {
1539  valid = (valid || model.getIllegalPlayer2Mask()).universalAbstract(model.getPlayer2Variables());
1540  } else {
1541  if (produceStrategiesInIteration) {
1542  storm::dd::Bdd<Type> newValidDistributions = valid && !consideredPlayer2States.get();
1543  player2StrategyBdd.get() = player2StrategyBdd.get() || newValidDistributions.existsAbstractRepresentative(model.getPlayer2Variables());
1544  }
1545 
1546  valid = valid.existsAbstract(model.getPlayer2Variables());
1547 
1548  if (produceStrategiesInIteration) {
1549  consideredPlayer2States.get() |= valid;
1550  }
1551  }
1552 
1553  player2Solution |= valid;
1554 
1555  // And do the same for player 1.
1556  if (player1Strategy == OptimizationDirection::Minimize) {
1557  valid = (valid || model.getIllegalPlayer1Mask()).universalAbstract(model.getPlayer1Variables());
1558  } else {
1559  if (produceStrategiesInIteration) {
1560  storm::dd::Bdd<Type> newValidDistributions = valid && !consideredPlayer1States.get();
1561  player1StrategyBdd.get() = player1StrategyBdd.get() || newValidDistributions.existsAbstractRepresentative(model.getPlayer1Variables());
1562  }
1563 
1564  valid = valid.existsAbstract(model.getPlayer1Variables());
1565 
1566  if (produceStrategiesInIteration) {
1567  consideredPlayer1States.get() |= valid;
1568  }
1569  }
1570 
1571  // Explicitly add psi states to result since they may have transitions going to some state that
1572  // does not have a reachability probability of 1.
1573  valid |= psiStates;
1574 
1575  // If no new states were added, we have found the current hypothesis for the states with
1576  // probability 1.
1577  if (valid == player1Solution) {
1578  solutionStatesDone = true;
1579  } else {
1580  player1Solution = valid;
1581  }
1582  ++solutionStateIterations;
1583  }
1584 
1585  // If the states with probability 1 and the potential probability 1 states coincide, we have found
1586  // the solution.
1587  if (player1Solution == maybePlayer1States) {
1588  maybePlayer2States = player2Solution;
1589  maybeStatesDone = true;
1590 
1591  // If we were asked to produce strategies, we propagate that by triggering another iteration.
1592  // We only do this if at least one strategy will be produced.
1593  produceStrategiesInIteration = !produceStrategiesInIteration && ((producePlayer1Strategy && player1Strategy == OptimizationDirection::Maximize) || (producePlayer2Strategy && player2Strategy == OptimizationDirection::Maximize));
1594  } else {
1595  // Otherwise, we use the current hypothesis for the states with probability 1 as the new maybe
1596  // state set.
1597  maybePlayer1States = player1Solution;
1598  }
1599  ++maybeStateIterations;
1600  }
1601 
1602  // From now on, the solution is stored in maybeStates (as it coincides with the previous solution).
1603 
1604  // If we were asked to produce strategies that do not need to pick a certain successor but are
1605  // 'arbitrary', do so now.
1606  bool strategiesToCompute = (producePlayer1Strategy && !player1StrategyBdd) || (producePlayer2Strategy && !player2StrategyBdd);
1607  if (strategiesToCompute) {
1608  storm::dd::Bdd<Type> relevantStates = (transitionMatrix && maybePlayer2States).existsAbstract(model.getColumnVariables());
1609  if (producePlayer2Strategy && !player2StrategyBdd) {
1610  player2StrategyBdd = relevantStates.existsAbstractRepresentative(model.getPlayer2Variables());
1611  }
1612  if (producePlayer1Strategy && !player1StrategyBdd) {
1613  relevantStates = (maybePlayer1States && relevantStates).existsAbstract(model.getPlayer2Variables());
1614  player1StrategyBdd = relevantStates.existsAbstractRepresentative(model.getPlayer1Variables());
1615  }
1616  }
1617 
1618  return SymbolicGameProb01Result<Type>(maybePlayer1States, maybePlayer2States, player1StrategyBdd, player2StrategyBdd);
1619  }

1622  template<typename T>
1623  void topologicalSortHelper(storm::storage::SparseMatrix<T> const& matrix, uint64_t state, std::vector<uint_fast64_t>& topologicalSort, std::vector<uint_fast64_t>& recursionStack, std::vector<typename storm::storage::SparseMatrix<T>::const_iterator>& iteratorRecursionStack, storm::storage::BitVector& visitedStates) {
1624  if (!visitedStates.get(state)) {
1625  recursionStack.push_back(state);
1626  iteratorRecursionStack.push_back(matrix.begin(state));
1627 
1628  recursionStepForward:
1629  while (!recursionStack.empty()) {
1630  uint_fast64_t currentState = recursionStack.back();
1631  typename storm::storage::SparseMatrix<T>::const_iterator successorIterator = iteratorRecursionStack.back();
1632 
1633  visitedStates.set(currentState, true);
1634 
1635  recursionStepBackward:
1636  for (; successorIterator != matrix.end(currentState); ++successorIterator) {
1637  if (!visitedStates.get(successorIterator->getColumn())) {
1638  // Put unvisited successor on top of our recursion stack and remember that.
1639  recursionStack.push_back(successorIterator->getColumn());
1640 
1641  // Also, put initial value for iterator on corresponding recursion stack.
1642  iteratorRecursionStack.push_back(matrix.begin(successorIterator->getColumn()));
1643 
1644  goto recursionStepForward;
1645  }
1646  }
1647 
1648  topologicalSort.push_back(currentState);
1649 
1650  // If we reach this point, we have completed the recursive descent for the current state.
1651  // That is, we need to pop it from the recursion stacks.
1652  recursionStack.pop_back();
1653  iteratorRecursionStack.pop_back();
1654 
1655  // If there is at least one state under the current one in our recursion stack, we need
1656  // to restore the topmost state as the current state and jump to the part after the
1657  // original recursive call.
1658  if (recursionStack.size() > 0) {
1659  currentState = recursionStack.back();
1660  successorIterator = iteratorRecursionStack.back();
1661 
1662  goto recursionStepBackward;
1663  }
1664  }
1665  }
1666  }

1669  std::vector<uint_fast64_t> getTopologicalSort(storm::storage::SparseMatrix<T> const& matrix, std::vector<uint64_t> const& firstStates) {
1670  if (matrix.getRowCount() != matrix.getColumnCount()) {
1671  STORM_LOG_ERROR("Provided matrix is required to be square.");
1672  throw storm::exceptions::InvalidArgumentException() << "Provided matrix is required to be square.";
1673  }
1674

1676

1678  std::vector<uint_fast64_t> topologicalSort;

1680

1682  std::vector<uint_fast64_t> recursionStack;

1684  std::vector<typename storm::storage::SparseMatrix<T>::const_iterator> iteratorRecursionStack;
1685  iteratorRecursionStack.reserve(numberOfStates);
1686 
1687  // Perform a depth-first search over the given transitions and record states in the reverse order they were visited.
1688  storm::storage::BitVector visitedStates(numberOfStates);
1689  for (auto const state : firstStates ) {
1690  topologicalSortHelper<T>(matrix, state, topologicalSort, recursionStack, iteratorRecursionStack, visitedStates);
1691  }
1692  for (uint_fast64_t state = 0; state < numberOfStates; ++state) {
1693  topologicalSortHelper<T>(matrix, state, topologicalSort, recursionStack, iteratorRecursionStack, visitedStates);
1694  }
1695
