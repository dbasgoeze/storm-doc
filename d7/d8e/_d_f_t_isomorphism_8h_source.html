15  struct GateGroupToHash {
16  static constexpr uint_fast64_t fivebitmask = (1 << 6) - 1;
17  static constexpr uint_fast64_t eightbitmask = (1 << 9) - 1;
18 
22  uint_fast64_t operator()(DFTElementType type, size_t nrChildren, size_t nrParents, size_t nrPDEPs, size_t rank) const {
23  // Sets first bit to 1
24  uint_fast64_t groupHash = static_cast<uint_fast64_t>(1) << 63;
25  //Assumes 5 bits for the rank,
26  groupHash |= (static_cast<uint_fast64_t>(rank) & fivebitmask) << (62 - 5);
27  // 8 bits for the nrChildren,
28  groupHash |= (static_cast<uint_fast64_t>(nrChildren) & eightbitmask) << (62 - 5 - 8);
29  // 5 bits for nrParents,
30  groupHash |= (static_cast<uint_fast64_t>(nrParents) & fivebitmask) << (62 - 5 - 8 - 5);
31  // 5 bits for nrPDEPs,
32  groupHash |= (static_cast<uint_fast64_t>(nrPDEPs) & fivebitmask) << (62 - 5 - 8 - 5 - 5);
33  // 5 bits for the type
34  groupHash |= (static_cast<uint_fast64_t>(type) & fivebitmask) << (62 - 5 - 8 - 5 - 5 - 5);
35  return groupHash;
36  }

38  };
