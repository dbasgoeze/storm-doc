233  STORM_LOG_THROW(model == nullptr || model->getNumberOfStates() == schedulerChoices.front().size(), storm::exceptions::InvalidOperationException, "The given model is not compatible with this scheduler.");

235  STORM_LOG_THROW(isMemorylessScheduler(), storm::exceptions::NotImplementedException, "Json export of schedulers with memory not implemented.");

237  for (uint64_t state = 0; state < schedulerChoices.front().size(); ++state) {

240  continue;

245  } else {

253  storm::json<storm::RationalNumber> choiceJson;
254  if (model && model->hasChoiceOrigins() && model->getChoiceOrigins()->getIdentifier(globalChoiceIndex) != model->getChoiceOrigins()->getIdentifierForChoicesWithNoOrigin()) {
255  choiceJson["origin"] = model->getChoiceOrigins()->getChoiceAsJson(globalChoiceIndex);

257  if (model && model->hasChoiceLabeling()) {

259  choiceJson["labels"] = std::vector<std::string>(choiceLabels.begin(), choiceLabels.end());
260  }

262  choiceJson["prob"] = storm::utility::convertNumber<storm::RationalNumber>(choiceProbPair.second);
263  choicesJson.push_back(std::move(choiceJson));
264  }
265  } else {
266  choicesJson = "undefined";
267  }

269  output.push_back(std::move(stateChoicesJson));
270  }
271  out << output.dump(4);
272  }
