393  std::shared_ptr<storm::logic::Formula const> FormulaParserGrammar::createMultiFormula(std::vector<std::shared_ptr<storm::logic::Formula const>> const& subformulas) {
394  bool isMultiDimensionalBoundedUntilFormula = !subformulas.empty();
395  for (auto const& subformula : subformulas) {
396  if (!subformula->isBoundedUntilFormula()) {
397  isMultiDimensionalBoundedUntilFormula = false;
398  break;
399  }
400  }
401 
402  if (isMultiDimensionalBoundedUntilFormula) {
403  std::vector<std::shared_ptr<storm::logic::Formula const>> leftSubformulas, rightSubformulas;
404  std::vector<boost::optional<storm::logic::TimeBound>> lowerBounds, upperBounds;
405  std::vector<storm::logic::TimeBoundReference> timeBoundReferences;
406  for (auto const& subformula : subformulas) {
407  auto const& f = subformula->asBoundedUntilFormula();
408  STORM_LOG_THROW(!f.isMultiDimensional(), storm::exceptions::WrongFormatException, "Composition of multidimensional bounded until formula must consist of single dimension subformulas. Got '" << f << "' instead.");
409  leftSubformulas.push_back(f.getLeftSubformula().asSharedPointer());
410  rightSubformulas.push_back(f.getRightSubformula().asSharedPointer());
411  if (f.hasLowerBound()) {
412  lowerBounds.emplace_back(storm::logic::TimeBound(f.isLowerBoundStrict(), f.getLowerBound()));
413  } else {
414  lowerBounds.emplace_back();
415  }
416  if (f.hasUpperBound()) {
417  upperBounds.emplace_back(storm::logic::TimeBound(f.isUpperBoundStrict(), f.getUpperBound()));
418  } else {
419  upperBounds.emplace_back();
420  }
421  timeBoundReferences.push_back(f.getTimeBoundReference());
422  }
423  return std::shared_ptr<storm::logic::Formula const>(new storm::logic::BoundedUntilFormula(leftSubformulas, rightSubformulas, lowerBounds, upperBounds, timeBoundReferences));
424  } else {
425  return std::shared_ptr<storm::logic::Formula const>(new storm::logic::MultiObjectiveFormula(subformulas));
426  }
427  }
