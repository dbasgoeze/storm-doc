146  std::string StateValuations::toString(storm::storage::sparse::state_type const& stateIndex, bool pretty, boost::optional<std::set<storm::expressions::Variable>> const& selectedVariables) const {
147  auto const& valueAssignment = at(stateIndex);
148  typename std::set<storm::expressions::Variable>::const_iterator setIt;
149  if (selectedVariables) {
150  setIt = selectedVariables->begin();
151  }
152  std::vector<std::string> assignments;
153  for (auto valIt = valueAssignment.begin(); valIt != valueAssignment.end(); ++valIt) {
154  if (selectedVariables && (*setIt != valIt.getVariable())) {
155  continue;
156  }
157 
158  std::stringstream stream;
159  if (pretty) {
160  if (valIt.isBoolean() && !valIt.getBooleanValue()) {
161  stream << "!";
162  }
163  stream << valIt.getName();
164  if (valIt.isInteger()) {
165  stream << "=" << valIt.getIntegerValue();
166  } else if (valIt.isRational()) {
167  stream << "=" << valIt.getRationalValue();
168  } else if (valIt.isLabelAssignment()) {
169  stream << "=" << valIt.getLabelValue();
170  } else {
171  STORM_LOG_THROW(valIt.isBoolean(), storm::exceptions::InvalidTypeException, "Unexpected variable type.");
172  }
173  } else {
174  if (valIt.isBoolean()) {
175  stream << std::boolalpha << valIt.getBooleanValue() << std::noboolalpha;
176  } else if (valIt.isInteger()) {
177  stream << valIt.getIntegerValue();
178  } else if (valIt.isRational()) {
179  stream << valIt.getRationalValue();
180  } else if (valIt.isLabelAssignment()) {
181  stream << valIt.getLabelValue();
182  }
183  }
184  assignments.push_back(stream.str());
185 
186  if (selectedVariables) {
187  // Go to next selected position
188  ++setIt;
189  }
190  }
191  STORM_LOG_ASSERT(!selectedVariables || setIt == selectedVariables->end(), "Valuation does not consider selected variable " << setIt->getName() << ".");
192 
193  if (pretty) {
194  return "[" + boost::join(assignments, "\t& ") + "]";
195  } else {
196  return "[" + boost::join(assignments, "\t") + "]";
197  }

230  bool StateValuations::assertValuation(StateValuation const& valuation) const {
231  storm::storage::BitVector foundBooleanValues(valuation.booleanValues.size(), false);
232  storm::storage::BitVector foundIntegerValues(valuation.integerValues.size(), false);
233  storm::storage::BitVector foundRationalValues(valuation.rationalValues.size(), false);
234  for (auto const& varIndex : variableToIndexMap) {
235  storm::storage::BitVector* bv;
236  if (varIndex.first.hasBooleanType()) {
237  bv = &foundBooleanValues;
238  } else if (varIndex.first.hasIntegerType()) {
239  bv = &foundIntegerValues;
240  } else if (varIndex.first.hasRationalType()) {
241  bv = &foundRationalValues;
242  } else {
243  STORM_LOG_THROW(false, storm::exceptions::InvalidTypeException, "Unexpected variable type.");
244  }
245  if (varIndex.second < bv->size()) {
246  if (bv->get(varIndex.second)) {
247  STORM_LOG_ERROR("Multiple variables refer to the same value index");
248  return false;
249  }
250  bv->set(varIndex.second, true);
251  } else {
252  STORM_LOG_ERROR("Valuation does not provide a value for all variables");
253  return false;
254  }
255  }
256  if (!(foundBooleanValues.full() && foundIntegerValues.full() && foundRationalValues.full())) {
257  STORM_LOG_ERROR("Valuation has too many entries.");
258  }
259  return true;
260  }
