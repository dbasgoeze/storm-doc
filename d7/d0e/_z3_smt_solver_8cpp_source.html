252  uint_fast64_t Z3SmtSolver::allSat(std::vector<storm::expressions::Variable> const& important, std::function<bool(storm::expressions::SimpleValuation&)> const& callback) {
253 #ifdef STORM_HAVE_Z3
254  for (storm::expressions::Variable const& variable : important) {
255  STORM_LOG_THROW(variable.hasBooleanType(), storm::exceptions::InvalidArgumentException, "The important atoms for AllSat must be boolean variables.");
256  }
257 
258  uint_fast64_t numberOfModels = 0;
259  bool proceed = true;
260 
261  // Save the current assertion stack, to be able to backtrack after the procedure.
262  this->push();
263 
264  // Enumerate models as long as the conjunction is satisfiable and the callback has not aborted the enumeration.
265  while (proceed && this->check() == CheckResult::Sat) {
266  ++numberOfModels;
267  z3::model model = this->solver->get_model();
268 
269  z3::expr modelExpr = this->context->bool_val(true);
270  storm::expressions::SimpleValuation valuation(this->getManager().getSharedPointer());
271 
272  for (storm::expressions::Variable const& importantAtom : important) {
273  z3::expr z3ImportantAtom = this->expressionAdapter->translateExpression(importantAtom.getExpression());
274  z3::expr z3ImportantAtomValuation = model.eval(z3ImportantAtom, true);
275  modelExpr = modelExpr && (z3ImportantAtom == z3ImportantAtomValuation);
276  valuation.setBooleanValue(importantAtom, this->expressionAdapter->translateExpression(z3ImportantAtomValuation).isTrue());
277  }
278 
279  // Check if we are required to proceed, and if so rule out the current model.
280  proceed = callback(valuation);
281  if (proceed) {
282  this->solver->add(!modelExpr);
283  }
284  }
285 
286  // Restore the old assertion stack and return.
287  this->pop();
288  return numberOfModels;
289 #else
290  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Storm is compiled without Z3 support.");
291 #endif
