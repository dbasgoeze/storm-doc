104  void createVariables(PermissiveSchedulerPenalties const& penalties, storm::storage::BitVector const& relevantStates) {
105  // We need the unique initial state later, so we get that one before looping.
106  STORM_LOG_ASSERT(this->mdp.getInitialStates().getNumberOfSetBits() == 1, "No unique initial state.");
107  uint_fast64_t initialStateIndex = this->mdp.getInitialStates().getNextSetIndex(0);
108 
109  storm::expressions::Variable var;
110  for(uint_fast64_t s : relevantStates) {
111  // Create x_s variables
112  // Notice that only the initial probability appears in the objective.
113  if(s == initialStateIndex) {
114  var = solver.addLowerBoundedContinuousVariable("x_" + std::to_string(s), 0.0, -1.0);
115  } else {
116  var = solver.addLowerBoundedContinuousVariable("x_" + std::to_string(s), 0.0);
117  }
118  mProbVariables[s] = var;
119  // Create alpha_s variables
120  var = solver.addBinaryVariable("alp_" + std::to_string(s));
121  mAlphaVariables[s] = var;
122  // Create gamma_s variables
123  var = solver.addBoundedContinuousVariable("gam_" + std::to_string(s), 0.0, 1.0);
124  mGammaVariables[s] = var;
125  for(uint_fast64_t a = 0; a < this->mdp.getNumberOfChoices(s); ++a) {
126  auto stateAndAction = storage::StateActionPair(s,a);
127 
128  // Create y_(s,a) variables
129  double penalty = penalties.get(stateAndAction);
130  var = solver.addBinaryVariable("y_" + std::to_string(s) + "_" + std::to_string(a), -penalty);
131  multistrategyVariables[stateAndAction] = var;
132 
133  // Create beta_(s,a,t) variables
134  // Iterate over successors of s via a.
135  for(auto const& entry : this->mdp.getTransitionMatrix().getRow(this->mdp.getNondeterministicChoiceIndices()[s]+a)) {
136  if(entry.getValue() != 0) {
137  storage::StateActionTarget sat = {s,a,entry.getColumn()};
138  var = solver.addBinaryVariable("beta_" + to_string(sat));
139  mBetaVariables[sat] = var;
140  }
141  }
142  }
143  }
144  solver.update();
145  }

150  void createConstraints(bool lowerBound, double boundary, storm::storage::BitVector const& relevantStates) {
151  // (5) and (7) are omitted on purpose (-- we currenty do not support controllability of actions -- )
152 
153  // (1)
154  STORM_LOG_ASSERT(this->mdp.getInitialStates().getNumberOfSetBits() == 1, "No unique initial state.");
155  uint_fast64_t initialStateIndex = this->mdp.getInitialStates().getNextSetIndex(0);
156  STORM_LOG_ASSERT(relevantStates[initialStateIndex], "Initial state not relevant.");
157  if(lowerBound) {
158  solver.addConstraint("c1", mProbVariables[initialStateIndex] >= solver.getConstant(boundary));
159  } else {
160  solver.addConstraint("c1", mProbVariables[initialStateIndex] <= solver.getConstant(boundary));
161  }
162  for(uint_fast64_t s : relevantStates) {
163  std::string stateString = std::to_string(s);
164  storm::expressions::Expression expr = solver.getConstant(0.0);
165  // (2)
166  for(uint_fast64_t a = 0; a < this->mdp.getNumberOfChoices(s); ++a) {
167  expr = expr + multistrategyVariables[storage::StateActionPair(s,a)];
168  }
169  solver.addConstraint("c2-" + stateString, solver.getConstant(1) <= expr);
170  // (5)
171  solver.addConstraint("c5-" + std::to_string(s), mProbVariables[s] <= mAlphaVariables[s]);
172 
173  // (3) For the relevant states.
174  for(uint_fast64_t a = 0; a < this->mdp.getNumberOfChoices(s); ++a) {
175  std::string sastring(stateString + "_" + std::to_string(a));
176  expr = solver.getConstant(0.0);
177  for(auto const& entry : this->mdp.getTransitionMatrix().getRow(this->mdp.getNondeterministicChoiceIndices()[s]+a)) {
178  if(entry.getValue() != 0 && relevantStates.get(entry.getColumn())) {
179  expr = expr + solver.getConstant(entry.getValue()) * mProbVariables[entry.getColumn()];
180  } else if (entry.getValue() != 0 && this->mGoals.get(entry.getColumn())) {
181  expr = expr + solver.getConstant(entry.getValue());
182  }
183  }
184  if(lowerBound) {
185  solver.addConstraint("c3-" + sastring, mProbVariables[s] <= (solver.getConstant(1) - multistrategyVariables[storage::StateActionPair(s,a)]) + expr);
186  } else {
187  solver.addConstraint("c3-" + sastring, mProbVariables[s] >= (solver.getConstant(1) - multistrategyVariables[storage::StateActionPair(s,a)]) + expr);
188  }
189  }
190 
191  for(uint_fast64_t a = 0; a < this->mdp.getNumberOfChoices(s); ++a) {
192  // (6)
193  std::string sastring(stateString + "_" + std::to_string(a));
194  expr = solver.getConstant(0.0);
195  for(auto const& entry : this->mdp.getTransitionMatrix().getRow(this->mdp.getNondeterministicChoiceIndices()[s]+a)) {
196  if(entry.getValue() != 0) {
197  storage::StateActionTarget sat = {s,a,entry.getColumn()};
198  expr = expr + mBetaVariables[sat];
199  }
200  }
201  solver.addConstraint("c6-" + sastring, multistrategyVariables[storage::StateActionPair(s,a)] == (solver.getConstant(1) - mAlphaVariables[s]) + expr);
202 
203  for(auto const& entry : this->mdp.getTransitionMatrix().getRow(this->mdp.getNondeterministicChoiceIndices()[s]+a)) {
204  if(entry.getValue() != 0) {
205  storage::StateActionTarget sat = {s,a,entry.getColumn()};
206  std::string satstring = to_string(sat);
207  // (8)
208  if(relevantStates[entry.getColumn()]) {
209  STORM_LOG_ASSERT(mGammaVariables.count(entry.getColumn()) > 0, "Entry not found.");
210  STORM_LOG_ASSERT(mGammaVariables.count(s) > 0, "Entry not found.");
211  STORM_LOG_ASSERT(mBetaVariables.count(sat) > 0, "Entry not found.");
212  solver.addConstraint("c8-" + satstring, mGammaVariables[entry.getColumn()] < mGammaVariables[s] + (solver.getConstant(1) - mBetaVariables[sat]));
213  }
214  }
215  }
216 
217  }
218  }
