268  // upper bound or the maximum value is below a lower bound, the property will definitely not hold and we can
269  // return the value.

297  }

412  template<typename ValueType>
413  ExplicitQuantitativeResult<ValueType> computeQuantitativeResult(Environment const& env, storm::OptimizationDirection player1Direction, storm::OptimizationDirection player2Direction, storm::storage::SparseMatrix<ValueType> const& transitionMatrix, std::vector<uint64_t> const& player1Groups, ExplicitQualitativeGameResultMinMax const& qualitativeResult, storm::storage::BitVector const& maybeStates, ExplicitGameStrategyPair& strategyPair, storm::dd::Odd const& odd, ExplicitQuantitativeResult<ValueType> const* startingQuantitativeResult = nullptr, ExplicitGameStrategyPair const* startingStrategyPair = nullptr, boost::optional<PreviousExplicitResult<ValueType>> const& previousResult = boost::none) {

415  bool player2Min = player2Direction == storm::OptimizationDirection::Minimize;
416  auto const& player1Prob1States = player2Min ? qualitativeResult.getProb1Min().asExplicitQualitativeGameResult().getPlayer1States() : qualitativeResult.getProb1Max().asExplicitQualitativeGameResult().getPlayer1States();
417  auto const& player2Prob0States = player2Min ? qualitativeResult.getProb0Min().asExplicitQualitativeGameResult().getPlayer2States() : qualitativeResult.getProb0Max().asExplicitQualitativeGameResult().getPlayer2States();

424  if (maybeStates.empty()) {
425  return result;
426  }
427 
428  // If there is a previous result, unpack the previous values with respect to the new ODD.
429  if (previousResult) {

438  // Otherwise, we need to solve a (sub)game.
439  STORM_LOG_TRACE("[" << player1Direction << ", " << player2Direction << "]: Solving " << maybeStates.getNumberOfSetBits()<< " maybe states.");
440 
441  // Create the game by selecting all maybe player 2 states (non-prob0/1) of all maybe player 1 states.
442  std::vector<uint64_t> subPlayer1Groups(maybeStates.getNumberOfSetBits() + 1);
443  uint64_t position = 0;
444  uint64_t previousPlayer2States = 0;
445  storm::storage::BitVector player2MaybeStates(transitionMatrix.getRowGroupCount());
446  for (auto state : maybeStates) {
447  subPlayer1Groups[position] = previousPlayer2States;

453  hasMaybePlayer2Successor = true;

455  }
456  }
457  STORM_LOG_ASSERT(hasMaybePlayer2Successor, "Player 1 maybe state has no player2 maybe successor.");
458  ++position;
459  }
460  subPlayer1Groups.back() = previousPlayer2States;

463  storm::storage::SparseMatrix<ValueType> submatrix = transitionMatrix.getSubmatrix(true, player2MaybeStates, maybeStates);
464  std::vector<ValueType> b = transitionMatrix.getConstrainedRowGroupSumVector(player2MaybeStates, player1Prob1States);
465 
466  // Set up game solver.
467  auto gameSolver = storm::solver::GameSolverFactory<ValueType>().create(env, subPlayer1Groups, submatrix);
468 
469  // Prepare the value storage for the maybe states. If the starting values were given, extract them now.
470  std::vector<ValueType> values(maybeStates.getNumberOfSetBits());
471  if (startingQuantitativeResult) {
472  storm::utility::vector::selectVectorValues(values, maybeStates, startingQuantitativeResult->getValues());
473  }
474  if (previousResult) {
475  STORM_LOG_ASSERT(!startingQuantitativeResult, "Cannot take two different hints.");
476  storm::utility::vector::selectVectorValues(values, maybeStates, result.getValues());
477  }
478 
479  // Prepare scheduler storage.
480  std::vector<uint64_t> player1Scheduler(subPlayer1Groups.size() - 1);
481  std::vector<uint64_t> player2Scheduler(submatrix.getRowGroupCount());
482  if (startingStrategyPair) {
483  // If the starting strategy pair was provided, we need to extract the choices of the maybe states here.
484  uint64_t maybeStatePosition = 0;
485  previousPlayer2States = 0;
486  for (auto state : maybeStates) {
487  uint64_t chosenPlayer2State = startingStrategyPair->getPlayer1Strategy().getChoice(state);
488 
489  uint64_t previousPlayer2MaybeStatesForState = 0;
490  for (uint64_t player2State = player1Groups[state]; player2State < player1Groups[state + 1]; ++player2State) {
491  if (player2MaybeStates.get(player2State)) {
492  if (player2State == chosenPlayer2State) {
493  player1Scheduler[maybeStatePosition] = previousPlayer2MaybeStatesForState;
494  }
495 
496  // Copy over the player 2 action (modulo making it local) as all rows for the player 2 state are taken.
497  if (startingStrategyPair->getPlayer2Strategy().hasDefinedChoice(player2State)) {
498  player2Scheduler[previousPlayer2States] = startingStrategyPair->getPlayer2Strategy().getChoice(player2State) - transitionMatrix.getRowGroupIndices()
499  [player2State];
500  } else {
501  player2Scheduler[previousPlayer2States] = 0;
502  }
503 
504  ++previousPlayer2MaybeStatesForState;
505  ++previousPlayer2States;
506  }
507  }
508 
509  ++maybeStatePosition;
510  }
511  STORM_LOG_ASSERT(previousPlayer2States == submatrix.getRowGroupCount(), "Expected correct number of player 2 states.");
512  }
513 
514  // Solve actual game and track schedulers.
515  gameSolver->solveGame(env, player1Direction, player2Direction, values, b, &player1Scheduler, &player2Scheduler);
516 
517  // Set values according to quantitative result (qualitative result has already been taken care of).
518  storm::utility::vector::setVectorValues(result.getValues(), maybeStates, values);
519 
520  // Obtain strategies from solver and fuse them with the pre-existing strategy pair for the qualitative result.
521  uint64_t previousPlayer1MaybeStates = 0;
522  uint64_t previousPlayer2MaybeStates = 0;
523  for (auto state : maybeStates) {
524  uint64_t previousPlayer2MaybeStatesForState = 0;
525  bool madePlayer1Choice = false;
526  for (uint64_t player2State = player1Groups[state]; player2State < player1Groups[state + 1]; ++player2State) {
527  if (player1Scheduler[previousPlayer1MaybeStates] == previousPlayer2MaybeStatesForState) {
528  strategyPair.getPlayer1Strategy().setChoice(state, player2State);
529  madePlayer1Choice = true;
530  }
531 
532  if (player2MaybeStates.get(player2State)) {
533  strategyPair.getPlayer2Strategy().setChoice(player2State, transitionMatrix.getRowGroupIndices()[player2State] + player2Scheduler[previousPlayer2MaybeStates]);
534 
535  ++previousPlayer2MaybeStatesForState;
536  ++previousPlayer2MaybeStates;
537  }
538  }
539  STORM_LOG_ASSERT(madePlayer1Choice, "[" << player1Direction << "]: player 1 state " << state << " did not make a choice, scheduler: " << player1Scheduler[previousPlayer1MaybeStates] << ".");
540 
541  ++previousPlayer1MaybeStates;
542  }
543 
544  return result;

645  template<storm::dd::DdType Type, typename ModelType>
646  std::unique_ptr<CheckResult> GameBasedMdpModelChecker<Type, ModelType>::performSymbolicAbstractionSolutionStep(Environment const& env, CheckTask<storm::logic::Formula, ValueType> const& checkTask, storm::abstraction::MenuGame<Type, ValueType> const& game, storm::OptimizationDirection player1Direction, storm::dd::Bdd<Type> const& initialStates, storm::dd::Bdd<Type> const& constraintStates, storm::dd::Bdd<Type> const& targetStates, storm::abstraction::MenuGameRefiner<Type, ValueType> const& refiner, boost::optional<SymbolicQualitativeGameResultMinMax<Type>>& previousQualitativeResult, boost::optional<SymbolicQuantitativeGameResult<Type, ValueType>>& previousMinQuantitativeResult) {

648  STORM_LOG_TRACE("Using dd-based solving.");
649 
650  // Prepare transition matrix BDD.
651  storm::dd::Bdd<Type> transitionMatrixBdd = game.getTransitionMatrix().toBdd();
652

685  }

688  if (!qualitativeRefinement) {

690  STORM_LOG_TRACE("Starting numerical solution step.");
691 
692  storm::dd::Add<Type, ValueType> initialStatesAdd = initialStates.template toAdd<ValueType>();
693 
694  SymbolicQuantitativeGameResultMinMax<Type, ValueType> quantitativeResult;

697  storm::utility::Stopwatch quantitativeWatch(true);
698  quantitativeResult.min = computeQuantitativeResult(env, player1Direction, storm::OptimizationDirection::Minimize, game, qualitativeResult, initialStatesAdd, maybeMin, reuseQuantitativeResults ? previousMinQuantitativeResult : boost::none);
699  quantitativeWatch.stop();

701  result = checkForResultAfterQuantitativeCheck<ValueType>(checkTask, storm::OptimizationDirection::Minimize, quantitativeResult.min.getInitialStatesRange());
702  if (result) {
703  totalSolutionWatch.add(quantitativeWatch);
704  return result;
705  }
706 
707  // (8) Solve the max values and check whether we can give the answer already.
708  quantitativeWatch.start();
709  quantitativeResult.max = computeQuantitativeResult(env, player1Direction, storm::OptimizationDirection::Maximize, game, qualitativeResult, initialStatesAdd, maybeMax, boost::make_optional(quantitativeResult.min));
710  quantitativeWatch.stop();
711  result = checkForResultAfterQuantitativeCheck<ValueType>(checkTask, storm::OptimizationDirection::Maximize, quantitativeResult.max.getInitialStatesRange());

728  }

743  // information about the game, but we could not yet answer the query. In this case, we need to refine.
744  storm::utility::Stopwatch refinementWatch(true);
745  refiner.refine(game, transitionMatrixBdd, quantitativeResult);
746  refinementWatch.stop();
747  totalRefinementWatch.add(refinementWatch);
748  STORM_LOG_INFO("Quantitative refinement completed in " << refinementWatch.getTimeInMilliseconds() << "ms.");
749  }
750 
751  // Return null to indicate no result has been found yet.
752  return nullptr;
