115  STORM_LOG_THROW(parsedStructure.count("type") == 1, storm::exceptions::InvalidJaniException, "A type must be given exactly once");

171  }
172  }
173

175  STORM_LOG_THROW(funDeclCount < 2, storm::exceptions::InvalidJaniException, "Model '" << name << "' has more than one list of functions");
176  FunctionsMap globalFuns;
177  scope.globalFunctions = &globalFuns;
178  if (funDeclCount > 0) {
179  // We require two passes through the function definitions array to allow referring to functions before they were defined.
180  std::vector<storm::jani::FunctionDefinition> dummyFunctionDefinitions;
181  for (auto const& funStructure : parsedStructure.at("functions")) {
182  // Skip parsing of function body
183  dummyFunctionDefinitions.push_back(parseFunctionDefinition(funStructure, scope.refine("functions[" + std::to_string(globalFuns.size()) + "] of model " + name),

189  STORM_LOG_THROW(unused, storm::exceptions::InvalidJaniException, "Multiple definitions of functions with the name " << funDef.getName() << " in " << scope.description);
190  }
191  for (auto const& funStructure : parsedStructure.at("functions")) {
192  // Actually parse the function body
193  storm::jani::FunctionDefinition funDef = parseFunctionDefinition(funStructure, scope.refine("functions[" + std::to_string(globalFuns.size()) + "] of model " + name),
194  false);
195  assert(globalFuns.count(funDef.getName()) == 1);

197  }

218

229  rewAccEliminator.eliminateRewardAccumulations(prop);
230  properties.push_back(prop);
231  } catch (storm::exceptions::NotSupportedException const& ex) {

233  } catch (storm::exceptions::NotImplementedException const& ex) {
234  STORM_LOG_WARN("Cannot handle property: " << ex.what());
235  }
236  }
237  }
238  return {model, properties};

412  model.addNonTrivialRewardExpression(rewardName, rewExpr);
413  }
414  return std::make_shared<storm::logic::RewardOperatorFormula>(std::make_shared<storm::logic::CumulativeRewardFormula>(bounds, boundReferences, rewardAccumulation), rewardName, opInfo);
415  } else {
416  time = !rewExpr.containsVariables() && storm::utility::isOne(rewExpr.evaluateAsRational());
417  std::shared_ptr<storm::logic::Formula const> subformula;
418  if (propertyStructure.count("reach") > 0) {
419  auto formulaContext = time ? storm::logic::FormulaContext::Time : storm::logic::FormulaContext::Reward;
420  subformula = std::make_shared<storm::logic::EventuallyFormula>(parseFormula(model, propertyStructure.at("reach"), formulaContext, scope.refine("Reach-expression of operator " + opString)), formulaContext, rewardAccumulation);
421  } else {
422  subformula = std::make_shared<storm::logic::TotalRewardFormula>(rewardAccumulation);
423  }
424  if (time) {
425  assert(subformula->isTotalRewardFormula() || subformula->isTimePathFormula());
426  return std::make_shared<storm::logic::TimeOperatorFormula>(subformula, opInfo);
427  } else {
428  if (!rewExpr.isVariable()) {
429  model.addNonTrivialRewardExpression(rewardName, rewExpr);
430  }
431  return std::make_shared<storm::logic::RewardOperatorFormula>(subformula, rewardName, opInfo);
432  }
433  }
434  } else if (opString == "Smin" || opString == "Smax") {
435  storm::logic::OperatorInformation opInfo;
436  opInfo.optimalityType = opString == "Smin" ? storm::solver::OptimizationDirection::Minimize : storm::solver::OptimizationDirection::Maximize;
437  opInfo.bound = bound;
438  // Reward accumulation is optional as it was not available in the early days...
439  boost::optional<storm::logic::RewardAccumulation> rewardAccumulation;
440  if (propertyStructure.count("accumulate") > 0) {
441  STORM_LOG_WARN_COND(model.getJaniVersion() == 1, "Unexpected accumulate field in " << scope.description << ".");
442  rewardAccumulation = parseRewardAccumulation(propertyStructure.at("accumulate"), scope.description);
443  }
444  STORM_LOG_THROW(propertyStructure.count("exp") > 0, storm::exceptions::InvalidJaniException, "Expected an expression at steady state property at " << scope.description);
445  auto exp = parseExpression(propertyStructure["exp"], scope.refine("steady-state operator"), true);
446  if (!exp.isInitialized() || exp.hasBooleanType()) {
447  STORM_LOG_THROW(!rewardAccumulation.is_initialized(), storm::exceptions::InvalidJaniException, "Long-run average probabilities are not allowed to have a reward accumulation at" << scope.description);
448  std::shared_ptr<storm::logic::Formula const> subformula = parseUnaryFormulaArgument(model, propertyStructure, formulaContext, opString, scope.refine("Steady-state operator"))[0];
449  return std::make_shared<storm::logic::LongRunAverageOperatorFormula>(subformula, opInfo);
450  }
451  STORM_LOG_THROW(exp.hasNumericalType(), storm::exceptions::InvalidJaniException, "Reward expression '" << exp << "' does not have numerical type in " << scope.description);
452  std::string rewardName = exp.toString();
453  if (!exp.isVariable()) {
454  model.addNonTrivialRewardExpression(rewardName, exp);
455  }
456  auto subformula = std::make_shared<storm::logic::LongRunAverageRewardFormula>(rewardAccumulation);
457  return std::make_shared<storm::logic::RewardOperatorFormula>(subformula, rewardName, opInfo);
458 
459  } else if (opString == "U" || opString == "F") {
460  assert(bound == boost::none);
461  std::vector<std::shared_ptr<storm::logic::Formula const>> args;
462  if (opString == "U") {
463  args = parseBinaryFormulaArguments(model, propertyStructure, formulaContext, opString, scope);
464  } else {
465  assert(opString == "F");
466  args = parseUnaryFormulaArgument(model, propertyStructure, formulaContext, opString, scope);
467  args.push_back(args[0]);
468  args[0] = storm::logic::BooleanLiteralFormula::getTrueFormula();
469  }
470 
471  std::vector<boost::optional<storm::logic::TimeBound>> lowerBounds, upperBounds;
472  std::vector<storm::logic::TimeBoundReference> tbReferences;
473  if (propertyStructure.count("step-bounds") > 0) {
474  STORM_LOG_WARN_COND(model.getJaniVersion() == 1, "Jani model not compliant: Contains step-bounds in " << scope.description << ".");
475  storm::jani::PropertyInterval pi = parsePropertyInterval(propertyStructure.at("step-bounds"), scope.refine("step-bounded until").clearVariables());
476  insertLowerUpperTimeBounds(lowerBounds, upperBounds, pi);
477  tbReferences.emplace_back(storm::logic::TimeBoundType::Steps);
478  }
479  if (propertyStructure.count("time-bounds") > 0) {
480  STORM_LOG_WARN_COND(model.getJaniVersion() == 1, "Jani model not compliant: Contains time-bounds in " << scope.description << ".");
481  storm::jani::PropertyInterval pi = parsePropertyInterval(propertyStructure.at("time-bounds"), scope.refine("time-bounded until").clearVariables());
482  insertLowerUpperTimeBounds(lowerBounds, upperBounds, pi);
483  tbReferences.emplace_back(storm::logic::TimeBoundType::Time);
484  }
485  if (propertyStructure.count("reward-bounds") > 0 ) {
486  for (auto const& rbStructure : propertyStructure.at("reward-bounds")) {
487  storm::jani::PropertyInterval pi = parsePropertyInterval(rbStructure.at("bounds"), scope.refine("reward-bounded until").clearVariables());
488  insertLowerUpperTimeBounds(lowerBounds, upperBounds, pi);
489  STORM_LOG_THROW(rbStructure.count("exp") == 1, storm::exceptions::InvalidJaniException, "Expecting reward-expression for operator " << opString << " in " << scope.description);
490  storm::expressions::Expression rewInstRewardModelExpression = parseExpression(rbStructure.at("exp"), scope.refine("Reward expression at reward-bounds"));
491  STORM_LOG_THROW(rewInstRewardModelExpression.hasNumericalType(), storm::exceptions::InvalidJaniException, "Reward expression '" << rewInstRewardModelExpression << "' does not have numerical type in " << scope.description);
492  storm::logic::RewardAccumulation boundRewardAccumulation = parseRewardAccumulation(rbStructure.at("accumulate"), scope.description);
493  bool steps = (boundRewardAccumulation.isStepsSet() || boundRewardAccumulation.isExitSet()) && boundRewardAccumulation.size() == 1;
494  bool time = boundRewardAccumulation.isTimeSet() && boundRewardAccumulation.size() == 1 && !model.isDiscreteTimeModel();
495  if ((steps || time) && !rewInstRewardModelExpression.containsVariables() && storm::utility::isOne(rewInstRewardModelExpression.evaluateAsRational())) {
496  tbReferences.emplace_back(steps ? storm::logic::TimeBoundType::Steps : storm::logic::TimeBoundType::Time);
497  } else {
498  std::string rewInstRewardModelName = rewInstRewardModelExpression.toString();
499  if (!rewInstRewardModelExpression.isVariable()) {
500  model.addNonTrivialRewardExpression(rewInstRewardModelName, rewInstRewardModelExpression);
501  }
502  tbReferences.emplace_back(rewInstRewardModelName, boundRewardAccumulation);
503  }
504  }
505  }
506  if (!tbReferences.empty()) {
507  return std::make_shared<storm::logic::BoundedUntilFormula const>(args[0], args[1], lowerBounds, upperBounds, tbReferences);
508  } else if (args[0]->isTrueFormula()) {
509  return std::make_shared<storm::logic::EventuallyFormula const>(args[1], formulaContext);
510  } else {
511  return std::make_shared<storm::logic::UntilFormula const>(args[0], args[1]);
512  }
513  } else if (opString == "G") {
514  assert(bound == boost::none);
515  std::vector<std::shared_ptr<storm::logic::Formula const>> args = parseUnaryFormulaArgument(model, propertyStructure, formulaContext, opString, scope.refine("Subformula of globally operator "));
516  if (propertyStructure.count("step-bounds") > 0) {
517  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Globally and step-bounds are not supported.");
518  } else if (propertyStructure.count("time-bounds") > 0) {
519  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Globally and time bounds are not supported.");
520  } else if (propertyStructure.count("reward-bounds") > 0 ) {
521  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Globally and reward bounded properties are not supported.");
522  }
523  return std::make_shared<storm::logic::GloballyFormula const>(args[0]);
524 
525  } else if (opString == "W") {
526  assert(bound == boost::none);
527  STORM_LOG_THROW(false, storm::exceptions::NotImplementedException, "Weak until is not supported");
528  } else if (opString == "R") {
529  assert(bound == boost::none);
530  STORM_LOG_THROW(false, storm::exceptions::NotImplementedException, "Release is not supported");
531  } else if (opString == "" || opString == "") {
532  assert(bound == boost::none);
533  std::vector<std::shared_ptr<storm::logic::Formula const>> args = parseBinaryFormulaArguments(model, propertyStructure, formulaContext, opString, scope);
534  assert(args.size() == 2);
535  storm::logic::BinaryBooleanStateFormula::OperatorType oper = opString == "" ? storm::logic::BinaryBooleanStateFormula::OperatorType::And : storm::logic::BinaryBooleanStateFormula::OperatorType::Or;
536  return std::make_shared<storm::logic::BinaryBooleanStateFormula const>(oper, args[0], args[1]);
537  } else if (opString == "") {
538  assert(bound == boost::none);
539  std::vector<std::shared_ptr<storm::logic::Formula const>> args = parseBinaryFormulaArguments(model, propertyStructure, formulaContext, opString, scope);
540  assert(args.size() == 2);
541  std::shared_ptr<storm::logic::UnaryBooleanStateFormula const> tmp = std::make_shared<storm::logic::UnaryBooleanStateFormula const>(storm::logic::UnaryBooleanStateFormula::OperatorType::Not, args[0]);
542  return std::make_shared<storm::logic::BinaryBooleanStateFormula const>(storm::logic::BinaryBooleanStateFormula::OperatorType::Or, tmp, args[1]);
543  } else if (opString == "") {
544  assert(bound == boost::none);
545  std::vector<std::shared_ptr<storm::logic::Formula const>> args = parseUnaryFormulaArgument(model, propertyStructure, formulaContext, opString, scope);
546  assert(args.size() == 1);
547  return std::make_shared<storm::logic::UnaryBooleanStateFormula const>(storm::logic::UnaryBooleanStateFormula::OperatorType::Not, args[0]);
548  } else if (!expr.isInitialized() && (opString == "" || opString == "" || opString == "<" || opString == ">" || opString == "=" || opString == "")) {
549  assert(bound == boost::none);
550  storm::logic::ComparisonType ct;
551  if(opString == "") {
552  ct = storm::logic::ComparisonType::GreaterEqual;
553  } else if (opString == "") {
554  ct = storm::logic::ComparisonType::LessEqual;
555  } else if (opString == "<") {
556  ct = storm::logic::ComparisonType::Less;
557  } else if (opString == ">") {
558  ct = storm::logic::ComparisonType::Greater;
559  }
560 
561  std::vector<std::string> const leftRight = {"left", "right"};
562  for (uint64_t i = 0; i < 2; ++i) {
563  if (propertyStructure.at(leftRight[i]).count("op") > 0) {
564  std::string propertyOperatorString = getString<ValueType>(propertyStructure.at(leftRight[i]).at("op"), "property-operator");
565  std::set<std::string> const propertyOperatorStrings = {"Pmin", "Pmax","Emin", "Emax", "Smin", "Smax"};
566  if (propertyOperatorStrings.count(propertyOperatorString) > 0) {
567  auto boundExpr = parseExpression(propertyStructure.at(leftRight[1-i]), scope.refine("Threshold for operator " + propertyStructure.at(leftRight[i]).at("op").template get<std::string>()));
568  if ((opString == "=" || opString == "")) {
569  STORM_LOG_THROW(!boundExpr.containsVariables(), storm::exceptions::NotSupportedException, "Comparison operators '=' or '' in property specifications are currently not supported.");
570  auto boundValue = boundExpr.evaluateAsRational();
571  if (storm::utility::isZero(boundValue)) {
572  if (opString == "=") {
573  ct = storm::logic::ComparisonType::LessEqual;
574  } else {
575  ct = storm::logic::ComparisonType::Greater;
576  }
577  } else if (storm::utility::isOne(boundValue) && (propertyOperatorString == "Pmin" || propertyOperatorString == "Pmax")) {
578  if (opString == "=") {
579  ct = storm::logic::ComparisonType::GreaterEqual;
580  } else {
581  ct = storm::logic::ComparisonType::Less;
582  }
583  } else {
584  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Comparison operators '=' or '' in property specifications are currently not supported in " << scope.description << ".");
585  }
586  }
587  return parseFormula(model, propertyStructure.at(leftRight[i]), formulaContext, scope, storm::logic::Bound(ct, boundExpr));
588  }
589  }
590  }
591  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "No complex comparisons for properties are supported.");
592  } else if (expr.isInitialized()) {
593  STORM_LOG_THROW(false, storm::exceptions::InvalidJaniException, "Non-trivial Expression '" << expr << "' contains a boolean transient variable. Can not translate to PRCTL-like formula at " << scope.description << ".");
594  } else {
595  STORM_LOG_THROW(false, storm::exceptions::InvalidJaniException, "Unknown operator " << opString);
596  }
597  } else {
598  STORM_LOG_THROW(false, storm::exceptions::InvalidJaniException, "Looking for operator for formula " << propertyStructure.dump() << ", but did not find one");
599  }
600  }

602  template <typename ValueType>
603  storm::jani::Property JaniParser<ValueType>::parseProperty(storm::jani::Model& model, Json const& propertyStructure, Scope const& scope) {
604  STORM_LOG_THROW(propertyStructure.count("name") == 1, storm::exceptions::InvalidJaniException, "Property must have a name");
605  // TODO check unique name
606  std::string name = getString<ValueType>(propertyStructure.at("name"), "property-name");
607  STORM_LOG_TRACE("Parsing property named: " << name);
608  std::string comment = "";
609  if (propertyStructure.count("comment") > 0) {
610  comment = getString<ValueType>(propertyStructure.at("comment"), "comment for property named '" + name + "'.");
611  }
612  STORM_LOG_THROW(propertyStructure.count("expression") == 1, storm::exceptions::InvalidJaniException, "Property must have an expression");
613  // Parse filter expression.
614  Json const& expressionStructure = propertyStructure.at("expression");
615 
616  STORM_LOG_THROW(expressionStructure.count("op") == 1, storm::exceptions::InvalidJaniException, "Expression in property must have an operation description");
617  STORM_LOG_THROW(expressionStructure.at("op") == "filter", storm::exceptions::InvalidJaniException, "Top level operation of a property must be a filter");
618  STORM_LOG_THROW(expressionStructure.count("fun") == 1, storm::exceptions::InvalidJaniException, "Filter must have a function descritpion");
619  std::string funDescr = getString<ValueType>(expressionStructure.at("fun"), "Filter function in property named " + name);
620  storm::modelchecker::FilterType ft;
621  if (funDescr == "min") {
622  ft = storm::modelchecker::FilterType::MIN;
623  } else if (funDescr == "max") {
624  ft = storm::modelchecker::FilterType::MAX;
625  } else if (funDescr == "sum") {
626  ft = storm::modelchecker::FilterType::SUM;
627  } else if (funDescr == "avg") {
628  ft = storm::modelchecker::FilterType::AVG;
629  } else if (funDescr == "count") {
630  ft = storm::modelchecker::FilterType::COUNT;
631  } else if (funDescr == "") {
632  ft = storm::modelchecker::FilterType::FORALL;
633  } else if (funDescr == "") {
634  ft = storm::modelchecker::FilterType::EXISTS;
635  } else if (funDescr == "argmin") {
636  ft = storm::modelchecker::FilterType::ARGMIN;
637  } else if (funDescr == "argmax") {
638  ft = storm::modelchecker::FilterType::ARGMAX;
639  } else if (funDescr == "values") {
640  ft = storm::modelchecker::FilterType::VALUES;
641  } else {
642  STORM_LOG_THROW(false, storm::exceptions::InvalidJaniException, "Unknown filter description " << funDescr << " in property named " << name);
643  }
644 
645  STORM_LOG_THROW(expressionStructure.count("states") == 1, storm::exceptions::InvalidJaniException, "Filter must have a states description");
646  std::shared_ptr<storm::logic::Formula const> statesFormula;
647  if (expressionStructure.at("states").count("op") > 0) {
648  std::string statesDescr = getString<ValueType>(expressionStructure.at("states").at("op"), "Filtered states in property named " + name);
649  if (statesDescr == "initial") {
650  statesFormula = std::make_shared<storm::logic::AtomicLabelFormula>("init");
651  }
652  }
653  if (!statesFormula) {
654  try {
655  // Try to parse the states as formula.
656  statesFormula = parseFormula(model, expressionStructure.at("states"), storm::logic::FormulaContext::Undefined, scope.refine("Values of property " + name));
657  } catch (storm::exceptions::NotSupportedException const& ex) {
658  throw ex;
659  } catch (storm::exceptions::NotImplementedException const& ex) {
660  throw ex;
661  }
662  }
663  STORM_LOG_THROW(statesFormula, storm::exceptions::NotImplementedException, "Could not derive states formula.");
664  STORM_LOG_THROW(expressionStructure.count("values") == 1, storm::exceptions::InvalidJaniException, "Values as input for a filter must be given");
665  auto formula = parseFormula(model, expressionStructure.at("values"), storm::logic::FormulaContext::Undefined, scope.refine("Values of property " + name));
666  return storm::jani::Property(name, storm::jani::FilterExpression(formula, ft, statesFormula), {}, comment);
667  }

669  template <typename ValueType>
670  std::shared_ptr<storm::jani::Constant> JaniParser<ValueType>::parseConstant(Json const& constantStructure, Scope const& scope) {
671  STORM_LOG_THROW(constantStructure.count("name") == 1, storm::exceptions::InvalidJaniException, "Variable (scope: " + scope.description + ") must have a name");
672  std::string name = getString<ValueType>(constantStructure.at("name"), "variable-name in " + scope.description + "-scope");
673  // TODO check existance of name.
674  // TODO store prefix in variable.
675  std::string exprManagerName = name;
676  size_t valueCount = constantStructure.count("value");
677  storm::expressions::Expression definingExpression;
678  STORM_LOG_THROW(valueCount < 2, storm::exceptions::InvalidJaniException, "Value for constant '" + name + "' (scope: " + scope.description + ") must be given at most once.");
679  if (valueCount == 1) {
680  // Read initial value before; that makes creation later on a bit easier, and has as an additional benefit that we do not need to check whether the variable occurs also on the assignment.
681  definingExpression = parseExpression(constantStructure.at("value"), scope.refine("Value of constant " + name));
682  assert(definingExpression.isInitialized());
683  }
684 
685  STORM_LOG_THROW(constantStructure.count("type") == 1, storm::exceptions::InvalidJaniException, "Constant '" + name + "' (scope: " + scope.description + ") must have a (single) type-declaration.");
686  ParsedType type;
687  parseType(type, constantStructure.at("type"), name, scope);
688 
689  STORM_LOG_THROW(type.basicType.is_initialized(), storm::exceptions::InvalidJaniException, "Constant '" + name + "' (scope: " + scope.description + ") has unexpected type");
690  storm::expressions::Variable var;
691  switch (type.basicType.get()) {
692  case ParsedType::BasicType::Bool:
693  var = expressionManager->declareBooleanVariable(exprManagerName);
694  break;
695  case ParsedType::BasicType::Int:
696  var = expressionManager->declareIntegerVariable(exprManagerName);
697  break;
698  case ParsedType::BasicType::Real:
699  var = expressionManager->declareRationalVariable(exprManagerName);
700  break;
701  }
702  storm::expressions::Expression constraintExpression;
703  if (type.bounds) {
704  if (type.bounds->first.isInitialized()) {
705  constraintExpression = var.getExpression() >= type.bounds->first;
706  if (type.bounds->second.isInitialized()) {
707  constraintExpression = constraintExpression && (var.getExpression() <= type.bounds->second);
708  }
709  } else if (type.bounds->second.isInitialized()) {
710  constraintExpression = var.getExpression() <= type.bounds->second;
711  }
712  }
713 
714  return std::make_shared<storm::jani::Constant>(name, std::move(var), definingExpression, constraintExpression);
715  }

717  template <typename ValueType>
718  void JaniParser<ValueType>::parseType(ParsedType& result, Json const& typeStructure, std::string variableName, Scope const& scope) {
719  if (typeStructure.is_string()) {
720  if (typeStructure == "real") {
721  result.basicType = ParsedType::BasicType::Real;
722  result.expressionType = expressionManager->getRationalType();
723  } else if (typeStructure == "bool") {
724  result.basicType = ParsedType::BasicType::Bool;
725  result.expressionType = expressionManager->getBooleanType();
726  } else if (typeStructure == "int") {
727  result.basicType = ParsedType::BasicType::Int;
728  result.expressionType = expressionManager->getIntegerType();
729  } else if(typeStructure == "clock") {
730  STORM_LOG_THROW(false, storm::exceptions::InvalidJaniException, "Unsupported type 'clock' for variable '" << variableName << "' (scope: " << scope.description << ").");
731  } else if(typeStructure == "continuous") {
732  STORM_LOG_THROW(false, storm::exceptions::InvalidJaniException, "Unsupported type 'continuous' for variable ''" << variableName << "' (scope: " << scope.description << ").");
733  } else {
734  STORM_LOG_THROW(false, storm::exceptions::InvalidJaniException, "Unsupported type " << typeStructure.dump() << " for variable '" << variableName << "' (scope: " << scope.description << ").");
735  }
736  } else if (typeStructure.is_object()) {
737  STORM_LOG_THROW(typeStructure.count("kind") == 1, storm::exceptions::InvalidJaniException, "For complex type as in variable " << variableName << "(scope: " << scope.description << ") kind must be given");
738  std::string kind = getString<ValueType>(typeStructure.at("kind"), "kind for complex type as in variable " + variableName + "(scope: " + scope.description + ") ");
739  if (kind == "bounded") {
740  STORM_LOG_THROW(typeStructure.count("lower-bound") + typeStructure.count("upper-bound") > 0, storm::exceptions::InvalidJaniException, "For bounded type as in variable " << variableName << "(scope: " << scope.description << ") lower-bound or upper-bound must be given");
741  storm::expressions::Expression lowerboundExpr;
742  if (typeStructure.count("lower-bound") > 0) {
743  lowerboundExpr = parseExpression(typeStructure.at("lower-bound"), scope.refine("Lower bound for variable " + variableName));
744  }
745  storm::expressions::Expression upperboundExpr;
746  if (typeStructure.count("upper-bound") > 0) {
747  upperboundExpr = parseExpression(typeStructure.at("upper-bound"), scope.refine("Upper bound for variable "+ variableName));
748  }
749  STORM_LOG_THROW(typeStructure.count("base") == 1, storm::exceptions::InvalidJaniException, "For bounded type as in variable " << variableName << "(scope: " << scope.description << ") base must be given");
750  std::string basictype = getString<ValueType>(typeStructure.at("base"), "base for bounded type as in variable " + variableName + "(scope: " + scope.description + ") ");
751  if (basictype == "int") {
752  STORM_LOG_THROW(!lowerboundExpr.isInitialized() || lowerboundExpr.hasIntegerType(), storm::exceptions::InvalidJaniException, "Lower bound for bounded integer variable " << variableName << "(scope: " << scope.description << ") must be integer-typed");
753  STORM_LOG_THROW(!upperboundExpr.isInitialized() || upperboundExpr.hasIntegerType(), storm::exceptions::InvalidJaniException, "Upper bound for bounded integer variable " << variableName << "(scope: " << scope.description << ") must be integer-typed");
754  if (lowerboundExpr.isInitialized() && upperboundExpr.isInitialized() && !lowerboundExpr.containsVariables() && !upperboundExpr.containsVariables()) {
755  STORM_LOG_THROW(lowerboundExpr.evaluateAsInt() <= upperboundExpr.evaluateAsInt(), storm::exceptions::InvalidJaniException, "Lower bound must not be larger than upper bound for bounded integer variable " << variableName << "(scope: " << scope.description << ").");
756  }
757  result.basicType = ParsedType::BasicType::Int;
758  result.expressionType = expressionManager->getIntegerType();
759  result.bounds = std::make_pair(lowerboundExpr, upperboundExpr);
760  } else if (basictype == "real") {
761  STORM_LOG_THROW(!lowerboundExpr.isInitialized() || lowerboundExpr.hasNumericalType(), storm::exceptions::InvalidJaniException, "Lower bound for bounded real variable " << variableName << "(scope: " << scope.description << ") must be numeric");
762  STORM_LOG_THROW(!upperboundExpr.isInitialized() || upperboundExpr.hasNumericalType(), storm::exceptions::InvalidJaniException, "Upper bound for bounded real variable " << variableName << "(scope: " << scope.description << ") must be numeric");
763  if (lowerboundExpr.isInitialized() && upperboundExpr.isInitialized() && !lowerboundExpr.containsVariables() && !upperboundExpr.containsVariables()) {
764  STORM_LOG_THROW(lowerboundExpr.evaluateAsRational() <= upperboundExpr.evaluateAsRational(), storm::exceptions::InvalidJaniException, "Lower bound must not be larger than upper bound for bounded integer variable " << variableName << "(scope: " << scope.description << ").");
765  }
766  result.basicType = ParsedType::BasicType::Real;
767  result.expressionType = expressionManager->getRationalType();
768  result.bounds = std::make_pair(lowerboundExpr, upperboundExpr);
769  } else {
770  STORM_LOG_THROW(false, storm::exceptions::InvalidJaniException, "Unsupported base " << basictype << " for bounded variable " << variableName << "(scope: " << scope.description << ").");
771  }
772  } else if (kind == "array") {
773  STORM_LOG_THROW(typeStructure.count("base") == 1, storm::exceptions::InvalidJaniException, "For array type as in variable " << variableName << "(scope: " << scope.description << ") base must be given");
774  result.arrayBase = std::make_unique<ParsedType>();
775  parseType(*result.arrayBase, typeStructure.at("base"), variableName, scope);
776  result.expressionType = expressionManager->getArrayType(result.arrayBase->expressionType);
777  } else {
778  STORM_LOG_THROW(false, storm::exceptions::InvalidJaniException, "Unsupported kind " << kind << " for complex type of variable " << variableName << "(scope: " << scope.description << ").");
779  }
780  }

789  parseType(type, functionDefinitionStructure.at("type"), functionName, scope);
790 
791  std::unordered_map<std::string, storm::expressions::Variable> parameterNameToVariableMap;
792  std::vector<storm::expressions::Variable> parameters;
793  if (!firstPass && functionDefinitionStructure.count("parameters") > 0) {
794  STORM_LOG_THROW(functionDefinitionStructure.count("parameters") == 1, storm::exceptions::InvalidJaniException, "Function definition '" + functionName + "' (scope: " + scope.description + ") must have exactly one list of parameters.");
795  for (auto const& parameterStructure : functionDefinitionStructure.at("parameters")) {
796  STORM_LOG_THROW(parameterStructure.count("name") == 1, storm::exceptions::InvalidJaniException, "Parameter declaration of parameter " + std::to_string(parameters.size()) + " of Function definition '" + functionName + "' (scope: " + scope.description + ") must have a name");

799  STORM_LOG_THROW(parameterStructure.count("type") == 1, storm::exceptions::InvalidJaniException, "Parameter declaration of parameter " + std::to_string(parameters.size()) + " of Function definition '" + functionName + "' (scope: " + scope.description + ") must have exactly one type.");

802 
803  std::string exprParameterName = parameterNamePrefix + functionName + VARIABLE_AUTOMATON_DELIMITER + parameterName;
804  parameters.push_back(expressionManager->declareVariable(exprParameterName, parameterType.expressionType));
805  parameterNameToVariableMap.emplace(parameterName, parameters.back());
806  }
807  }
808 
809  STORM_LOG_THROW(functionDefinitionStructure.count("body") == 1, storm::exceptions::InvalidJaniException, "Function definition '" + functionName + "' (scope: " + scope.description + ") must have a (single) body.");

811  if (!firstPass) {
812  functionBody = parseExpression(functionDefinitionStructure.at("body"), scope.refine("body of function definition " + functionName), false, parameterNameToVariableMap);
813  STORM_LOG_WARN_COND(functionBody.getType() == type.expressionType, "Type of body of function " + functionName + "' (scope: " + scope.description + ") has type " << functionBody.getType() << " although the function type is given as " << type.expressionType);
814  }
815  return storm::jani::FunctionDefinition(functionName, type.expressionType, parameters, functionBody);
816  }

819  template <typename ValueType>
820  std::shared_ptr<storm::jani::Variable> JaniParser<ValueType>::parseVariable(Json const& variableStructure, bool requireInitialValues, Scope const& scope, std::string const& namePrefix) {
821  STORM_LOG_THROW(variableStructure.count("name") == 1, storm::exceptions::InvalidJaniException, "Variable (scope: " + scope.description + ") must have a name");
822  std::string name = getString<ValueType>(variableStructure.at("name"), "variable-name in " + scope.description + "-scope");
823  // TODO check existance of name.
824  // TODO store prefix in variable.
825  std::string exprManagerName = namePrefix + name;
826  bool transientVar = defaultVariableTransient; // Default value for variables.
827  size_t tvarcount = variableStructure.count("transient");
828  STORM_LOG_THROW(tvarcount <= 1, storm::exceptions::InvalidJaniException, "Multiple definitions of transient not allowed in variable '" + name + "' (scope: " + scope.description + ").");
829  if(tvarcount == 1) {
830  transientVar = getBoolean<ValueType>(variableStructure.at("transient"), "transient-attribute in variable '" + name + "' (scope: " + scope.description + ").");
831  }
832  STORM_LOG_THROW(variableStructure.count("type") == 1, storm::exceptions::InvalidJaniException, "Variable '" + name + "' (scope: " + scope.description + ") must have a (single) type-declaration.");
833  ParsedType type;
834  parseType(type, variableStructure.at("type"), name, scope);
835 
836  size_t initvalcount = variableStructure.count("initial-value");
837  if(transientVar) {
838  STORM_LOG_THROW(initvalcount == 1, storm::exceptions::InvalidJaniException, "Initial value must be given once for transient variable '" + name + "' (scope: " + scope.description + ") "+ name + "' (scope: " + scope.description + ").");
839  } else {
840  STORM_LOG_THROW(initvalcount <= 1, storm::exceptions::InvalidJaniException, "Initial value can be given at most one for variable " + name + "' (scope: " + scope.description + ").");
841  }
842  boost::optional<storm::expressions::Expression> initVal;
843  if (initvalcount == 1 && !variableStructure.at("initial-value").is_null()) {
844  initVal = parseExpression(variableStructure.at("initial-value"), scope.refine("Initial value for variable " + name));
845  } else {
846  assert(!transientVar);
847  }
848 
849  bool setInitValFromDefault = !initVal.is_initialized() && requireInitialValues;
850  if (type.basicType) {
851  switch (type.basicType.get()) {
852  case ParsedType::BasicType::Real:
853  STORM_LOG_WARN_COND(!type.bounds.is_initialized(), "Bounds for rational variable " + name + "(scope " + scope.description + ") will be ignored.");
854  if (setInitValFromDefault) {
855  initVal = expressionManager->rational(defaultRationalInitialValue);
856  }
857  if (initVal) {
858  STORM_LOG_THROW(initVal.get().hasRationalType() || initVal.get().hasIntegerType(), storm::exceptions::InvalidJaniException, "Initial value for rational variable " + name + "(scope " + scope.description + ") should be a rational");
859  return std::make_shared<storm::jani::RealVariable>(name, expressionManager->declareRationalVariable(exprManagerName), initVal.get(), transientVar);
860  } else {
861  return std::make_shared<storm::jani::RealVariable>(name, expressionManager->declareRationalVariable(exprManagerName));
862  }
863  case ParsedType::BasicType::Int:
864  if (setInitValFromDefault) {
865  if (type.bounds) {
866  storm::expressions::Expression takeDefaultCondition;
867  if (type.bounds->first.isInitialized()) {
868  takeDefaultCondition = type.bounds->first < defaultIntegerInitialValue;
869  if (type.bounds->second.isInitialized()) {
870  takeDefaultCondition = takeDefaultCondition && type.bounds->second >= defaultIntegerInitialValue;
871  }
872  } else {
873  STORM_LOG_ASSERT(type.bounds->second.isInitialized(), "Expected to have either a lower or an upper bound");
874  takeDefaultCondition = type.bounds->second >= defaultIntegerInitialValue;
875  }
876  initVal = storm::expressions::ite(takeDefaultCondition, expressionManager->integer(defaultIntegerInitialValue), type.bounds->first);
877  } else {
878  initVal = expressionManager->integer(defaultIntegerInitialValue);
879  }
880  }
881  if (initVal) {
882  STORM_LOG_THROW(initVal.get().hasIntegerType(), storm::exceptions::InvalidJaniException, "Initial value for integer variable " + name + "(scope " + scope.description + ") should be an integer");
883  if (type.bounds) {
884  return storm::jani::makeBoundedIntegerVariable(name, expressionManager->declareIntegerVariable(exprManagerName), initVal, transientVar, type.bounds->first, type.bounds->second);
885  } else {
886  return std::make_shared<storm::jani::UnboundedIntegerVariable>(name, expressionManager->declareIntegerVariable(exprManagerName), initVal.get(), transientVar);
887  }
888  } else {
889  if (type.bounds) {
890  return storm::jani::makeBoundedIntegerVariable(name, expressionManager->declareIntegerVariable(exprManagerName), boost::none, false, type.bounds->first, type.bounds->second);
891  } else {
892  return std::make_shared<storm::jani::UnboundedIntegerVariable>(name, expressionManager->declareIntegerVariable(exprManagerName));
893  }
894  }
895  break;
896  case ParsedType::BasicType::Bool:
897  if (setInitValFromDefault) {
898  initVal = expressionManager->boolean(defaultBooleanInitialValue);
899  }
900  if (initVal) {
901  STORM_LOG_THROW(initVal.get().hasBooleanType(), storm::exceptions::InvalidJaniException, "Initial value for boolean variable " + name + "(scope " + scope.description + ") should be a Boolean");
902  if (transientVar) {
903  labels.insert(name);
904  }
905  return std::make_shared<storm::jani::BooleanVariable>(name, expressionManager->declareBooleanVariable(exprManagerName), initVal.get(), transientVar);
906  } else {
907  return std::make_shared<storm::jani::BooleanVariable>(name, expressionManager->declareBooleanVariable(exprManagerName));
908  }
909  }
910  } else if (type.arrayBase) {
911  STORM_LOG_THROW(type.arrayBase->basicType, storm::exceptions::InvalidJaniException, "Array base type for variable " + name + "(scope " + scope.description + ") should be a BasicType or a BoundedType.");
912  storm::jani::ArrayVariable::ElementType elementType;
913  storm::expressions::Type exprVariableType = type.expressionType;
914  switch (type.arrayBase->basicType.get()) {
915  case ParsedType::BasicType::Real:
916  elementType = storm::jani::ArrayVariable::ElementType::Real;
917  break;
918  case ParsedType::BasicType::Bool:
919  elementType = storm::jani::ArrayVariable::ElementType::Bool;
920  break;
921  case ParsedType::BasicType::Int:
922  elementType = storm::jani::ArrayVariable::ElementType::Int;
923  break;
924  default:
925  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Unsupported type");
926  }
927  if (setInitValFromDefault) {
928  initVal = storm::expressions::ValueArrayExpression(*expressionManager, exprVariableType, {}).toExpression();
929  }
930  std::shared_ptr<storm::jani::ArrayVariable> result;
931  if (initVal) {
932  STORM_LOG_THROW(initVal->getType().isArrayType(), storm::exceptions::InvalidJaniException, "Initial value for array variable " + name + "(scope " + scope.description + ") should be an Array");
933  result = std::make_shared<storm::jani::ArrayVariable>(name, expressionManager->declareArrayVariable(exprManagerName, exprVariableType.getElementType()), elementType, initVal.get(), transientVar);
934  } else {
935  result = std::make_shared<storm::jani::ArrayVariable>(name, expressionManager->declareArrayVariable(exprManagerName, exprVariableType.getElementType()), elementType);
936  }
937  if (type.arrayBase->bounds) {
938  auto const& bounds = type.arrayBase->bounds.get();
939  if (bounds.first.isInitialized()) {
940  result->setLowerElementTypeBound(bounds.first);
941  }
942  if (bounds.second.isInitialized()) {
943  result->setUpperElementTypeBound(bounds.second);
944  }
945  }
946  return result;
947  }
948 
949  STORM_LOG_THROW(false, storm::exceptions::InvalidJaniException, "Unknown type description, " << variableStructure.at("type").dump() << " for variable '" << name << "' (scope: " << scope.description << ").");
950  }

1059  template <typename ValueType>
1060  storm::expressions::Expression JaniParser<ValueType>::parseExpression(Json const& expressionStructure, Scope const& scope, bool returnNoneInitializedOnUnknownOperator, std::unordered_map<std::string, storm::expressions::Variable> const& auxiliaryVariables) {
1061  if (expressionStructure.is_boolean()) {
1062  if (expressionStructure.template get<bool>()) {
1063  return expressionManager->boolean(true);
1064  } else {
1065  return expressionManager->boolean(false);
1066  }
1067  } else if (expressionStructure.is_number_integer()) {
1068  return expressionManager->integer(expressionStructure.template get<int64_t>());
1069  } else if (expressionStructure.is_number_float()) {
1070  return expressionManager->rational(storm::utility::convertNumber<storm::RationalNumber>(expressionStructure.template get<ValueType>()));
1071  } else if (expressionStructure.is_string()) {
1072  std::string ident = expressionStructure.template get<std::string>();
1073  return storm::expressions::Expression(getVariableOrConstantExpression(ident, scope, auxiliaryVariables));
1074  } else if (expressionStructure.is_object()) {
1075  if (expressionStructure.count("distribution") == 1) {
1076  STORM_LOG_THROW(false, storm::exceptions::InvalidJaniException, "Distributions are not supported by storm expressions, cannot import " << expressionStructure.dump() << " in " << scope.description << ".");
1077  }
1078  if (expressionStructure.count("op") == 1) {
1079  std::string opstring = getString<ValueType>(expressionStructure.at("op"), scope.description);
1080  std::vector<storm::expressions::Expression> arguments = {};
1081  if(opstring == "ite") {
1082  STORM_LOG_THROW(expressionStructure.count("if") == 1, storm::exceptions::InvalidJaniException, "If operator required");
1083  STORM_LOG_THROW(expressionStructure.count("else") == 1, storm::exceptions::InvalidJaniException, "Else operator required");
1084  STORM_LOG_THROW(expressionStructure.count("then") == 1, storm::exceptions::InvalidJaniException, "Then operator required");
1085  arguments.push_back(parseExpression(expressionStructure.at("if"), scope.refine("if-formula"), returnNoneInitializedOnUnknownOperator, auxiliaryVariables));
1086  arguments.push_back(parseExpression(expressionStructure.at("then"), scope.refine("then-formula"), returnNoneInitializedOnUnknownOperator, auxiliaryVariables));
1087  arguments.push_back(parseExpression(expressionStructure.at("else"), scope.refine("else-formula"), returnNoneInitializedOnUnknownOperator, auxiliaryVariables));
1088  ensureNumberOfArguments(3, arguments.size(), opstring, scope.description);
1089  assert(arguments.size() == 3);
1090  ensureBooleanType(arguments[0], opstring, 0, scope.description);
1091  return storm::expressions::ite(arguments[0], arguments[1], arguments[2]);
1092  } else if (opstring == "") {
1093  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1094  assert(arguments.size() == 2);
1095  if(!arguments[0].isInitialized() || !arguments[1].isInitialized()) {
1096  return storm::expressions::Expression();
1097  }
1098  ensureBooleanType(arguments[0], opstring, 0, scope.description);
1099  ensureBooleanType(arguments[1], opstring, 1, scope.description);
1100  return arguments[0] || arguments[1];
1101  } else if (opstring == "") {
1102  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1103  assert(arguments.size() == 2);
1104  if(!arguments[0].isInitialized() || !arguments[1].isInitialized()) {
1105  return storm::expressions::Expression();
1106  }
1107  ensureBooleanType(arguments[0], opstring, 0, scope.description);
1108  ensureBooleanType(arguments[1], opstring, 1, scope.description);
1109  return arguments[0] && arguments[1];
1110  } else if (opstring == "") {
1111  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1112  assert(arguments.size() == 2);
1113  if(!arguments[0].isInitialized() || !arguments[1].isInitialized()) {
1114  return storm::expressions::Expression();
1115  }
1116  ensureBooleanType(arguments[0], opstring, 0, scope.description);
1117  ensureBooleanType(arguments[1], opstring, 1, scope.description);
1118  return (!arguments[0]) || arguments[1];
1119  } else if (opstring == "") {
1120  arguments = parseUnaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1121  assert(arguments.size() == 1);
1122  if(!arguments[0].isInitialized()) {
1123  return storm::expressions::Expression();
1124  }
1125  ensureBooleanType(arguments[0], opstring, 0, scope.description);
1126  return !arguments[0];
1127  } else if (opstring == "=") {
1128  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1129  assert(arguments.size() == 2);
1130  if(!arguments[0].isInitialized() || !arguments[1].isInitialized()) {
1131  return storm::expressions::Expression();
1132  }
1133  if(arguments[0].hasBooleanType()) {
1134  ensureBooleanType(arguments[1], opstring, 1, scope.description);
1135  return storm::expressions::iff(arguments[0], arguments[1]);
1136  } else {
1137  ensureNumericalType(arguments[1], opstring, 1, scope.description);
1138  return arguments[0] == arguments[1];
1139  }
1140  } else if (opstring == "") {
1141  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1142  assert(arguments.size() == 2);
1143  if(!arguments[0].isInitialized() || !arguments[1].isInitialized()) {
1144  return storm::expressions::Expression();
1145  }
1146  if(arguments[0].hasBooleanType()) {
1147  ensureBooleanType(arguments[1], opstring, 1, scope.description);
1148  return storm::expressions::xclusiveor(arguments[0], arguments[1]);
1149  } else {
1150  ensureNumericalType(arguments[1], opstring, 1, scope.description);
1151  return arguments[0] != arguments[1];
1152  }
1153  } else if (opstring == "<") {
1154  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1155  assert(arguments.size() == 2);
1156  if(!arguments[0].isInitialized() || !arguments[1].isInitialized()) {
1157  return storm::expressions::Expression();
1158  }
1159  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1160  ensureNumericalType(arguments[1], opstring, 1, scope.description);
1161  return arguments[0] < arguments[1];
1162  } else if (opstring == "") {
1163  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1164  assert(arguments.size() == 2);
1165  if(!arguments[0].isInitialized() || !arguments[1].isInitialized()) {
1166  return storm::expressions::Expression();
1167  }
1168  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1169  ensureNumericalType(arguments[1], opstring, 1, scope.description);
1170  return arguments[0] <= arguments[1];
1171  } else if (opstring == ">") {
1172  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1173  assert(arguments.size() == 2);
1174  if(!arguments[0].isInitialized() || !arguments[1].isInitialized()) {
1175  return storm::expressions::Expression();
1176  }
1177  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1178  ensureNumericalType(arguments[1], opstring, 1, scope.description);
1179  return arguments[0] > arguments[1];
1180  } else if (opstring == "") {
1181  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1182  assert(arguments.size() == 2);
1183  if(!arguments[0].isInitialized() || !arguments[1].isInitialized()) {
1184  return storm::expressions::Expression();
1185  }
1186  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1187  ensureNumericalType(arguments[1], opstring, 1, scope.description);
1188  return arguments[0] >= arguments[1];
1189  } else if (opstring == "+") {
1190  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1191  assert(arguments.size() == 2);
1192  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1193  ensureNumericalType(arguments[1], opstring, 1, scope.description);
1194  return arguments[0] + arguments[1];
1195  } else if (opstring == "-" && expressionStructure.count("left") > 0) {
1196  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1197  assert(arguments.size() == 2);
1198  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1199  ensureNumericalType(arguments[1], opstring, 1, scope.description);
1200  return arguments[0] - arguments[1];
1201  } else if (opstring == "-") {
1202  arguments = parseUnaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1203  assert(arguments.size() == 1);
1204  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1205  return -arguments[0];
1206  } else if (opstring == "*") {
1207  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1208  assert(arguments.size() == 2);
1209  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1210  ensureNumericalType(arguments[1], opstring, 1, scope.description);
1211  return arguments[0] * arguments[1];
1212  } else if (opstring == "/") {
1213  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1214  assert(arguments.size() == 2);
1215  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1216  ensureNumericalType(arguments[1], opstring, 1, scope.description);
1217  return arguments[0] / arguments[1];
1218  } else if (opstring == "%") {
1219  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1220  assert(arguments.size() == 2);
1221  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1222  ensureNumericalType(arguments[1], opstring, 1, scope.description);
1223  return arguments[0] % arguments[1];
1224  } else if (opstring == "max") {
1225  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1226  assert(arguments.size() == 2);
1227  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1228  ensureNumericalType(arguments[1], opstring, 1, scope.description);
1229  return storm::expressions::maximum(arguments[0],arguments[1]);
1230  } else if (opstring == "min") {
1231  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1232  assert(arguments.size() == 2);
1233  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1234  ensureNumericalType(arguments[1], opstring, 1, scope.description);
1235  return storm::expressions::minimum(arguments[0],arguments[1]);
1236  } else if (opstring == "floor") {
1237  arguments = parseUnaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1238  assert(arguments.size() == 1);
1239  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1240  return storm::expressions::floor(arguments[0]);
1241  } else if (opstring == "ceil") {
1242  arguments = parseUnaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1243  assert(arguments.size() == 1);
1244  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1245  return storm::expressions::ceil(arguments[0]);
1246  } else if (opstring == "abs") {
1247  arguments = parseUnaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1248  assert(arguments.size() == 1);
1249  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1250  return storm::expressions::abs(arguments[0]);
1251  } else if (opstring == "sgn") {
1252  arguments = parseUnaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1253  assert(arguments.size() == 1);
1254  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1255  return storm::expressions::sign(arguments[0]);
1256  } else if (opstring == "trc") {
1257  arguments = parseUnaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1258  assert(arguments.size() == 1);
1259  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1260  return storm::expressions::truncate(arguments[0]);
1261  } else if (opstring == "pow") {
1262  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1263  assert(arguments.size() == 2);
1264  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1265  ensureNumericalType(arguments[1], opstring, 1, scope.description);
1266  return arguments[0]^arguments[1];
1267  } else if (opstring == "exp") {
1268  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1269  assert(arguments.size() == 2);
1270  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1271  ensureNumericalType(arguments[1], opstring, 1, scope.description);
1272  // TODO implement
1273  STORM_LOG_THROW(false, storm::exceptions::NotImplementedException, "exp operation is not yet implemented");
1274  } else if (opstring == "log") {
1275  arguments = parseBinaryExpressionArguments(expressionStructure, opstring, scope, returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1276  assert(arguments.size() == 2);
1277  ensureNumericalType(arguments[0], opstring, 0, scope.description);
1278  ensureNumericalType(arguments[1], opstring, 1, scope.description);
1279  // TODO implement
1280  STORM_LOG_THROW(false, storm::exceptions::NotImplementedException, "log operation is not yet implemented");
1281  } else if (opstring == "aa") {
1282  STORM_LOG_THROW(expressionStructure.count("exp") == 1, storm::exceptions::InvalidJaniException, "Array access operator requires exactly one exp (at " + scope.description + ").");
1283  storm::expressions::Expression exp = parseExpression(expressionStructure.at("exp"), scope.refine("'exp' of array access operator"), returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1284  STORM_LOG_THROW(expressionStructure.count("index") == 1, storm::exceptions::InvalidJaniException, "Array access operator requires exactly one index (at " + scope.description + ").");
1285  storm::expressions::Expression index = parseExpression(expressionStructure.at("index"), scope.refine("index of array access operator"), returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1286  ensureArrayType(exp, opstring, 0, scope.description);
1287  ensureIntegerType(index, opstring, 1, scope.description);
1288  return std::make_shared<storm::expressions::ArrayAccessExpression>(exp.getManager(), exp.getType().getElementType(), exp.getBaseExpressionPointer(), index.getBaseExpressionPointer())->toExpression();
1289  } else if (opstring == "av") {
1290  STORM_LOG_THROW(expressionStructure.count("elements") == 1, storm::exceptions::InvalidJaniException, "Array value operator requires exactly one 'elements' (at " + scope.description + ").");
1291  std::vector<std::shared_ptr<storm::expressions::BaseExpression const>> elements;
1292  storm::expressions::Type commonType;
1293  bool first = true;
1294  for (auto const& element : expressionStructure.at("elements")) {
1295  elements.push_back(parseExpression(element, scope.refine("element " + std::to_string(elements.size()) + " of array value expression"), returnNoneInitializedOnUnknownOperator, auxiliaryVariables).getBaseExpressionPointer());
1296  if (first) {
1297  commonType = elements.back()->getType();
1298  first = false;
1299  } else if (!(commonType == elements.back()->getType())) {
1300  if (commonType.isIntegerType() && elements.back()->getType().isRationalType()) {
1301  commonType = elements.back()->getType();
1302  } else {
1303  STORM_LOG_THROW(false, storm::exceptions::InvalidJaniException, "Incompatible element types " << commonType << " and " << elements.back()->getType() << " of array value expression at " << scope.description);
1304  }
1305  }
1306  }
1307  return std::make_shared<storm::expressions::ValueArrayExpression>(*expressionManager, expressionManager->getArrayType(commonType), elements)->toExpression();
1308  } else if (opstring == "ac") {
1309  STORM_LOG_THROW(expressionStructure.count("length") == 1, storm::exceptions::InvalidJaniException, "Array access operator requires exactly one length (at " + scope.description + ").");
1310  storm::expressions::Expression length = parseExpression(expressionStructure.at("length"), scope.refine("index of array constructor expression"), returnNoneInitializedOnUnknownOperator, auxiliaryVariables);
1311  ensureIntegerType(length, opstring, 1, scope.description);
1312  STORM_LOG_THROW(expressionStructure.count("var") == 1, storm::exceptions::InvalidJaniException, "Array access operator requires exactly one var (at " + scope.description + ").");
1313  std::string indexVarName = getString<ValueType>(expressionStructure.at("var"), "Field 'var' of Array access operator (at " + scope.description + ").");
1314  STORM_LOG_THROW(auxiliaryVariables.find(indexVarName) == auxiliaryVariables.end(), storm::exceptions::InvalidJaniException, "Index variable " << indexVarName << " is already defined as an auxiliary variable (at " + scope.description + ").");
1315  auto newAuxVars = auxiliaryVariables;
1316  storm::expressions::Variable indexVar = expressionManager->declareFreshIntegerVariable(false, "ac_" + indexVarName);
1317  newAuxVars.emplace(indexVarName, indexVar);
1318  STORM_LOG_THROW(expressionStructure.count("exp") == 1, storm::exceptions::InvalidJaniException, "Array constructor operator requires exactly one exp (at " + scope.description + ").");
1319  storm::expressions::Expression exp = parseExpression(expressionStructure.at("exp"), scope.refine("exp of array constructor"), returnNoneInitializedOnUnknownOperator, newAuxVars);
1320  return std::make_shared<storm::expressions::ConstructorArrayExpression>(*expressionManager, expressionManager->getArrayType(exp.getType()), length.getBaseExpressionPointer(), indexVar, exp.getBaseExpressionPointer())->toExpression();
1321  } else if (opstring == "call") {
1322  STORM_LOG_THROW(expressionStructure.count("function") == 1, storm::exceptions::InvalidJaniException, "Function call operator requires exactly one function (at " + scope.description + ").");
1323  std::string functionName = getString<ValueType>(expressionStructure.at("function"), "in function call operator (at " + scope.description + ").");
1324  storm::jani::FunctionDefinition const* functionDefinition;
1325  if (scope.localFunctions != nullptr && scope.localFunctions->count(functionName) > 0) {
1326  functionDefinition = scope.localFunctions->at(functionName);
1327  } else if (scope.globalFunctions != nullptr && scope.globalFunctions->count(functionName) > 0){
1328  functionDefinition = scope.globalFunctions->at(functionName);
1329  } else {
1330  STORM_LOG_THROW(false, storm::exceptions::InvalidJaniException, "Function call operator calls unknown function '" + functionName + "' (at " + scope.description + ").");
1331  }
1332  STORM_LOG_THROW(expressionStructure.count("args") == 1, storm::exceptions::InvalidJaniException, "Function call operator requires exactly one args (at " + scope.description + ").");
1333  std::vector<std::shared_ptr<storm::expressions::BaseExpression const>> args;
1334  if (expressionStructure.count("args") > 0) {
1335  STORM_LOG_THROW(expressionStructure.count("args") == 1, storm::exceptions::InvalidJaniException, "Function call operator requires exactly one args (at " + scope.description + ").");
1336  for (auto const& arg : expressionStructure.at("args")) {
1337  args.push_back(parseExpression(arg, scope.refine("argument " + std::to_string(args.size()) + " of function call expression"), returnNoneInitializedOnUnknownOperator, auxiliaryVariables).getBaseExpressionPointer());
1338  }
1339  }
1340  return std::make_shared<storm::expressions::FunctionCallExpression>(*expressionManager, functionDefinition->getType(), functionName, args)->toExpression();
1341  } else if (unsupportedOpstrings.count(opstring) > 0) {
1342  STORM_LOG_THROW(false, storm::exceptions::InvalidJaniException, "Opstring " + opstring + " is not supported by storm");
1343  } else {
1344  if(returnNoneInitializedOnUnknownOperator) {
1345  return storm::expressions::Expression();
1346  }
1347  STORM_LOG_THROW(false, storm::exceptions::InvalidJaniException, "Unknown operator " << opstring << " in " << scope.description << ".");
1348  }
1349  }
1350  STORM_LOG_THROW(false, storm::exceptions::InvalidJaniException, "No supported operator declaration found for complex expressions as " << expressionStructure.dump() << " in " << scope.description << ".");
1351  }
1352  STORM_LOG_THROW(false, storm::exceptions::InvalidJaniException, "No supported expression found at " << expressionStructure.dump() << " in " << scope.description << ".");
1353  // Silly warning suppression.
1354  return storm::expressions::Expression();
1355 
1356  }

1370  template <typename ValueType>
1371  storm::jani::Automaton JaniParser<ValueType>::parseAutomaton(Json const &automatonStructure, storm::jani::Model const& parentModel, Scope const& globalScope) {
1372  STORM_LOG_THROW(automatonStructure.count("name") == 1, storm::exceptions::InvalidJaniException, "Each automaton must have a name");
1373  std::string name = getString<ValueType>(automatonStructure.at("name"), " the name field for automaton");
1374  Scope scope = globalScope.refine(name);
1375  storm::jani::Automaton automaton(name, expressionManager->declareIntegerVariable("_loc_" + name));
1376 
1377  uint64_t varDeclCount = automatonStructure.count("variables");
1378  STORM_LOG_THROW(varDeclCount < 2, storm::exceptions::InvalidJaniException, "Automaton '" << name << "' has more than one list of variables");
1379  VariablesMap localVars;
1380  scope.localVars = &localVars;
1381  if (varDeclCount > 0) {
1382  bool requireInitialValues = automatonStructure.count("restrict-initial") == 0;
1383  for(auto const& varStructure : automatonStructure.at("variables")) {
1384  std::shared_ptr<storm::jani::Variable> var = parseVariable(varStructure, requireInitialValues, scope.refine("variables[" + std::to_string(localVars.size()) + "] of automaton " + name), name + VARIABLE_AUTOMATON_DELIMITER);
1385  assert(localVars.count(var->getName()) == 0);
1386  localVars.emplace(var->getName(), &automaton.addVariable(*var));
1387  }
1388  }
1389 
1390  uint64_t funDeclCount = automatonStructure.count("functions");
1391  STORM_LOG_THROW(funDeclCount < 2, storm::exceptions::InvalidJaniException, "Automaton '" << name << "' has more than one list of functions");
1392  FunctionsMap localFuns;
1393  scope.localFunctions = &localFuns;
1394  if (funDeclCount > 0) {
1395  // We require two passes through the function definitions array to allow referring to functions before they were defined.
1396  std::vector<storm::jani::FunctionDefinition> dummyFunctionDefinitions;
1397  for (auto const& funStructure : automatonStructure.at("functions")) {
1398  // Skip parsing of function body
1399  dummyFunctionDefinitions.push_back(parseFunctionDefinition(funStructure, scope.refine("functions[" + std::to_string(localFuns.size()) + "] of automaton " + name), true));
1400  }
1401  // Store references to the dummy function definitions. This needs to happen in a separate loop since otherwise, references to FunDefs can be invalidated after calling dummyFunctionDefinitions.push_back
1402  for (auto const& funDef : dummyFunctionDefinitions) {
1403  bool unused = localFuns.emplace(funDef.getName(), &funDef).second;
1404  STORM_LOG_THROW(unused, storm::exceptions::InvalidJaniException, "Multiple definitions of functions with the name " << funDef.getName() << " in " << scope.description);
1405  }
1406  for (auto const& funStructure : automatonStructure.at("functions")) {
1407  // Actually parse the function body
1408  storm::jani::FunctionDefinition funDef = parseFunctionDefinition(funStructure, scope.refine("functions[" + std::to_string(localFuns.size()) + "] of automaton " + name),
1409  false, name + VARIABLE_AUTOMATON_DELIMITER);
1410  assert(localFuns.count(funDef.getName()) == 1);
1411  localFuns[funDef.getName()] = &automaton.addFunctionDefinition(funDef);
1412  }
1413  }
1414 
1415  STORM_LOG_THROW(automatonStructure.count("locations") > 0, storm::exceptions::InvalidJaniException, "Automaton '" << name << "' does not have locations.");
1416  std::unordered_map<std::string, uint64_t> locIds;
1417  for(auto const& locEntry : automatonStructure.at("locations")) {
1418  STORM_LOG_THROW(locEntry.count("name") == 1, storm::exceptions::InvalidJaniException, "Locations for automaton '" << name << "' must have exactly one name");
1419  std::string locName = getString<ValueType>(locEntry.at("name"), "location of automaton " + name);
1420  STORM_LOG_THROW(locIds.count(locName) == 0, storm::exceptions::InvalidJaniException, "Location with name '" + locName + "' already exists in automaton '" + name + "'");
1421  STORM_LOG_THROW(locEntry.count("invariant") == 0, storm::exceptions::InvalidJaniException, "Invariants in locations as in '" + locName + "' in automaton '" + name + "' are not supported");
1422  //STORM_LOG_THROW(locEntry.count("invariant") > 0 && !supportsInvariants(parentModel.getModelType()), storm::exceptions::InvalidJaniException, "Invariants are not supported in the model type " + to_string(parentModel.getModelType()));
1423  std::vector<storm::jani::Assignment> transientAssignments;
1424  if(locEntry.count("transient-values") > 0) {
1425  for(auto const& transientValueEntry : locEntry.at("transient-values")) {
1426  STORM_LOG_THROW(transientValueEntry.count("ref") == 1, storm::exceptions::InvalidJaniException, "Transient values in location " << locName << " need exactly one ref that is assigned to");
1427  STORM_LOG_THROW(transientValueEntry.count("value") == 1, storm::exceptions::InvalidJaniException, "Transient values in location " << locName << " need exactly one assigned value");
1428  storm::jani::LValue lValue = parseLValue(transientValueEntry.at("ref"), scope.refine("LHS of assignment in location " + locName));
1429  STORM_LOG_THROW(lValue.isTransient(), storm::exceptions::InvalidJaniException, "Assigned non-transient variable " << lValue << " in location " + locName + " (automaton: '" + name + "').");
1430  storm::expressions::Expression rhs = parseExpression(transientValueEntry.at("value"), scope.refine("Assignment of lValue in location " + locName));
1431  transientAssignments.emplace_back(lValue, rhs);
1432  }
1433  }
1434  uint64_t id = automaton.addLocation(storm::jani::Location(locName, transientAssignments));
1435  locIds.emplace(locName, id);
1436  }
1437  STORM_LOG_THROW(automatonStructure.count("initial-locations") == 1, storm::exceptions::InvalidJaniException, "Automaton '" << name << "' does not have initial locations.");
1438  for(Json const& initLocStruct : automatonStructure.at("initial-locations")) {
1439  automaton.addInitialLocation(getString<ValueType>(initLocStruct, "Initial locations for automaton '" + name + "'."));
1440  }
1441  STORM_LOG_THROW(automatonStructure.count("restrict-initial") < 2, storm::exceptions::InvalidJaniException, "Automaton '" << name << "' has multiple initial value restrictions");
1442  storm::expressions::Expression initialValueRestriction = expressionManager->boolean(true);
1443  if(automatonStructure.count("restrict-initial") > 0) {
1444  STORM_LOG_THROW(automatonStructure.at("restrict-initial").count("exp") == 1, storm::exceptions::InvalidJaniException, "Automaton '" << name << "' needs an expression inside the initial restricion");
1445  initialValueRestriction = parseExpression(automatonStructure.at("restrict-initial").at("exp"), scope.refine("Initial value restriction"));
1446  }
1447  automaton.setInitialStatesRestriction(initialValueRestriction);
1448 
1449 
1450  STORM_LOG_THROW(automatonStructure.count("edges") > 0, storm::exceptions::InvalidJaniException, "Automaton '" << name << "' must have a list of edges");
1451  for (auto const& edgeEntry : automatonStructure.at("edges")) {
1452  // source location
1453  STORM_LOG_THROW(edgeEntry.count("location") == 1, storm::exceptions::InvalidJaniException, "Each edge in automaton '" << name << "' must have a source");
1454  std::string sourceLoc = getString<ValueType>(edgeEntry.at("location"), "source location for edge in automaton '" + name + "'");
1455  STORM_LOG_THROW(locIds.count(sourceLoc) == 1, storm::exceptions::InvalidJaniException, "Source of edge has unknown location '" << sourceLoc << "' in automaton '" << name << "'.");
1456  // action
1457  STORM_LOG_THROW(edgeEntry.count("action") < 2, storm::exceptions::InvalidJaniException, "Edge from " << sourceLoc << " in automaton " << name << " has multiple actions");
1458  std::string action = storm::jani::Model::SILENT_ACTION_NAME; // def is tau
1459  if(edgeEntry.count("action") > 0) {
1460  action = getString<ValueType>(edgeEntry.at("action"), "action name in edge from '" + sourceLoc + "' in automaton '" + name + "'");
1461  // TODO check if action is known
1462  assert(action != "");
1463  }
1464  // rate
1465  STORM_LOG_THROW(edgeEntry.count("rate") < 2, storm::exceptions::InvalidJaniException, "Edge from '" << sourceLoc << "' in automaton '" << name << "' has multiple rates");
1466  storm::expressions::Expression rateExpr;
1467  if(edgeEntry.count("rate") > 0) {
1468  STORM_LOG_THROW(edgeEntry.at("rate").count("exp") == 1, storm::exceptions::InvalidJaniException, "Rate in edge from '" << sourceLoc << "' in automaton '" << name << "' must have a defing expression.");
1469  rateExpr = parseExpression(edgeEntry.at("rate").at("exp"), scope.refine("rate expression in edge from '" + sourceLoc));
1470  STORM_LOG_THROW(rateExpr.hasNumericalType(), storm::exceptions::InvalidJaniException, "Rate '" << rateExpr << "' has not a numerical type");
1471  STORM_LOG_THROW(rateExpr.containsVariables() || rateExpr.evaluateAsRational() > storm::utility::zero<storm::RationalNumber>(), storm::exceptions::InvalidJaniException, "Only positive rates are allowed but rate '" << rateExpr << " was found.");
1472  }
1473  // guard
1474  STORM_LOG_THROW(edgeEntry.count("guard") <= 1, storm::exceptions::InvalidJaniException, "Guard can be given at most once in edge from '" << sourceLoc << "' in automaton '" << name << "'");
1475  storm::expressions::Expression guardExpr = expressionManager->boolean(true);
1476  if (edgeEntry.count("guard") == 1) {
1477  STORM_LOG_THROW(edgeEntry.at("guard").count("exp") == 1, storm::exceptions::InvalidJaniException, "Guard in edge from '" + sourceLoc + "' in automaton '" + name + "' must have one expression");
1478  guardExpr = parseExpression(edgeEntry.at("guard").at("exp"), scope.refine("guard expression in edge from '" + sourceLoc));
1479  STORM_LOG_THROW(guardExpr.hasBooleanType(), storm::exceptions::InvalidJaniException, "Guard " << guardExpr << " does not have Boolean type.");
1480  }
1481  assert(guardExpr.isInitialized());
1482  std::shared_ptr<storm::jani::TemplateEdge> templateEdge = std::make_shared<storm::jani::TemplateEdge>(guardExpr);
1483 
1484  // edge assignments
1485  if (edgeEntry.count("assignments") > 0) {
1486  STORM_LOG_THROW(edgeEntry.count("assignments") == 1, storm::exceptions::InvalidJaniException, "Multiple edge assignments in edge from '" + sourceLoc + "' in automaton '" + name + "'.");
1487  for (auto const& assignmentEntry : edgeEntry.at("assignments")) {
1488  // ref
1489  STORM_LOG_THROW(assignmentEntry.count("ref") == 1, storm::exceptions::InvalidJaniException, "Assignment in edge from '" << sourceLoc << "' in automaton '" << name << "'must have one ref field");
1490  storm::jani::LValue lValue = parseLValue(assignmentEntry.at("ref"), scope.refine("Assignment variable in edge from '" + sourceLoc + "' in automaton '" + name + "'"));
1491  // value
1492  STORM_LOG_THROW(assignmentEntry.count("value") == 1, storm::exceptions::InvalidJaniException, "Assignment in edge from '" << sourceLoc << "' in automaton '" << name << "' must have one value field");
1493  storm::expressions::Expression assignmentExpr = parseExpression(assignmentEntry.at("value"), scope.refine("assignment in edge from '" + sourceLoc + "' in automaton '" + name + "'"));
1494  // TODO check types
1495  // index
1496  int64_t assignmentIndex = 0; // default.
1497  if(assignmentEntry.count("index") > 0) {
1498  assignmentIndex = getSignedInt<ValueType>(assignmentEntry.at("index"), "assignment index in edge from '" + sourceLoc + "' in automaton '" + name + "'");
1499  }
1500  templateEdge->getAssignments().add(storm::jani::Assignment(lValue, assignmentExpr, assignmentIndex));
1501  }
1502  }
1503 
1504  // destinations
1505  STORM_LOG_THROW(edgeEntry.count("destinations") == 1, storm::exceptions::InvalidJaniException, "A single list of destinations must be given in edge from '" << sourceLoc << "' in automaton '" << name << "'");
1506  std::vector<std::pair<uint64_t, storm::expressions::Expression>> destinationLocationsAndProbabilities;
1507  for(auto const& destEntry : edgeEntry.at("destinations")) {
1508  // target location
