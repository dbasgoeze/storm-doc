262  std::vector<std::string> SmtlibSmtSolver::readSolverOutput(bool waitForOutput){
263 #ifndef WINDOWS
264  if (processIdOfSolver==0){
265  STORM_LOG_DEBUG("failed to read solver output as the solver is not running");
266  return std::vector<std::string>();
267  }
268  int bytesReadable;
269  if(waitForOutput){
270  bytesReadable=1;//just assume that there are bytes readable
271  }
272  else if(ioctl(fromSolver, FIONREAD, &bytesReadable)< 0){ //actually obtain the readable bytes
273  STORM_LOG_ERROR("Could not check if the solver has output");
274  return std::vector<std::string>();
275  }
276  std::string solverOutput="";
277  const ssize_t MAX_CHUNK_SIZE=256;
278  char chunk[MAX_CHUNK_SIZE];
279  while(bytesReadable>0){
280  ssize_t chunkSize = read(fromSolver, chunk, MAX_CHUNK_SIZE);
281  STORM_LOG_THROW(chunkSize>=0, storm::exceptions::UnexpectedException, "failed to read solver output");
282  solverOutput += std::string(chunk, chunkSize);
283  if(ioctl(fromSolver, FIONREAD, &bytesReadable)< 0){//obtain the new amount of readable bytes
284  STORM_LOG_ERROR("Could not check if the solver has output");
285  return std::vector<std::string>();
286  }
287  if(bytesReadable==0 && solverOutput.back()!='\n'){
288  STORM_LOG_DEBUG("Solver Output '" << solverOutput << "' did not end with newline symbol (\\n). Since we assume that this should be the case, we will wait for more output from the solver");
289  bytesReadable=1; //we expect more output!
290  }
291  if(bytesReadable>0){
292  pid_t w=waitpid(processIdOfSolver, nullptr, WNOHANG);
293  if(w!=0){
294  STORM_LOG_WARN_COND(w>0, "Error when checking whether the solver is still running. Will assume that solver has terminated.");
295  STORM_LOG_WARN("The solver exited unexpectedly when reading output: " << solverOutput);
296  solverOutput+="terminated";
297  this->needsRestart=true;
298  this->processIdOfSolver=0;
299  bytesReadable=0;
300  }
301  }
302  }
303  checkForErrorMessage(solverOutput);
304  std::vector<std::string> solverOutputAsVector;
305  if(solverOutput.length()>0){
306  solverOutput=solverOutput.substr(0,solverOutput.length()-1);
307  boost::split(solverOutputAsVector, solverOutput, boost::is_any_of("\n"));
308  //note: this is a little bit unsafe as \n can be contained within a solver response
309  }
310  return solverOutputAsVector;
311 #endif

314  void SmtlibSmtSolver::checkForErrorMessage(const std::string message){
315  size_t errorOccurrance = message.find("error");
316  //do not throw an exception for timeout or memout errors
317  if(message.find("timeout")!=std::string::npos){
318  STORM_LOG_INFO("SMT solver answered: '" << message << "' and I am interpreting this as timeout ");
319  this->needsRestart=true;
320  this->processIdOfSolver=0;
321  }
322  else if(message.find("memory")!=std::string::npos){
323  STORM_LOG_INFO("SMT solver answered: '" << message << "' and I am interpreting this as out of memory ");
324  this->needsRestart=true;
325  this->processIdOfSolver=0;
326  }
327  else if(errorOccurrance!=std::string::npos){
328  this->needsRestart=true;
329  std::string errorMsg = "An error was detected while checking the solver output. ";
330  STORM_LOG_DEBUG("Detected an error message in the solver response:\n" + message);
331  size_t firstQuoteSign = message.find('\"',errorOccurrance);
332  if(firstQuoteSign!=std::string::npos && message.find("\\\"", firstQuoteSign-1)!=firstQuoteSign-1){
333  size_t secondQuoteSign = message.find('\"', firstQuoteSign+1);
334  while(secondQuoteSign!=std::string::npos && message.find("\\\"", secondQuoteSign-1)==secondQuoteSign-1){
335  secondQuoteSign = message.find('\"',secondQuoteSign+1);
336  }
337  if(secondQuoteSign!=std::string::npos){
338  errorMsg += "The error message was: <<" + message.substr(errorOccurrance,secondQuoteSign-errorOccurrance+1) + ">>.";
339  STORM_LOG_THROW(false, storm::exceptions::UnexpectedException, errorMsg);
340  return;
341  }
342  }
343  errorMsg += "The error message could not be parsed correctly. Snippet:\n" + message.substr(errorOccurrance,200);
344  STORM_LOG_THROW(false, storm::exceptions::UnexpectedException, errorMsg);
345  }
346  }
