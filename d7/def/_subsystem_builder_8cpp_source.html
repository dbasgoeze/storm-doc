62  SubsystemBuilderReturnType<ValueType, RewardModelType> internalBuildSubsystem(storm::models::sparse::Model<ValueType, RewardModelType> const& originalModel, storm::storage::BitVector const& subsystemStates, storm::storage::BitVector const& subsystemActions, SubsystemBuilderOptions const& options ) {
63 
64  auto const& groupIndices = originalModel.getTransitionMatrix().getRowGroupIndices();
65  SubsystemBuilderReturnType<ValueType, RewardModelType> result;
66  uint_fast64_t subsystemStateCount = subsystemStates.getNumberOfSetBits();
67  if (options.buildStateMapping) {
68  result.newToOldStateIndexMapping.reserve(subsystemStateCount);
69  }
70  if (options.buildActionMapping) {
71  result.newToOldActionIndexMapping.reserve(subsystemActions.getNumberOfSetBits());
72  }
73  storm::storage::BitVector deadlockStates;
74  if (options.fixDeadlocks) {
75  deadlockStates.resize(subsystemStates.size(), false);
76  }
77  // Get the set of actions that stay in the subsystem.
78  // Also establish the mappings if requested.
79  storm::storage::BitVector keptActions(originalModel.getTransitionMatrix().getRowCount(), false);
80  for (auto subsysState : subsystemStates) {
81  if (options.buildStateMapping) {
82  result.newToOldStateIndexMapping.push_back(subsysState);
83  }
84  bool hasDeadlock = true;
85  for (uint_fast64_t row = subsystemActions.getNextSetIndex(groupIndices[subsysState]); row < groupIndices[subsysState+1]; row = subsystemActions.getNextSetIndex(row + 1)) {
86  bool allRowEntriesStayInSubsys = true;
87  if (options.checkTransitionsOutside) {
88  for (auto const &entry : originalModel.getTransitionMatrix().getRow(row)) {
89  if (!subsystemStates.get(entry.getColumn())) {
90  allRowEntriesStayInSubsys = false;
91  break;
92  }
93  }
94  }
95  if (allRowEntriesStayInSubsys) {
96  if (options.buildActionMapping) {
97  result.newToOldActionIndexMapping.push_back(row);
98  }
99  keptActions.set(row, true);
100  hasDeadlock = false;
101  }
102  }
103  if (hasDeadlock) {
104  STORM_LOG_THROW(options.fixDeadlocks, storm::exceptions::InvalidOperationException, "Expected that in each state, at least one action is selected. Got a deadlock state instead. (violated at " << subsysState << ")");
105  if (options.buildActionMapping) {
106  result.newToOldActionIndexMapping.push_back(std::numeric_limits<uint64_t>::max());
107  }
108  deadlockStates.set(subsysState);
109  }
110  }
111  // If we have deadlock states we keep an action in each rowgroup of a deadlock states.
112  bool hasDeadlockStates = !deadlockStates.empty();
113  if (options.buildKeptActions) {
114  // store them now, before changing them.
115  result.keptActions = keptActions;
116  }
117  for (auto const& deadlockState : deadlockStates) {
118  keptActions.set(groupIndices[deadlockState], true);
119  }
120 
121  // Transform the components of the model
122  storm::storage::sparse::ModelComponents<ValueType, RewardModelType> components;
123  if (hasDeadlockStates) {
124  // make deadlock choices a selfloop
125  components.transitionMatrix = originalModel.getTransitionMatrix();
126  components.transitionMatrix.makeRowGroupsAbsorbing(deadlockStates);
127  components.transitionMatrix.dropZeroEntries();
128  components.transitionMatrix = components.transitionMatrix.getSubmatrix(false, keptActions, subsystemStates);
129  } else {
130  components.transitionMatrix = originalModel.getTransitionMatrix().getSubmatrix(false, keptActions, subsystemStates);
131  }
132  components.stateLabeling = originalModel.getStateLabeling().getSubLabeling(subsystemStates);
133  for (auto const& rewardModel : originalModel.getRewardModels()){
134  components.rewardModels.insert(std::make_pair(rewardModel.first, transformRewardModel(rewardModel.second, subsystemStates, keptActions)));
135  }
136  if (originalModel.hasChoiceLabeling()) {
137  components.choiceLabeling = originalModel.getChoiceLabeling().getSubLabeling(keptActions);
138  }
139  if (originalModel.hasStateValuations()) {
140  components.stateValuations = originalModel.getStateValuations().selectStates(subsystemStates);
141  }
142  if (originalModel.hasChoiceOrigins()) {
143  components.choiceOrigins = originalModel.getChoiceOrigins()->selectChoices(keptActions);
144  }
145 
146  if (hasDeadlockStates) {
147  auto subDeadlockStates = deadlockStates % subsystemStates;
148  assert(deadlockStates.getNumberOfSetBits() == subDeadlockStates.getNumberOfSetBits());
149  // erase rewards, choice labels, choice origins
150  for (auto& rewModel : components.rewardModels) {
151  for (auto const& state : subDeadlockStates) {
152  rewModel.second.clearRewardAtState(state, components.transitionMatrix);
153  }
154  }
155  if (components.choiceLabeling) {
156  storm::storage::BitVector nonDeadlockChoices(components.transitionMatrix.getRowCount(), true);
157  for (auto const& state : subDeadlockStates) {
158  auto const& choice = components.transitionMatrix.getRowGroupIndices()[state];
159  nonDeadlockChoices.set(choice, false);
160  }
161  for (auto const& label : components.choiceLabeling.get().getLabels()) {
162  components.choiceLabeling->setChoices(label, components.choiceLabeling->getChoices(label) & nonDeadlockChoices);
163  }
164  }
165  if (components.choiceOrigins) {
166  for (auto const& state : subDeadlockStates) {
167  auto const& choice = components.transitionMatrix.getRowGroupIndices()[state];
168  components.choiceOrigins.get()->clearOriginOfChoice(choice);
169  }
170  }
171  // get a unique label for the deadlock states
172  result.deadlockLabel = "deadl";
173  while (components.stateLabeling.containsLabel(result.deadlockLabel.get())) {
174  result.deadlockLabel->push_back('_');
175  }
176  components.stateLabeling.addLabel(result.deadlockLabel.get(), std::move(subDeadlockStates));
177  }
178 
179  transformModelSpecificComponents<ValueType, RewardModelType>(originalModel, subsystemStates, components);
180 
181  result.model = storm::utility::builder::buildModelFromComponents(originalModel.getType(), std::move(components));
182 
183  STORM_LOG_DEBUG("Subsystem Builder is done. Resulting model has " << result.model->getNumberOfStates() << " states.");
184  return result;
185  }
