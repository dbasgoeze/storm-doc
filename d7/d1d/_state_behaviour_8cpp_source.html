116  template <typename IndexType, typename ValueType>
117  void StateBehaviour<IndexType, ValueType>::reduceMarkovAutomaton() {
118  // If the model we build is a Markov Automaton, we reduce the choices by summing all Markovian choices
119  // and making the Markovian choice the very first one (if there is any).
120  bool foundPreviousMarkovianChoice = false;
121  uint64_t numberOfChoicesToDelete = 0;
122 
123  for (uint_fast64_t index = 0; index + numberOfChoicesToDelete < this->size();) {
124  Choice<IndexType, ValueType>& choice = choices[index];
125 
126  if (choice.isMarkovian()) {
127  if (foundPreviousMarkovianChoice) {
128  // If there was a previous Markovian choice, we need to sum them. Note that we can assume
129  // that the previous Markovian choice is the very first one in the choices vector.
130  choices.front().add(std::move(choice));
131 
132  // Swap the choice to the end to indicate it can be removed (if it's not already there).
133  if (index != this->size() - 1) {
134  choice = std::move(choices[choices.size() - 1 - numberOfChoicesToDelete]);
135  }
136  ++numberOfChoicesToDelete;
137  } else {
138  // If there is no previous Markovian choice, just move the Markovian choice to the front.
139  if (index != 0) {
140  std::swap(choices.front(), choice);
141  }
142  foundPreviousMarkovianChoice = true;
143  ++index;
144  }
145  } else {
146  ++index;
147  }
148  }
149 
150  // Finally remove the choices that were added to other Markovian choices.
151  if (numberOfChoicesToDelete > 0) {
152  choices.resize(choices.size() - numberOfChoicesToDelete);
153  }
154  }
