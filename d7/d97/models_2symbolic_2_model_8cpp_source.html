126  template<storm::dd::DdType Type, typename ValueType>
127  storm::dd::Bdd<Type> Model<Type, ValueType>::getStates(storm::expressions::Expression const& expression) const {
128  if (expression.isTrue()) {
129  return this->getReachableStates();
130  } else if (expression.isFalse()) {
131  return manager->getBddZero();
132  }
133 
134  // Look up the string equivalent of the expression.
135  std::stringstream stream;
136  stream << expression;
137  auto bddIt = labelToBddMap.find(stream.str());
138  if (bddIt != labelToBddMap.end()) {
139  return bddIt->second;
140  }
141 
142  // Finally try to translate the expression with an adapter.
143  STORM_LOG_THROW(rowExpressionAdapter != nullptr, storm::exceptions::InvalidOperationException, "Cannot create BDD for expression without expression adapter.");
144  return rowExpressionAdapter->translateExpression(expression).toBdd() && this->reachableStates;
145  }
