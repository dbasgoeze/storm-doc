318  bool checkConvergence(std::vector<ValueType> const& lower, std::vector<ValueType> const& upper, boost::optional<storm::storage::BitVector> const& relevantValues, ValueType const& epsilon, bool relative, ValueType& kappa) {

320  if (!relative) {
321  if (relevantValues) {
322  return storm::utility::vector::equalModuloPrecision(lower, upper, relevantValues.get(), epsilon * (storm::utility::one<ValueType>() - kappa), false);
323  } else {
324  return storm::utility::vector::equalModuloPrecision(lower, upper, epsilon * (storm::utility::one<ValueType>() - kappa), false);
325  }
326  }
327  ValueType truncationError = epsilon * kappa;
328  for (uint64_t i = 0; i < lower.size(); ++i) {
329  if (relevantValues) {
330  i = relevantValues->getNextSetIndex(i);
331  if (i == lower.size()) {
332  break;
333  }
334  }
335  if (lower[i] == upper[i]) {
336  continue;
337  }
338  if (lower[i] <= truncationError) {
339  return false;
340  }

347  }

472  // * a matrix aProbabilisticToMarkovian with all (non-discretized) transitions from probabilistic non-goal states to all Markovian non-goal states.
473  typename storm::storage::SparseMatrix<ValueType> aMarkovian = transitionMatrix.getSubmatrix(true, markovianNonGoalStates, markovianNonGoalStates, true);
474 
475  bool existProbabilisticStates = !probabilisticNonGoalStates.empty();

477  typename storm::storage::SparseMatrix<ValueType> aProbabilistic;
478  typename storm::storage::SparseMatrix<ValueType> aProbabilisticToMarkovian;
479  if (existProbabilisticStates) {
480  aMarkovianToProbabilistic = transitionMatrix.getSubmatrix(true, markovianNonGoalStates, probabilisticNonGoalStates);
481  aProbabilistic = transitionMatrix.getSubmatrix(true, probabilisticNonGoalStates, probabilisticNonGoalStates);
482  aProbabilisticToMarkovian = transitionMatrix.getSubmatrix(true, probabilisticNonGoalStates, markovianNonGoalStates);
483  }
484 
485  // The matrices with transitions from Markovian states need to be digitized.
486  // Digitize aMarkovian. Based on whether the transition is a self-loop or not, we apply the two digitization rules.
487  uint64_t rowIndex = 0;
488  for (auto state : markovianNonGoalStates) {
489  for (auto& element : aMarkovian.getRow(rowIndex)) {
490  ValueType eTerm = std::exp(-exitRates[state] * delta);
491  if (element.getColumn() == rowIndex) {
492  element.setValue((storm::utility::one<ValueType>() - eTerm) * element.getValue() + eTerm);
493  } else {
494  element.setValue((storm::utility::one<ValueType>() - eTerm) * element.getValue());
495  }
496  }
497  ++rowIndex;
498  }
499 
500  // Digitize aMarkovianToProbabilistic. As there are no self-loops in this case, we only need to apply the digitization formula for regular successors.
501  if (existProbabilisticStates) {
502  rowIndex = 0;
503  for (auto state : markovianNonGoalStates) {
504  for (auto& element : aMarkovianToProbabilistic.getRow(rowIndex)) {
505  element.setValue((1 - std::exp(-exitRates[state] * delta)) * element.getValue());
506  }
507  ++rowIndex;
508  }
509  }
510 
511  // Initialize the two vectors that hold the variable one-step probabilities to all target states for probabilistic and Markovian (non-goal) states.
512  std::vector<ValueType> bProbabilistic(existProbabilisticStates ? aProbabilistic.getRowCount() : 0);
513  std::vector<ValueType> bMarkovian(markovianNonGoalStates.getNumberOfSetBits());
514 
515  // Compute the two fixed right-hand side vectors, one for Markovian states and one for the probabilistic ones.
516  std::vector<ValueType> bProbabilisticFixed;
517  if (existProbabilisticStates) {
518  bProbabilisticFixed = transitionMatrix.getConstrainedRowGroupSumVector(probabilisticNonGoalStates, goalStates);
519  }
520  std::vector<ValueType> bMarkovianFixed;
521  bMarkovianFixed.reserve(markovianNonGoalStates.getNumberOfSetBits());
522  for (auto state : markovianNonGoalStates) {
523  bMarkovianFixed.push_back(storm::utility::zero<ValueType>());
524 
525  for (auto& element : transitionMatrix.getRowGroup(state)) {
526  if (goalStates.get(element.getColumn())) {
527  bMarkovianFixed.back() += (1 - std::exp(-exitRates[state] * delta)) * element.getValue();
528  }
529  }
530  }
531 
532  // Create a solver object (only if there are actually transitions between probabilistic states)
533  auto solverEnv = env;
534  solverEnv.solver().setForceExact(true);
535  auto solver = setUpProbabilisticStatesSolver(solverEnv, dir, aProbabilistic);
536 
537  // Perform the actual value iteration
538  // * loop until the step bound has been reached
539  // * in the loop:
540  // * perform value iteration using A_PSwG, v_PS and the vector b where b = (A * 1_G)|PS + A_PStoMS * v_MS
541  // and 1_G being the characteristic vector for all goal states.
542  // * perform one timed-step using v_MS := A_MSwG * v_MS + A_MStoPS * v_PS + (A * 1_G)|MS
543  std::vector<ValueType> markovianNonGoalValuesSwap(markovianNonGoalValues);
544  for (uint64_t currentStep = 0; currentStep < numberOfSteps; ++currentStep) {
545  if (existProbabilisticStates) {
546  // Start by (re-)computing bProbabilistic = bProbabilisticFixed + aProbabilisticToMarkovian * vMarkovian.
547  aProbabilisticToMarkovian.multiplyWithVector(markovianNonGoalValues, bProbabilistic);
548  storm::utility::vector::addVectors(bProbabilistic, bProbabilisticFixed, bProbabilistic);
549 
550  // Now perform the inner value iteration for probabilistic states.
551  if (solver) {
552  solver->solveEquations(solverEnv, dir, probabilisticNonGoalValues, bProbabilistic);
553  } else {
554  storm::utility::vector::reduceVectorMinOrMax(dir, bProbabilistic, probabilisticNonGoalValues, aProbabilistic.getRowGroupIndices());
555  }
556 
557  // (Re-)compute bMarkovian = bMarkovianFixed + aMarkovianToProbabilistic * vProbabilistic.
558  aMarkovianToProbabilistic.multiplyWithVector(probabilisticNonGoalValues, bMarkovian);
559  storm::utility::vector::addVectors(bMarkovian, bMarkovianFixed, bMarkovian);
560  }
561 
562  aMarkovian.multiplyWithVector(markovianNonGoalValues, markovianNonGoalValuesSwap);
563  std::swap(markovianNonGoalValues, markovianNonGoalValuesSwap);
564  if (existProbabilisticStates) {
565  storm::utility::vector::addVectors(markovianNonGoalValues, bMarkovian, markovianNonGoalValues);
566  } else {
567  storm::utility::vector::addVectors(markovianNonGoalValues, bMarkovianFixed, markovianNonGoalValues);
568  }
569  if (storm::utility::resources::isTerminate()) {
570  break;
571  }
572  }
573 
574  if (existProbabilisticStates) {
575  // After the loop, perform one more step of the value iteration for PS states.
576  aProbabilisticToMarkovian.multiplyWithVector(markovianNonGoalValues, bProbabilistic);
577  storm::utility::vector::addVectors(bProbabilistic, bProbabilisticFixed, bProbabilistic);
578  if (solver) {
579  solver->solveEquations(solverEnv, dir, probabilisticNonGoalValues, bProbabilistic);
580  } else {
581  storm::utility::vector::reduceVectorMinOrMax(dir, bProbabilistic, probabilisticNonGoalValues, aProbabilistic.getRowGroupIndices());
582  }
583  }

587  std::vector<ValueType> computeBoundedUntilProbabilitiesImca(Environment const& env, OptimizationDirection dir, storm::storage::SparseMatrix<ValueType> const& transitionMatrix, std::vector<ValueType> const& exitRateVector, storm::storage::BitVector const& markovianStates, storm::storage::BitVector const& psiStates, std::pair<double, double> const& boundsPair) {
588  STORM_LOG_TRACE("Using IMCA's technique to compute bounded until probabilities.");
589 
590  uint64_t numberOfStates = transitionMatrix.getRowGroupCount();
591 
592  // 'Unpack' the bounds to make them more easily accessible.
593  double lowerBound = boundsPair.first;
594  double upperBound = boundsPair.second;
595 
596  // (1) Compute the accuracy we need to achieve the required error bound.
597  ValueType maxExitRate = 0;
598  for (auto value : exitRateVector) {
599  maxExitRate = std::max(maxExitRate, value);
600  }
601  ValueType delta = (2.0 * storm::utility::convertNumber<ValueType>(env.solver().timeBounded().getPrecision())) / (upperBound * maxExitRate * maxExitRate);
602 
603  // (2) Compute the number of steps we need to make for the interval.
604  uint64_t numberOfSteps = static_cast<uint64_t>(std::ceil((upperBound - lowerBound) / delta));
605  STORM_LOG_INFO("Performing " << numberOfSteps << " iterations (delta=" << delta << ") for interval [" << lowerBound << ", " << upperBound << "]." << std::endl);
606 
607  // (3) Compute the non-goal states and initialize two vectors
608  // * vProbabilistic holds the probability values of probabilistic non-goal states.
609  // * vMarkovian holds the probability values of Markovian non-goal states.
610  storm::storage::BitVector const& markovianNonGoalStates = markovianStates & ~psiStates;
611  storm::storage::BitVector const& probabilisticNonGoalStates = ~markovianStates & ~psiStates;
612  std::vector<ValueType> vProbabilistic(probabilisticNonGoalStates.getNumberOfSetBits());
613  std::vector<ValueType> vMarkovian(markovianNonGoalStates.getNumberOfSetBits());
614 
615  computeBoundedReachabilityProbabilitiesImca(env, dir, transitionMatrix, exitRateVector, psiStates, markovianNonGoalStates, probabilisticNonGoalStates, vMarkovian, vProbabilistic, delta, numberOfSteps);
616 
617  // (4) If the lower bound of interval was non-zero, we need to take the current values as the starting values for a subsequent value iteration.
618  if (lowerBound != storm::utility::zero<ValueType>()) {
619  std::vector<ValueType> vAllProbabilistic((~markovianStates).getNumberOfSetBits());
620  std::vector<ValueType> vAllMarkovian(markovianStates.getNumberOfSetBits());
621 
622  // Create the starting value vectors for the next value iteration based on the results of the previous one.
623  storm::utility::vector::setVectorValues<ValueType>(vAllProbabilistic, psiStates % ~markovianStates, storm::utility::one<ValueType>());
624  storm::utility::vector::setVectorValues<ValueType>(vAllProbabilistic, ~psiStates % ~markovianStates, vProbabilistic);
625  storm::utility::vector::setVectorValues<ValueType>(vAllMarkovian, psiStates % markovianStates, storm::utility::one<ValueType>());
626  storm::utility::vector::setVectorValues<ValueType>(vAllMarkovian, ~psiStates % markovianStates, vMarkovian);
627 
628  // Compute the number of steps to reach the target interval.
629  numberOfSteps = static_cast<uint64_t>(std::ceil(lowerBound / delta));
630  STORM_LOG_INFO("Performing " << numberOfSteps << " iterations (delta=" << delta << ") for interval [0, " << lowerBound << "]." << std::endl);
631 
632  // Compute the bounded reachability for interval [0, b-a].
633  computeBoundedReachabilityProbabilitiesImca(env, dir, transitionMatrix, exitRateVector, storm::storage::BitVector(numberOfStates), markovianStates, ~markovianStates, vAllMarkovian, vAllProbabilistic, delta, numberOfSteps);
634 
635  // Create the result vector out of vAllProbabilistic and vAllMarkovian and return it.
636  std::vector<ValueType> result(numberOfStates, storm::utility::zero<ValueType>());
637  storm::utility::vector::setVectorValues(result, ~markovianStates, vAllProbabilistic);
638  storm::utility::vector::setVectorValues(result, markovianStates, vAllMarkovian);
639 
640  return result;
641  } else {
642  // Create the result vector out of 1_G, vProbabilistic and vMarkovian and return it.
643  std::vector<ValueType> result(numberOfStates);
644  storm::utility::vector::setVectorValues<ValueType>(result, psiStates, storm::utility::one<ValueType>());
645  storm::utility::vector::setVectorValues(result, probabilisticNonGoalStates, vProbabilistic);
646  storm::utility::vector::setVectorValues(result, markovianNonGoalStates, vMarkovian);
647  return result;
648  }
