937  ddRowVariableIndices.push_back(ddVariable.getIndex());

943  DdMetaVariable<LibraryType> const& metaVariable = this->getDdManager().getMetaVariable(variable);
944  for (auto const& ddVariable : metaVariable.getDdVariables()) {
945  ddColumnVariableIndices.push_back(ddVariable.getIndex());
946  }
947  rowAndColumnMetaVariables.insert(variable);
948  }
949  std::sort(ddColumnVariableIndices.begin(), ddColumnVariableIndices.end());
950  for (auto const& variable : groupMetaVariables) {
951  DdMetaVariable<LibraryType> const& metaVariable = this->getDdManager().getMetaVariable(variable);
952  for (auto const& ddVariable : metaVariable.getDdVariables()) {
953  ddGroupVariableIndices.push_back(ddVariable.getIndex());
954  }
955  }
956  std::sort(ddGroupVariableIndices.begin(), ddGroupVariableIndices.end());

958  Bdd<LibraryType> columnVariableCube = Bdd<LibraryType>::getCube(this->getDdManager(), columnMetaVariables);
959 
960  // Transform the row group sizes to the actual row group indices.
961  rowGroupIndices.resize(rowGroupIndices.size() + 1);
962  uint_fast64_t tmp = 0;
963  uint_fast64_t tmp2 = 0;
964  for (uint_fast64_t i = 1; i < rowGroupIndices.size(); ++i) {
965  tmp2 = rowGroupIndices[i];
966  rowGroupIndices[i] = rowGroupIndices[i - 1] + tmp;
967  std::swap(tmp, tmp2);
968  }
969  rowGroupIndices[0] = 0;
970 
971  // Create the explicit vectors we need to fill later.
972  std::vector<std::vector<ValueType>> explicitVectors(vectors.size());
973  for (auto& v : explicitVectors) {
974  v.resize(rowGroupIndices.back());
975  }
976 
977  // Next, we split the matrix into one for each group. Note that this only works if the group variables are at the very top.
978  std::vector<std::vector<Add<LibraryType, ValueType>>> groups;
979  if (vectors.size() == 1) {
980  // This version potentially has slightly reduced overhead
981  std::vector<std::pair<InternalAdd<LibraryType, ValueType>, InternalAdd<LibraryType, ValueType>>> internalAddGroups = internalAdd.splitIntoGroups(vectors.front(), ddGroupVariableIndices);
982  for (auto const& internalAdd : internalAddGroups) {
983  groups.push_back({Add<LibraryType, ValueType>(this->getDdManager(), internalAdd.second, rowMetaVariables), Add<LibraryType, ValueType>(this->getDdManager(), internalAdd.first, rowAndColumnMetaVariables)});
984  }
985  } else {
986  std::vector<InternalAdd<LibraryType, ValueType>> internalVectors;
987  for (Add<LibraryType, ValueType> const& v : vectors) {
988  internalVectors.push_back(v.getInternalAdd());
989  }
990  std::vector<std::vector<InternalAdd<LibraryType, ValueType>>> internalAddGroups = internalAdd.splitIntoGroups(internalVectors, ddGroupVariableIndices);
991  for (auto const& internalAddGroup : internalAddGroups) {
992  STORM_LOG_ASSERT(internalAddGroup.size() == vectors.size() + 1, "Unexpected group size.");
993  std::vector<Add<LibraryType, ValueType>> group;
994  for (uint64_t vectorIndex = 0; vectorIndex < vectors.size(); ++vectorIndex) {
995  group.push_back(Add<LibraryType, ValueType>(this->getDdManager(), internalAddGroup[vectorIndex], rowMetaVariables));
996  }
997  // The last group member corresponds to the matrix.
998  group.push_back(Add<LibraryType, ValueType>(this->getDdManager(), internalAddGroup.back(), rowAndColumnMetaVariables));
999  groups.push_back(std::move(group));
1000  }
1001  }
1002 
1003  // Create the actual storage for the non-zero entries.
1004  std::vector<storm::storage::MatrixEntry<uint_fast64_t, ValueType>> columnsAndValues(this->getNonZeroCount());
1005 
1006  // Now compute the indices at which the individual rows start.
1007  std::vector<uint_fast64_t> rowIndications(rowGroupIndices.back() + 1);
1008 
1009  std::vector<InternalAdd<LibraryType, uint_fast64_t>> statesWithGroupEnabled(groups.size());
1010  InternalAdd<LibraryType, uint_fast64_t> stateToRowGroupCount = this->getDdManager().template getAddZero<uint_fast64_t>();
1011  for (uint_fast64_t i = 0; i < groups.size(); ++i) {
1012  std::vector<Add<LibraryType, ValueType>> const& group = groups[i];
1013  Bdd<LibraryType> matrixDdNotZero = group.back().notZero();
1014 
1015  std::vector<uint64_t> tmpRowIndications = matrixDdNotZero.template toAdd<uint_fast64_t>().sumAbstract(columnMetaVariables).toVector(rowOdd);
1016  for (uint64_t offset = 0; offset < tmpRowIndications.size(); ++offset) {
1017  rowIndications[rowGroupIndices[offset]] += tmpRowIndications[offset];
1018  }
1019 
1020  Bdd<LibraryType> vectorDdNotZero = this->getDdManager().getBddZero();
1021  for (uint64_t vectorIndex = 0; vectorIndex < vectors.size(); ++vectorIndex) {
1022  vectorDdNotZero |= group[vectorIndex].notZero();
1023  group[vectorIndex].internalAdd.composeWithExplicitVector(rowOdd, ddRowVariableIndices, rowGroupIndices, explicitVectors[vectorIndex], std::plus<ValueType>());
1024  }
1025 
1026  statesWithGroupEnabled[i] = (matrixDdNotZero.existsAbstract(columnMetaVariables) || vectorDdNotZero).template toAdd<uint_fast64_t>();
1027  stateToRowGroupCount += statesWithGroupEnabled[i];
1028  statesWithGroupEnabled[i].composeWithExplicitVector(rowOdd, ddRowVariableIndices, rowGroupIndices, std::plus<uint_fast64_t>());
1029  }
1030 
1031  // Since we modified the rowGroupIndices, we need to restore the correct values.
1032  stateToRowGroupCount.composeWithExplicitVector(rowOdd, ddRowVariableIndices, rowGroupIndices, std::minus<uint_fast64_t>());
1033 
1034  // Now that we computed the number of entries in each row, compute the corresponding offsets in the entry vector.
1035  tmp = 0;
1036  tmp2 = 0;
1037  for (uint_fast64_t i = 1; i < rowIndications.size(); ++i) {
1038  tmp2 = rowIndications[i];
1039  rowIndications[i] = rowIndications[i - 1] + tmp;
1040  std::swap(tmp, tmp2);
1041  }
1042  rowIndications[0] = 0;
1043 
1044  // Now actually fill the entry vector.
1045  for (uint_fast64_t i = 0; i < groups.size(); ++i) {
1046  auto const& dd = groups[i].back();
1047 
1048  dd.internalAdd.toMatrixComponents(rowGroupIndices, rowIndications, columnsAndValues, rowOdd, columnOdd, ddRowVariableIndices, ddColumnVariableIndices, true);
1049  statesWithGroupEnabled[i].composeWithExplicitVector(rowOdd, ddRowVariableIndices, rowGroupIndices, std::plus<uint_fast64_t>());
1050  }
1051 
1052  // Since we modified the rowGroupIndices, we need to restore the correct values.
1053  stateToRowGroupCount.composeWithExplicitVector(rowOdd, ddRowVariableIndices, rowGroupIndices, std::minus<uint_fast64_t>());
1054 
1055  // Since the last call to toMatrixRec modified the rowIndications, we need to restore the correct values.
1056  for (uint_fast64_t i = rowIndications.size() - 1; i > 0; --i) {
1057  rowIndications[i] = rowIndications[i - 1];
1058  }
1059  rowIndications[0] = 0;
1060 
1061  return std::make_pair(storm::storage::SparseMatrix<ValueType>(columnOdd.getTotalOffset(), std::move(rowIndications), std::move(columnsAndValues), std::move(rowGroupIndices)), std::move(explicitVectors));
