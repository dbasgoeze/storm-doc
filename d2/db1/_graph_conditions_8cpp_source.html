36  void ConstraintCollector<ValueType>::wellformedRequiresNonNegativeEntries(std::vector<ValueType> const& vec) {
37  for(auto const& entry : vec) {
38  if (!storm::utility::isConstant(entry)) {
39  auto const& transitionVars = entry.gatherVariables();
40  variableSet.insert(transitionVars.begin(), transitionVars.end());
41  if (entry.denominator().isConstant()) {
42  assert(entry.denominator().constantPart() != 0);
43  if (entry.denominator().constantPart() > 0) {
44  wellformedConstraintSet.emplace(entry.nominator().polynomialWithCoefficient(), storm::CompareRelation::GEQ);
45  } else if (entry.denominator().constantPart() < 0) {
46  wellformedConstraintSet.emplace(entry.nominator().polynomialWithCoefficient(), storm::CompareRelation::LEQ);
47  } else {
48  STORM_LOG_THROW(false, storm::exceptions::UnexpectedException, "Should have failed before.");
49  }
50  } else {
51  wellformedConstraintSet.emplace(entry.denominator().polynomialWithCoefficient(), storm::CompareRelation::NEQ);
52  wellformedConstraintSet.emplace(carl::FormulaType::ITE, typename ConstraintType<ValueType>::val(entry.denominator().polynomialWithCoefficient(), storm::CompareRelation::GREATER), typename ConstraintType<ValueType>::val(entry.nominator().polynomialWithCoefficient(), storm::CompareRelation::GEQ), typename ConstraintType<ValueType>::val(entry.nominator().polynomialWithCoefficient(), storm::CompareRelation::LEQ));
53  }

59  void ConstraintCollector<ValueType>::process(storm::models::sparse::Model<ValueType> const& model) {
60 
61  if (model.getType() != storm::models::ModelType::Ctmc) {
62 
63  for(uint_fast64_t action = 0; action < model.getTransitionMatrix().getRowCount(); ++action) {
64  ValueType sum = storm::utility::zero<ValueType>();
65 
66  for (auto transitionIt = model.getTransitionMatrix().begin(action); transitionIt != model.getTransitionMatrix().end(action); ++transitionIt) {
67  auto const& transition = *transitionIt;
68  sum += transition.getValue();
69  if (!storm::utility::isConstant(transition.getValue())) {
70  auto const& transitionVars = transition.getValue().gatherVariables();
71  variableSet.insert(transitionVars.begin(), transitionVars.end());
72  // Assert: 0 <= transition <= 1
73  if (transition.getValue().denominator().isConstant()) {
74  assert(transition.getValue().denominator().constantPart() != 0);
75  if (transition.getValue().denominator().constantPart() > 0) {
76  // Assert: nom <= denom
77  wellformedConstraintSet.emplace((transition.getValue().nominator() - transition.getValue().denominator()).polynomialWithCoefficient(), storm::CompareRelation::LEQ);
78  // Assert: nom >= 0
79  wellformedConstraintSet.emplace(transition.getValue().nominator().polynomialWithCoefficient(), storm::CompareRelation::GEQ);
80  } else if (transition.getValue().denominator().constantPart() < 0) {
81  // Assert nom >= denom
82  wellformedConstraintSet.emplace((transition.getValue().nominator() - transition.getValue().denominator()).polynomialWithCoefficient(), storm::CompareRelation::GEQ);
83  // Assert: nom <= 0
84  wellformedConstraintSet.emplace(transition.getValue().nominator().polynomialWithCoefficient(), storm::CompareRelation::LEQ);
85  } else {
86  STORM_LOG_ASSERT(false, "Denominator must not equal 0.");
87  }
88  } else {
89  // Assert: denom != 0
90  wellformedConstraintSet.emplace(transition.getValue().denominator().polynomialWithCoefficient(), storm::CompareRelation::NEQ);
91  // Assert: transition >= 0 <==> if denom > 0 then nom >= 0 else nom <= 0
92  wellformedConstraintSet.emplace(carl::FormulaType::ITE, typename ConstraintType<ValueType>::val(transition.getValue().denominator().polynomialWithCoefficient(), storm::CompareRelation::GREATER), typename ConstraintType<ValueType>::val(transition.getValue().nominator().polynomialWithCoefficient(), storm::CompareRelation::GEQ), typename ConstraintType<ValueType>::val(transition.getValue().nominator().polynomialWithCoefficient(), storm::CompareRelation::LEQ));
93  // TODO: Assert: transition <= 1 <==> if denom > 0 then nom - denom <= 0 else nom - denom >= 0
94  }
95  // Assert: transition > 0
96  graphPreservingConstraintSet.emplace(transition.getValue().nominator().polynomialWithCoefficient(), storm::CompareRelation::NEQ);
97  }
98  }
99  STORM_LOG_ASSERT(!storm::utility::isConstant(sum) || storm::utility::isOne(sum), "If the sum is a constant, it must be equal to 1.");
100  if(!storm::utility::isConstant(sum)) {
101  // Assert: sum == 1
102  wellformedConstraintSet.emplace((sum.nominator() - sum.denominator()).polynomialWithCoefficient(), storm::CompareRelation::EQ);
103  }
104  }
105  } else {
106  for (auto const& transition : model.getTransitionMatrix()) {
107  if(!transition.getValue().isConstant()) {
108  if (transition.getValue().denominator().isConstant()) {
109  assert(transition.getValue().denominator().constantPart() != 0);
110  if (transition.getValue().denominator().constantPart() > 0) {
111  wellformedConstraintSet.emplace(transition.getValue().nominator().polynomialWithCoefficient(), storm::CompareRelation::GEQ);
112  } else if (transition.getValue().denominator().constantPart() < 0) {
113  wellformedConstraintSet.emplace(transition.getValue().nominator().polynomialWithCoefficient(), storm::CompareRelation::LEQ);
114  } else {
115  STORM_LOG_THROW(false, storm::exceptions::UnexpectedException, "Should have failed before.");
116  }
117  } else {
118  wellformedConstraintSet.emplace(transition.getValue().denominator().polynomialWithCoefficient(), storm::CompareRelation::NEQ);
119  wellformedConstraintSet.emplace(carl::FormulaType::ITE, typename ConstraintType<ValueType>::val(transition.getValue().denominator().polynomialWithCoefficient(), storm::CompareRelation::GREATER), typename ConstraintType<ValueType>::val(transition.getValue().nominator().polynomialWithCoefficient(), storm::CompareRelation::GEQ), typename ConstraintType<ValueType>::val(transition.getValue().nominator().polynomialWithCoefficient(), storm::CompareRelation::LEQ));
120  }
121  graphPreservingConstraintSet.emplace(transition.getValue().nominator().polynomialWithCoefficient(), storm::CompareRelation::NEQ);
122  }
123  }
124  }
125 
126  if (model.getType() == storm::models::ModelType::Ctmc) {
127  auto const& exitRateVector = static_cast<storm::models::sparse::Ctmc<ValueType> const&>(model).getExitRateVector();
128  wellformedRequiresNonNegativeEntries(exitRateVector);
129  } else if (model.getType() == storm::models::ModelType::MarkovAutomaton) {
130  auto const& exitRateVector = static_cast<storm::models::sparse::MarkovAutomaton<ValueType> const&>(model).getExitRates();
131  wellformedRequiresNonNegativeEntries(exitRateVector);
132  }
133 
134  for(auto const& rewModelEntry : model.getRewardModels()) {
135  if (rewModelEntry.second.hasStateRewards()) {
136  wellformedRequiresNonNegativeEntries(rewModelEntry.second.getStateRewardVector());
137  }
138  if (rewModelEntry.second.hasStateActionRewards()) {
139  wellformedRequiresNonNegativeEntries(rewModelEntry.second.getStateActionRewardVector());
140  }
141  if (rewModelEntry.second.hasTransitionRewards()) {
142  for (auto const& entry : rewModelEntry.second.getTransitionRewardMatrix()) {
143  if(!entry.getValue().isConstant()) {
144  if (entry.getValue().denominator().isConstant()) {
145  assert(entry.getValue().denominator().constantPart() != 0);
146  if (entry.getValue().denominator().constantPart() > 0) {
147  wellformedConstraintSet.emplace(entry.getValue().nominator().polynomialWithCoefficient(), storm::CompareRelation::GEQ);
148  } else if (entry.getValue().denominator().constantPart() < 0) {
149  wellformedConstraintSet.emplace(entry.getValue().nominator().polynomialWithCoefficient(), storm::CompareRelation::LEQ);
150  } else {
151  STORM_LOG_THROW(false, storm::exceptions::UnexpectedException, "Should have failed before.");
152  }

154  wellformedConstraintSet.emplace(entry.getValue().denominator().polynomialWithCoefficient(), storm::CompareRelation::NEQ);
