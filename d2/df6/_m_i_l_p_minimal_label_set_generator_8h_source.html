124  static struct ChoiceInformation determineRelevantAndProblematicChoices(storm::models::sparse::Mdp<T> const& mdp, std::vector<storm::storage::FlatSet<uint_fast64_t>> const& labelSets, StateInformation const& stateInformation, storm::storage::BitVector const& psiStates) {
125  // Create result and shortcuts to needed data for convenience.
126  ChoiceInformation result;
127  storm::storage::SparseMatrix<T> const& transitionMatrix = mdp.getTransitionMatrix();
128  std::vector<uint_fast64_t> const& nondeterministicChoiceIndices = mdp.getNondeterministicChoiceIndices();
129 
130  // Now traverse all choices of all relevant states and check whether there is a relevant target state.
131  // If so, the associated labels become relevant. Also, if a choice of relevant state has at least one
132  // relevant successor, the choice is considered to be relevant.
133  for (auto state : stateInformation.relevantStates) {
134  result.relevantChoicesForRelevantStates.emplace(state, std::list<uint_fast64_t>());
135  if (stateInformation.problematicStates.get(state)) {
136  result.problematicChoicesForProblematicStates.emplace(state, std::list<uint_fast64_t>());
137  }
138  for (uint_fast64_t row = nondeterministicChoiceIndices[state]; row < nondeterministicChoiceIndices[state + 1]; ++row) {
139  bool currentChoiceRelevant = false;
140  bool allSuccessorsProblematic = true;
141  for (auto const& successorEntry : transitionMatrix.getRow(row)) {
142  // If there is a relevant successor, we need to add the labels of the current choice.
143  if (stateInformation.relevantStates.get(successorEntry.getColumn()) || psiStates.get(successorEntry.getColumn())) {
144  for (auto const& label : labelSets[row]) {
145  result.allRelevantLabels.insert(label);
146  }
147  if (!currentChoiceRelevant) {
148  currentChoiceRelevant = true;
149  result.relevantChoicesForRelevantStates[state].push_back(row);
150  }
151  }
152  if (!stateInformation.problematicStates.get(successorEntry.getColumn())) {
153  allSuccessorsProblematic = false;
154  }
155  }
156 
157  // If all successors of a problematic state are problematic themselves, we record this choice
158  // as being problematic.
159  if (stateInformation.problematicStates.get(state) && allSuccessorsProblematic) {
160  result.problematicChoicesForProblematicStates[state].push_back(row);
161  }
162  }
163  }
164 
165  // Finally, determine the set of labels that are known to be taken.
166  result.knownLabels = storm::counterexamples::getGuaranteedLabelSet(mdp, labelSets, psiStates, result.allRelevantLabels);
167  STORM_LOG_DEBUG("Found " << result.allRelevantLabels.size() << " relevant labels and " << result.knownLabels.size() << " known labels.");
168 
169  return result;
170  }

363  static VariableInformation createVariables(storm::solver::LpSolver<double>& solver, storm::models::sparse::Mdp<T> const& mdp, StateInformation const& stateInformation, ChoiceInformation const& choiceInformation) {

369  result.labelToVariableMap = std::move(labelVariableResult.first);
370  result.numberOfVariables += labelVariableResult.second;

372

374  std::pair<std::unordered_map<uint_fast64_t, std::list<storm::expressions::Variable>>, uint_fast64_t> schedulerVariableResult = createSchedulerVariables(solver, stateInformation, choiceInformation);

381  result.initialStateToChoiceVariableMap = std::move(initialChoiceVariableResult.first);
382  result.numberOfVariables += initialChoiceVariableResult.second;

384 
385  // Create variables for probabilities for all relevant states.

387  result.stateToProbabilityVariableMap = std::move(probabilityVariableResult.first);
388  result.numberOfVariables += probabilityVariableResult.second;
389  STORM_LOG_DEBUG("Created variables for the reachability probabilities.");

392  std::pair<storm::expressions::Variable, uint_fast64_t> virtualInitialStateVariableResult = createVirtualInitialStateVariable(solver);
393  result.virtualInitialStateVariable = virtualInitialStateVariableResult.first;
394  result.numberOfVariables += virtualInitialStateVariableResult.second;
395  STORM_LOG_DEBUG("Created variables for the virtual initial state.");
396 
397  // Create variables for problematic states.

400  result.numberOfVariables += problematicStateVariableResult.second;
401  STORM_LOG_DEBUG("Created variables for the problematic states.");
402 
403  // Create variables for problematic choices.

405  result.problematicTransitionToVariableMap = problematicTransitionVariableResult.first;
406  result.numberOfVariables += problematicTransitionVariableResult.second;
407  STORM_LOG_DEBUG("Created variables for the problematic choices.");
408 
409  // Finally, we need to update the model to make the new variables usable.
410  solver.update();
411  STORM_LOG_INFO("Successfully created " << result.numberOfVariables << " MILP variables.");
412 
413  // Finally, return variable information struct.
414  return result;
415  }

447  // Assert that the policy chooses at most one action in each state of the system.
448  uint_fast64_t numberOfConstraintsCreated = 0;

472  ++numberOfConstraintsCreated;

630  * @param variableInformation A struct with information about the variables of the model.
631  * @return The total number of constraints that were created.
632  */
633  static uint_fast64_t assertKnownLabels(storm::solver::LpSolver<double>& solver, ChoiceInformation const& choiceInformation, VariableInformation const& variableInformation) {
634  uint_fast64_t numberOfConstraintsCreated = 0;
635 
636  for (auto label : choiceInformation.knownLabels) {
637  storm::expressions::Expression constraint = variableInformation.labelToVariableMap.at(label) == solver.getConstant(1);
638  solver.addConstraint("KnownLabels" + std::to_string(numberOfConstraintsCreated), constraint);
639  ++numberOfConstraintsCreated;
640  }
641 
642  return numberOfConstraintsCreated;

656  static uint_fast64_t assertSchedulerCuts(storm::solver::LpSolver<double>& solver, storm::models::sparse::Mdp<T> const& mdp, storm::storage::BitVector const& psiStates, StateInformation const& stateInformation, ChoiceInformation const& choiceInformation, VariableInformation const& variableInformation) {
657  storm::storage::SparseMatrix<T> backwardTransitions = mdp.getBackwardTransitions();
658  uint_fast64_t numberOfConstraintsCreated = 0;
659 
660  for (auto state : stateInformation.relevantStates) {
661  // Assert that all states, that select an action, this action either has a non-zero probability to
662  // go to a psi state directly, or in the successor states, at least one action is selected as well.
663  std::list<storm::expressions::Variable>::const_iterator choiceVariableIterator = variableInformation.stateToChoiceVariablesMap.at(state).begin();
664  for (auto choice : choiceInformation.relevantChoicesForRelevantStates.at(state)) {
665  bool psiStateReachableInOneStep = false;
666  for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(choice)) {
667  if (psiStates.get(successorEntry.getColumn())) {
668  psiStateReachableInOneStep = true;
669  }
670  }
671 
672  if (!psiStateReachableInOneStep) {
673  storm::expressions::Expression constraint = *choiceVariableIterator;
674  for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(choice)) {
675  if (state != successorEntry.getColumn() && stateInformation.relevantStates.get(successorEntry.getColumn())) {
676  std::list<storm::expressions::Variable> const& successorChoiceVariableIndices = variableInformation.stateToChoiceVariablesMap.at(successorEntry.getColumn());
677 
678  for (auto const& choiceVariable : successorChoiceVariableIndices) {
679  constraint = constraint - choiceVariable;
680  }
681  }
682  }
683  constraint = constraint <= solver.getConstant(1);
684 
685  solver.addConstraint("SchedulerCuts" + std::to_string(numberOfConstraintsCreated), constraint);
686  ++numberOfConstraintsCreated;
687  }
688 
689  ++choiceVariableIterator;
690  }
691 
692  // For all states assert that there is either a selected incoming transition in the subsystem or the
693  // state is the chosen initial state if there is one selected action in the current state.
694  storm::expressions::Expression constraint = solver.getConstant(0);
695 
696  for (auto const& choiceVariable : variableInformation.stateToChoiceVariablesMap.at(state)) {
697  constraint = constraint + choiceVariable;
698  }
699 
700  // Compute the set of predecessors.
701  std::unordered_set<uint_fast64_t> predecessors;
702  for (auto const& predecessorEntry : backwardTransitions.getRow(state)) {
703  if (state != predecessorEntry.getColumn()) {
704  predecessors.insert(predecessorEntry.getColumn());
705  }
706  }
707 
708  for (auto predecessor : predecessors) {
709  // If the predecessor is not a relevant state, we need to skip it.
710  if (!stateInformation.relevantStates.get(predecessor)) {
711  continue;
712  }
713 
714  std::list<storm::expressions::Variable>::const_iterator choiceVariableIterator = variableInformation.stateToChoiceVariablesMap.at(predecessor).begin();
715  for (auto relevantChoice : choiceInformation.relevantChoicesForRelevantStates.at(predecessor)) {
716  bool choiceTargetsCurrentState = false;
717 
718  // Check if the current choice targets the current state.
719  for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(relevantChoice)) {
720  if (state == successorEntry.getColumn()) {
721  choiceTargetsCurrentState = true;
722  break;
723  }
724  }
725 
726  // If it does, we can add the choice to the sum.
727  if (choiceTargetsCurrentState) {
728  constraint = constraint - *choiceVariableIterator;
729  }
730  ++choiceVariableIterator;
731  }
732  }
733 
734  // If the current state is an initial state and is selected as a successor state by the virtual
735  // initial state, then this also justifies making a choice in the current state.
736  if (mdp.getStates("init").get(state)) {
737  constraint = constraint - variableInformation.initialStateToChoiceVariableMap.at(state);
738  }
739  constraint = constraint <= solver.getConstant(0);
740 
741  solver.addConstraint("SchedulerCuts" + std::to_string(numberOfConstraintsCreated), constraint);
742  ++numberOfConstraintsCreated;
743  }
744 
745  // Assert that at least one initial state selects at least one action.
746  storm::expressions::Expression constraint = solver.getConstant(0);
747  for (auto initialState : mdp.getStates("init")) {
748  for (auto const& choiceVariable : variableInformation.stateToChoiceVariablesMap.at(initialState)) {
749  constraint = constraint + choiceVariable;
750  }
751  }
752  constraint = constraint >= solver.getConstant(1);
753  solver.addConstraint("SchedulerCuts" + std::to_string(numberOfConstraintsCreated), constraint);
754  ++numberOfConstraintsCreated;
755 
756  // Add constraints that ensure at least one choice is selected that targets a psi state.
757  constraint = solver.getConstant(0);
758  std::unordered_set<uint_fast64_t> predecessors;
759  for (auto psiState : psiStates) {
760  // Compute the set of predecessors.
761  for (auto const& predecessorEntry : backwardTransitions.getRow(psiState)) {
762  if (psiState != predecessorEntry.getColumn()) {
763  predecessors.insert(predecessorEntry.getColumn());
764  }
765  }
766  }
767 
768  for (auto predecessor : predecessors) {
769  // If the predecessor is not a relevant state, we need to skip it.
770  if (!stateInformation.relevantStates.get(predecessor)) {
771  continue;
772  }
773 
774  std::list<storm::expressions::Variable>::const_iterator choiceVariableIterator = variableInformation.stateToChoiceVariablesMap.at(predecessor).begin();
775  for (auto relevantChoice : choiceInformation.relevantChoicesForRelevantStates.at(predecessor)) {
776  bool choiceTargetsPsiState = false;
777 
778  // Check if the current choice targets the current state.
779  for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(relevantChoice)) {
780  if (psiStates.get(successorEntry.getColumn())) {
781  choiceTargetsPsiState = true;
782  break;
783  }
784  }
785 
786  // If it does, we can add the choice to the sum.
787  if (choiceTargetsPsiState) {
788  constraint = constraint + *choiceVariableIterator;
789  }
790  ++choiceVariableIterator;
791  }
792  }
793  constraint = constraint >= solver.getConstant(1);
794 
795  solver.addConstraint("SchedulerCuts" + std::to_string(numberOfConstraintsCreated), constraint);
796  ++numberOfConstraintsCreated;
797 
798  return numberOfConstraintsCreated;
799  }

816  static void buildConstraintSystem(storm::solver::LpSolver<double>& solver, storm::models::sparse::Mdp<T> const& mdp, std::vector<storm::storage::FlatSet<uint_fast64_t>> const& labelSets, storm::storage::BitVector const& psiStates, StateInformation const& stateInformation, ChoiceInformation const& choiceInformation, VariableInformation const& variableInformation, double probabilityThreshold, bool strictBound, bool includeSchedulerCuts = false) {
817  // Assert that the reachability probability in the subsystem exceeds the given threshold.
818  uint_fast64_t numberOfConstraints = assertProbabilityGreaterThanThreshold(solver, variableInformation, probabilityThreshold, strictBound);
819  STORM_LOG_DEBUG("Asserted that reachability probability exceeds threshold.");
820 
821  // Add constraints that assert the policy takes at most one action in each state.
822  numberOfConstraints += assertValidPolicy(solver, stateInformation, variableInformation);
823  STORM_LOG_DEBUG("Asserted that policy is valid.");
824 
825  // Add constraints that assert the labels that belong to some taken choices are taken as well.
826  numberOfConstraints += assertChoicesImplyLabels(solver, mdp, labelSets, stateInformation, choiceInformation, variableInformation);
827  STORM_LOG_DEBUG("Asserted that labels implied by choices are taken.");
828 
829  // Add constraints that encode that the reachability probability from states which do not pick any action
830  // is zero.
831  numberOfConstraints += assertZeroProbabilityWithoutChoice(solver, stateInformation, variableInformation);
832  STORM_LOG_DEBUG("Asserted that reachability probability is zero if no choice is taken.");
833 
834  // Add constraints that encode the reachability probabilities for states.
835  numberOfConstraints += assertReachabilityProbabilities(solver, mdp, psiStates, stateInformation, choiceInformation, variableInformation);
836  STORM_LOG_DEBUG("Asserted constraints for reachability probabilities.");
837 
838  // Add constraints that ensure the reachability of an unproblematic state from each problematic state.
839  numberOfConstraints += assertUnproblematicStateReachable(solver, mdp, stateInformation, choiceInformation, variableInformation);
840  STORM_LOG_DEBUG("Asserted that unproblematic state reachable from problematic states.");
841 
842  // Add constraints that express that certain labels are already known to be taken.
843  numberOfConstraints += assertKnownLabels(solver, choiceInformation, variableInformation);
844  STORM_LOG_DEBUG("Asserted known labels are taken.");
845 
846  // If required, assert additional constraints that reduce the number of possible policies.
847  if (includeSchedulerCuts) {
848  numberOfConstraints += assertSchedulerCuts(solver, mdp, psiStates, stateInformation, choiceInformation, variableInformation);
849  STORM_LOG_DEBUG("Asserted scheduler cuts.");
850  }
851 
852  // Finally, we can tell the solver to incorporate the latest changes.
853  solver.update();
854 
855  STORM_LOG_INFO("Successfully created " << numberOfConstraints << " MILP constraints.");
856  }

926  static storm::storage::FlatSet<uint_fast64_t> getMinimalLabelSet(Environment const& env,storm::models::sparse::Mdp<T> const& mdp, std::vector<storm::storage::FlatSet<uint_fast64_t>> const& labelSets, storm::storage::BitVector const& phiStates, storm::storage::BitVector const& psiStates, double probabilityThreshold, bool strictBound, bool checkThresholdFeasible = false, bool includeSchedulerCuts = false) {
927  // (0) Check whether the label sets are valid
928  STORM_LOG_THROW(mdp.getNumberOfChoices() == labelSets.size(), storm::exceptions::InvalidArgumentException, "The given number of labels does not match the number of choices.");
929 
930  // (1) Check whether its possible to exceed the threshold if checkThresholdFeasible is set.
931  double maximalReachabilityProbability = 0;
932  if (checkThresholdFeasible) {
933  storm::modelchecker::helper::SparseMdpPrctlHelper<T> modelcheckerHelper;
934  std::vector<T> result = std::move(modelcheckerHelper.computeUntilProbabilities(env, false, mdp.getTransitionMatrix(), mdp.getBackwardTransitions(), phiStates, psiStates, false, false).values);
935  for (auto state : mdp.getInitialStates()) {
936  maximalReachabilityProbability = std::max(maximalReachabilityProbability, result[state]);
937  }
938  STORM_LOG_THROW((strictBound && maximalReachabilityProbability >= probabilityThreshold) || (!strictBound && maximalReachabilityProbability > probabilityThreshold), storm::exceptions::InvalidArgumentException, "Given probability threshold " << probabilityThreshold << " can not be " << (strictBound ? "achieved" : "exceeded") << " in model with maximal reachability probability of " << maximalReachabilityProbability << ".");
939  std::cout << std::endl << "Maximal reachability in model is " << maximalReachabilityProbability << "." << std::endl << std::endl;
940  }
941 
942  // (2) Identify relevant and problematic states.
943  StateInformation stateInformation = determineRelevantAndProblematicStates(mdp, phiStates, psiStates);
944 
945  // (3) Determine sets of relevant labels and problematic choices.
946  ChoiceInformation choiceInformation = determineRelevantAndProblematicChoices(mdp, labelSets, stateInformation, psiStates);
947 
948  // (4) Encode resulting system as MILP problem.
949  std::shared_ptr<storm::solver::LpSolver<double>> solver = storm::utility::solver::getLpSolver<double>("MinimalLabelSetCounterexample");
950 
951  // (4.1) Create variables.
952  VariableInformation variableInformation = createVariables(*solver, mdp, stateInformation, choiceInformation);
953 
954  // (4.2) Construct constraint system.
955  buildConstraintSystem(*solver, mdp, labelSets, psiStates, stateInformation, choiceInformation, variableInformation, probabilityThreshold, strictBound, includeSchedulerCuts);
956 
957  // (4.3) Optimize the model.
958  solver->optimize();
959 
960  // (4.4) Read off result from variables.
961  storm::storage::FlatSet<uint_fast64_t> usedLabelSet = getUsedLabelsInSolution(*solver, variableInformation);
962  usedLabelSet.insert(choiceInformation.knownLabels.begin(), choiceInformation.knownLabels.end());
963 
964  // (5) Return result.
965  return usedLabelSet;
966  }

975  static std::shared_ptr<HighLevelCounterexample> computeCounterexample(Environment const& env, storm::storage::SymbolicModelDescription const& symbolicModel, storm::models::sparse::Mdp<T> const& mdp, std::shared_ptr<storm::logic::Formula const> const& formula) {
976  std::cout << std::endl << "Generating minimal label counterexample for formula " << *formula << std::endl;
977 
978  // Check whether there are choice origins available
979  STORM_LOG_THROW(mdp.hasChoiceOrigins(), storm::exceptions::InvalidArgumentException, "Restriction to minimal command set is impossible for model without choice origns.");
980  STORM_LOG_THROW(mdp.getChoiceOrigins()->isPrismChoiceOrigins() || mdp.getChoiceOrigins()->isJaniChoiceOrigins(), storm::exceptions::InvalidArgumentException, "Restriction to command set is impossible for model without PRISM or JANI choice origins.");
981 
982  STORM_LOG_THROW(formula->isProbabilityOperatorFormula(), storm::exceptions::InvalidPropertyException, "Counterexample generation does not support this kind of formula. Expecting a probability operator as the outermost formula element.");
983  storm::logic::ProbabilityOperatorFormula const& probabilityOperator = formula->asProbabilityOperatorFormula();
984  STORM_LOG_THROW(probabilityOperator.hasBound(), storm::exceptions::InvalidPropertyException, "Counterexample generation only supports bounded formulas.");
985  storm::logic::ComparisonType comparisonType = probabilityOperator.getComparisonType();
986  STORM_LOG_THROW(comparisonType == storm::logic::ComparisonType::Less || comparisonType == storm::logic::ComparisonType::LessEqual, storm::exceptions::InvalidPropertyException, "Counterexample generation only supports formulas with an upper probability bound.");
987  STORM_LOG_THROW(probabilityOperator.getSubformula().isUntilFormula() || probabilityOperator.getSubformula().isEventuallyFormula(), storm::exceptions::InvalidPropertyException, "Path formula is required to be of the form 'phi U psi' for counterexample generation.");
988 
989  bool strictBound = comparisonType == storm::logic::ComparisonType::Less;
990  double threshold = probabilityOperator.getThresholdAs<double>();
991 
992  storm::storage::BitVector phiStates;
993  storm::storage::BitVector psiStates;
994  storm::modelchecker::SparsePropositionalModelChecker<storm::models::sparse::Mdp<T>> modelchecker(mdp);
995 
996  if (probabilityOperator.getSubformula().isUntilFormula()) {
997  storm::logic::UntilFormula const& untilFormula = probabilityOperator.getSubformula().asUntilFormula();
998 
999  std::unique_ptr<storm::modelchecker::CheckResult> leftResult = modelchecker.check(env, untilFormula.getLeftSubformula());
1000  std::unique_ptr<storm::modelchecker::CheckResult> rightResult = modelchecker.check(env, untilFormula.getRightSubformula());
1001 
1002  storm::modelchecker::ExplicitQualitativeCheckResult const& leftQualitativeResult = leftResult->asExplicitQualitativeCheckResult();
1003  storm::modelchecker::ExplicitQualitativeCheckResult const& rightQualitativeResult = rightResult->asExplicitQualitativeCheckResult();
1004 
1005  phiStates = leftQualitativeResult.getTruthValuesVector();
1006  psiStates = rightQualitativeResult.getTruthValuesVector();
1007  } else if (probabilityOperator.getSubformula().isEventuallyFormula()) {
1008  storm::logic::EventuallyFormula const& eventuallyFormula = probabilityOperator.getSubformula().asEventuallyFormula();
1009 
1010  std::unique_ptr<storm::modelchecker::CheckResult> subResult = modelchecker.check(env, eventuallyFormula.getSubformula());
1011 
1012  storm::modelchecker::ExplicitQualitativeCheckResult const& subQualitativeResult = subResult->asExplicitQualitativeCheckResult();
1013 
1014  phiStates = storm::storage::BitVector(mdp.getNumberOfStates(), true);
1015  psiStates = subQualitativeResult.getTruthValuesVector();
1016  }
1017 
1018  // Obtain the label sets for each choice.
1019  std::vector<storm::storage::FlatSet<uint_fast64_t>> labelSets(mdp.getNumberOfChoices());
1020  if (mdp.getChoiceOrigins()->isPrismChoiceOrigins()) {
1021  storm::storage::sparse::PrismChoiceOrigins const& choiceOrigins = mdp.getChoiceOrigins()->asPrismChoiceOrigins();
1022  for (uint_fast64_t choice = 0; choice < mdp.getNumberOfChoices(); ++choice) {
1023  labelSets[choice] = choiceOrigins.getCommandSet(choice);
1024  }
1025  } else {
1026  storm::storage::sparse::JaniChoiceOrigins const& choiceOrigins = mdp.getChoiceOrigins()->asJaniChoiceOrigins();
1027 
1028  // The choice origins are known to be JANI ones at this point.
1029  for (uint_fast64_t choice = 0; choice < mdp.getNumberOfChoices(); ++choice) {
1030  labelSets[choice] = choiceOrigins.getEdgeIndexSet(choice);
1031  }
1032  }
1033 
1034  // Delegate the actual computation work to the function of equal name.
1035  auto startTime = std::chrono::high_resolution_clock::now();
1036  storm::storage::FlatSet<uint_fast64_t> usedLabelSet = getMinimalLabelSet(env, mdp, labelSets, phiStates, psiStates, threshold, strictBound, true, storm::settings::getModule<storm::settings::modules::CounterexampleGeneratorSettings>().isUseSchedulerCutsSet());
1037  auto endTime = std::chrono::high_resolution_clock::now();
1038  std::cout << std::endl << "Computed minimal command set of size " << usedLabelSet.size() << " in " << std::chrono::duration_cast<std::chrono::milliseconds>(endTime - startTime).count() << "ms." << std::endl;
1039 
1040 
1041  if (symbolicModel.isPrismProgram()) {
1042  return std::make_shared<HighLevelCounterexample>(symbolicModel.asPrismProgram().restrictCommands(usedLabelSet));
1043  } else {
1044  STORM_LOG_ASSERT(symbolicModel.isJaniModel(), "Unknown symbolic model description type.");
1045  return std::make_shared<HighLevelCounterexample>(symbolicModel.asJaniModel().restrictEdges(usedLabelSet));
1046  }
1047  }
1048
