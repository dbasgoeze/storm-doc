69  std::string shellQuoteSingleIfNecessary(const std::string& arg) {
70  // quote empty argument
71  if (arg.empty()) {
72  return "''";
73  }
74 
75  if (arg.find_first_not_of("abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789-_./=") != std::string::npos) {
76  // contains potentially unsafe character, needs quoting
77  if (arg.find('\'') != std::string::npos) {
78  // contains ', we have to replace all ' with '\''
79  std::string escaped(arg);
80  boost::replace_all(escaped, "'", "'\\''");
81  return "'" + escaped + "'";
82  } else {
83  return "'" + arg + "'";
84  }
85  }
86 
87  return arg;
