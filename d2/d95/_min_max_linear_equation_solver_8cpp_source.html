207  if (method == MinMaxMethod::ValueIteration || method == MinMaxMethod::PolicyIteration || method == MinMaxMethod::RationalSearch || method == MinMaxMethod::IntervalIteration || method == MinMaxMethod::SoundValueIteration || method == MinMaxMethod::OptimisticValueIteration || method == MinMaxMethod::ViToPi) {
208  result = std::make_unique<IterativeMinMaxLinearEquationSolver<ValueType>>(std::make_unique<GeneralLinearEquationSolverFactory<ValueType>>());

211  } else if (method == MinMaxMethod::TopologicalCuda) {
212  result = std::make_unique<TopologicalCudaMinMaxLinearEquationSolver<ValueType>>();
213  } else if (method == MinMaxMethod::LinearProgramming) {
214  result = std::make_unique<LpMinMaxLinearEquationSolver<ValueType>>(std::make_unique<storm::utility::solver::LpSolverFactory<ValueType>>());
215  } else if (method == MinMaxMethod::Acyclic) {
216  result = std::make_unique<AcyclicMinMaxLinearEquationSolver<ValueType>>();
217  } else {
218  STORM_LOG_THROW(false, storm::exceptions::InvalidSettingsException, "Unsupported technique.");
