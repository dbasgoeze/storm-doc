51  uint_fast64_t lhsNumStates = this->getTransitionMatrix().size();
52  uint_fast64_t rhsNumStates = rhs.getTransitionMatrix().size();
53  uint_fast64_t resNumStates = lhsNumStates * rhsNumStates;
54 
55  // Transition matrix
56  TransitionMatrix resultTransitions(resNumStates, std::vector<boost::optional<storm::storage::BitVector>>(resNumStates));
57  uint_fast64_t resState = 0;
58  for (uint_fast64_t lhsState = 0; lhsState < lhsNumStates; ++lhsState) {
59  for (uint_fast64_t rhsState = 0; rhsState < rhsNumStates; ++rhsState) {
60  assert (resState == (lhsState * rhsNumStates) + rhsState);
61  auto& resStateTransitions = resultTransitions[resState];
62  for (uint_fast64_t lhsTransitionTarget = 0; lhsTransitionTarget < lhsNumStates; ++lhsTransitionTarget) {
63  auto& lhsTransition = this->getTransitionMatrix()[lhsState][lhsTransitionTarget];
64  if (lhsTransition) {
65  for (uint_fast64_t rhsTransitionTarget = 0; rhsTransitionTarget < rhsNumStates; ++rhsTransitionTarget) {
66  auto& rhsTransition = rhs.getTransitionMatrix()[rhsState][rhsTransitionTarget];
67  if (rhsTransition) {
68  uint_fast64_t resTransitionTarget = (lhsTransitionTarget * rhsNumStates) + rhsTransitionTarget;
69  resStateTransitions[resTransitionTarget] = rhsTransition.get() & lhsTransition.get();
70  // If it is not possible to take the considered transition w.r.t. the considered model, we can delete it.
71  if (resStateTransitions[resTransitionTarget]->empty()) {
72  resStateTransitions[resTransitionTarget] = boost::none;
73  }
74  }
75  }
76  }
77  }
78  ++resState;
79  }
80  }
81 
82  // State Labels
83  storm::models::sparse::StateLabeling resultLabeling(resNumStates);
84  for (std::string lhsLabel : this->getStateLabeling().getLabels()) {
85  storm::storage::BitVector const& lhsLabeledStates = this->getStateLabeling().getStates(lhsLabel);
86  storm::storage::BitVector resLabeledStates(resNumStates, false);
87  for (auto const& lhsState : lhsLabeledStates) {
88  for (uint_fast64_t rhsState = 0; rhsState < rhsNumStates; ++rhsState) {
89  resState = (lhsState * rhsNumStates) + rhsState;
90  resLabeledStates.set(resState, true);
91  }
92  }

94  }

97  storm::storage::BitVector const& rhsLabeledStates = rhs.getStateLabeling().getStates(rhsLabel);
98  storm::storage::BitVector resLabeledStates(resNumStates, false);
99  for (auto const& rhsState : rhsLabeledStates) {
100  for (uint_fast64_t lhsState = 0; lhsState < lhsNumStates; ++lhsState) {
101  resState = (lhsState * rhsNumStates) + rhsState;
102  resLabeledStates.set(resState, true);
103  }
104  }
105  resultLabeling.addLabel(rhsLabel, std::move(resLabeledStates));
106  }
107

109  std::vector<uint_fast64_t> resultInitialMemoryStates;
110  STORM_LOG_THROW(this->getInitialMemoryStates().size() == rhs.getInitialMemoryStates().size(), storm::exceptions::InvalidOperationException, "Tried to build the product of two memory structures that consider a different number of initial model states.");
111  resultInitialMemoryStates.reserve(this->getInitialMemoryStates().size());

113  auto rhsStateIt = rhs.getInitialMemoryStates().begin();
114  for (; lhsStateIt != this->getInitialMemoryStates().end(); ++lhsStateIt, ++rhsStateIt) {
115  resultInitialMemoryStates.push_back(*lhsStateIt * rhsNumStates + *rhsStateIt);
116  }
