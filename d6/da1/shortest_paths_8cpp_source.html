143  template <typename T>
144  void ShortestPathsGenerator<T>::performDijkstra() {
145  // the existing Dijkstra isn't working anyway AND
146  // doesn't fully meet our requirements, so let's roll our own
147 
148  T inftyDistance = zero<T>();
149  T zeroDistance = one<T>();
150  shortestPathDistances.resize(numStates, inftyDistance);
151  shortestPathPredecessors.resize(numStates, boost::optional<state_t>());
152 
153  // set serves as priority queue with unique membership
154  // default comparison on pair actually works fine if distance is the first entry
155  std::set<std::pair<T, state_t>, std::greater<std::pair<T, state_t>>> dijkstraQueue;
156 
157  for (state_t initialState : initialStates) {
158  shortestPathDistances[initialState] = zeroDistance;
159  dijkstraQueue.emplace(zeroDistance, initialState);
160  }
161 
162  while (!dijkstraQueue.empty()) {
163  state_t currentNode = (*dijkstraQueue.begin()).second;
164  dijkstraQueue.erase(dijkstraQueue.begin());
165 
166  if (!isMetaTargetPredecessor(currentNode)) {
167  // non-target node, treated normally
168  for (auto const& transition : transitionMatrix.getRowGroup(currentNode)) {
169  state_t otherNode = transition.getColumn();
170 
171  // note that distances are probabilities, thus they are multiplied and larger is better
172  T alternateDistance = shortestPathDistances[currentNode] * convertDistance(currentNode, otherNode, transition.getValue());
173  assert((zero<T>() <= alternateDistance) && (alternateDistance <= one<T>()));
174  if (alternateDistance > shortestPathDistances[otherNode]) {
175  shortestPathDistances[otherNode] = alternateDistance;
176  shortestPathPredecessors[otherNode] = boost::optional<state_t>(currentNode);
177  dijkstraQueue.emplace(alternateDistance, otherNode);
178  }
179  }
180  } else {
181  // node only has one "virtual edge" (with prob as per targetProbMap) to meta-target
182  T alternateDistance = shortestPathDistances[currentNode] * targetProbMap[currentNode];
183  if (alternateDistance > shortestPathDistances[metaTarget]) {
184  shortestPathDistances[metaTarget] = alternateDistance;
185  shortestPathPredecessors[metaTarget] = boost::optional<state_t>(currentNode);
186  }
187  // no need to enqueue meta-target
188  }
189  }

206  kShortestPaths.resize(numStates);
207 
208  // BFS in Dijkstra-SP order
209  std::queue<state_t> bfsQueue;
210  for (state_t initialState : initialStates) {
211  bfsQueue.push(initialState);
212  }
213 
214  while (!bfsQueue.empty()) {
215  state_t currentNode = bfsQueue.front();
216  bfsQueue.pop();

219  continue; // already visited
220  }
221

223  bfsQueue.push(successorNode);
224  }

227  // if current node is an initial state
228  // in this case, the boost::optional copy of an uninitialized optional is hopefully also uninitialized
229  kShortestPaths[currentNode].push_back(Path<T> {

232  shortestPathDistances[currentNode]

261  void ShortestPathsGenerator<T>::computeNextPath(state_t node, unsigned long k) {
262  assert(k >= 2); // Dijkstra is used for k=1
263  assert(kShortestPaths[node].size() == k - 1); // if not, the previous SP must not exist
264 
265  // TODO: I could extract the candidate generation to make this function more succinct
266  if (k == 2) {
267  // Step B.1 in J&M paper
268 
269  Path<T> shortestPathToNode = kShortestPaths[node][1 - 1]; // never forget index shift :-|
270 
271  for (state_t predecessor : graphPredecessors[node]) {
272  // add shortest paths to predecessors plus edge to current node
273  Path<T> pathToPredecessorPlusEdge = {
274  boost::optional<state_t>(predecessor),
275  1,
276  shortestPathDistances[predecessor] * getEdgeDistance(predecessor, node)
277  };
278  candidatePaths[node].insert(pathToPredecessorPlusEdge);
279 
280  // ... but not the actual shortest path
281  auto it = std::find(candidatePaths[node].begin(), candidatePaths[node].end(), shortestPathToNode);
282  if (it != candidatePaths[node].end()) {
283  candidatePaths[node].erase(it);
284  }
285  }
286  }
287 
288  if (not (k == 2 && isInitialState(node))) {
289  // Steps B.2-5 in J&M paper
290 
291  // the (k-1)th shortest path (i.e., one better than the one we want to compute)
292  Path<T> previousShortestPath = kShortestPaths[node][k - 1 - 1]; // oh god, I forgot index shift AGAIN
293 
294  // the predecessor node on that path
295  state_t predecessor = previousShortestPath.predecessorNode.get();
296  // the path to that predecessor was the `tailK`-shortest
297  unsigned long tailK = previousShortestPath.predecessorK;
298 
299  // i.e. source ~~tailK-shortest path~~> predecessor --> node
300 
301  // compute one-worse-shortest path to the predecessor (if it hasn't yet been computed)
302  if (kShortestPaths[predecessor].size() < tailK + 1) {
303  // TODO: investigate recursion depth and possible iterative alternative
304  computeNextPath(predecessor, tailK + 1);
305  }
306 
307  if (kShortestPaths[predecessor].size() >= tailK + 1) {
308  // take that path, add an edge to the current node; that's a candidate
309  Path<T> pathToPredecessorPlusEdge = {
310  boost::optional<state_t>(predecessor),
311  tailK + 1,
312  kShortestPaths[predecessor][tailK + 1 - 1].distance * getEdgeDistance(predecessor, node)
313  };
314  candidatePaths[node].insert(pathToPredecessorPlusEdge);
315  }
316  // else there was no path; TODO: does this need handling? -- yes, but not here (because the step B.1 may have added candidates)
317  }
318 
319  // Step B.6 in J&M paper
320  if (!candidatePaths[node].empty()) {
321  Path<T> minDistanceCandidate = *(candidatePaths[node].begin());
322  for (auto path : candidatePaths[node]) {
323  if (path.distance > minDistanceCandidate.distance) {
324  minDistanceCandidate = path;
325  }
326  }
327 
328  candidatePaths[node].erase(std::find(candidatePaths[node].begin(), candidatePaths[node].end(), minDistanceCandidate));
329  kShortestPaths[node].push_back(minDistanceCandidate);
330  } else {
331  // TODO: kSP does not exist. this is handled later, but it would be nice to catch it as early as possble, wouldn't it?
332  STORM_LOG_TRACE("KSP: no candidates, this will trigger nonexisting ksp after exiting these recursions. TODO: handle here");
333  }

337  void ShortestPathsGenerator<T>::computeKSP(unsigned long k) {
338  if (k == 0) {
339  throw std::invalid_argument("Index 0 is invalid, since we use 1-based indices (sorry)!");
340  }
341 
342  unsigned long alreadyComputedK = kShortestPaths[metaTarget].size();
343 
344  for (unsigned long nextK = alreadyComputedK + 1; nextK <= k; nextK++) {
345  computeNextPath(metaTarget, nextK);
346  if (kShortestPaths[metaTarget].size() < nextK) {
347  unsigned long lastExistingK = nextK - 1;
348  STORM_LOG_DEBUG("KSP throws (as expected) due to nonexistence -- maybe this is unhandled and causes the Python interface to segfault?");
349  STORM_LOG_DEBUG("last existing k-SP has k=" + std::to_string(lastExistingK));
350  STORM_LOG_DEBUG("maybe this is unhandled and causes the Python interface to segfault?");
351  throw std::invalid_argument("k-SP does not exist for k=" + std::to_string(k));
