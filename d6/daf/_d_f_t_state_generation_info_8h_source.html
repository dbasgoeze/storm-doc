116  void generateSymmetries(storm::storage::DFTIndependentSymmetries const& symmetries) {
117  // Iterate over possible children
118  for (size_t i = 0; i < mSymmetries.size(); ++i) {
119  size_t childStart = mSymmetries[i].second[0];
120  size_t childLength = mSymmetries[i].first;
121  // Iterate over possible parents
122  for (size_t j = i + 1; j < mSymmetries.size(); ++j) {
123  size_t parentStart = mSymmetries[j].second[0];
124  size_t parentLength = mSymmetries[j].first;
125  // Check if child lies in parent
126  if (parentStart <= childStart && childStart + childLength < parentStart + parentLength) {
127  // We add the symmetry of the child to all symmetric elements in the parent
128  std::vector<std::vector<size_t>> newSymmetries;
129  // Start iteration at 1, because symmetry for child at 0 is already included
130  for (size_t index = 1; index < mSymmetries[j].second.size(); ++index) {
131  std::vector<size_t> newStarts;
132  // Apply child symmetry to all symmetric elements of parent
133  for (size_t symmetryStarts : mSymmetries[i].second) {
134  // Get symmetric element by applying the bijection
135  size_t symmetryOffset = symmetryStarts - parentStart;
136  newStarts.push_back(mSymmetries[j].second[index] + symmetryOffset);
137  }
138  newSymmetries.push_back(newStarts);
139  }
140  // Insert new symmetry after child
141  for (size_t index = 0; index < newSymmetries.size(); ++index) {
142  mSymmetries.insert(mSymmetries.begin() + i + 1 + index, std::make_pair(childLength, newSymmetries[index]));
143  }
144  i += newSymmetries.size();
145  break;
146  }
147  }
148  }
149  }
