116  return true;
117  }

125  std::pair<Point, bool> optimizationResOverApprox = this->overApproximation->optimize(weights);
126  if (optimizationResOverApprox.second) {
127  std::pair<Point, bool> optimizationResUnderApprox = this->underApproximation->optimize(weights);
128  if (optimizationResUnderApprox.second) {
129  GeometryValueType distance = storm::utility::vector::dotProduct(optimizationResOverApprox.first, weights) - storm::utility::vector::dotProduct(optimizationResUnderApprox.first, weights);
130  STORM_LOG_ASSERT(distance >= storm::utility::zero<GeometryValueType>(), "Negative distance between under- and over approximation was not expected");

133  distance /= GeometryValueType(2);

135  }
136  }

141  GeometryValueType SparsePcaaQuantitativeQuery<SparseModelType, GeometryValueType>::improveSolution(Environment const& env) {
142  STORM_LOG_THROW(env.modelchecker().multi().getPrecisionType() == MultiObjectiveModelCheckerEnvironment::PrecisionType::Absolute, storm::exceptions::IllegalArgumentException, "Unhandled multiobjective precision type.");
143  this->diracWeightVectorsToBeChecked.clear(); // Only check weight vectors that can actually improve the solution
144 
145  WeightVector directionOfOptimizingObjective(this->objectives.size(), storm::utility::zero<GeometryValueType>());
146  directionOfOptimizingObjective[indexOfOptimizingObjective] = storm::utility::one<GeometryValueType>();
147 
148  // Improve the found solution.
149  // Note that we do not care whether a threshold is strict anymore, because the resulting optimum should be
150  // the supremum over all strategies. Hence, one could combine a scheduler inducing the optimum value (but possibly violating strict
151  // thresholds) and (with very low probability) a scheduler that satisfies all (possibly strict) thresholds.
152  GeometryValueType result = storm::utility::zero<GeometryValueType>();
153  while(!this->maxStepsPerformed(env) && !storm::utility::resources::isTerminate()) {
154  if (this->refinementSteps.empty()) {
155  // We did not make any refinement steps during the checkAchievability phase (e.g., because there is only one objective).
156  this->weightVectorChecker->setWeightedPrecision(storm::utility::convertNumber<typename SparseModelType::ValueType>(env.modelchecker().multi().getPrecision()));
157  WeightVector separatingVector = directionOfOptimizingObjective;
158  this->performRefinementStep(env, std::move(separatingVector));
159  }
160  std::pair<Point, bool> optimizationRes = this->underApproximation->intersection(thresholdsAsPolytope)->optimize(directionOfOptimizingObjective);
161  STORM_LOG_THROW(optimizationRes.second, storm::exceptions::UnexpectedException, "The underapproximation is either unbounded or empty.");
162  result = optimizationRes.first[indexOfOptimizingObjective];
163  STORM_LOG_DEBUG("Best solution found so far is ~" << storm::utility::convertNumber<double>(result) << ".");
164  //Compute an upper bound for the optimum and check for convergence
165  optimizationRes = this->overApproximation->intersection(thresholdsAsPolytope)->optimize(directionOfOptimizingObjective);
166  if (optimizationRes.second) {
167  GeometryValueType precisionOfResult = optimizationRes.first[indexOfOptimizingObjective] - result;
168  if (precisionOfResult < storm::utility::convertNumber<GeometryValueType>(env.modelchecker().multi().getPrecision())) {
169  // Goal precision reached!
170  return result;
171  } else {
172  STORM_LOG_DEBUG("Solution can be improved by at most " << storm::utility::convertNumber<double>(precisionOfResult));
173  thresholds[indexOfOptimizingObjective] = optimizationRes.first[indexOfOptimizingObjective];
174  }
175  } else {
176  thresholds[indexOfOptimizingObjective] = result + storm::utility::one<GeometryValueType>();
177  }
178  WeightVector separatingVector = this->findSeparatingVector(thresholds);
179  this->updateWeightedPrecisionInImprovingPhase(env, separatingVector);
180  this->performRefinementStep(env, std::move(separatingVector));
181  }
182  STORM_LOG_ERROR("Could not reach the desired precision: Termination requested or maximum number of refinement steps exceeded.");
183  return result;
184  }

188  void SparsePcaaQuantitativeQuery<SparseModelType, GeometryValueType>::updateWeightedPrecisionInImprovingPhase(Environment const& env, WeightVector const& weights) {
189  STORM_LOG_THROW(!storm::utility::isZero(weights[this->indexOfOptimizingObjective]), exceptions::UnexpectedException, "The chosen weight-vector gives zero weight for the objective that is to be optimized.");
190  STORM_LOG_THROW(env.modelchecker().multi().getPrecisionType() == MultiObjectiveModelCheckerEnvironment::PrecisionType::Absolute, storm::exceptions::IllegalArgumentException, "Unhandled multiobjective precision type.");
191  // If weighs[indexOfOptimizingObjective] is low, the computation of the weightVectorChecker needs to be more precise.
192  // Our heuristic ensures that if p is the new vertex of the under-approximation, then max{ eps | p' = p + (0..0 eps 0..0) is in the over-approximation } <= multiobjective_precision/0.9
193  GeometryValueType weightedPrecision = weights[this->indexOfOptimizingObjective] * storm::utility::convertNumber<GeometryValueType>(env.modelchecker().multi().getPrecision());
194  // Normalize by division with the Euclidean Norm of the weight-vector
195  weightedPrecision /= storm::utility::sqrt(storm::utility::vector::dotProduct(weights, weights));
196  weightedPrecision *= storm::utility::convertNumber<GeometryValueType>(0.9);
197 
198  this->weightVectorChecker->setWeightedPrecision(storm::utility::convertNumber<typename SparseModelType::ValueType>(weightedPrecision));
