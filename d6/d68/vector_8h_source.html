1047  // Reserve the known amount of slots for the resulting vector.
1048  std::vector<uint_fast64_t> subVector(constraint.getNumberOfSetBits() + 1);
1049  uint_fast64_t currentRowCount = 0;
1050  uint_fast64_t currentIndexCount = 1;
1051 
1052  // Set the first element as this will clearly begin at offset 0.

1056  // to the resulting vector.

1058  subVector[currentIndexCount] = currentRowCount + offsetVector[index + 1] - offsetVector[index];
1059  currentRowCount += offsetVector[index + 1] - offsetVector[index];
1060  ++currentIndexCount;
1061  }
1062 
1063  // Put a sentinel element at the end.
1064  subVector[constraint.getNumberOfSetBits()] = currentRowCount;

1096  template<typename ValueType, typename TargetValueType>
1097  typename std::enable_if<std::is_same<ValueType, storm::RationalNumber>::value, std::pair<std::vector<TargetValueType>, ValueType>>::type
1098  toIntegralVector(std::vector<ValueType> const& vec) {
1099 
1100  // Collect the numbers occurring in the input vector
1101  std::set<ValueType> occurringNonZeroNumbers;
1102  for (auto const& v : vec) {
1103  if (!storm::utility::isZero(v)) {
1104  occurringNonZeroNumbers.insert(v);
1105  }
1106  }
1107 
1108  // Compute the scaling factor
1109  ValueType factor;
1110  if (occurringNonZeroNumbers.empty()) {
1111  factor = storm::utility::one<ValueType>();
1112  } else if (occurringNonZeroNumbers.size() == 1) {
1113  factor = *occurringNonZeroNumbers.begin();
1114  } else {
1115  // Obtain the least common multiple of the denominators of the occurring numbers.
1116  // We can then multiply the numbers with the lcm to obtain integers.
1117  auto numberIt = occurringNonZeroNumbers.begin();
1118  ValueType lcm = storm::utility::asFraction(*numberIt).second;
1119  for (++numberIt; numberIt != occurringNonZeroNumbers.end(); ++numberIt) {
1120  lcm = carl::lcm(lcm, storm::utility::asFraction(*numberIt).second);
1121  }
1122  // Multiply all values with the lcm. To reduce the range of considered integers, we also obtain the gcd of the results.
1123  numberIt = occurringNonZeroNumbers.begin();
1124  ValueType gcd = *numberIt * lcm;
1125  for (++numberIt; numberIt != occurringNonZeroNumbers.end(); ++numberIt) {
1126  gcd = carl::gcd(gcd, static_cast<ValueType>(*numberIt * lcm));
1127  }
1128 
1129  factor = gcd / lcm;
1130 
1131  }
1132 
1133  // Build the result
1134  std::vector<TargetValueType> result;
1135  result.reserve(vec.size());
1136  for (auto const& v : vec) {
1137  ValueType vScaled = v / factor;
1138  STORM_LOG_ASSERT(storm::utility::isInteger(vScaled), "Resulting number '(" << v << ")/(" << factor << ") = " << vScaled << "' is not integral.");
1139  result.push_back(storm::utility::convertNumber<TargetValueType, ValueType>(vScaled));
1140  }
1141  return std::make_pair(std::move(result), std::move(factor));
1142 
1143  }
