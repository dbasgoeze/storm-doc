496  auto optionalPoint = lpChecker->check(env, overApproximation);
497  STORM_LOG_THROW(optionalPoint.is_initialized(), storm::exceptions::UnexpectedException, "Unable to find a point in the current overapproximation.");

501  p.setOnFacet();
502  // Adapt the overapproximation
503  GeometryValueType offset = storm::utility::vector::dotProduct(weightVector, p.get());
504  addHalfspaceToOverApproximation(env, weightVector, offset);
505  pointset.addPoint(env, std::move(p));
506  }

508  auto initialHalfspaces = pointset.downwardClosure()->getHalfspaces();
509  for (auto& h : initialHalfspaces) {
510  Facet f(std::move(h));
511  for (auto const& p : pointset) {
512  if (f.getHalfspace().isPointOnBoundary(p.second.get())) {
513  f.addPoint(p.first, p.second);
514  }
515  }
516  STORM_LOG_ASSERT(std::count(f.getHalfspace().normalVector().begin(), f.getHalfspace().normalVector().end(), storm::utility::zero<GeometryValueType>()) + f.getNumberOfPoints() >= objectives.size(), "Not enough points on facet.");
517 
518  unprocessedFacets.push(std::move(f));

539  if (!wvChecker) {

544  return;

559  for (auto const& infeasableArea : res.second) {
560  addUnachievableArea(env, infeasableArea);
561  }

563  pointset.addPoint(env, Point(std::move(achievablePoint)));

578  template <class SparseModelType, typename GeometryValueType>
579  bool DeterministicSchedsParetoExplorer<SparseModelType, GeometryValueType>::optimizeAndSplitFacet(Environment const& env, Facet& f) {
580 
581  // Invoke optimization and insert the explored points
582  boost::optional<PointId> optPointId;
583  std::vector<GeometryValueType> point;
584  if (wvChecker) {
585  wvChecker->check(env, storm::utility::vector::convertNumericVector<ModelValueType>(f.getHalfspace().normalVector()));
586  point = storm::utility::vector::convertNumericVector<GeometryValueType>(wvChecker->getUnderApproximationOfInitialStateResults());
587  negateMinObjectives(point);
588  } else {
589  auto currentArea = overApproximation->intersection(f.getHalfspace().invert());
590  auto optionalPoint = lpChecker->check(env, currentArea);
591  if (optionalPoint.is_initialized()) {
592  point = std::move(optionalPoint.get());
593  } else {
594  // As we did not find any feasable solution in the given area, we take a point that lies on the facet
595  point = pointset.getPoint(f.getPoints().front()).get();
596  }
597  }
598  Point p(point);
599  p.setOnFacet();
600  GeometryValueType offset = storm::utility::vector::dotProduct(f.getHalfspace().normalVector(), p.get());
601  addHalfspaceToOverApproximation(env, f.getHalfspace().normalVector(), offset);
602  optPointId = pointset.addPoint(env, std::move(p));
603 
604  // Potentially generate new facets
605  if (optPointId) {
606  auto const& optPoint = pointset.getPoint(*optPointId);
607  if (f.getHalfspace().contains(optPoint.get())) {
608  // The point is contained in the halfspace which means that no more splitting is possible.
609  return false;
610  } else {
611  // Collect the new point with its neighbors.
612  // Also check whether the remamining area is already sufficiently small.
613  storm::storage::geometry::PolytopeTree<GeometryValueType> remainingArea(overApproximation->intersection(f.getHalfspace().invert()));
614  std::vector<std::vector<GeometryValueType>> vertices;
615  vertices.push_back(optPoint.get());
616  auto minmaxPrec = storm::utility::convertNumber<GeometryValueType>(env.solver().minMax().getPrecision());
617  minmaxPrec += minmaxPrec;
618  for (auto const& pId : f.getPoints()) {
619  vertices.push_back(pointset.getPoint(pId).get());
620  remainingArea.substractDownwardClosure(vertices.back(), eps);
621  STORM_LOG_WARN_COND((std::is_same<ModelValueType, storm::RationalNumber>::value) || !closePoints(vertices.front(), vertices.back(), minmaxPrec), "Found Pareto optimal points that are close to each other. This can be due to numerical issues. Maybe try exact mode?");
622  }
623  if (remainingArea.isEmpty()) {
624  return false;
625  }
626 
627  // We need to generate new facets
628  auto newHalfspaceCandidates = storm::storage::geometry::Polytope<GeometryValueType>::createSelectiveDownwardClosure(vertices, storm::utility::vector::filterZero(f.getHalfspace().normalVector()))->getHalfspaces();
629  for (auto& h : newHalfspaceCandidates) {
630  if (!storm::utility::vector::hasNegativeEntry(h.normalVector())) {
631  STORM_LOG_ASSERT(h.isPointOnBoundary(optPoint.get()), "Unexpected facet found while splitting.");
632  Facet fNew(std::move(h));
633  fNew.addPoint(optPointId.get(), optPoint);
634  auto vertexIt = vertices.begin();
635  ++vertexIt;
636  for (auto const& pId : f.getPoints()) {
637  assert(pointset.getPoint(pId).get() == *vertexIt);
638  if (fNew.getHalfspace().isPointOnBoundary(*vertexIt)) {
639  fNew.addPoint(pId, pointset.getPoint(pId));
640  }
641  ++vertexIt;
642  }
643  assert(vertexIt == vertices.end());
644  unprocessedFacets.push(std::move(fNew));
645  }
646  }
647  return true;
648  }
649  } else {
650  // If the 'optimal point' was dominated by an existing point, we can not split the facet any further.
651  return false;
652  }
653  }
