116  if(elem->rank() == static_cast<decltype(elem->rank())>(-1)) {

204  STORM_LOG_THROW(visited[n] != topoSortColour::GREY, storm::exceptions::WrongFormatException, "DFT is cyclic");
205  if(visited[n] == topoSortColour::WHITE) {
206  if(n->isGate()) {
207  visited[n] = topoSortColour::GREY;
208  for (DFTElementPointer const& c : std::static_pointer_cast<storm::storage::DFTGate<ValueType>>(n)->children()) {
209  topoVisit(c, visited, L);
210  }
211  }
212  // TODO restrictions and dependencies have no parents, so this can be done more efficiently.
213  else if(n->isRestriction()) {
214  visited[n] = topoSortColour::GREY;
215  for (DFTElementPointer const& c : std::static_pointer_cast<storm::storage::DFTRestriction<ValueType>>(n)->children()) {
216  topoVisit(c, visited, L);
217  }
218  }
219  else if(n->isDependency()) {
220  visited[n] = topoSortColour::GREY;
221  for (DFTElementPointer const& c : std::static_pointer_cast<storm::storage::DFTDependency<ValueType>>(n)->dependentEvents()) {
222  topoVisit(c, visited, L);
223  }
224  topoVisit(std::static_pointer_cast<storm::storage::DFTDependency<ValueType>>(n)->triggerEvent(), visited, L);
225  }
226  visited[n] = topoSortColour::BLACK;
227  L.push_back(n);
