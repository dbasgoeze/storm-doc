90  storm::storage::SchedulerClass MultiObjectiveSettings::getSchedulerRestriction() const {
91  storm::storage::SchedulerClass result;
92  result.setIsDeterministic(true);
93 
94  std::string pattern = this->getOption(schedulerRestrictionOptionName).getArgumentByName("memorypattern").getValueAsString();
95  uint64_t states = this->getOption(schedulerRestrictionOptionName).getArgumentByName("memorystates").getValueAsUnsignedInteger();
96  if (pattern == "positional") {
97  result.setPositional();
98  STORM_LOG_THROW(states <=1 , storm::exceptions::IllegalArgumentException, "The number of memory states should not be provided for the given memory pattern.");
99  } else if (pattern == "goalmemory") {
100  result.setMemoryPattern(storm::storage::SchedulerClass::MemoryPattern::GoalMemory);
101  STORM_LOG_THROW(states == 0, storm::exceptions::IllegalArgumentException, "The number of memory states should not be provided for the given memory pattern.");
102  } else if (pattern == "arbitrary") {
103  STORM_LOG_THROW(states > 0, storm::exceptions::IllegalArgumentException, "Invalid number of memory states for provided pattern. Please specify a positive number.");
104  result.setMemoryPattern(storm::storage::SchedulerClass::MemoryPattern::Arbitrary);
105  result.setMemoryStates(states);
106  } else if (pattern == "counter") {
107  STORM_LOG_THROW(states > 0, storm::exceptions::IllegalArgumentException, "Invalid number of memory states for provided pattern. Please specify a positive number.");
108  result.setMemoryPattern(storm::storage::SchedulerClass::MemoryPattern::Counter);
109  result.setMemoryStates(states);
110  } else {
111  STORM_LOG_THROW(false, storm::exceptions::IllegalArgumentException, "Invalid memory pattern: " + pattern + ".");
112  }
113  return result;
