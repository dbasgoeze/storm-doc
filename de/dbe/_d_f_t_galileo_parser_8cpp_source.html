41  const std::regex commentRegex("(/\\*([^*]|(\\*+[^*/]))*\\*+/)|(//.*)");
42

45

50  try {
51  while (storm::utility::getline(file, line)) {
52  ++lineNo;
53  // First consider comments
54  if (comment) {
55  // Line is part of multiline comment -> search for end of this comment
56  size_t commentEnd = line.find("*/");
57  if (commentEnd == std::string::npos) {
58  continue;
59  } else {
60  // Remove comment
61  line = line.substr(commentEnd + 2);
62  comment = false;
63  }
64  }
65  // Remove comments
66  line = std::regex_replace(line, commentRegex, "");
67  // Check if multiline comment starts
68  size_t commentStart = line.find("/*");
69  if (commentStart != std::string::npos) {
70  // Remove comment
71  line = line.substr(0, commentStart);
72  comment = true;
73  }
74 
75  boost::trim(line);
76  if (line.empty()) {
77  // Empty line
78  continue;
79  }
80 
81  // Remove semicolon
82  STORM_LOG_THROW(line.back() == ';', storm::exceptions::WrongFormatException, "Semicolon expected at the end of line " << lineNo << ".");
83  line.pop_back();
84 
85  // Split line into tokens
86  boost::trim(line);
87  std::vector<std::string> tokens;
88  boost::split(tokens, line, boost::is_any_of(" \t"), boost::token_compress_on);
89 
90  if (tokens[0] == "toplevel") {
91  // Top level indicator
92  STORM_LOG_THROW(toplevelId == "", storm::exceptions::WrongFormatException, "Toplevel element already defined.");
93  STORM_LOG_THROW(tokens.size() == 2, storm::exceptions::WrongFormatException, "Expected element id after 'toplevel' in line " << lineNo << ".");
94  toplevelId = parseName(tokens[1]);
95  } else if (tokens[0] == "param") {
96  // Parameters
97  STORM_LOG_THROW(tokens.size() == 2, storm::exceptions::WrongFormatException, "Expected parameter name after 'param' in line " << lineNo << ".");
98  STORM_LOG_THROW((std::is_same<ValueType, storm::RationalFunction>::value), storm::exceptions::NotSupportedException, "Parameters only allowed when using rational functions.");
99  valueParser.addParameter(parseName(tokens[1]));
100  } else {
101  // DFT element
102  std::string name = parseName(tokens[0]);
103 
104  std::vector<std::string> childNames;
105  for(size_t i = 2; i < tokens.size(); ++i) {
106  childNames.push_back(parseName(tokens[i]));
107  }
108  bool success = true;
109 
110  // Add element according to type
111  std::string type = tokens[1];
112  if (type == "and") {
113  success = builder.addAndElement(name, childNames);
114  } else if (type == "or") {
115  success = builder.addOrElement(name, childNames);
116  } else if (boost::starts_with(type, "vot")) {
117  size_t threshold = storm::parser::parseNumber<size_t>(type.substr(3));
118  success = builder.addVotElement(name, threshold, childNames);
119  } else if (type.find("of") != std::string::npos) {
120  size_t pos = type.find("of");
121  size_t threshold = storm::parser::parseNumber<size_t>(type.substr(0, pos));
122  size_t count = storm::parser::parseNumber<size_t>(type.substr(pos + 2));
123  STORM_LOG_THROW(count == childNames.size(), storm::exceptions::WrongFormatException, "Voting gate number " << count << " does not correspond to number of children " << childNames.size() << " in line " << lineNo << ".");
124  success = builder.addVotElement(name, threshold, childNames);
125  } else if (type == "pand") {
126  success = builder.addPandElement(name, childNames, defaultInclusive);
127  } else if (type == "pand-inc") {
128  success = builder.addPandElement(name, childNames, true);
129  } else if (type == "pand-ex") {
130  success = builder.addPandElement(name, childNames, false);
131  } else if (type == "por") {
132  success = builder.addPorElement(name, childNames, defaultInclusive);
133  } else if (type == "por-ex") {
134  success = builder.addPorElement(name, childNames, false);
135  } else if (type == "por-inc") {
136  success = builder.addPorElement(name, childNames, true);
137  } else if (type == "wsp" || type == "csp" || type == "hsp") {
138  success = builder.addSpareElement(name, childNames);
139  } else if (type == "seq") {
140  success = builder.addSequenceEnforcer(name, childNames);
141  } else if (type == "mutex") {
142  success = builder.addMutex(name, childNames);
143  } else if (type == "fdep") {
144  STORM_LOG_THROW(childNames.size() >= 2, storm::exceptions::WrongFormatException, "FDEP gate needs at least two children in line " << lineNo << ".");
145  success = builder.addDepElement(name, childNames, storm::utility::one<ValueType>());
146  } else if (boost::starts_with(type, "pdep=")) {
147  STORM_LOG_THROW(childNames.size() >= 2, storm::exceptions::WrongFormatException, "PDEP gate needs at least two children in line " << lineNo << ".");
148  ValueType probability = valueParser.parseValue(type.substr(5));
149  success = builder.addDepElement(name, childNames, probability);
150  } else if (type.find("=") != std::string::npos) {
151  success = parseBasicElement(tokens[0], line, lineNo, builder, valueParser);
152  } else if (type.find("insp") != std::string::npos) {
153  // Inspection as defined by DFTCalc
154  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Inspections (defined in line " << lineNo << ") are not supported.");
155  } else {
156  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Type name: " << type << " in line " << lineNo << " not recognized.");
157  success = false;
158  }
159  STORM_LOG_THROW(success, storm::exceptions::FileIoException, "Error while adding element '" << name << "' in line " << lineNo << ".");
160  }
161  }
162  } catch (storm::exceptions::BaseException const& exception) {
163  STORM_LOG_THROW(false, storm::exceptions::FileIoException, "A parsing exception occurred in line " << lineNo << ": " << exception.what());
164  }
165  if (!builder.setTopLevel(toplevelId)) {
166  STORM_LOG_THROW(false, storm::exceptions::FileIoException, "Top level id '" << toplevelId << "' unknown.");
167  }
168  storm::utility::closeFile(file);
169 
170  // Build DFT
171  storm::storage::DFT<ValueType> dft = builder.build();
172  STORM_LOG_DEBUG("DFT Elements:" << std::endl << dft.getElementsString());
173  STORM_LOG_DEBUG("Spare Modules:" << std::endl << dft.getSpareModulesString());
174  return dft;

210  bool DFTGalileoParser<ValueType>::parseBasicElement(std::string const& name, std::string const& input, size_t lineNo, storm::builder::DFTBuilder<ValueType>& builder, ValueParser<ValueType>& valueParser) {
211  // Default values
212  Distribution distribution = Distribution::None;
213  ValueType firstValDistribution = storm::utility::zero<ValueType>();
214  ValueType secondValDistribution = storm::utility::zero<ValueType>();
215  ValueType dormancyFactor = storm::utility::one<ValueType>();
216  size_t replication = 1;
217  size_t erlangPhases = 1;
218  // Remove name from input
219  std::regex regexName("\"?" + name + "\"?");
220  std::string line = std::regex_replace(input, regexName, "");
221 
222  // Parse properties and determine distribution
223  // Constant distribution
224  std::pair<bool, ValueType> result = parseValue("prob", line, valueParser);
225  if (result.first) {
226  STORM_LOG_THROW(distribution == Distribution::None, storm::exceptions::WrongFormatException, "A different distribution was already defined for basic element '" << name << "' in line " << lineNo << ".");
227  firstValDistribution = result.second;
228  distribution = Distribution::Constant;
229  }
230  // Exponential distribution
231  result = parseValue("lambda", line, valueParser);
232  if (result.first) {
233  STORM_LOG_THROW(distribution == Distribution::None || distribution == Distribution::Erlang, storm::exceptions::WrongFormatException, "A different distribution was already defined for basic element '" << name << "' in line " << lineNo << ".");
234  firstValDistribution = result.second;
235  if (distribution == Distribution::None) {
236  distribution = Distribution::Exponential;
237  }
238  }
239  // Erlang distribution
240  std::pair<bool, size_t> resultNum = parseNumber("phases", line);
241  if (resultNum.first) {
242  STORM_LOG_THROW(distribution == Distribution::None || distribution == Distribution::Exponential, storm::exceptions::WrongFormatException, "A different distribution was already defined for basic element '" << name << "' in line " << lineNo << ".");
243  erlangPhases = resultNum.second;
244  distribution = Distribution::Erlang;
245  }
246  // Weibull distribution
247  result = parseValue("rate", line, valueParser);
248  if (result.first) {
249  STORM_LOG_THROW(distribution == Distribution::None || distribution == Distribution::Weibull, storm::exceptions::WrongFormatException, "A different distribution was already defined for basic element '" << name << "' in line " << lineNo << ".");
250  firstValDistribution = result.second;
251  distribution = Distribution::Weibull;
252  }
253  result = parseValue("shape", line, valueParser);
254  if (result.first) {
255  STORM_LOG_THROW(distribution == Distribution::None || distribution == Distribution::Weibull, storm::exceptions::WrongFormatException, "A different distribution was already defined for basic element '" << name << "' in line " << lineNo << ".");
256  secondValDistribution = result.second;
257  distribution = Distribution::Weibull;
258  }
259  // Lognormal distribution
260  result = parseValue("mean", line, valueParser);
261  if (result.first) {
262  STORM_LOG_THROW(distribution == Distribution::None || distribution == Distribution::LogNormal, storm::exceptions::WrongFormatException, "A different distribution was already defined for basic element '" << name << "' in line " << lineNo << ".");
263  firstValDistribution = result.second;
264  distribution = Distribution::LogNormal;
265  }
266  result = parseValue("stddev", line, valueParser);
267  if (result.first) {
268  STORM_LOG_THROW(distribution == Distribution::None || distribution == Distribution::LogNormal, storm::exceptions::WrongFormatException, "A different distribution was already defined for basic element '" << name << "' in line " << lineNo << ".");
269  secondValDistribution = result.second;
270  distribution = Distribution::LogNormal;
271  }
272  // Additional arguments
273  result = parseValue("cov", line, valueParser);
274  if (result.first) {
275  STORM_LOG_WARN("Coverage is not supported and will be ignored for basic element '" << name << "' in line " << lineNo << ".");
276  }
277  result = parseValue("res", line, valueParser);
278  if (result.first) {
279  STORM_LOG_WARN("Restoration is not supported and will be ignored for basic element '" << name << "' in line " << lineNo << ".");
280  }
281  resultNum = parseNumber("repl", line);
282  if (resultNum.first) {
283  replication = resultNum.second;
284  STORM_LOG_THROW(replication == 1, storm::exceptions::NotSupportedException, "Replication > 1 is not supported for basic element '" << name << "' in line " << lineNo << ".");
285  }
286  result = parseValue("interval", line, valueParser);
287  if (result.first) {
288  STORM_LOG_WARN("Interval is not supported and will be ignored for basic element '" << name << "' in line " << lineNo << ".");
289  }
290  result = parseValue("dorm", line, valueParser);
291  if (result.first) {
292  dormancyFactor = result.second;
293  } else {
294  STORM_LOG_WARN("No dormancy factor was provided for basic element '" << name << "' in line " << lineNo << ". Assuming dormancy factor of 1.");
295  }
296  boost::trim(line);
297  if (line != "") {
298  STORM_LOG_THROW(false, storm::exceptions::WrongFormatException, "Unknown arguments for basic element '" << name << "' in line " << lineNo << ": " << line);
299  }
300 
301  switch (distribution) {
302  case Constant:
303  if (storm::utility::isZero(firstValDistribution) || storm::utility::isOne(firstValDistribution)) {
304  return builder.addBasicElementProbability(parseName(name), firstValDistribution, dormancyFactor, false); // TODO set transient BEs
305  } else {
306  // Model constant BEs with probability 0 < p < 1
307  bool success = true;
308  if (!builder.nameInUse("constantBeTrigger")) {
309  // Use a unique constantly failed element that triggers failsafe elements probabilistically
310  success = success && builder.addBasicElementProbability("constantBeTrigger",
311  storm::utility::one<ValueType>(),
312  storm::utility::one<ValueType>(),
313  false);
314  }
315  std::vector<std::string> childNames;
316  childNames.push_back("constantBeTrigger");
317  success = success &&
318  builder.addBasicElementProbability(parseName(name), storm::utility::zero<ValueType>(),
319  storm::utility::one<ValueType>(), false);
320  childNames.push_back(parseName(name));
321  return success &&
322  builder.addDepElement(parseName(name) + "_pdep", childNames, firstValDistribution);
323  }
324  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Constant distribution is not supported for basic element '" << name << "' in line " << lineNo << ".");
325  break;
326  case Exponential:
327  return builder.addBasicElementExponential(parseName(name), firstValDistribution, dormancyFactor, false); // TODO set transient BEs
328  break;
329  case Erlang:
330  if (erlangPhases == 1) {
331  // Erlang distribution reduces to exponential distribution
332  return builder.addBasicElementExponential(parseName(name), firstValDistribution, dormancyFactor, false); // TODO set transient BEs
333  } else {
334  // Model Erlang distribution by using SEQ over BEs instead.
335  // For each phase a BE is added, then the SEQ ensures the ordered failure.
336  STORM_LOG_WARN("Erlang distribution for basic element '" << name << "' in line " << lineNo << " is modelled by SEQ gate and BEs.");
337  std::string origName = parseName(name);
338  std::vector<std::string> childNames;
339  bool success = builder.addBasicElementExponential(origName, firstValDistribution, dormancyFactor, false); // TODO set transient BEs
340  for (size_t i = 0; i < erlangPhases - 1; ++i) {
341  std::string beName = origName + "_" + std::to_string(i);
342  childNames.push_back(beName);
343  success = success && builder.addBasicElementExponential(beName, firstValDistribution, dormancyFactor, false); // TODO set transient BEs
344  }
345  childNames.push_back(origName);
346  return success && builder.addSequenceEnforcer(origName + "_seq", childNames);
347  }
348  break;
349  case Weibull:
350  if (storm::utility::isOne<ValueType>(secondValDistribution)) {
351  // Weibull distribution reduces to exponential distribution
352  return builder.addBasicElementExponential(parseName(name), firstValDistribution, dormancyFactor, false); // TODO set transient BEs
353  } else {
354  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Weibull distribution is not supported for basic element '" << name << "' in line " << lineNo << ".");
355  }
356  break;
357  case LogNormal:
358  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "LogNormal distribution is not supported for basic element '" << name << "' in line " << lineNo << ".");
359  break;
360  case None:
361  // go-through
362  default:
363  STORM_LOG_THROW(false, storm::exceptions::WrongFormatException, "No distribution defined for basic element '" << name << "' in line " << lineNo << ".");
364  break;
365  }
366  return false;
367 
368  }
