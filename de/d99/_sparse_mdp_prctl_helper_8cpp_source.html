338  if (type == SolutionType::UntilProbabilities) {
339  requirements.clearBounds();
340  } else if (type == SolutionType::ExpectedRewards) {
341  requirements.clearLowerBounds();
342  }
343  if (requirements.upperBounds()) {
344  result.computeUpperBounds = true;
345  requirements.clearUpperBounds();
346  }
347  STORM_LOG_THROW(!requirements.hasEnabledCriticalRequirement(), storm::exceptions::UncheckedRequirementException, "Solver requirements " + requirements.getEnabledRequirementsAsString() + " not checked.");

349  STORM_LOG_DEBUG("Solver has no requirements.");
350  }

353  // hints from the provided hint.

358  }
359

361  if (!result.hasLowerResultBound()) {
362  result.lowerResultBound = storm::utility::zero<ValueType>();
363  }
364  if (!result.hasUpperResultBound() && type == SolutionType::UntilProbabilities) {
365  result.upperResultBound = storm::utility::one<ValueType>();
366  }
367 
368  // If we received an upper bound, we can drop the requirement to compute one.
369  if (result.hasUpperResultBound()) {
370  result.computeUpperBounds = false;
371  }
372
