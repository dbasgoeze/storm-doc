99  // Initialize which DFT elements have dynamic behavior
100  dft.setDynamicBehaviorInfo();
101 
102  std::vector<std::pair<uint64_t, uint64_t>> fdepConflicts = storm::dft::utility::FDEPConflictFinder<ValueType>::getDependencyConflicts(dft, useSMT, solverTimeout);
103 
104  for (auto const& pair: fdepConflicts) {
105  STORM_LOG_DEBUG("Conflict between " << dft.getElement(pair.first)->name() << " and " << dft.getElement(pair.second)->name());
106  }
107 
108  // Set the conflict map of the dft
109  std::set<size_t> conflict_set;
110  for (auto const& conflict : fdepConflicts) {
111  conflict_set.insert(size_t(conflict.first));
112  conflict_set.insert(size_t(conflict.second));
113  }
114  for (size_t depId : dft.getDependencies()) {
115  if (!conflict_set.count(depId)) {
116  dft.setDependencyNotInConflict(depId);
117  }
118  }
119  return !fdepConflicts.empty();
120  }
