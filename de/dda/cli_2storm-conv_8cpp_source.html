175

187  // Substitution of constants can only be done after conversion in order to preserve formula definitions in which

257  void processJaniInput() {
258  auto parsingTime = startStopwatch("Parsing JANI input ... " );
259 
260  auto const& input = storm::settings::getModule<storm::settings::modules::ConversionInputSettings>();
261 
262  // Parse the jani model and selected properties
263  boost::optional<std::vector<std::string>> janiPropertyFilter;
264  if (input.isJaniPropertiesSet()) {
265  if (input.areJaniPropertiesSelected()) {
266  janiPropertyFilter = input.getSelectedJaniProperties();
267  } else {
268  janiPropertyFilter = boost::none;
269  }
270  } else {
271  if (input.isPropertyInputSet()) {
272  janiPropertyFilter = std::vector<std::string>();
273  } else {
274  // If no properties are selected, take the ones from the jani file.
275  janiPropertyFilter = boost::none;
276  }
277  }
278  auto janiModelProperties = storm::api::parseJaniModel(input.getJaniInputFilename(), storm::jani::getAllKnownModelFeatures(), janiPropertyFilter);
279 
280  // Parse additional properties given from command line
281  std::vector<storm::jani::Property> properties = std::move(janiModelProperties.second);
282  if (input.isPropertyInputSet()) {
283  boost::optional<std::set<std::string>> propertyFilter = storm::api::parsePropertyFilter(input.getPropertyInputFilter());
284  auto additionalProperties = storm::api::parsePropertiesForSymbolicModelDescription(input.getPropertyInput(), janiModelProperties.first, propertyFilter);
285  properties.insert(properties.end(), additionalProperties.begin(), additionalProperties.end());
286  }
287 
288  storm::storage::SymbolicModelDescription symbDescr(janiModelProperties.first);
289 
290  // Substitute constant definitions in model and properties.
291  std::string constantDefinitionString = input.getConstantDefinitionString();
292  auto constantDefinitions = symbDescr.parseConstantDefinitions(constantDefinitionString);
293  auto janiModel = janiModelProperties.first.defineUndefinedConstants(constantDefinitions).substituteConstants();
294  if (!properties.empty()) {
295  properties = storm::api::substituteConstantsInProperties(properties, constantDefinitions);
296  }
297  stopStopwatch(parsingTime);
298 
299  // Branch on the type of output
300  auto const& output = storm::settings::getModule<storm::settings::modules::ConversionOutputSettings>();
301  if (output.isJaniOutputSet()) {
302  processJaniInputJaniOutput(janiModel, properties);
303  } else {
304  STORM_LOG_THROW(!output.isPrismOutputSet(), storm::exceptions::InvalidSettingsException, "Conversion from Jani to Prism is not supported.");
305  STORM_LOG_THROW(false, storm::exceptions::InvalidSettingsException, "There is either no outputformat specified or the provided combination of input and output format is not compatible.");
306  }
