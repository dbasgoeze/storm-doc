115  void AddIterator<DdType::Sylvan, ValueType>::treatNewCube() {
116  this->relevantDontCareDdVariables.clear();
117 
118  // Now loop through the bits of all meta variables and check whether they need to be set, not set or are
119  // don't cares. In the latter case, we add them to a special list, so we can iterate over their concrete
120  // valuations later.
121  for (auto const& metaVariable : *this->metaVariables) {
122  bool metaVariableAppearsInCube = false;
123  std::vector<std::tuple<storm::expressions::Variable, uint_fast64_t>> localRelenvantDontCareDdVariables;
124  auto const& ddMetaVariable = this->ddManager->getMetaVariable(metaVariable);
125  if (ddMetaVariable.getType() == MetaVariableType::Bool) {
126  if (this->cube[globalToLocalIndexMap.at(ddMetaVariable.getDdVariables().front().getIndex())] == 0) {
127  metaVariableAppearsInCube = true;
128  currentValuation.setBooleanValue(metaVariable, false);
129  } else if (this->cube[globalToLocalIndexMap.at(ddMetaVariable.getDdVariables().front().getIndex())] == 1) {
130  metaVariableAppearsInCube = true;
131  currentValuation.setBooleanValue(metaVariable, true);
132  } else {
133  localRelenvantDontCareDdVariables.push_back(std::make_tuple(metaVariable, 0));
134  }
135  } else {
136  int_fast64_t intValue = 0;
137  for (uint_fast64_t bitIndex = 0; bitIndex < ddMetaVariable.getNumberOfDdVariables(); ++bitIndex) {
138  if (cube[globalToLocalIndexMap.at(ddMetaVariable.getDdVariables()[bitIndex].getIndex())] == 0) {
139  // Leave bit unset.
140  metaVariableAppearsInCube = true;
141  } else if (cube[globalToLocalIndexMap.at(ddMetaVariable.getDdVariables()[bitIndex].getIndex())] == 1) {
142  intValue |= 1ull << (ddMetaVariable.getNumberOfDdVariables() - bitIndex - 1);
143  metaVariableAppearsInCube = true;
144  } else {
145  // Temporarily leave bit unset so we can iterate trough the other option later.
146  // Add the bit to the relevant don't care bits.
147  localRelenvantDontCareDdVariables.push_back(std::make_tuple(metaVariable, ddMetaVariable.getNumberOfDdVariables() - bitIndex - 1));
148  }
149  }
150  if (this->enumerateDontCareMetaVariables || metaVariableAppearsInCube) {
151  currentValuation.setBitVectorValue(metaVariable, intValue + ddMetaVariable.getLow());
152  }
153  }
154 
155  // If all meta variables are to be enumerated or the meta variable appeared in the cube, we register the
156  // missing bits to later enumerate all possible valuations.
157  if (this->enumerateDontCareMetaVariables || metaVariableAppearsInCube) {
158  relevantDontCareDdVariables.insert(relevantDontCareDdVariables.end(), localRelenvantDontCareDdVariables.begin(), localRelenvantDontCareDdVariables.end());
159  }
160  }
161 
162  // Finally, reset the cube counter.
163  this->cubeCounter = 0;
164  }

173  if (ddMetaVariable.getType() == MetaVariableType::Bool) {

175  currentValuation.setBooleanValue(std::get<0>(this->relevantDontCareDdVariables[index]), true);

178  }
