126  void TransientVariableInformation<ValueType>::createVariablesForVariableSet(storm::jani::VariableSet const& variableSet, bool global) {
127  for (auto const& variable : variableSet.getBooleanVariables()) {
128  if (variable.isTransient()) {
129  booleanVariableInformation.emplace_back(variable.getExpressionVariable(), variable.getInitExpression().evaluateAsBool(), global);
130  }
131  }

142  integerVariableInformation.emplace_back(variable.getExpressionVariable(), lowerBound, upperBound, variable.getInitExpression().evaluateAsInt(), global);
143  }
144  }

146  if (variable.isTransient()) {

148  }
