319  for (uint_fast64_t i = 0; i < matrix.getColumnCount(); ++i) {
320  auto row = matrix.getRow(i);
321  // only enter if you are in a state with at least two successors (so there must be successors,
322  // and first prob shouldn't be 1)
323  if (row.begin() != row.end() && !row.begin()->getValue().isOne()) {
324  std::map<uint_fast64_t, ValueType> transitions;

329  for (auto const& entry : row) {
330  if (!entry.getValue().isConstant()) {
331  // only analyse take non constant transitions
332  transitions.insert(std::pair<uint_fast64_t, ValueType>(entry.getColumn(), entry.getValue()));
333  for (auto const& var:entry.getValue().gatherVariables()) {
334  vars.insert(var);
335  states->set(entry.getColumn());
336  }
337  }
338  }
339 
340  // Copy info from checkOnSamples
341  if (checkSamples) {
342  for (auto var : vars) {
343  assert (resultCheckOnSamples.find(var) != resultCheckOnSamples.end());
344  if (varsMonotone.find(var) == varsMonotone.end()) {
345  varsMonotone[var].first = resultCheckOnSamples[var].first;
346  varsMonotone[var].second = resultCheckOnSamples[var].second;
347  } else {

349  varsMonotone[var].second &= resultCheckOnSamples[var].second;
350  }
351  }
352  } else {

354  if (varsMonotone.find(var) == varsMonotone.end()) {
355  varsMonotone[var].first = true;
356  varsMonotone[var].second = true;
357  }
358  }
359  }
360 
361 
362

364  auto sortedStates = order->sortStates(states);
365  if (sortedStates[sortedStates.size() - 1] == matrix.getColumnCount()) {
366  // If the states are not all sorted, we still might obtain some monotonicity
367  for (auto var: vars) {
368  // current value of monotonicity
369  std::pair<bool, bool> *value = &varsMonotone.find(var)->second;
370 
371  // Go over all transitions to successor states, compare all of them
372  for (auto itr2 = transitions.begin(); (value->first || value->second)
373  && itr2 != transitions.end(); ++itr2) {
374  for (auto itr3 = transitions.begin(); (value->first || value->second)
375  && itr3 != transitions.end(); ++itr3) {
376  if (itr2->first < itr3->first) {

379  auto derivative3 = getDerivative(itr3->second, var);
380 
381  auto compare = order->compare(itr2->first, itr3->first);
382 
383  if (compare == Order::ABOVE) {
384  // As the first state (itr2) is above the second state (itr3) it
385  // is sufficient to look at the derivative of itr2.
386  std::pair<bool, bool> mon2;
387  mon2 = checkDerivative(derivative2, region);
388  value->first &= mon2.first;
389  value->second &= mon2.second;
390  } else if (compare == Order::BELOW) {
391  // As the second state (itr3) is above the first state (itr2) it
392  // is sufficient to look at the derivative of itr3.
393  std::pair<bool, bool> mon3;
394 
395  mon3 = checkDerivative(derivative3, region);
396  value->first &= mon3.first;
397  value->second &= mon3.second;
398  } else if (compare == Order::SAME) {
399  // Behaviour doesn't matter, as the states are at the same level.
400  } else {
401  // only if derivatives are the same we can continue
402  if (derivative2 != derivative3) {
403  // As the relation between the states is unknown, we can't claim
404  // anything about the monotonicity.
405  value->first = false;
406  value->second = false;
407  }
408 
409  }
410  }
411  }
412  }
413  }
414  } else {
415  // The states are all sorted
416  for (auto var : vars) {
417  std::pair<bool, bool> *value = &varsMonotone.find(var)->second;
418  bool change = false;
419  for (auto const &i : sortedStates) {
420  auto res = checkDerivative(getDerivative(transitions[i], var), region);
421  change = change || (!(value->first && value->second) // they do not hold both
422  && ((value->first && !res.first)
423  || (value->second && !res.second)));
424 
425  if (change) {
426  value->first &= res.second;
427  value->second &= res.first;
428  } else {
429  value->first &= res.first;
430  value->second &= res.second;
431  }
432  if (!value->first && !value->second) {
433  break;
434  }
435  }
436 
437  }
438  }
439  }
440  }
