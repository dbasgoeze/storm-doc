504  typename SparseDtmcPrctlHelper<ValueType, RewardModelType>::BaierTransformedModel SparseDtmcPrctlHelper<ValueType, RewardModelType>::computeBaierTransformation(Environment const& env, storm::storage::SparseMatrix<ValueType> const& transitionMatrix, storm::storage::SparseMatrix<ValueType> const& backwardTransitions, storm::storage::BitVector const& targetStates, storm::storage::BitVector const& conditionStates, boost::optional<std::vector<ValueType>> const& stateRewards) {
505 
506  BaierTransformedModel result;
507 
508  // Start by computing all 'before' states, i.e. the states for which the conditional probability is defined.
509  std::vector<ValueType> probabilitiesToReachConditionStates = computeUntilProbabilities(env, storm::solver::SolveGoal<ValueType>(), transitionMatrix, backwardTransitions, storm::storage::BitVector(transitionMatrix.getRowCount(), true), conditionStates, false);
510 
511  result.beforeStates = storm::storage::BitVector(targetStates.size(), true);
512  uint_fast64_t state = 0;
513  uint_fast64_t beforeStateIndex = 0;
514  for (auto const& value : probabilitiesToReachConditionStates) {
515  if (value == storm::utility::zero<ValueType>()) {
516  result.beforeStates.set(state, false);
517  } else {
518  probabilitiesToReachConditionStates[beforeStateIndex] = value;
519  ++beforeStateIndex;
520  }
521  ++state;
522  }
523  probabilitiesToReachConditionStates.resize(beforeStateIndex);
524 
525  if (targetStates.empty()) {
526  result.noTargetStates = true;
527  return result;
528  } else if (!result.beforeStates.empty()) {
529  // If there are some states for which the conditional probability is defined and there are some
530  // states that can reach the target states without visiting condition states first, we need to
531  // do more work.
532 
533  // First, compute the relevant states and some offsets.
534  storm::storage::BitVector allStates(targetStates.size(), true);
535  std::vector<uint_fast64_t> numberOfBeforeStatesUpToState = result.beforeStates.getNumberOfSetBitsBeforeIndices();
536  storm::storage::BitVector statesWithProbabilityGreater0 = storm::utility::graph::performProbGreater0(backwardTransitions, allStates, targetStates);
537  statesWithProbabilityGreater0 &= storm::utility::graph::getReachableStates(transitionMatrix, conditionStates, allStates, targetStates);
538  uint_fast64_t normalStatesOffset = result.beforeStates.getNumberOfSetBits();
539  std::vector<uint_fast64_t> numberOfNormalStatesUpToState = statesWithProbabilityGreater0.getNumberOfSetBitsBeforeIndices();
540 
541  // All transitions going to states with probability zero, need to be redirected to a deadlock state.
542  bool addDeadlockState = false;
543  uint_fast64_t deadlockState = normalStatesOffset + statesWithProbabilityGreater0.getNumberOfSetBits();
544 
545  // Now, we create the matrix of 'before' and 'normal' states.
546  storm::storage::SparseMatrixBuilder<ValueType> builder;
547 
548  // Start by creating the transitions of the 'before' states.
549  uint_fast64_t currentRow = 0;
550  for (auto beforeState : result.beforeStates) {
551  if (conditionStates.get(beforeState)) {
552  // For condition states, we move to the 'normal' states.
553  ValueType zeroProbability = storm::utility::zero<ValueType>();
554  for (auto const& successorEntry : transitionMatrix.getRow(beforeState)) {
555  if (statesWithProbabilityGreater0.get(successorEntry.getColumn())) {
556  builder.addNextValue(currentRow, normalStatesOffset + numberOfNormalStatesUpToState[successorEntry.getColumn()], successorEntry.getValue());
557  } else {
558  zeroProbability += successorEntry.getValue();
559  }
560  }
561  if (!storm::utility::isZero(zeroProbability)) {
562  builder.addNextValue(currentRow, deadlockState, zeroProbability);
563  }
564  } else {
565  // For non-condition states, we scale the probabilities going to other before states.
566  for (auto const& successorEntry : transitionMatrix.getRow(beforeState)) {
567  if (result.beforeStates.get(successorEntry.getColumn())) {
568  builder.addNextValue(currentRow, numberOfBeforeStatesUpToState[successorEntry.getColumn()], successorEntry.getValue() * probabilitiesToReachConditionStates[numberOfBeforeStatesUpToState[successorEntry.getColumn()]] / probabilitiesToReachConditionStates[currentRow]);
569  }
570  }
571  }
572  ++currentRow;
573  }
574 
575  // Then, create the transitions of the 'normal' states.
576  for (auto state : statesWithProbabilityGreater0) {
577  ValueType zeroProbability = storm::utility::zero<ValueType>();
578  for (auto const& successorEntry : transitionMatrix.getRow(state)) {
579  if (statesWithProbabilityGreater0.get(successorEntry.getColumn())) {
580  builder.addNextValue(currentRow, normalStatesOffset + numberOfNormalStatesUpToState[successorEntry.getColumn()], successorEntry.getValue());
581  } else {
582  zeroProbability += successorEntry.getValue();
583  }
584  }
585  if (!storm::utility::isZero(zeroProbability)) {
586  addDeadlockState = true;
587  builder.addNextValue(currentRow, deadlockState, zeroProbability);
588  }
589  ++currentRow;
590  }
591  if (addDeadlockState) {
592  builder.addNextValue(deadlockState, deadlockState, storm::utility::one<ValueType>());
593  }
594 
595  // Build the new transition matrix and the new targets.
596  result.transitionMatrix = builder.build(addDeadlockState ? (deadlockState + 1) : deadlockState);
597  storm::storage::BitVector newTargetStates = targetStates % result.beforeStates;
598  newTargetStates.resize(result.transitionMatrix.get().getRowCount());
599  for (auto state : targetStates % statesWithProbabilityGreater0) {
600  newTargetStates.set(normalStatesOffset + state, true);
601  }
602  result.targetStates = std::move(newTargetStates);
603 
604  // If a reward model was given, we need to compute the rewards for the transformed model.
605  if (stateRewards) {
606  std::vector<ValueType> newStateRewards(result.beforeStates.getNumberOfSetBits());
607  storm::utility::vector::selectVectorValues(newStateRewards, result.beforeStates, stateRewards.get());
608 
609  newStateRewards.reserve(result.transitionMatrix.get().getRowCount());
610  for (auto state : statesWithProbabilityGreater0) {
611  newStateRewards.push_back(stateRewards.get()[state]);
612  }
613  // Add a zero reward to the deadlock state.
614  if (addDeadlockState) {
615  newStateRewards.push_back(storm::utility::zero<ValueType>());
616  }
617  result.stateRewards = std::move(newStateRewards);
618  }
619  }
620 
621  return result;
622  }
