103  std::vector<ValueType> analyzeTrivialMdpEpochModel(OptimizationDirection dir, EpochModel<ValueType, true>& epochModel) {
104  // Assert that the epoch model is indeed trivial
105  assert(epochModel.epochMatrix.getEntryCount() == 0);
106 
107  std::vector<ValueType> epochResult;
108  epochResult.reserve(epochModel.epochInStates.getNumberOfSetBits());
109 
110  auto stepSolutionIt = epochModel.stepSolutions.begin();
111  auto stepChoiceIt = epochModel.stepChoices.begin();
112  for (auto const& state : epochModel.epochInStates) {
113  // Obtain the best choice for this state
114  ValueType bestValue;
115  uint64_t lastChoice = epochModel.epochMatrix.getRowGroupIndices()[state + 1];
116  bool isFirstChoice = true;
117  for (uint64_t choice = epochModel.epochMatrix.getRowGroupIndices()[state]; choice < lastChoice; ++choice) {
118  while (*stepChoiceIt < choice) {
119  ++stepChoiceIt;
120  ++stepSolutionIt;
121  }
122 
123  ValueType choiceValue = storm::utility::zero<ValueType>();
124  if (epochModel.objectiveRewardFilter.front().get(choice)) {
125  choiceValue += epochModel.objectiveRewards.front()[choice];
126  }
127  if (*stepChoiceIt == choice) {
128  choiceValue += *stepSolutionIt;
129  }
130 
131  if (isFirstChoice) {
132  bestValue = std::move(choiceValue);
133  isFirstChoice = false;
134  } else {
135  if (storm::solver::minimize(dir)) {
136  if (choiceValue < bestValue) {
137  bestValue = std::move(choiceValue);
138  }
139  } else {
140  if (choiceValue > bestValue) {
141  bestValue = std::move(choiceValue);
142  }
143  }
144  }
145  }
146  // Insert the solution w.r.t. this choice
147  epochResult.push_back(std::move(bestValue));
148  }
149  return epochResult;
150  }
