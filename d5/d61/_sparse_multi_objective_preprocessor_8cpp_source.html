242  storm::logic::OperatorInformation getOperatorInformation(storm::logic::OperatorFormula const& formula, bool considerComplementaryEvent) {
243  storm::logic::OperatorInformation opInfo;
244  if (formula.hasBound()) {
245  opInfo.bound = formula.getBound();
246  // Invert the bound (if necessary)
247  if (considerComplementaryEvent) {
248  opInfo.bound->threshold = opInfo.bound->threshold.getManager().rational(storm::utility::one<storm::RationalNumber>()) - opInfo.bound->threshold;
249  switch (opInfo.bound->comparisonType) {
250  case storm::logic::ComparisonType::Greater:
251  opInfo.bound->comparisonType = storm::logic::ComparisonType::Less;
252  break;
253  case storm::logic::ComparisonType::GreaterEqual:
254  opInfo.bound->comparisonType = storm::logic::ComparisonType::LessEqual;
255  break;
256  case storm::logic::ComparisonType::Less:
257  opInfo.bound->comparisonType = storm::logic::ComparisonType::Greater;
258  break;
259  case storm::logic::ComparisonType::LessEqual:
260  opInfo.bound->comparisonType = storm::logic::ComparisonType::GreaterEqual;
261  break;
262  default:
263  STORM_LOG_THROW(false, storm::exceptions::InvalidPropertyException, "Current objective " << formula << " has unexpected comparison type");
264  }
265  }
266  if (storm::logic::isLowerBound(opInfo.bound->comparisonType)) {
267  opInfo.optimalityType = storm::solver::OptimizationDirection::Maximize;
268  } else {
269  opInfo.optimalityType = storm::solver::OptimizationDirection::Minimize;
270  }
271  STORM_LOG_WARN_COND(!formula.hasOptimalityType(), "Optimization direction of formula " << formula << " ignored as the formula also specifies a threshold.");
272  } else if (formula.hasOptimalityType()){
273  opInfo.optimalityType = formula.getOptimalityType();
274  // Invert the optimality type (if necessary)
275  if (considerComplementaryEvent) {
276  opInfo.optimalityType = storm::solver::invert(opInfo.optimalityType.get());
277  }
278  } else {
279  STORM_LOG_THROW(false, storm::exceptions::InvalidPropertyException, "Objective " << formula << " does not specify whether to minimize or maximize");
280  }
281  return opInfo;
282  }

423  for (uint_fast64_t row = data.model->getTransitionMatrix().getRowGroupIndices()[state]; row < data.model->getTransitionMatrix().getRowGroupIndices()[state + 1]; ++row) {

425  }
426  }

429  if (subformula == nullptr) {
430  subformula = std::make_shared<storm::logic::TotalRewardFormula>();
431  }
432  data.objectives.back()->formula = std::make_shared<storm::logic::RewardOperatorFormula>(subformula, rewardModelName, opInfo);
433  } else {
434  STORM_LOG_INFO("Objective " << *data.objectives.back()->originalFormula << " can not be transformed to an expected total/cumulative reward property.");
435  data.objectives.back()->formula = std::make_shared<storm::logic::ProbabilityOperatorFormula>(formula.asSharedPointer(), opInfo);
436  }
