58  for (auto const& variable : variables) {
59  blocksToMerge.insert(getBlockIndexOfVariable(variable));

63

70  return identity;

77  std::map<uint64_t, uint64_t> LocalExpressionInformation<DdType>::mergeBlocks(std::set<uint_fast64_t> const& blocksToMerge) {
78  std::map<uint64_t, uint64_t> oldToNewIndices;
79 
80  // Merge all blocks into the block to keep.
81  std::vector<std::set<storm::expressions::Variable>> newVariableBlocks;
82  std::vector<std::set<uint_fast64_t>> newExpressionBlocks;
83 
84  std::set<uint_fast64_t>::const_iterator blocksToMergeIt = blocksToMerge.begin();
85  std::set<uint_fast64_t>::const_iterator blocksToMergeIte = blocksToMerge.end();
86 
87  // Determine which block to keep (to merge the other blocks into).
88  uint_fast64_t blockToKeep = *blocksToMergeIt;
89  ++blocksToMergeIt;
90 
91  for (uint_fast64_t blockIndex = 0; blockIndex < variableBlocks.size(); ++blockIndex) {
92  // If the block is the next one to merge into the block to keep, do so now.
93  if (blocksToMergeIt != blocksToMergeIte && *blocksToMergeIt == blockIndex && blockIndex != blockToKeep) {
94  // Adjust the mapping for all variables of the old block.
95  for (auto const& variable : variableBlocks[blockIndex]) {
96  variableToBlockMapping[variable] = blockToKeep;
97  }
98  oldToNewIndices[blockIndex] = blockToKeep;
99 
100  newVariableBlocks[blockToKeep].insert(variableBlocks[blockIndex].begin(), variableBlocks[blockIndex].end());
101  newExpressionBlocks[blockToKeep].insert(expressionBlocks[blockIndex].begin(), expressionBlocks[blockIndex].end());
102  ++blocksToMergeIt;
103  } else {
104  // Otherwise just move the current block to the new partition.
105  oldToNewIndices[blockIndex] = newVariableBlocks.size();
106 
107  // Adjust the mapping for all variables of the old block.
108  for (auto const& variable : variableBlocks[blockIndex]) {
109  variableToBlockMapping[variable] = newVariableBlocks.size();
110  }
111 
112  newVariableBlocks.emplace_back(std::move(variableBlocks[blockIndex]));
113  newExpressionBlocks.emplace_back(std::move(expressionBlocks[blockIndex]));
114  }
115  }
116 
117  variableBlocks = std::move(newVariableBlocks);
118  expressionBlocks = std::move(newExpressionBlocks);
119 
120  return oldToNewIndices;
121  }
