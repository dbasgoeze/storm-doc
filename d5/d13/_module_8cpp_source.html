151  void Module::createMappings() {
152  // Clear the current mappings.
153  this->actionIndicesToCommandIndexMap.clear();
154  this->booleanVariableToIndexMap.clear();
155  this->integerVariableToIndexMap.clear();
156 
157  // Create the mappings for the variables.
158  for (uint_fast64_t i = 0; i < this->booleanVariables.size(); ++i) {
159  this->booleanVariableToIndexMap[this->getBooleanVariables()[i].getName()] = i;
160  }
161  for (uint_fast64_t i = 0; i < this->integerVariables.size(); ++i) {
162  this->integerVariableToIndexMap[this->getIntegerVariables()[i].getName()] = i;
163  }
164  for (uint_fast64_t i = 0; i < this->clockVariables.size(); ++i) {
165  this->booleanVariableToIndexMap[this->getClockVariables()[i].getName()] = i;
166  }
167 
168  // Add the mapping for all commands.
169  for (uint_fast64_t i = 0; i < this->commands.size(); i++) {
170  if (this->commands[i].isLabeled()) {
171  uint_fast64_t actionIndex = this->commands[i].getActionIndex();
172  if (this->actionIndicesToCommandIndexMap.find(actionIndex) == this->actionIndicesToCommandIndexMap.end()) {
173  this->actionIndicesToCommandIndexMap.emplace(actionIndex, std::set<uint_fast64_t>());
174  }
175  this->actionIndicesToCommandIndexMap[actionIndex].insert(i);
176 
177  // Only take the command into the set if it's non-synchronizing.
178  if (actionIndex != 0) {
179  this->synchronizingActionIndices.insert(actionIndex);
180  }
181  }
182  }
183 
184  // For all actions that are "in the module", but for which no command exists, we add the mapping to an empty
185  // set of commands.
186  for (auto const& actionIndex : this->synchronizingActionIndices) {
187  if (this->actionIndicesToCommandIndexMap.find(actionIndex) == this->actionIndicesToCommandIndexMap.end()) {
188  this->actionIndicesToCommandIndexMap[actionIndex] = std::set<uint_fast64_t>();
189  }
190  }
191  }
