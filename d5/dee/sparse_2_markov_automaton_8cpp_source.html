121  template <typename ValueType, typename RewardModelType>
122  void MarkovAutomaton<ValueType, RewardModelType>::close() {
123  if (!closed) {
124  // Get the choices that we will keep
125  storm::storage::BitVector keptChoices(this->getNumberOfChoices(), true);
126  for(auto state : this->getMarkovianStates()) {
127  if(this->getTransitionMatrix().getRowGroupSize(state) > 1) {
128  // The state is hybrid, hence, we remove the first choice.
129  keptChoices.set(this->getTransitionMatrix().getRowGroupIndices()[state], false);
130  // Afterwards, the state will no longer be Markovian.
131  this->markovianStates.set(state, false);
132  exitRates[state] = storm::utility::zero<ValueType>();
133  }
134  }
135 
136  if (!keptChoices.full()) {
137  *this = std::move(*storm::transformer::buildSubsystem(*this, storm::storage::BitVector(this->getNumberOfStates(), true), keptChoices, false).model->template as<MarkovAutomaton<ValueType, RewardModelType>>());
138  }
139 
140  // Mark the automaton as closed.
141  closed = true;
142  }
143  }

145  template <typename ValueType, typename RewardModelType>
146  void MarkovAutomaton<ValueType, RewardModelType>::turnRatesToProbabilities() {
147  bool assertRates = (this->exitRates.size() == this->getNumberOfStates());
148  if (!assertRates) {
149  STORM_LOG_THROW(this->exitRates.empty(), storm::exceptions::InvalidArgumentException, "The specified exit rate vector has an unexpected size.");
150  this->exitRates.reserve(this->getNumberOfStates());
151  }
152 
153  storm::utility::ConstantsComparator<ValueType> comparator;
154  for (uint_fast64_t state = 0; state< this->getNumberOfStates(); ++state) {
155  uint_fast64_t row = this->getTransitionMatrix().getRowGroupIndices()[state];
156  if (this->markovianStates.get(state)) {
157  if (assertRates) {
158  STORM_LOG_THROW(this->exitRates[state] == this->getTransitionMatrix().getRowSum(row), storm::exceptions::InvalidArgumentException, "The specified exit rate is inconsistent with the rate matrix. Difference is " << (this->exitRates[state] - this->getTransitionMatrix().getRowSum(row)) << ".");
159  } else {
160  this->exitRates.push_back(this->getTransitionMatrix().getRowSum(row));
161  }
162  for (auto& transition : this->getTransitionMatrix().getRow(row)) {
163  transition.setValue(transition.getValue() / this->exitRates[state]);
164  }
165  ++row;
166  } else {
167  if (assertRates) {
168  STORM_LOG_THROW(comparator.isZero(this->exitRates[state]), storm::exceptions::InvalidArgumentException, "The specified exit rate for (non-Markovian) choice should be 0.");
169  } else {
170  this->exitRates.push_back(storm::utility::zero<ValueType>());
171  }
172  }
173  for (; row < this->getTransitionMatrix().getRowGroupIndices()[state+1]; ++row) {
174  STORM_LOG_THROW(comparator.isOne(this->getTransitionMatrix().getRowSum(row)), storm::exceptions::InvalidArgumentException, "Entries of transition matrix do not sum up to one for (non-Markovian) choice " << row << " of state " << state << " (sum is " << this->getTransitionMatrix().getRowSum(row) << ").");
175  }
176  }
177  }

184  template <typename ValueType, typename RewardModelType>
185  bool MarkovAutomaton<ValueType, RewardModelType>::hasOnlyTrivialNondeterminism() const {
186  // Check every state
187  for (uint_fast64_t state = 0; state < this->getNumberOfStates(); ++state) {
188  // Get number of choices in current state
189  uint_fast64_t numberChoices = this->getTransitionMatrix().getRowGroupIndices()[state + 1] - this->getTransitionMatrix().getRowGroupIndices()[state];
190  if (isMarkovianState(state)) {
191  STORM_LOG_ASSERT(numberChoices == 1, "Wrong number of choices for Markovian state.");
192  }
193  if (numberChoices > 1) {
194  STORM_LOG_ASSERT(isProbabilisticState(state), "State is not probabilistic.");
195  return false;
196  }
197  }
198  return true;
199  }

212  std::shared_ptr<storm::models::sparse::Ctmc<ValueType, RewardModelType>> MarkovAutomaton<ValueType, RewardModelType>::convertToCtmc() const {
213  if (isClosed() && markovianStates.full()) {
214  storm::storage::sparse::ModelComponents<ValueType, RewardModelType> components(this->getTransitionMatrix(), this->getStateLabeling(), this->getRewardModels(), false);
215  components.transitionMatrix.makeRowGroupingTrivial();
216  components.exitRates = this->getExitRates();
217  if (this->hasChoiceLabeling()) {
218  components.choiceLabeling = this->getChoiceLabeling();
219  }
220  if (this->hasStateValuations()) {
221  components.stateValuations = this->getStateValuations();
222  }
223  if (this->hasChoiceOrigins()) {
224  components.choiceOrigins = this->getChoiceOrigins();
225  }
226  return std::make_shared<storm::models::sparse::Ctmc<ValueType, RewardModelType>>(std::move(components));
227  }
228 
229  // Eliminate all probabilistic states by state elimination
230  // Initialize
231  storm::storage::FlexibleSparseMatrix<ValueType> flexibleMatrix(this->getTransitionMatrix());
232  storm::storage::FlexibleSparseMatrix<ValueType> flexibleBackwardTransitions(this->getTransitionMatrix().transpose());
233  storm::solver::stateelimination::StateEliminator<ValueType> stateEliminator(flexibleMatrix, flexibleBackwardTransitions);
234 
235  for (uint_fast64_t state = 0; state < this->getNumberOfStates(); ++state) {
236  STORM_LOG_ASSERT(!this->isHybridState(state), "State is hybrid.");
237  if (this->isProbabilisticState(state)) {
238  // Eliminate this probabilistic state
239  stateEliminator.eliminateState(state, true);
240  }
241  }
242 
243  // Create the rate matrix for the CTMC
244  storm::storage::SparseMatrixBuilder<ValueType> transitionMatrixBuilder(0, 0, 0, false, false);
245  // Remember state to keep
246  storm::storage::BitVector keepStates(this->getNumberOfStates(), true);
247  for (uint_fast64_t state = 0; state < this->getNumberOfStates(); ++state) {
248  if (storm::utility::isZero(flexibleMatrix.getRowSum(state))) {
249  // State is eliminated and can be discarded
250  keepStates.set(state, false);
251  } else {
252  STORM_LOG_ASSERT(this->isMarkovianState(state), "State is not Markovian.");
253  // Copy transitions
254  for (uint_fast64_t row = flexibleMatrix.getRowGroupIndices()[state]; row < flexibleMatrix.getRowGroupIndices()[state + 1]; ++row) {
255  for (auto const& entry : flexibleMatrix.getRow(row)) {
256  // Convert probabilities into rates
257  transitionMatrixBuilder.addNextValue(state, entry.getColumn(), entry.getValue() * exitRates[state]);
258  }
259  }
260  }
261  }
262 
263  storm::storage::SparseMatrix<ValueType> rateMatrix = transitionMatrixBuilder.build();
264  rateMatrix = rateMatrix.getSubmatrix(false, keepStates, keepStates, false);
265  // Construct CTMC
266  storm::models::sparse::StateLabeling stateLabeling = this->getStateLabeling().getSubLabeling(keepStates);
267 
268  //TODO update reward models and choice labels according to kept states
269  STORM_LOG_WARN_COND(this->getRewardModels().empty(), "Conversion of MA to CTMC does not preserve rewards.");
270  STORM_LOG_WARN_COND(!this->hasChoiceLabeling(), "Conversion of MA to CTMC does not preserve choice labels.");
271  STORM_LOG_WARN_COND(!this->hasStateValuations(), "Conversion of MA to CTMC does not preserve state valuations.");
272  STORM_LOG_WARN_COND(!this->hasChoiceOrigins(), "Conversion of MA to CTMC does not preserve choice origins.");
273  return std::make_shared<storm::models::sparse::Ctmc<ValueType, RewardModelType>>(std::move(rateMatrix), std::move(stateLabeling));
274  }
