372  template<typename ValueType>
373  void InternalAdd<DdType::CUDD, ValueType>::exportToDot(std::string const &filename, std::vector <std::string> const &ddVariableNamesAsStrings, bool showVariablesIfPossible) const {
374  // Build the name input of the DD.
375  std::vector<char *> ddNames;
376  std::string ddName("f");
377  ddNames.push_back(new char[ddName.size() + 1]);
378  std::copy(ddName.c_str(), ddName.c_str() + 2, ddNames.back());
379 
380  // Now build the variables names.
381  std::vector<char *> ddVariableNames;
382  for (auto const &element : ddVariableNamesAsStrings) {
383  ddVariableNames.push_back(new char[element.size() + 1]);
384  std::copy(element.c_str(), element.c_str() + element.size() + 1, ddVariableNames.back());
385  }
386 
387  // Open the file, dump the DD and close it again.
388  FILE *filePointer = fopen(filename.c_str(), "w");
389  std::vector <cudd::ADD> cuddAddVector = {this->getCuddAdd()};
390  if (showVariablesIfPossible) {
391  ddManager->getCuddManager().DumpDot(cuddAddVector, ddVariableNames.data(), &ddNames[0], filePointer);
392  } else {
393  ddManager->getCuddManager().DumpDot(cuddAddVector, nullptr, &ddNames[0], filePointer);
394  }
395  fclose(filePointer);
396 
397  // Finally, delete the names.
398  for (char *element : ddNames) {
399  delete[] element;
400  }
401  for (char *element : ddVariableNames) {
402  delete[] element;
403  }
404  }

442  Odd InternalAdd<DdType::CUDD, ValueType>::createOdd(std::vector<uint_fast64_t> const& ddVariableIndices) const {
443  // Prepare a unique table for each level that keeps the constructed ODD nodes unique.
444  std::vector<std::unordered_map<DdNode*, std::shared_ptr<Odd>>> uniqueTableForLevels(ddVariableIndices.size() + 1);
445 
446  // Now construct the ODD structure from the ADD.
447  std::shared_ptr<Odd> rootOdd = createOddRec(this->getCuddDdNode(), ddManager->getCuddManager(), 0, ddVariableIndices.size(), ddVariableIndices, uniqueTableForLevels);

453  template<typename ValueType>
454  std::shared_ptr<Odd> InternalAdd<DdType::CUDD, ValueType>::createOddRec(DdNode* dd, cudd::Cudd const& manager, uint_fast64_t currentLevel, uint_fast64_t maxLevel, std::vector<uint_fast64_t> const& ddVariableIndices, std::vector<std::unordered_map<DdNode*, std::shared_ptr<Odd>>>& uniqueTableForLevels) {
455  // Check whether the ODD for this node has already been computed (for this level) and if so, return this instead.
456  auto const& iterator = uniqueTableForLevels[currentLevel].find(dd);
457  if (iterator != uniqueTableForLevels[currentLevel].end()) {
458  return iterator->second;
459  } else {
460  // Otherwise, we need to recursively compute the ODD.
461 
462  // If we are already past the maximal level that is to be considered, we can simply create an Odd without
463  // successors
464  if (currentLevel == maxLevel) {
465  uint_fast64_t elseOffset = 0;
466  uint_fast64_t thenOffset = 0;
467 
468  // If the DD is not the zero leaf, then the then-offset is 1.
469  if (dd != Cudd_ReadZero(manager.getManager())) {
470  thenOffset = 1;
471  }
472 
473  auto oddNode = std::make_shared<Odd>(nullptr, elseOffset, nullptr, thenOffset);
474  uniqueTableForLevels[currentLevel].emplace(dd, oddNode);
475  return oddNode;
476  } else if (ddVariableIndices[currentLevel] < Cudd_NodeReadIndex(dd)) {
477  // If we skipped the level in the DD, we compute the ODD just for the else-successor and use the same
478  // node for the then-successor as well.
479  std::shared_ptr<Odd> elseNode = createOddRec(dd, manager, currentLevel + 1, maxLevel, ddVariableIndices, uniqueTableForLevels);
480  std::shared_ptr<Odd> thenNode = elseNode;
481  auto oddNode = std::make_shared<Odd>(elseNode, elseNode->getElseOffset() + elseNode->getThenOffset(), thenNode, thenNode->getElseOffset() + thenNode->getThenOffset());
482  uniqueTableForLevels[currentLevel].emplace(dd, oddNode);
483  return oddNode;
484  } else {
485  // Otherwise, we compute the ODDs for both the then- and else successors.
486  std::shared_ptr<Odd> elseNode = createOddRec(Cudd_E(dd), manager, currentLevel + 1, maxLevel, ddVariableIndices, uniqueTableForLevels);
487  std::shared_ptr<Odd> thenNode = createOddRec(Cudd_T(dd), manager, currentLevel + 1, maxLevel, ddVariableIndices, uniqueTableForLevels);
488 
489  uint_fast64_t totalElseOffset = elseNode->getElseOffset() + elseNode->getThenOffset();
490  uint_fast64_t totalThenOffset = thenNode->getElseOffset() + thenNode->getThenOffset();
491 
492  auto oddNode = std::make_shared<Odd>(elseNode, totalElseOffset, thenNode, totalThenOffset);
493  uniqueTableForLevels[currentLevel].emplace(dd, oddNode);
494  return oddNode;
495  }
496  }
497  }

522  template<typename ValueType>
523  void InternalAdd<DdType::CUDD, ValueType>::forEachRec(DdNode const* dd, uint_fast64_t currentLevel, uint_fast64_t maxLevel, uint_fast64_t currentOffset, Odd const& odd, std::vector<uint_fast64_t> const& ddVariableIndices, std::function<void (uint64_t const&, ValueType const&)> const& function) const {
524  // For the empty DD, we do not need to add any entries.
525  if (dd == Cudd_ReadZero(ddManager->getCuddManager().getManager())) {
526  return;
527  }
528 
529  // If we are at the maximal level, the value to be set is stored as a constant in the DD.
530  if (currentLevel == maxLevel) {
531  function(currentOffset, storm::utility::convertNumber<ValueType>(Cudd_V(dd)));
532  } else if (ddVariableIndices[currentLevel] < Cudd_NodeReadIndex(dd)) {
533  // If we skipped a level, we need to enumerate the explicit entries for the case in which the bit is set
534  // and for the one in which it is not set.
535  forEachRec(dd, currentLevel + 1, maxLevel, currentOffset, odd.getElseSuccessor(), ddVariableIndices, function);
536  forEachRec(dd, currentLevel + 1, maxLevel, currentOffset + odd.getElseOffset(), odd.getThenSuccessor(), ddVariableIndices, function);
537  } else {
538  // Otherwise, we simply recursively call the function for both (different) cases.
539  forEachRec(Cudd_E_const(dd), currentLevel + 1, maxLevel, currentOffset, odd.getElseSuccessor(), ddVariableIndices, function);
540  forEachRec(Cudd_T_const(dd), currentLevel + 1, maxLevel, currentOffset + odd.getElseOffset(), odd.getThenSuccessor(), ddVariableIndices, function);
541  }

696  template<typename ValueType>
697  void InternalAdd<DdType::CUDD, ValueType>::toMatrixComponentsRec(DdNode const* dd, std::vector<uint_fast64_t> const& rowGroupOffsets, std::vector<uint_fast64_t>& rowIndications, std::vector<storm::storage::MatrixEntry<uint_fast64_t, ValueType>>& columnsAndValues, Odd const& rowOdd, Odd const& columnOdd, uint_fast64_t currentRowLevel, uint_fast64_t currentColumnLevel, uint_fast64_t maxLevel, uint_fast64_t currentRowOffset, uint_fast64_t currentColumnOffset, std::vector<uint_fast64_t> const& ddRowVariableIndices, std::vector<uint_fast64_t> const& ddColumnVariableIndices, bool generateValues) const {
698  // For the empty DD, we do not need to add any entries.
699  if (dd == Cudd_ReadZero(ddManager->getCuddManager().getManager())) {
700  return;
701  }
702 
703  // If we are at the maximal level, the value to be set is stored as a constant in the DD.
704  if (currentRowLevel + currentColumnLevel == maxLevel) {
705  if (generateValues) {
706  columnsAndValues[rowIndications[rowGroupOffsets[currentRowOffset]]] = storm::storage::MatrixEntry<uint_fast64_t, ValueType>(currentColumnOffset, storm::utility::convertNumber<ValueType>(Cudd_V(dd)));
707  }
708  ++rowIndications[rowGroupOffsets[currentRowOffset]];
709  } else {
710  DdNode const* elseElse;
711  DdNode const* elseThen;
712  DdNode const* thenElse;
713  DdNode const* thenThen;
714 
715  if (ddColumnVariableIndices[currentColumnLevel] < Cudd_NodeReadIndex(dd)) {
716  elseElse = elseThen = thenElse = thenThen = dd;
717  } else if (ddRowVariableIndices[currentColumnLevel] < Cudd_NodeReadIndex(dd)) {
718  elseElse = thenElse = Cudd_E_const(dd);
719  elseThen = thenThen = Cudd_T_const(dd);
720  } else {
721  DdNode const* elseNode = Cudd_E_const(dd);
722  if (ddColumnVariableIndices[currentColumnLevel] < Cudd_NodeReadIndex(elseNode)) {
723  elseElse = elseThen = elseNode;
724  } else {
725  elseElse = Cudd_E_const(elseNode);
726  elseThen = Cudd_T_const(elseNode);
727  }
728 
729  DdNode const* thenNode = Cudd_T_const(dd);
730  if (ddColumnVariableIndices[currentColumnLevel] < Cudd_NodeReadIndex(thenNode)) {
731  thenElse = thenThen = thenNode;
732  } else {
733  thenElse = Cudd_E_const(thenNode);
734  thenThen = Cudd_T_const(thenNode);
735  }
736  }
737 
738  // Visit else-else.
739  toMatrixComponentsRec(elseElse, rowGroupOffsets, rowIndications, columnsAndValues, rowOdd.getElseSuccessor(), columnOdd.getElseSuccessor(), currentRowLevel + 1, currentColumnLevel + 1, maxLevel, currentRowOffset, currentColumnOffset, ddRowVariableIndices, ddColumnVariableIndices, generateValues);
740  // Visit else-then.
741  toMatrixComponentsRec(elseThen, rowGroupOffsets, rowIndications, columnsAndValues, rowOdd.getElseSuccessor(), columnOdd.getThenSuccessor(), currentRowLevel + 1, currentColumnLevel + 1, maxLevel, currentRowOffset, currentColumnOffset + columnOdd.getElseOffset(), ddRowVariableIndices, ddColumnVariableIndices, generateValues);
742  // Visit then-else.
743  toMatrixComponentsRec(thenElse, rowGroupOffsets, rowIndications, columnsAndValues, rowOdd.getThenSuccessor(), columnOdd.getElseSuccessor(), currentRowLevel + 1, currentColumnLevel + 1, maxLevel, currentRowOffset + rowOdd.getElseOffset(), currentColumnOffset, ddRowVariableIndices, ddColumnVariableIndices, generateValues);
744  // Visit then-then.
745  toMatrixComponentsRec(thenThen, rowGroupOffsets, rowIndications, columnsAndValues, rowOdd.getThenSuccessor(), columnOdd.getThenSuccessor(), currentRowLevel + 1, currentColumnLevel + 1, maxLevel, currentRowOffset + rowOdd.getElseOffset(), currentColumnOffset + columnOdd.getElseOffset(), ddRowVariableIndices, ddColumnVariableIndices, generateValues);
746  }

755  template<typename ValueType>
756  DdNode* InternalAdd<DdType::CUDD, ValueType>::fromVectorRec(::DdManager* manager, uint_fast64_t& currentOffset, uint_fast64_t currentLevel, uint_fast64_t maxLevel, std::vector<ValueType> const& values, Odd const& odd, std::vector<uint_fast64_t> const& ddVariableIndices) {
757  if (currentLevel == maxLevel) {
758  // If we are in a terminal node of the ODD, we need to check whether the then-offset of the ODD is one
759  // (meaning the encoding is a valid one) or zero (meaning the encoding is not valid). Consequently, we
760  // need to copy the next value of the vector iff the then-offset is greater than zero.
761  if (odd.getThenOffset() > 0) {
762  return Cudd_addConst(manager, values[currentOffset++]);
763  } else {
764  return Cudd_ReadZero(manager);
765  }
766  } else {
767  // If the total offset is zero, we can just return the constant zero DD.
768  if (odd.getThenOffset() + odd.getElseOffset() == 0) {
769  return Cudd_ReadZero(manager);
770  }
771 
772  // Determine the new else-successor.
773  DdNode* elseSuccessor = nullptr;
774  if (odd.getElseOffset() > 0) {
775  elseSuccessor = fromVectorRec(manager, currentOffset, currentLevel + 1, maxLevel, values, odd.getElseSuccessor(), ddVariableIndices);
776  } else {
777  elseSuccessor = Cudd_ReadZero(manager);
778  }
779  Cudd_Ref(elseSuccessor);
780 
781  // Determine the new then-successor.
782  DdNode* thenSuccessor = nullptr;
783  if (odd.getThenOffset() > 0) {
784  thenSuccessor = fromVectorRec(manager, currentOffset, currentLevel + 1, maxLevel, values, odd.getThenSuccessor(), ddVariableIndices);
785  } else {
786  thenSuccessor = Cudd_ReadZero(manager);
787  }
788  Cudd_Ref(thenSuccessor);
789 
790  // Create a node representing ITE(currentVar, thenSuccessor, elseSuccessor);
791  DdNode* result = Cudd_addIthVar(manager, static_cast<int>(ddVariableIndices[currentLevel]));
792  Cudd_Ref(result);
793  DdNode* newResult = Cudd_addIte(manager, result, thenSuccessor, elseSuccessor);
794  Cudd_Ref(newResult);
795 
796  // Dispose of the intermediate results
797  Cudd_RecursiveDeref(manager, result);
798  Cudd_RecursiveDeref(manager, thenSuccessor);
799  Cudd_RecursiveDeref(manager, elseSuccessor);
800 
801  // Before returning, we remove the protection imposed by the previous call to Cudd_Ref.
802  Cudd_Deref(newResult);
803 
804  return newResult;
805  }
806  }
