90  template <typename ValueType>
91  SampleInformation<ValueType> parseSamples(std::shared_ptr<storm::models::ModelBase> const& model, std::string const& sampleString, bool graphPreserving) {
92  STORM_LOG_THROW(!model || model->isSparseModel(), storm::exceptions::NotSupportedException, "Sampling is only supported for sparse models.");
93 
94  SampleInformation<ValueType> sampleInfo(graphPreserving);
95  if (sampleString.empty()) {
96  return sampleInfo;
97  }
98 
99  // Get all parameters from the model.
100  std::set<typename utility::parametric::VariableType<ValueType>::type> modelParameters;
101  auto const& sparseModel = *model->as<storm::models::sparse::Model<ValueType>>();
102  modelParameters = storm::models::sparse::getProbabilityParameters(sparseModel);
103  auto rewParameters = storm::models::sparse::getRewardParameters(sparseModel);
104  modelParameters.insert(rewParameters.begin(), rewParameters.end());
105 
106  std::vector<std::string> cartesianProducts;
107  boost::split(cartesianProducts, sampleString, boost::is_any_of(";"));
108  for (auto& product : cartesianProducts) {
109  boost::trim(product);
110 
111  // Get the values string for each variable.
112  std::vector<std::string> valuesForVariables;
113  boost::split(valuesForVariables, product, boost::is_any_of(","));
114  for (auto& values : valuesForVariables) {
115  boost::trim(values);
116  }
117 
118  std::set<typename utility::parametric::VariableType<ValueType>::type> encounteredParameters;
119  sampleInfo.cartesianProducts.emplace_back();
120  auto& newCartesianProduct = sampleInfo.cartesianProducts.back();
121  for (auto const& varValues : valuesForVariables) {
122  auto equalsPosition = varValues.find("=");
123  STORM_LOG_THROW(equalsPosition != varValues.npos, storm::exceptions::WrongFormatException, "Incorrect format of samples.");
124  std::string variableName = varValues.substr(0, equalsPosition);
125  boost::trim(variableName);
126  std::string values = varValues.substr(equalsPosition + 1);
127  boost::trim(values);
128 
129  bool foundParameter = false;
130  typename utility::parametric::VariableType<ValueType>::type theParameter;
131  for (auto const& parameter : modelParameters) {
132  std::stringstream parameterStream;
133  parameterStream << parameter;
134  if (parameterStream.str() == variableName) {
135  foundParameter = true;
136  theParameter = parameter;
137  encounteredParameters.insert(parameter);
138  }
139  }
140  STORM_LOG_THROW(foundParameter, storm::exceptions::WrongFormatException, "Unknown parameter '" << variableName << "'.");
141 
142  std::vector<std::string> splitValues;
143  boost::split(splitValues, values, boost::is_any_of(":"));
144  STORM_LOG_THROW(!splitValues.empty(), storm::exceptions::WrongFormatException, "Expecting at least one value per parameter.");
145 
146  auto& list = newCartesianProduct[theParameter];
147 
148  for (auto& value : splitValues) {
149  boost::trim(value);
150  list.push_back(storm::utility::convertNumber<typename utility::parametric::CoefficientType<ValueType>::type>(value));
151  }
152  }
153 
154  STORM_LOG_THROW(encounteredParameters == modelParameters, storm::exceptions::WrongFormatException, "Variables for all parameters are required when providing samples.");
155  }
156 
157  return sampleInfo;
158  }

307  template<template<typename, typename> class ModelCheckerType, typename ModelType, typename ValueType, typename SolveValueType = double>
308  void verifyPropertiesAtSamplePoints(ModelType const& model, SymbolicInput const& input, SampleInformation<ValueType> const& samples) {
309 
310  // When samples are provided, we create an instantiation model checker.
311  ModelCheckerType<ModelType, SolveValueType> modelchecker(model);
312 
313  for (auto const& property : input.properties) {
314  storm::cli::printModelCheckingProperty(property);
315 
316  modelchecker.specifyFormula(storm::api::createTask<ValueType>(property.getRawFormula(), true));
317  modelchecker.setInstantiationsAreGraphPreserving(samples.graphPreserving);
318 
319  storm::utility::parametric::Valuation<ValueType> valuation;
320 
321  std::vector<typename utility::parametric::VariableType<ValueType>::type> parameters;
322  std::vector<typename std::vector<typename utility::parametric::CoefficientType<ValueType>::type>::const_iterator> iterators;
323  std::vector<typename std::vector<typename utility::parametric::CoefficientType<ValueType>::type>::const_iterator> iteratorEnds;
324 
325  storm::utility::Stopwatch watch(true);
326  for (auto const& product : samples.cartesianProducts) {
327  parameters.clear();
328  iterators.clear();
329  iteratorEnds.clear();
330 
331  for (auto const& entry : product) {
332  parameters.push_back(entry.first);
333  iterators.push_back(entry.second.cbegin());
334  iteratorEnds.push_back(entry.second.cend());
335  }
336 
337  bool done = false;
338  while (!done) {
339  // Read off valuation.
340  for (uint64_t i = 0; i < parameters.size(); ++i) {
341  valuation[parameters[i]] = *iterators[i];
342  }
343 
344  storm::utility::Stopwatch valuationWatch(true);
345  std::unique_ptr<storm::modelchecker::CheckResult> result = modelchecker.check(Environment(), valuation);
346  valuationWatch.stop();
347 
348  if (result) {
349  result->filter(storm::modelchecker::ExplicitQualitativeCheckResult(model.getInitialStates()));
350  }
351  printInitialStatesResult<ValueType>(result, property, &valuationWatch, &valuation);
352 
353  for (uint64_t i = 0; i < parameters.size(); ++i) {
354  ++iterators[i];
355  if (iterators[i] == iteratorEnds[i]) {
356  // Reset iterator and proceed to move next iterator.
357  iterators[i] = product.at(parameters[i]).cbegin();
358 
359  // If the last iterator was removed, we are done.
360  if (i == parameters.size() - 1) {
361  done = true;
362  }
363  } else {
364  // If an iterator was moved but not reset, we have another valuation to check.
365  break;
366  }
367  }
368 
369  }
370  }
371 
372  watch.stop();
373  STORM_PRINT_AND_LOG("Overall time for sampling all instances: " << watch << std::endl << std::endl);
374  }
375  }

568  auto buildSettings = storm::settings::getModule<storm::settings::modules::BuildSettings>();
569  auto parSettings = storm::settings::getModule<storm::settings::modules::ParametricSettings>();
570  auto monSettings = storm::settings::getModule<storm::settings::modules::MonotonicitySettings>();
571 
572  STORM_LOG_THROW(mpi.engine == storm::utility::Engine::Sparse || mpi.engine == storm::utility::Engine::Hybrid || mpi.engine == storm::utility::Engine::Dd, storm::exceptions::InvalidSettingsException, "The selected engine is not supported for parametric models.");
573 
574  std::shared_ptr<storm::models::ModelBase> model;
575  if (!buildSettings.isNoBuildModelSet()) {
576  model = storm::cli::buildModel<DdType, ValueType>(input, ioSettings, mpi);
577  }
578 
579  if (model) {
580  model->printModelInformationToStream(std::cout);
581  }
582 
583  STORM_LOG_THROW(model || input.properties.empty(), storm::exceptions::InvalidSettingsException, "No input model.");
584 
585  if (monSettings.isMonotonicityAnalysisSet()) {
586  // Simplify the model
587  storm::utility::Stopwatch simplifyingWatch(true);
588  if (model->isOfType(storm::models::ModelType::Dtmc)) {
589  auto consideredModel = (model->as<storm::models::sparse::Dtmc<ValueType>>());
590  auto simplifier = storm::transformer::SparseParametricDtmcSimplifier<storm::models::sparse::Dtmc<ValueType>>(*consideredModel);
591 
592  std::vector<std::shared_ptr<storm::logic::Formula const>> formulas = storm::api::extractFormulasFromProperties(input.properties);
593  STORM_LOG_THROW(formulas.begin()!=formulas.end(), storm::exceptions::NotSupportedException, "Only one formula at the time supported");
594 
595  if (!simplifier.simplify(*(formulas[0]))) {
596  STORM_LOG_THROW(false, storm::exceptions::UnexpectedException, "Simplifying the model was not successfull.");
597  }
598  model = simplifier.getSimplifiedModel();
599  } else if (model->isOfType(storm::models::ModelType::Mdp)) {
600  auto consideredModel = (model->as<storm::models::sparse::Mdp<ValueType>>());
601  auto simplifier = storm::transformer::SparseParametricMdpSimplifier<storm::models::sparse::Mdp<ValueType>>(*consideredModel);
602 
603  std::vector<std::shared_ptr<storm::logic::Formula const>> formulas = storm::api::extractFormulasFromProperties(input.properties);
604  STORM_LOG_THROW(formulas.begin()!=formulas.end(), storm::exceptions::NotSupportedException, "Only one formula at the time supported");
605 
606  if (!simplifier.simplify(*(formulas[0]))) {
607  STORM_LOG_THROW(false, storm::exceptions::UnexpectedException, "Simplifying the model was not successfull.");
608  }
609  model = simplifier.getSimplifiedModel();
610  } else {
611  STORM_LOG_THROW(false, storm::exceptions::InvalidOperationException, "Unable to perform monotonicity analysis on the provided model type.");
612  }
613 
614  simplifyingWatch.stop();
615  STORM_PRINT(std::endl << "Time for model simplification: " << simplifyingWatch << "." << std::endl << std::endl);
616  model->printModelInformationToStream(std::cout);
617  }
618 
619  if (model) {
620  auto preprocessingResult = storm::pars::preprocessModel<DdType, ValueType>(model, input, mpi);
621  if (preprocessingResult.changed) {
622  model = preprocessingResult.model;
623 
624  if (preprocessingResult.formulas) {
625  std::vector<storm::jani::Property> newProperties;
626  for (size_t i = 0; i < preprocessingResult.formulas.get().size(); ++i) {
627  auto formula = preprocessingResult.formulas.get().at(i);
628  STORM_LOG_ASSERT(i < input.properties.size(), "Index " << i << " greater than number of properties.");
629  storm::jani::Property property = input.properties.at(i);
630  newProperties.push_back(storm::jani::Property(property.getName(), formula, property.getUndefinedConstants(), property.getComment()));
631  }
632  input.properties = newProperties;
633  }
634 
635  model->printModelInformationToStream(std::cout);
636  }
637  }
638 
639 
640  if (model && monSettings.isSccEliminationSet()) {
641  storm::utility::Stopwatch eliminationWatch(true);
642  if (model->isOfType(storm::models::ModelType::Dtmc)) {
643  STORM_PRINT("Applying scc elimination" << std::endl);
644  auto sparseModel = model->as<storm::models::sparse::Model<ValueType>>();
645  auto matrix = sparseModel->getTransitionMatrix();
646  auto backwardsTransitionMatrix = matrix.transpose();
647 
648  storm::storage::StronglyConnectedComponentDecompositionOptions const options;
649  auto decomposition = storm::storage::StronglyConnectedComponentDecomposition<ValueType>(matrix, options);
650 
651  storm::storage::BitVector selectedStates(matrix.getRowCount());
652  storm::storage::BitVector selfLoopStates(matrix.getRowCount());
653  for (size_t i = 0; i < decomposition.size(); ++i) {
654  auto scc = decomposition.getBlock(i);
655  if (scc.size() > 1) {
656  auto statesScc = scc.getStates();
657  std::vector<uint_fast64_t> entryStates;
658  for (auto state : statesScc) {
659  auto row = backwardsTransitionMatrix.getRow(state);
660  bool found = false;
661  for (auto backState : row) {
662  if (!scc.containsState(backState.getColumn())) {
663  found = true;
664  }
665  }
666  if (found) {
667  entryStates.push_back(state);
668  selfLoopStates.set(state);
669  } else {
670  selectedStates.set(state);
671  }
672  }
673 
674  if (entryStates.size() != 1) {
675  STORM_LOG_THROW(entryStates.size() > 1, storm::exceptions::NotImplementedException,
676  "state elimination not implemented for scc with more than 1 entry points");
677  }
678  }
679  }
680 
681  storm::storage::FlexibleSparseMatrix<ValueType> flexibleMatrix(matrix);
682  storm::storage::FlexibleSparseMatrix<ValueType> flexibleBackwardTransitions(backwardsTransitionMatrix, true);
683  auto actionRewards = std::vector<ValueType>(matrix.getRowCount(), storm::utility::zero<ValueType>());
684  storm::solver::stateelimination::NondeterministicModelStateEliminator<ValueType> stateEliminator(flexibleMatrix, flexibleBackwardTransitions, actionRewards);
685  for(auto state : selectedStates) {
686  stateEliminator.eliminateState(state, true);
687  }
688  for (auto state : selfLoopStates) {
689  auto row = flexibleMatrix.getRow(state);
690  stateEliminator.eliminateLoop(state);
691  }
692  selectedStates.complement();
693  auto keptRows = matrix.getRowFilter(selectedStates);
694  storm::storage::SparseMatrix<ValueType> newTransitionMatrix = flexibleMatrix.createSparseMatrix(keptRows, selectedStates);
695  // TODO: note that rewards get lost
696  model = std::make_shared<storm::models::sparse::Dtmc<ValueType>>(std::move(newTransitionMatrix), sparseModel->getStateLabeling().getSubLabeling(selectedStates));
697 
698  eliminationWatch.stop();
699  STORM_PRINT(std::endl << "Time for scc elimination: " << eliminationWatch << "." << std::endl << std::endl);
700  model->printModelInformationToStream(std::cout);
701  } else if (model->isOfType(storm::models::ModelType::Mdp)) {
702  STORM_LOG_THROW(false, storm::exceptions::NotImplementedException, "Unable to perform SCC elimination for monotonicity analysis on MDP: Not mplemented");
703  } else {
704  STORM_LOG_THROW(false, storm::exceptions::InvalidOperationException, "Unable to perform monotonicity analysis on the provided model type.");
705  }
706 
707  }
708 
709  std::vector<storm::storage::ParameterRegion<ValueType>> regions = parseRegions<ValueType>(model);
710 
711  if (monSettings.isMonotonicityAnalysisSet()) {
712  std::vector<std::shared_ptr<storm::logic::Formula const>> formulas = storm::api::extractFormulasFromProperties(input.properties);
713  // Monotonicity
714  storm::utility::Stopwatch monotonicityWatch(true);
715 
716  STORM_LOG_THROW(regions.size() <= 1, storm::exceptions::InvalidArgumentException, "Monotonicity analysis only allowed on single region");
717  storm::analysis::MonotonicityChecker<ValueType> monotonicityChecker = storm::analysis::MonotonicityChecker<ValueType>(model, formulas, regions, monSettings.isValidateAssumptionsSet(), monSettings.getNumberOfSamples(), monSettings.getMonotonicityAnalysisPrecision());
718  if (ioSettings.isExportMonotonicitySet()) {
719  std::ofstream outfile;
720  utility::openFile(ioSettings.getExportMonotonicityFilename(), outfile);
721  monotonicityChecker.checkMonotonicity(outfile);
722  utility::closeFile(outfile);
723  } else {
724  monotonicityChecker.checkMonotonicity(std::cout);
725  }
726  monotonicityWatch.stop();
727  STORM_PRINT(std::endl << "Total time for monotonicity checking: " << monotonicityWatch << "." << std::endl
728  << std::endl);
729  return;
730  }
731 
732  std::string samplesAsString = parSettings.getSamples();
733  SampleInformation<ValueType> samples;
734  if (!samplesAsString.empty()) {
735  samples = parseSamples<ValueType>(model, samplesAsString, parSettings.isSamplesAreGraphPreservingSet());
736  samples.exact = parSettings.isSampleExactSet();
737  }
738 
739  if (model) {
740  storm::cli::exportModel<DdType, ValueType>(model, input);
741  }
742 
743  if (parSettings.onlyObtainConstraints()) {
744  STORM_LOG_THROW(parSettings.exportResultToFile(), storm::exceptions::InvalidSettingsException, "When computing constraints, export path has to be specified.");
745  storm::api::exportParametricResultToFile<ValueType>(boost::none, storm::analysis::ConstraintCollector<ValueType>(*(model->as<storm::models::sparse::Model<ValueType>>())), parSettings.exportResultPath());
746  return;
747  }
748 
749  if (model) {
750  verifyParametricModel<DdType, ValueType>(model, input, regions, samples);
751  }
752  }
