205  template<typename ValueType>
206  void Z3LpSolver<ValueType>::optimize() const {
207  // First incorporate all recent changes.
208  this->update();
209 
210  // Invoke push() as we want to be able to erase the current optimization function after checking
211  solver->push();
212 
213  // Solve the optimization problem depending on the optimization direction
214  storm::expressions::Expression optimizationFunction = storm::expressions::sum(optimizationSummands);
215  z3::optimize::handle optFuncHandle = this->getOptimizationDirection() == OptimizationDirection::Minimize ? solver->minimize(expressionAdapter->translateExpression(optimizationFunction)) : solver->maximize(expressionAdapter->translateExpression(optimizationFunction));
216  z3::check_result chkRes = solver->check();
217  STORM_LOG_THROW(chkRes != z3::unknown, storm::exceptions::InvalidStateException, "Unable to solve LP problem with Z3: Check result is unknown.");
218 
219  // We need to store the resulting information at this point. Otherwise, the information would be lost after calling pop() ...
220 
221  // Check feasibility
222  lastCheckInfeasible = (chkRes == z3::unsat);
223  if (lastCheckInfeasible) {
224  lastCheckUnbounded = false;
225  } else {
226  // Get objective result
227  lastCheckObjectiveValue = std::make_unique<z3::expr>(solver->upper(optFuncHandle));
228  // Check boundedness
229  STORM_LOG_ASSERT(lastCheckObjectiveValue->is_app(), "Failed to convert Z3 expression. Encountered unknown expression type.");
230  lastCheckUnbounded = (lastCheckObjectiveValue->decl().decl_kind() != Z3_OP_ANUM);
231  if (lastCheckUnbounded) {
232  lastCheckObjectiveValue.reset(nullptr);
233  } else {
234  // Assert that the upper approximation equals the lower one
235  STORM_LOG_ASSERT(std::string(Z3_get_numeral_string(*context, *lastCheckObjectiveValue)) == std::string(Z3_get_numeral_string(*context, solver->lower(optFuncHandle))), "Lower and Upper Approximation of z3LPSolver result do not match.");
236  lastCheckModel = std::make_unique<z3::model>(solver->get_model());
237  }
238  }
239 
240  solver->pop(); // removes current optimization function
241  this->currentModelHasBeenOptimized = true;
242  }
