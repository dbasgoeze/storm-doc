375  ValueType maxReward = std::max(totalRewardVector.getMax(), -totalRewardVector.getMin());
376 
377  // If all rewards are zero, the result is the constant zero vector.
378  if (storm::utility::isZero(maxReward)) {
379  return std::unique_ptr<CheckResult>(new SymbolicQuantitativeCheckResult<DdType, ValueType>(model.getReachableStates(), model.getManager().template getAddZero<ValueType>()));
380  }
381 
382  // Set the possible (absolute) error allowed for truncation (epsilon for fox-glynn)
383  ValueType epsilon = storm::utility::convertNumber<ValueType>(env.solver().timeBounded().getPrecision());
384  if (env.solver().timeBounded().getRelativeTerminationCriterion()) {
385  // Be more precise, if the value is very small (this still gives no sound guarantee)
386  epsilon *= std::min(storm::utility::one<ValueType>(), maxReward);
387  } else {
388  // Be more precise, if the maximal possible value is very large
389  epsilon /= std::max(storm::utility::one<ValueType>(), maxReward * timeBound);
390  }
391 
392  storm::storage::BitVector relevantValues;
393  if (onlyInitialStatesRelevant) {
394  relevantValues = model.getInitialStates().toVector(odd);
395  } else {
396  relevantValues = storm::storage::BitVector(explicitTotalRewardVector.size(), true);
397  }
398 
399  std::vector<ValueType> result;
400  // Finally, compute the transient probabilities.
401  // Loop until the desired precision is reached.
402  do {
403  result = storm::modelchecker::helper::SparseCtmcCslHelper::computeTransientProbabilities<ValueType, true>(env, explicitUniformizedMatrix, nullptr, timeBound, uniformizationRate, explicitTotalRewardVector, epsilon);
404  } while (storm::modelchecker::helper::SparseCtmcCslHelper::checkAndUpdateTransientProbabilityEpsilon(env, epsilon, result, relevantValues));
405 
406  return std::unique_ptr<CheckResult>(new HybridQuantitativeCheckResult<DdType, ValueType>(model.getReachableStates(), model.getManager().getBddZero(), model.getManager().template getAddZero<ValueType>(), model.getReachableStates(), std::move(odd), std::move(result)));
407  }

415  storm::dd::Add<DdType, ValueType> HybridCtmcCslHelper::computeUniformizedMatrix(storm::models::symbolic::Ctmc<DdType, ValueType> const& model, storm::dd::Add<DdType, ValueType> const& transitionMatrix, storm::dd::Add<DdType, ValueType> const& exitRateVector, storm::dd::Bdd<DdType> const& maybeStates, ValueType uniformizationRate) {
416  STORM_LOG_DEBUG("Computing uniformized matrix using uniformization rate " << uniformizationRate << ".");
417  STORM_LOG_DEBUG("Keeping " << maybeStates.getNonZeroCount() << " rows.");
418 
419  // Cut all non-maybe rows/columns from the transition matrix.
420  storm::dd::Add<DdType, ValueType> uniformizedMatrix = transitionMatrix * maybeStates.template toAdd<ValueType>() * maybeStates.swapVariables(model.getRowColumnMetaVariablePairs()).template toAdd<ValueType>();
421 
422  // Now perform the uniformization.
423  uniformizedMatrix = uniformizedMatrix / model.getManager().getConstant(uniformizationRate);
424  storm::dd::Add<DdType, ValueType> diagonal = model.getRowColumnIdentity() * maybeStates.template toAdd<ValueType>();
425  storm::dd::Add<DdType, ValueType> diagonalOffset = diagonal;
426  diagonalOffset -= diagonal * (exitRateVector / model.getManager().getConstant(uniformizationRate));
427  uniformizedMatrix += diagonalOffset;
428 
429  return uniformizedMatrix;
430  }
