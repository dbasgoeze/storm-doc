238  // To detect cyclic dependencies between function bodies, we need to eliminate the functions in a topological order
239  enum class FunctionDefinitionStatus {Unprocessed, Current, Processed};
240  void eliminateFunctionsInFunctionBodies(FunctionEliminationExpressionVisitor& eliminationVisitor, std::unordered_map<std::string, FunctionDefinition>& functions, std::unordered_map<std::string, FunctionDefinitionStatus>& status, std::string const& current) {
241  status[current] = FunctionDefinitionStatus::Current;
242  FunctionDefinition& funDef = functions.find(current)->second;
243  auto calledFunctions = getOccurringFunctionCalls(funDef.getFunctionBody());
244  for (auto const& calledFunction : calledFunctions) {
245  STORM_LOG_THROW(calledFunction != current, storm::exceptions::NotSupportedException, "Function '" << calledFunction << "' calls itself. This is not supported.");
246  auto calledStatus = status.find(calledFunction);
247  // Check whether the called function belongs to the ones that actually needed processing
248  if (calledStatus != status.end()) {
249  STORM_LOG_THROW(calledStatus->second != FunctionDefinitionStatus::Current, storm::exceptions::NotSupportedException, "Found cyclic dependencies between functions '" << calledFunction << "' and '" << current << "'. This is not supported.");
250  if (calledStatus->second == FunctionDefinitionStatus::Unprocessed) {
251  eliminateFunctionsInFunctionBodies(eliminationVisitor, functions, status, calledFunction);
252  }
253  }
254  }
255  // At this point, all called functions are processed already. So we can finally process this one.
256  funDef.setFunctionBody(eliminationVisitor.eliminate(funDef.getFunctionBody()));
257  status[current] = FunctionDefinitionStatus::Processed;
258  }
259  void eliminateFunctionsInFunctionBodies(FunctionEliminationExpressionVisitor& eliminationVisitor, std::unordered_map<std::string, FunctionDefinition>& functions) {
260 
261  std::unordered_map<std::string, FunctionDefinitionStatus> status;
262  for (auto const& f : functions) {
263  status.emplace(f.first, FunctionDefinitionStatus::Unprocessed);
264  }
265  for (auto const& f : functions) {
266  if (status[f.first] == FunctionDefinitionStatus::Unprocessed) {
267  eliminateFunctionsInFunctionBodies(eliminationVisitor, functions, status, f.first);
268  }
269  }
