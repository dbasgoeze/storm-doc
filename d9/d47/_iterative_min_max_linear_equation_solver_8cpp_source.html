280  }

286  } else if (method == MinMaxMethod::ViToPi) {

288  if (!this->hasUniqueSolution()) {
289  requirements.requireUniqueSolution();
290  }
291  } else {

507  template<typename ValueType>
508  bool IterativeMinMaxLinearEquationSolver<ValueType>::solveEquationsIntervalIteration(Environment const& env, OptimizationDirection dir, std::vector<ValueType>& x, std::vector<ValueType> const& b) const {
509  STORM_LOG_THROW(this->hasUpperBound(), storm::exceptions::UnmetRequirementException, "Solver requires upper bound, but none was given.");
510 
511  if (!this->multiplierA) {
512  this->multiplierA = storm::solver::MultiplierFactory<ValueType>().create(env, *this->A);
513  }
514 
515  if (!auxiliaryRowGroupVector) {
516  auxiliaryRowGroupVector = std::make_unique<std::vector<ValueType>>(this->A->getRowGroupCount());
517  }
518 
519  // Allow aliased multiplications.
520  bool useGaussSeidelMultiplication = env.solver().minMax().getMultiplicationStyle() == storm::solver::MultiplicationStyle::GaussSeidel;
521 
522  std::vector<ValueType>* lowerX = &x;
523  this->createLowerBoundsVector(*lowerX);
524  this->createUpperBoundsVector(this->auxiliaryRowGroupVector, this->A->getRowGroupCount());
525  std::vector<ValueType>* upperX = this->auxiliaryRowGroupVector.get();
526 
527  std::vector<ValueType>* tmp = nullptr;
528  if (!useGaussSeidelMultiplication) {
529  auxiliaryRowGroupVector2 = std::make_unique<std::vector<ValueType>>(lowerX->size());
530  tmp = auxiliaryRowGroupVector2.get();
531  }
532 
533  // Proceed with the iterations as long as the method did not converge or reach the maximum number of iterations.
534  uint64_t iterations = 0;
535 
536  SolverStatus status = SolverStatus::InProgress;
537  bool doConvergenceCheck = true;
538  bool useDiffs = this->hasRelevantValues() && !env.solver().minMax().isSymmetricUpdatesSet();
539  std::vector<ValueType> oldValues;
540  if (useGaussSeidelMultiplication && useDiffs) {
541  oldValues.resize(this->getRelevantValues().getNumberOfSetBits());
542  }
543  ValueType maxLowerDiff = storm::utility::zero<ValueType>();
544  ValueType maxUpperDiff = storm::utility::zero<ValueType>();
545  bool relative = env.solver().minMax().getRelativeTerminationCriterion();
546  ValueType precision = storm::utility::convertNumber<ValueType>(env.solver().minMax().getPrecision());
547  if (!relative) {
548  precision *= storm::utility::convertNumber<ValueType>(2.0);
549  }
550  this->startMeasureProgress();
551  while (status == SolverStatus::InProgress && iterations < env.solver().minMax().getMaximalNumberOfIterations()) {
552  // Remember in which directions we took steps in this iteration.
553  bool lowerStep = false;
554  bool upperStep = false;
555 
556  // In every thousandth iteration, we improve both bounds.
557  if (iterations % 1000 == 0 || maxLowerDiff == maxUpperDiff) {
558  lowerStep = true;
559  upperStep = true;
560  if (useGaussSeidelMultiplication) {
561  if (useDiffs) {
562  preserveOldRelevantValues(*lowerX, this->getRelevantValues(), oldValues);
563  }
564  this->multiplierA->multiplyAndReduceGaussSeidel(env, dir, *lowerX, &b);
565  if (useDiffs) {
566  maxLowerDiff = computeMaxAbsDiff(*lowerX, this->getRelevantValues(), oldValues);
567  preserveOldRelevantValues(*upperX, this->getRelevantValues(), oldValues);
568  }
569  this->multiplierA->multiplyAndReduceGaussSeidel(env, dir, *upperX, &b);
570  if (useDiffs) {
571  maxUpperDiff = computeMaxAbsDiff(*upperX, this->getRelevantValues(), oldValues);
572  }
573  } else {
574  this->multiplierA->multiplyAndReduce(env, dir, *lowerX, &b, *tmp);
575  if (useDiffs) {
576  maxLowerDiff = computeMaxAbsDiff(*lowerX, *tmp, this->getRelevantValues());
577  }
578  std::swap(lowerX, tmp);
579  this->multiplierA->multiplyAndReduce(env, dir, *upperX, &b, *tmp);
580  if (useDiffs) {
581  maxUpperDiff = computeMaxAbsDiff(*upperX, *tmp, this->getRelevantValues());
582  }
583  std::swap(upperX, tmp);
584  }
585  } else {
586  // In the following iterations, we improve the bound with the greatest difference.
587  if (useGaussSeidelMultiplication) {
588  if (maxLowerDiff >= maxUpperDiff) {
589  if (useDiffs) {
590  preserveOldRelevantValues(*lowerX, this->getRelevantValues(), oldValues);
591  }
592  this->multiplierA->multiplyAndReduceGaussSeidel(env, dir, *lowerX, &b);
593  if (useDiffs) {
594  maxLowerDiff = computeMaxAbsDiff(*lowerX, this->getRelevantValues(), oldValues);
595  }
596  lowerStep = true;
597  } else {
598  if (useDiffs) {
599  preserveOldRelevantValues(*upperX, this->getRelevantValues(), oldValues);
600  }
601  this->multiplierA->multiplyAndReduceGaussSeidel(env, dir, *upperX, &b);
602  if (useDiffs) {
603  maxUpperDiff = computeMaxAbsDiff(*upperX, this->getRelevantValues(), oldValues);
604  }
605  upperStep = true;
606  }
607  } else {
608  if (maxLowerDiff >= maxUpperDiff) {
609  this->multiplierA->multiplyAndReduce(env, dir, *lowerX, &b, *tmp);
610  if (useDiffs) {
611  maxLowerDiff = computeMaxAbsDiff(*lowerX, *tmp, this->getRelevantValues());
612  }
613  std::swap(tmp, lowerX);
614  lowerStep = true;
615  } else {
616  this->multiplierA->multiplyAndReduce(env, dir, *upperX, &b, *tmp);
617  if (useDiffs) {
618  maxUpperDiff = computeMaxAbsDiff(*upperX, *tmp, this->getRelevantValues());
619  }
620  std::swap(tmp, upperX);
621  upperStep = true;
622  }
623  }
624  }
625  STORM_LOG_ASSERT(maxLowerDiff >= storm::utility::zero<ValueType>(), "Expected non-negative lower diff.");
626  STORM_LOG_ASSERT(maxUpperDiff >= storm::utility::zero<ValueType>(), "Expected non-negative upper diff.");
627  if (iterations % 1000 == 0) {
628  STORM_LOG_TRACE("Iteration " << iterations << ": lower difference: " << maxLowerDiff << ", upper difference: " << maxUpperDiff << ".");
629  }
630 
631  if (doConvergenceCheck) {
632  // Determine whether the method converged.
633  if (this->hasRelevantValues()) {
634  status = storm::utility::vector::equalModuloPrecision<ValueType>(*lowerX, *upperX, this->getRelevantValues(), precision, relative) ? SolverStatus::Converged : status;
635  } else {
636  status = storm::utility::vector::equalModuloPrecision<ValueType>(*lowerX, *upperX, precision, relative) ? SolverStatus::Converged : status;
637  }
638  }
639 
640  // Update environment variables.
641  ++iterations;
642  doConvergenceCheck = !doConvergenceCheck;
643  if (lowerStep) {
644  status = this->updateStatus(status, *lowerX, SolverGuarantee::LessOrEqual, iterations, env.solver().minMax().getMaximalNumberOfIterations());
645  }
646  if (upperStep) {
647  status = this->updateStatus(status, *upperX, SolverGuarantee::GreaterOrEqual, iterations, env.solver().minMax().getMaximalNumberOfIterations());
648  }
649 
650  // Potentially show progress.
651  this->showProgressIterative(iterations);
652  }
653 
654  this->reportStatus(status, iterations);
655 
656  // We take the means of the lower and upper bound so we guarantee the desired precision.
657  ValueType two = storm::utility::convertNumber<ValueType>(2.0);
658  storm::utility::vector::applyPointwise<ValueType, ValueType, ValueType>(*lowerX, *upperX, *lowerX, [&two] (ValueType const& a, ValueType const& b) -> ValueType { return (a + b) / two; });
659 
660  // Since we shuffled the pointer around, we need to write the actual results to the input/output vector x.
661  if (&x == tmp) {
662  std::swap(x, *tmp);
663  } else if (&x == this->auxiliaryRowGroupVector.get()) {
664  std::swap(x, *this->auxiliaryRowGroupVector);
665  }
666 
667  // If requested, we store the scheduler for retrieval.
668  if (this->isTrackSchedulerSet()) {
669  this->schedulerChoices = std::vector<uint_fast64_t>(this->A->getRowGroupCount());
670  this->multiplierA->multiplyAndReduce(env, dir, x, &b, *this->auxiliaryRowGroupVector, &this->schedulerChoices.get());
671  }
672 
673  if (!this->isCachingEnabled()) {
674  clearCache();
675  }
676 
677  return status == SolverStatus::Converged;
678  }
