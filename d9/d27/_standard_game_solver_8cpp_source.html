417  bool StandardGameSolver<ValueType>::extractChoices(Environment const& env, OptimizationDirection player1Dir, OptimizationDirection player2Dir, std::vector<ValueType> const& x, std::vector<ValueType> const& b, std::vector<ValueType>& player2ChoiceValues, std::vector<uint_fast64_t>& player1Choices, std::vector<uint_fast64_t>& player2Choices) const {

447  if (valueImproved(player2Dir, comparator, *currentValueIt, choiceValue)) {
448  schedulerImproved = true;
449  player2Choices[p2Group] = p2Choice;
450  *currentValueIt = std::move(choiceValue);
451  }
452  }
453  }
454 
455  // Now extract the choices of player 1.
456  if (this->player1RepresentedByMatrix()) {
457  // Player 1 represented by matrix.
458  for (uint_fast64_t p1Group = 0; p1Group < this->getPlayer1Matrix().getRowGroupCount(); ++p1Group) {
459  uint_fast64_t firstRowInGroup = this->getPlayer1Matrix().getRowGroupIndices()[p1Group];
460  uint_fast64_t rowGroupSize = this->getPlayer1Matrix().getRowGroupIndices()[p1Group + 1] - firstRowInGroup;
461  uint_fast64_t currentChoice = player1Choices[p1Group];
462  ValueType currentValue = player2ChoiceValues[this->getPlayer1Matrix().getRow(firstRowInGroup + currentChoice).begin()->getColumn()];
463  for (uint_fast64_t p1Choice = 0; p1Choice < rowGroupSize; ++p1Choice) {
464  // If the choice is the currently selected one, we can skip it.
465  if (p1Choice == currentChoice) {
466  continue;
467  }
468  ValueType const& choiceValue = player2ChoiceValues[this->getPlayer1Matrix().getRow(firstRowInGroup + p1Choice).begin()->getColumn()];
469  if (valueImproved(player1Dir, comparator, currentValue, choiceValue)) {
470  schedulerImproved = true;
471  player1Choices[p1Group] = p1Choice;
472  currentValue = choiceValue;
473  }
474  }
475  }
476  } else {
477  // Player 1 represented by grouping of player 2 states (vector).
478  for (uint64_t player1State = 0; player1State < this->getPlayer1Grouping().size() - 1; ++player1State) {
479  uint64_t currentChoice = player1Choices[player1State];
480  ValueType currentValue = player2ChoiceValues[this->getPlayer1Grouping()[player1State] + currentChoice];
481  uint64_t numberOfPlayer2Successors = this->getPlayer1Grouping()[player1State + 1] - this->getPlayer1Grouping()[player1State];
482  for (uint64_t player2State = 0; player2State < numberOfPlayer2Successors; ++player2State) {
483  // If the choice is the currently selected one, we can skip it.
484  if (currentChoice == player2State) {
485  continue;
486  }
487 
488  ValueType const& choiceValue = player2ChoiceValues[this->getPlayer1Grouping()[player1State] + player2State];
489  if (valueImproved(player1Dir, comparator, currentValue, choiceValue)) {
490  schedulerImproved = true;
491  player1Choices[player1State] = player2State;
492  currentValue = choiceValue;
493  }
494  }
495  }
496  }
497 
498  return schedulerImproved;
499  }

502  void StandardGameSolver<ValueType>::getInducedMatrixVector(std::vector<ValueType>&, std::vector<ValueType> const& b, std::vector<uint_fast64_t> const& player1Choices, std::vector<uint_fast64_t> const& player2Choices, storm::storage::SparseMatrix<ValueType>& inducedMatrix, std::vector<ValueType>& inducedVector) const {
503  // Get the rows of the player 2 matrix that are selected by the schedulers.
504  // Note that rows can be selected more than once and in an arbitrary order.
505  std::vector<storm::storage::sparse::state_type> selectedRows;
506  if (this->player1RepresentedByMatrix()) {
507  // Player 1 is represented by a matrix.
508  selectedRows.reserve(this->getPlayer1Matrix().getRowGroupCount());
509  uint_fast64_t player1State = 0;
510  for (auto const& player1Choice : player1Choices) {
511  auto const& player1Row = this->getPlayer1Matrix().getRow(player1State, player1Choice);
512  STORM_LOG_ASSERT(player1Row.getNumberOfEntries() == 1, "It is assumed that rows of player one have one entry, but this is not the case.");
513  uint_fast64_t player2State = player1Row.begin()->getColumn();
514  selectedRows.push_back(player2Matrix.getRowGroupIndices()[player2State] + player2Choices[player2State]);
515  ++player1State;
516  }
517  } else {
518  // Player 1 is represented by the grouping of player 2 states (vector).
519  selectedRows.reserve(this->player2Matrix.getRowGroupCount());
520  for (uint64_t player1State = 0; player1State < this->getPlayer1Grouping().size() - 1; ++player1State) {
521  uint64_t player2State = this->getPlayer1Grouping()[player1State] + player1Choices[player1State];
522  selectedRows.emplace_back(player2Matrix.getRowGroupIndices()[player2State] + player2Choices[player2State]);
523  }
524  }
525 
526  // Get the matrix and the vector induced by this selection and add entries on the diagonal in the process.
527  inducedMatrix = player2Matrix.selectRowsFromRowIndexSequence(selectedRows, true);
528  inducedVector.resize(inducedMatrix.getRowCount());
529  storm::utility::vector::selectVectorValues<ValueType>(inducedVector, selectedRows, b);
530  }
