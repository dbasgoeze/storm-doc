12  uint64_t nrNonMarkovian = 0;

14

16  for (size_t i = 0; i < dft.nrElements(); ++i) {
17  std::shared_ptr<storm::storage::DFTElement<double> const> element = dft.getElement(i);
18  if (element->isBasicElement()) {
19  auto be = std::static_pointer_cast<storm::storage::DFTBE<double> const>(element);
20  if (be->hasIngoingDependencies()) {
21  ++nrDepEvents;
22  }
23  }
24  }
25 
26  // Only need to check as long as bound candidate + nr of non-Markovians to check is smaller than number of dependent events
27  while (nrNonMarkovian <= nrDepEvents && boundCandidate >= 0) {
28  if (nrNonMarkovian == 0 and boundCandidate == 0) {
29  nrNonMarkovian = 1;
30  }
31  STORM_LOG_TRACE(
32  "Lower bound correction - check possible bound " << std::to_string(boundCandidate)
33  << " with "
34  << std::to_string(nrNonMarkovian)
35  << " non-Markovian states");
36  // The uniqueness transformation for constantly failed BEs guarantees that a DFT never fails
37  // in step 0 without intermediate non-Markovians, thus forcibly set nrNonMarkovian
38  smtchecker->setSolverTimeout(timeout * 1000);
39  storm::solver::SmtSolver::CheckResult tmp_res =
40  smtchecker->checkFailsLeqWithEqNonMarkovianState(boundCandidate + nrNonMarkovian,
41  nrNonMarkovian);
42  smtchecker->unsetSolverTimeout();
43  switch (tmp_res) {
44  case storm::solver::SmtSolver::CheckResult::Sat:
45  /* If SAT, there is a sequence where only boundCandidate-many BEs fail directly and rest is nonMarkovian.
46  * Bound candidate is vaild, therefore check the next one */
47  STORM_LOG_TRACE("Lower bound correction - SAT");
48  // Prevent integer underflow
49  if (boundCandidate == 0) {
50  STORM_LOG_DEBUG("Lower bound correction - corrected bound to 0");
51  return 0;
52  }
53  --boundCandidate;
54  break;
55  case storm::solver::SmtSolver::CheckResult::Unknown:
56  // If any query returns unknown, we cannot be sure about the bound and fall back to the naive one
57  STORM_LOG_DEBUG("Lower bound correction - Solver returned 'Unknown', corrected to 1");
58  return 1;
59  default:
60  // if query is UNSAT, increase number of non-Markovian states and try again
61  STORM_LOG_TRACE("Lower bound correction - UNSAT");
62  ++nrNonMarkovian;
63  break;
64  }
65  }
66  // if for one candidate all queries are UNSAT, it is not valid. Return last valid candidate
67  STORM_LOG_DEBUG("Lower bound correction - corrected bound to " << std::to_string(boundCandidate + 1));
