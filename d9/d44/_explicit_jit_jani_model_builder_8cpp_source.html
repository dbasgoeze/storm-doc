805  template <typename ValueType, typename RewardModelType>
806  void ExplicitJitJaniModelBuilder<ValueType, RewardModelType>::generateInitialStates(cpptempl::data_map& modelData) {
807  cpptempl::data_list initialStates;
808 
809  // Prepare an SMT solver to enumerate all initial states.
810  storm::utility::solver::SmtSolverFactory factory;
811  std::unique_ptr<storm::solver::SmtSolver> solver = factory.create(model.getExpressionManager());
812 
813  std::vector<storm::expressions::Expression> rangeExpressions = model.getAllRangeExpressions();
814  for (auto const& expression : rangeExpressions) {
815  solver->add(expression);
816  }
817  solver->add(model.getInitialStatesExpression(parallelAutomata));
818 
819  // Proceed as long as the solver can still enumerate initial states.
820  while (solver->check() == storm::solver::SmtSolver::CheckResult::Sat) {
821  // Create fresh state.
822  cpptempl::data_list initialStateAssignment;
823 
824  // Read variable assignment from the solution of the solver. Also, create an expression we can use to
825  // prevent the variable assignment from being enumerated again.
826  storm::expressions::Expression blockingExpression;
827  std::shared_ptr<storm::solver::SmtSolver::ModelReference> model = solver->getModel();
828  for (auto const& variable : this->model.getGlobalVariables().getBooleanVariables()) {
829  if (variable.isTransient()) {
830  continue;
831  }
832 
833  storm::expressions::Variable const& expressionVariable = variable.getExpressionVariable();
834  bool variableValue = model->getBooleanValue(expressionVariable);
835  initialStateAssignment.push_back(generateAssignment(variable, variableValue));
836 
837  storm::expressions::Expression localBlockingExpression = variableValue ? !expressionVariable : expressionVariable;
838  blockingExpression = blockingExpression.isInitialized() ? blockingExpression || localBlockingExpression : localBlockingExpression;
839  }
840  for (auto const& variable : this->model.getGlobalVariables().getBoundedIntegerVariables()) {
841  if (variable.isTransient()) {
842  continue;
843  }
844 
845  storm::expressions::Variable const& expressionVariable = variable.getExpressionVariable();
846  int_fast64_t variableValue = model->getIntegerValue(expressionVariable);
847  initialStateAssignment.push_back(generateAssignment(variable, variableValue));
848 
849  storm::expressions::Expression localBlockingExpression = expressionVariable != model->getManager().integer(variableValue);
850  blockingExpression = blockingExpression.isInitialized() ? blockingExpression || localBlockingExpression : localBlockingExpression;
851  }
852  for (auto const& automatonRef : parallelAutomata) {
853  storm::jani::Automaton const& automaton = automatonRef.get();
854  for (auto const& variable : automaton.getVariables().getBooleanVariables()) {
855  if (variable.isTransient()) {
856  continue;
857  }
858 
859  storm::expressions::Variable const& expressionVariable = variable.getExpressionVariable();
860  bool variableValue = model->getBooleanValue(expressionVariable);
861  initialStateAssignment.push_back(generateAssignment(variable, variableValue));
862 
863  storm::expressions::Expression localBlockingExpression = variableValue ? !expressionVariable : expressionVariable;
864  blockingExpression = blockingExpression.isInitialized() ? blockingExpression || localBlockingExpression : localBlockingExpression;
865  }
866  for (auto const& variable : automaton.getVariables().getBoundedIntegerVariables()) {
867  if (variable.isTransient()) {
868  continue;
869  }
870 
871  storm::expressions::Variable const& expressionVariable = variable.getExpressionVariable();
872  int_fast64_t variableValue = model->getIntegerValue(expressionVariable);
873  initialStateAssignment.push_back(generateAssignment(variable, variableValue));
874 
875  storm::expressions::Expression localBlockingExpression = expressionVariable != model->getManager().integer(variableValue);
876  blockingExpression = blockingExpression.isInitialized() ? blockingExpression || localBlockingExpression : localBlockingExpression;
877  }
878  }
879 
880  // Gather iterators to the initial locations of all the automata.
881  std::vector<std::set<uint64_t>::const_iterator> initialLocationsIterators;
882  for (auto const& automaton : parallelAutomata) {
883  initialLocationsIterators.push_back(automaton.get().getInitialLocationIndices().cbegin());
884  }
885 
886  // Now iterate through all combinations of initial locations.
887  while (true) {
888  cpptempl::data_list completeAssignment(initialStateAssignment);
889 
890  for (uint64_t index = 0; index < initialLocationsIterators.size(); ++index) {
891  storm::jani::Automaton const& automaton = parallelAutomata[index].get();
892  if (automaton.getNumberOfLocations() > 1) {
893  completeAssignment.push_back(generateLocationAssignment(automaton, *initialLocationsIterators[index]));
894  }
895  }
896  initialStates.push_back(cpptempl::make_data(completeAssignment));
897 
898  uint64_t index = 0;
899  for (; index < initialLocationsIterators.size(); ++index) {
900  ++initialLocationsIterators[index];
901  if (initialLocationsIterators[index] == parallelAutomata[index].get().getInitialLocationIndices().cend()) {
902  initialLocationsIterators[index] = parallelAutomata[index].get().getInitialLocationIndices().cbegin();
903  } else {
904  break;
905  }
906  }
907 
908  // If we are at the end, leave the loop.
909  if (index == initialLocationsIterators.size()) {
910  break;
911  }
912  }
913 
914  // Block the current initial state to search for the next one.
915  if (!blockingExpression.isInitialized()) {
916  break;
917  }
918  solver->add(blockingExpression);
919  }
920 
921  modelData["initialStates"] = cpptempl::make_data(initialStates);
922  }

924  template <typename ValueType, typename RewardModelType>
925  void ExplicitJitJaniModelBuilder<ValueType, RewardModelType>::generateRewards(cpptempl::data_map& modelData) {
926  // Extract the reward models from the program based on the names we were given.
927  std::vector<storm::expressions::Variable> rewardVariables;
928  if (this->options.isBuildAllRewardModelsSet()) {
929  for (auto const& rewExpr : model.getAllRewardModelExpressions()) {
930  STORM_LOG_ERROR_COND(rewExpr.second.isVariable(), "The jit builder can not build the non-trivial reward expression '" << rewExpr.second << "'.");
931  rewardVariables.push_back(rewExpr.second.getBaseExpression().asVariableExpression().getVariable());
932  }
933  } else {
934  for (auto const& rewardModelName : this->options.getRewardModelNames()) {
935  auto const& rewExpr = model.getRewardModelExpression(rewardModelName);
936  STORM_LOG_ERROR_COND(rewExpr.isVariable(), "The jit builder can not build the non-trivial reward expression '" << rewExpr << "'.");
937  rewardVariables.push_back(rewExpr.getBaseExpression().asVariableExpression().getVariable());
938  }
939  }
940  // Sort the reward variables to match the order in the ordered assignments
941  std::sort(rewardVariables.begin(), rewardVariables.end());
942 
943  std::vector<storm::builder::RewardModelInformation> rewardModels;
944  cpptempl::data_list rewards;
945  cpptempl::data_list locationRewards;
946  cpptempl::data_list edgeRewards;
947  cpptempl::data_list destinationRewards;
948  uint64_t rewardModelIndex = 0;
949  uint64_t stateActionRewardCount = 0;
950  for (auto const& variable : rewardVariables) {
951  bool hasLocationRewards = false;
952  bool hasEdgeRewards = false;
953  bool hasDestinationRewards = false;
954 
955  for (auto const& automatonRef : parallelAutomata) {
956  storm::jani::Automaton const& automaton = automatonRef.get();
957 
958  for (auto const& location : automaton.getLocations()) {
959  for (auto const& assignment : location.getAssignments()) {
960  if (assignment.getExpressionVariable() == variable) {
961  hasLocationRewards = true;
962  break;
963  }
964  }
965  }
966 
967  for (auto const& edge : automaton.getEdges()) {
968  for (auto const& assignment : edge.getAssignments()) {
969  if (assignment.getExpressionVariable() == variable) {
970  hasEdgeRewards = true;
971  }
972  }
973 
974  for (auto const& destination : edge.getDestinations()) {
975  for (auto const& assignment : destination.getOrderedAssignments()) {
976  if (assignment.getExpressionVariable() == variable) {
977  hasDestinationRewards = true;
978  }
979  }
980  }
981  }
982  }
983 
984  rewardModels.emplace_back(variable.getName(), hasLocationRewards, hasEdgeRewards || hasDestinationRewards, false);
985 
986  if (hasEdgeRewards || hasDestinationRewards) {
987  ++stateActionRewardCount;
988  }
989 
990  cpptempl::data_map reward;
991  reward["name"] = variable.getName();
992  reward["location_rewards"] = asString(hasLocationRewards);
993  reward["edge_rewards"] = asString(hasEdgeRewards);
994  reward["destination_rewards"] = asString(hasDestinationRewards);
995  rewards.push_back(reward);
996 
997  if (hasLocationRewards) {
998  cpptempl::data_map locationReward;
999  locationReward["variable"] = variable.getName() + JIT_VARIABLE_EXTENSION;
1000  locationRewards.push_back(locationReward);
1001  }
1002  if (hasEdgeRewards) {
1003  cpptempl::data_map edgeReward;
1004  edgeReward["variable"] = variable.getName() + JIT_VARIABLE_EXTENSION;
1005  edgeReward["index"] = asString(rewardModelIndex);
1006  edgeRewards.push_back(edgeReward);
1007  }
1008  if (hasDestinationRewards) {
1009  cpptempl::data_map destinationReward;
1010  destinationReward["index"] = asString(rewardModelIndex);
1011  destinationReward["variable"] = variable.getName() + JIT_VARIABLE_EXTENSION;
1012  destinationRewards.push_back(destinationReward);
1013  }
1014  ++rewardModelIndex;
1015  }
1016 
1017  modelData["location_rewards"] = cpptempl::make_data(locationRewards);
1018  modelData["edge_rewards"] = cpptempl::make_data(edgeRewards);
1019  modelData["destination_rewards"] = cpptempl::make_data(destinationRewards);
1020  modelData["rewards"] = cpptempl::make_data(rewards);
1021  modelData["edge_destination_rewards_count"] = asString(stateActionRewardCount);
1022  }

1070  }
1071  }
