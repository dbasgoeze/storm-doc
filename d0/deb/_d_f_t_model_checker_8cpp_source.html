67

75  nrM = dfts.size();
76  break;
77  case storm::storage::DFTElementType::OR:
78  STORM_LOG_TRACE("top modularisation called OR");
79  dfts = dft.topModularisation();
80  nrK = 0;
81  nrM = dfts.size();
82  invResults = true;
83  break;
84  case storm::storage::DFTElementType::VOT:
85  STORM_LOG_TRACE("top modularisation called VOT");
86  dfts = dft.topModularisation();
87  nrK = std::static_pointer_cast<storm::storage::DFTVot<ValueType> const>(
88  dft.getTopLevelGate())->threshold();
89  nrM = dfts.size();
90  if (nrK <= nrM / 2) {
91  nrK -= 1;
92  invResults = true;
93  }
94  break;
95  default:
96  // No static gate -> no modularisation applicable
97  break;
98  }
99  }
100 
101  // Perform modularisation
102  if (dfts.size() > 1) {
103  STORM_LOG_DEBUG("Modularisation of " << dft.getTopLevelGate()->name() << " into " << dfts.size() << " submodules.");
104  // TODO: compute simultaneously
105  dft_results results;
106  for (auto property : properties) {
107  if (!property->isProbabilityOperatorFormula()) {
108  STORM_LOG_WARN("Could not check property: " << *property);
109  } else {
110  // Recursively call model checking
111  std::vector<ValueType> res;
112  for (auto const ft : dfts) {
113  // TODO: allow approximation in modularisation
114  dft_results ftResults = checkHelper(ft, {property}, symred, true, relevantEvents, 0.0);
115  STORM_LOG_ASSERT(ftResults.size() == 1, "Wrong number of results");
116  res.push_back(boost::get<ValueType>(ftResults[0]));
117  }
118

121  << (invResults ? "On" : "Off"));
122  ValueType result = storm::utility::zero<ValueType>();
123  int limK = invResults ? -1 : nrM + 1;
124  int chK = invResults ? -1 : 1;
125  for (int cK = nrK; cK != limK; cK += chK) {
126  STORM_LOG_ASSERT(cK >= 0, "ck negative.");
127  uint64_t permutation = smallestIntWithNBitsSet(static_cast<uint64_t>(cK));
128  do {
129  STORM_LOG_TRACE("Permutation=" << permutation);
130  ValueType permResult = storm::utility::one<ValueType>();
131  for (size_t i = 0; i < res.size(); ++i) {
132  if (permutation & (1ul << i)) {
133  permResult *= res[i];
134  } else {
135  permResult *= storm::utility::one<ValueType>() - res[i];
136  }
137  }
138  STORM_LOG_TRACE("Result for permutation:" << permResult);
139  permutation = nextBitPermutation(permutation);
140  result += permResult;
141  } while (permutation < (1ul << nrM) && permutation != 0);
142  }
143  if (invResults) {
144  result = storm::utility::one<ValueType>() - result;
145  }
146  results.push_back(result);
147  }
148  }
149  return results;
150  } else {
151  // No modularisation was possible
152  return checkDFT(dft, properties, symred, relevantEvents, approximationError, approximationHeuristic, eliminateChains, labelBehavior);
153  }
