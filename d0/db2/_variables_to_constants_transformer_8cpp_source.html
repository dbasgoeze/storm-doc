27  // It does not make sense to do this for transient variables.

41  void VariablesToConstantsTransformer::transform(Model& model) {
42  // It is dangerous to erase a variable from a set of variables while iterating over the set.

44  std::vector<storm::expressions::Variable> variablesToErase;
45  for (auto const& globalVar : model.getGlobalVariables()) {
46  if (detail::canTransformVariable(globalVar, model)) {
