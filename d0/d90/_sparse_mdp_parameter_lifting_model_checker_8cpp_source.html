243  std::unique_ptr<CheckResult> SparseMdpParameterLiftingModelChecker<SparseModelType, ConstantType>::computeQuantitativeValues(Environment const& env, storm::storage::ParameterRegion<typename SparseModelType::ValueType> const& region, storm::solver::OptimizationDirection const& dirForParameters) {
244 
245  if (maybeStates.empty()) {
246  return std::make_unique<storm::modelchecker::ExplicitQuantitativeCheckResult<ConstantType>>(resultsForNonMaybeStates);
247  }
248 
249  parameterLifter->specifyRegion(region, dirForParameters);
250 
251  // Set up the solver
252  auto solver = solverFactory->create(env, player1Matrix, parameterLifter->getMatrix());
253  if (lowerResultBound) solver->setLowerBound(lowerResultBound.get());
254  if (upperResultBound) solver->setUpperBound(upperResultBound.get());
255  if (applyPreviousResultAsHint) {
256  solver->setTrackSchedulers(true);
257  x.resize(maybeStates.getNumberOfSetBits(), storm::utility::zero<ConstantType>());

259  if(storm::solver::maximize(dirForParameters) && maxSchedChoices && player1SchedChoices) solver->setSchedulerHints(std::move(player1SchedChoices.get()), std::move(maxSchedChoices.get()));
260  } else {
261  x.assign(maybeStates.getNumberOfSetBits(), storm::utility::zero<ConstantType>());

265  std::unique_ptr<storm::solver::TerminationCondition<ConstantType>> termCond;
266  storm::storage::BitVector relevantStatesInSubsystem = this->currentCheckTask->isOnlyInitialStatesRelevantSet() ? this->parametricModel->getInitialStates() % maybeStates : storm::storage::BitVector(maybeStates.getNumberOfSetBits(), true);
267  if (storm::solver::minimize(dirForParameters)) {
268  // Terminate if the value for ALL relevant states is already below the threshold
269  termCond = std::make_unique<storm::solver::TerminateIfFilteredExtremumBelowThreshold<ConstantType>> (relevantStatesInSubsystem, true, this->currentCheckTask->getBoundThreshold(), false);
270  } else {
271  // Terminate if the value for ALL relevant states is already above the threshold
272  termCond = std::make_unique<storm::solver::TerminateIfFilteredExtremumExceedsThreshold<ConstantType>> (relevantStatesInSubsystem, true, this->currentCheckTask->getBoundThreshold(), true);
273  }
274  solver->setTerminationCondition(std::move(termCond));
275  }
276

278  if (stepBound) {
279  STORM_LOG_ASSERT(*stepBound > 0, "Expected positive step bound.");
280  solver->repeatedMultiply(env, this->currentCheckTask->getOptimizationDirection(), dirForParameters, x, &parameterLifter->getVector(), *stepBound);
281  } else {
282  solver->solveGame(env, this->currentCheckTask->getOptimizationDirection(), dirForParameters, x, parameterLifter->getVector());

284  if(storm::solver::minimize(dirForParameters)) {
285  minSchedChoices = solver->getPlayer2SchedulerChoices();
286  } else {
287  maxSchedChoices = solver->getPlayer2SchedulerChoices();
288  }
289  player1SchedChoices = solver->getPlayer1SchedulerChoices();
290  }
291  }
292 
293  // Get the result for the complete model (including maybestates)
294  std::vector<ConstantType> result = resultsForNonMaybeStates;
295  auto maybeStateResIt = x.begin();
296  for (auto const& maybeState : maybeStates) {
297  result[maybeState] = *maybeStateResIt;
298  ++maybeStateResIt;
299  }

304  template <typename SparseModelType, typename ConstantType>
305  void SparseMdpParameterLiftingModelChecker<SparseModelType, ConstantType>::computePlayer1Matrix(boost::optional<storm::storage::BitVector> const& selectedRows) {
306  uint_fast64_t n = 0;
307  if (selectedRows) {
308  // only count selected rows
309  n = selectedRows->getNumberOfSetBits();
310  } else {
311  for (auto const& maybeState : maybeStates) {
312  n += this->parametricModel->getTransitionMatrix().getRowGroupSize(maybeState);
313  }
314  }
315 
316  // The player 1 matrix is the identity matrix of size n with the row groups as given by the original matrix (potentially without unselected rows)
317  storm::storage::SparseMatrixBuilder<storm::storage::sparse::state_type> matrixBuilder(n, n, n, true, true, maybeStates.getNumberOfSetBits());
318  uint_fast64_t p1MatrixRow = 0;
319  for (auto maybeState : maybeStates){
320  matrixBuilder.newRowGroup(p1MatrixRow);
321  if (selectedRows) {
322  for (uint_fast64_t row = selectedRows->getNextSetIndex(this->parametricModel->getTransitionMatrix().getRowGroupIndices()[maybeState]); row < this->parametricModel->getTransitionMatrix().getRowGroupIndices()[maybeState + 1]; row = selectedRows->getNextSetIndex(row + 1)) {
323  matrixBuilder.addNextValue(p1MatrixRow, p1MatrixRow, storm::utility::one<storm::storage::sparse::state_type>());
324  ++p1MatrixRow;
325  }
326  } else {
327  for (uint_fast64_t endOfGroup = p1MatrixRow + this->parametricModel->getTransitionMatrix().getRowGroupSize(maybeState); p1MatrixRow < endOfGroup; ++p1MatrixRow){
328  matrixBuilder.addNextValue(p1MatrixRow, p1MatrixRow, storm::utility::one<storm::storage::sparse::state_type>());
329  }
330  }
331  }
332  player1Matrix = matrixBuilder.build();
333  }
