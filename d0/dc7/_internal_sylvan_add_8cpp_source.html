839  Odd InternalAdd<DdType::Sylvan, ValueType>::createOdd(std::vector<uint_fast64_t> const& ddVariableIndices) const {

850  template<typename ValueType>
851  std::shared_ptr<Odd> InternalAdd<DdType::Sylvan, ValueType>::createOddRec(BDD dd, uint_fast64_t currentLevel, uint_fast64_t maxLevel, std::vector<uint_fast64_t> const& ddVariableIndices, std::vector<std::unordered_map<BDD, std::shared_ptr<Odd>>>& uniqueTableForLevels) {
852  // Check whether the ODD for this node has already been computed (for this level) and if so, return this instead.
853  auto const& iterator = uniqueTableForLevels[currentLevel].find(dd);
854  if (iterator != uniqueTableForLevels[currentLevel].end()) {
855  return iterator->second;
856  } else {
857  // Otherwise, we need to recursively compute the ODD.
858 
859  // If we are already past the maximal level that is to be considered, we can simply create an Odd without
860  // successors
861  if (currentLevel == maxLevel) {
862  uint_fast64_t elseOffset = 0;
863  uint_fast64_t thenOffset = 0;
864 
865  STORM_LOG_ASSERT(mtbdd_isleaf(dd), "Expected leaf at last level.");
866 
867  // If the DD is not the zero leaf, then the then-offset is 1.
868  if (!mtbdd_iszero(dd)) {
869  thenOffset = 1;
870  }
871 
872  auto oddNode = std::make_shared<Odd>(nullptr, elseOffset, nullptr, thenOffset);
873  uniqueTableForLevels[currentLevel].emplace(dd, oddNode);
874  return oddNode;
875  } else if (mtbdd_isleaf(dd) || ddVariableIndices[currentLevel] < mtbdd_getvar(dd)) {
876  // If we skipped the level in the DD, we compute the ODD just for the else-successor and use the same
877  // node for the then-successor as well.
878  std::shared_ptr<Odd> elseNode = createOddRec(dd, currentLevel + 1, maxLevel, ddVariableIndices, uniqueTableForLevels);
879  std::shared_ptr<Odd> thenNode = elseNode;
880  auto oddNode = std::make_shared<Odd>(elseNode, elseNode->getElseOffset() + elseNode->getThenOffset(), thenNode, thenNode->getElseOffset() + thenNode->getThenOffset());
