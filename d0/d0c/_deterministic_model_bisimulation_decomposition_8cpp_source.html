41  void DeterministicModelBisimulationDecomposition<ModelType>::splitOffDivergentStates() {
42  std::vector<storm::storage::sparse::state_type> stateStack;
43  stateStack.reserve(this->model.getNumberOfStates());
44  storm::storage::BitVector nondivergentStates(this->model.getNumberOfStates());
45 
46  uint_fast64_t currentSize = this->partition.size();
47  for (uint_fast64_t blockIndex = 0; blockIndex < currentSize; ++blockIndex) {
48  auto& block = *this->partition.getBlocks()[blockIndex];
49  nondivergentStates.clear();
50 
51  for (auto stateIt = this->partition.begin(block), stateIte = this->partition.end(block); stateIt != stateIte; ++stateIt) {
52  if (nondivergentStates.get(*stateIt)) {
53  continue;
54  }
55 
56  // Now traverse the forward transitions of the current state and check whether there is a
57  // transition to some other block.
58  bool isDirectlyNonDivergent = false;
59  for (auto const& successor : this->model.getRows(*stateIt)) {
60  // If there is such a transition, then we can mark all states in the current block that can
61  // reach the state as non-divergent.
62  if (this->partition.getBlock(successor.getColumn()) != block) {
63  isDirectlyNonDivergent = true;
64  break;
65  }
66  }
67 
68  if (isDirectlyNonDivergent) {
69  stateStack.push_back(*stateIt);
70 
71  while (!stateStack.empty()) {
72  storm::storage::sparse::state_type currentState = stateStack.back();
73  stateStack.pop_back();
74  nondivergentStates.set(currentState);
75 
76  for (auto const& predecessor : this->backwardTransitions.getRow(currentState)) {
77  if (this->partition.getBlock(predecessor.getColumn()) == block && !nondivergentStates.get(predecessor.getColumn())) {
78  stateStack.push_back(predecessor.getColumn());
79  }
80  }
81  }
82  }
83  }
84 
85  if (!nondivergentStates.empty() && nondivergentStates.getNumberOfSetBits() != block.getNumberOfStates()) {
86  // After performing the split, the current block will contain the divergent states only.
87  this->partition.splitStates(block, nondivergentStates);
88 
89  // Since the remaining states in the block are divergent, we can mark the block as absorbing.
90  // This also guarantees that the self-loop will be added to the state of the quotient
91  // representing this block of states.
92  block.data().setAbsorbing(true);
93  } else if (nondivergentStates.empty()) {
94  // If there are only diverging states in the block, we need to make it absorbing.
95  block.data().setAbsorbing(true);
96  }
97  }
98  }

258  template <typename ModelType>
259  void DeterministicModelBisimulationDecomposition<ModelType>::moveStateInSplitter(storm::storage::sparse::state_type predecessor, bisimulation::Block<BlockDataType>& predecessorBlock, storm::storage::sparse::state_type currentPositionInSplitter, uint_fast64_t& elementsToSkip) {
260  storm::storage::sparse::state_type predecessorPosition = this->partition.getPosition(predecessor);
261 
262  // If the predecessors of the given predecessor were already explored, we can move it easily.
263  if (predecessorPosition <= currentPositionInSplitter + elementsToSkip) {
264  this->partition.swapStates(predecessor, this->partition.getState(predecessorBlock.data().marker1()));
265  predecessorBlock.data().incrementMarker1();
266  } else {
267  // Otherwise, we need to move the predecessor, but we need to make sure that we explore its
268  // predecessors later. We do this by moving it to a range at the beginning of the block that will hold
269  // all predecessors in the splitter whose predecessors have yet to be explored.
270  if (predecessorBlock.data().marker2() == predecessorBlock.data().marker1()) {
271  this->partition.swapStatesAtPositions(predecessorBlock.data().marker2(), predecessorPosition);
272  this->partition.swapStatesAtPositions(predecessorPosition, currentPositionInSplitter + elementsToSkip + 1);
273  } else {
274  this->partition.swapStatesAtPositions(predecessorBlock.data().marker2(), predecessorPosition);
275  this->partition.swapStatesAtPositions(predecessorPosition, predecessorBlock.data().marker1());
276  this->partition.swapStatesAtPositions(predecessorPosition, currentPositionInSplitter + elementsToSkip + 1);
277  }
278 
279  // Since we had to move an already explored state to the right of the current position,
280  ++elementsToSkip;
281  predecessorBlock.data().incrementMarker1();
282  predecessorBlock.data().incrementMarker2();
283  }
284  }

286  template <typename ModelType>
287  void DeterministicModelBisimulationDecomposition<ModelType>::exploreRemainingStatesOfSplitter(bisimulation::Block<BlockDataType>& splitter, std::list<bisimulation::Block<BlockDataType>*>& predecessorBlocks) {
288  for (auto splitterIt = this->partition.begin(splitter), splitterIte = this->partition.begin(splitter) + (splitter.data().marker2() - splitter.getBeginIndex()); splitterIt != splitterIte; ++splitterIt) {
289  storm::storage::sparse::state_type currentState = *splitterIt;
290 
291  for (auto const& predecessorEntry : this->backwardTransitions.getRow(currentState)) {
292  storm::storage::sparse::state_type predecessor = predecessorEntry.getColumn();
293  Block<BlockDataType>& predecessorBlock = this->partition.getBlock(predecessor);
294 
295  // If the block does not need to be refined, we skip it.
296  if (!possiblyNeedsRefinement(predecessorBlock)) {
297  continue;
298  }
299 
300  // If we are computing a weak bisimulation on CTMCs and the predecessor block is the splitter, we
301  // need to ignore it and proceed to the next predecessor.
302  if (this->options.getType() == BisimulationType::Weak && this->model.getType() == storm::models::ModelType::Ctmc && predecessorBlock == splitter) {
303  continue;
304  }
305 
306  // We keep track of the probability of the predecessor moving to the splitter.
307  increaseProbabilityToSplitter(predecessor, predecessorBlock, predecessorEntry.getValue());
308 
309  // Only move the state if it has not been seen as a predecessor before.
310  storm::storage::sparse::state_type predecessorPosition = this->partition.getPosition(predecessor);
311  if (predecessorPosition >= predecessorBlock.data().marker1()) {
312  moveStateToMarker1(predecessor, predecessorBlock);
313  }
314 
315  // We must not insert the the splitter itself if we are not computing a weak bisimulation on CTMCs.
316  if (this->options.getType() != BisimulationType::Weak || this->model.getType() != storm::models::ModelType::Ctmc || predecessorBlock != splitter) {
317  insertIntoPredecessorList(predecessorBlock, predecessorBlocks);
318  }
319  }
320  }
321 
322  // Finally, we can reset the second marker.
323  splitter.data().setMarker2(splitter.getBeginIndex());
324  }

476  void DeterministicModelBisimulationDecomposition<ModelType>::refinePartitionBasedOnSplitter(bisimulation::Block<BlockDataType>& splitter, std::vector<bisimulation::Block<BlockDataType>*>& splitterQueue) {
477  STORM_LOG_TRACE("Refining partition based on splitter " << splitter.getId());
478 
479  // The outline of the refinement is as follows.
480  //
481  // We iterate over all states of the splitter and determine for each predecessor the state the probability
482  // entering the splitter. These probabilities are written to a member vector so that after the iteration
483  // process we have the probabilities of all predecessors of the splitter of entering the splitter in one
484  // step. To directly separate the states having a transition into the splitter from the ones who do not,
485  // we move the states to certain locations. That is, on encountering a predecessor of the splitter, it is
486  // moved to the beginning of its block. If the predecessor is in the splitter itself, we have to be a bit
487  // careful about where to move states.
488  //
489  // After this iteration, there may be states of the splitter whose predecessors have not yet been explored,
490  // so this needs to be done now.
491  //
492  // Finally, we use the information obtained in the first part for the actual splitting process in which all
493  // predecessor blocks of the splitter are split based on the probabilities computed earlier.
494  std::list<Block<BlockDataType>*> predecessorBlocks;
495  storm::storage::sparse::state_type currentPosition = splitter.getBeginIndex();
496  bool splitterIsPredecessorBlock = false;
497  for (auto splitterIt = this->partition.begin(splitter), splitterIte = this->partition.end(splitter); splitterIt != splitterIte; ++splitterIt, ++currentPosition) {
498  storm::storage::sparse::state_type currentState = *splitterIt;
499 
500  uint_fast64_t elementsToSkip = 0;
501  for (auto const& predecessorEntry : this->backwardTransitions.getRow(currentState)) {
502  storm::storage::sparse::state_type predecessor = predecessorEntry.getColumn();
503  storm::storage::sparse::state_type predecessorPosition = this->partition.getPosition(predecessor);
504  Block<BlockDataType>& predecessorBlock = this->partition.getBlock(predecessor);
505 
506  // If the block does not need to be refined, we skip it.
507  if (!possiblyNeedsRefinement(predecessorBlock)) {
508  continue;
509  }
510 
511  // If we are computing a weak bisimulation on CTMCs and the predecessor block is the splitter, we
512  // need to ignore it and proceed to the next predecessor.
513  if (this->options.getType() == BisimulationType::Weak && this->model.getType() == storm::models::ModelType::Ctmc && predecessorBlock == splitter) {
514  continue;
515  }
516 
517  // We keep track of the probability of the predecessor moving to the splitter.
518  increaseProbabilityToSplitter(predecessor, predecessorBlock, predecessorEntry.getValue());
519 
520  // We only need to move the predecessor if its not already known as a predecessor already.
521  if (predecessorPosition >= predecessorBlock.data().marker1()) {
522  // If the predecessor block is not the splitter, we can move the state easily.
523  if (predecessorBlock != splitter) {
524  moveStateToMarker1(predecessor, predecessorBlock);
525  } else {
526  // If the predecessor is in the splitter, we need to be a bit more careful.
527  splitterIsPredecessorBlock = true;
528  moveStateInSplitter(predecessor, predecessorBlock, currentPosition, elementsToSkip);
529  }
530 
531  insertIntoPredecessorList(predecessorBlock, predecessorBlocks);
532  }
533  }
534 
535  // If, as a consequence of shifting states, we need to skip some elements, do so now.
536  splitterIt += elementsToSkip;
537  currentPosition += elementsToSkip;
538  }
539 
540  // If the splitter was a predecessor block of itself, we potentially need to explore some states that have
541  // not been explored yet.
542  if (splitterIsPredecessorBlock) {
543  exploreRemainingStatesOfSplitter(splitter, predecessorBlocks);
544  }
545 
546  // Finally, we split the block based on the precomputed probabilities and the chosen bisimulation type.
547  if (this->options.getType() == BisimulationType::Strong || this->model.getType() == storm::models::ModelType::Ctmc) {
548  // In the case of CTMCs and weak bisimulation, we still call the "splitStrong" method, but we already have
549  // taken care of not adding the splitter to the predecessor blocks, so this is safe.
550  refinePredecessorBlocksOfSplitterStrong(predecessorBlocks, splitterQueue);
551  } else {
552  // If the splitter is a predecessor of we can use the computed probabilities to update the silent
553  // probabilities.
554  if (splitterIsPredecessorBlock) {
555  updateSilentProbabilitiesBasedOnProbabilitiesToSplitter(splitter);
556  }
557 
558  refinePredecessorBlocksOfSplitterWeak(splitter, predecessorBlocks, splitterQueue);
559  }
560  }

562  template<typename ModelType>
563  void DeterministicModelBisimulationDecomposition<ModelType>::buildQuotient() {
564  // In order to create the quotient model, we need to construct
565  // (a) the new transition matrix,
566  // (b) the new labeling,
567  // (c) the new reward structures.
568 
569  // Prepare a matrix builder for (a).
570  storm::storage::SparseMatrixBuilder<ValueType> builder(this->size(), this->size());
571 
572  // Prepare the new state labeling for (b).
573  storm::models::sparse::StateLabeling newLabeling(this->size());
574  std::set<std::string> atomicPropositionsSet = this->options.respectedAtomicPropositions.get();
575  atomicPropositionsSet.insert("init");
576  std::vector<std::string> atomicPropositions = std::vector<std::string>(atomicPropositionsSet.begin(), atomicPropositionsSet.end());
577  for (auto const& ap : atomicPropositions) {
578  newLabeling.addLabel(ap);
579  }
580 
581  // If the model had state rewards, we need to build the state rewards for the quotient as well.
582  boost::optional<std::vector<ValueType>> stateRewards;
583  if (this->options.getKeepRewards() && this->model.hasRewardModel()) {
584  stateRewards = std::vector<ValueType>(this->blocks.size());
585  }
586 
587  // Now build (a) and (b) by traversing all blocks.
588  for (uint_fast64_t blockIndex = 0; blockIndex < this->blocks.size(); ++blockIndex) {
589  auto const& block = this->blocks[blockIndex];
590 
591  // Pick one representative state. For strong bisimulation it doesn't matter which state it is, because
592  // they all behave equally.
593  storm::storage::sparse::state_type representativeState = *block.begin();
594 
595  // However, for weak bisimulation, we need to make sure the representative state is a non-silent one (if
596  // there is any such state).
597  if (this->options.getType() == BisimulationType::Weak && this->model.getType() == storm::models::ModelType::Dtmc) {
598  for (auto const& state : block) {
599  if (!isSilent(state)) {
600  representativeState = state;
601  break;
602  }
603  }
604  }
605 
606  Block<BlockDataType> const& oldBlock = this->partition.getBlock(representativeState);
607 
608  // If the block is absorbing, we simply add a self-loop.
609  if (oldBlock.data().absorbing()) {
610  builder.addNextValue(blockIndex, blockIndex, storm::utility::one<ValueType>());
611 
612  // If the block has a special representative state, we retrieve it now.
613  if (oldBlock.data().hasRepresentativeState()) {
614  representativeState = oldBlock.data().representativeState();
615  }
616 
617  // Add all of the selected atomic propositions that hold in the representative state to the state
618  // representing the block.
619  for (auto const& ap : atomicPropositions) {
620  if (this->model.getStateLabeling().getStateHasLabel(ap, representativeState)) {
621  newLabeling.addLabelToState(ap, blockIndex);
622  }
623  }
624  } else {
625  // Compute the outgoing transitions of the block.
626  std::map<storm::storage::sparse::state_type, ValueType> blockProbability;
627  for (auto const& entry : this->model.getTransitionMatrix().getRow(representativeState)) {
628  storm::storage::sparse::state_type targetBlock = this->partition.getBlock(entry.getColumn()).getId();
629 
630  // If we are computing a weak bisimulation quotient, there is no need to add self-loops.
631  if ((this->options.getType() == BisimulationType::Weak) && targetBlock == blockIndex && !oldBlock.data().hasRewards()) {
632  continue;
633  }
634 
635  auto probIterator = blockProbability.find(targetBlock);
636  if (probIterator != blockProbability.end()) {
637  probIterator->second += entry.getValue();
638  } else {
639  blockProbability[targetBlock] = entry.getValue();
640  }
641  }
642 
643  // Now add them to the actual matrix.
644  for (auto const& probabilityEntry : blockProbability) {
645  if (this->options.getType() == BisimulationType::Weak && this->model.getType() == storm::models::ModelType::Dtmc && !oldBlock.data().hasRewards()) {
646  builder.addNextValue(blockIndex, probabilityEntry.first, probabilityEntry.second / (storm::utility::one<ValueType>() - getSilentProbability(representativeState)));
647  } else {
648  builder.addNextValue(blockIndex, probabilityEntry.first, probabilityEntry.second);
649  }
650  }
651 
652  // Otherwise add all atomic propositions to the equivalence class that the representative state
653  // satisfies.
654  for (auto const& ap : atomicPropositions) {
655  if (this->model.getStateLabeling().getStateHasLabel(ap, representativeState)) {
656  newLabeling.addLabelToState(ap, blockIndex);
657  }
658  }
659  }
660 
661  // If the model has state rewards, we simply copy the state reward of the representative state, because
662  // all states in a block are guaranteed to have the same state reward.
663  if (this->options.getKeepRewards() && this->model.hasRewardModel()) {
664  auto const& rewardModel = this->model.getUniqueRewardModel();
665  if (rewardModel.hasStateRewards()) {
666  stateRewards.get()[blockIndex] = rewardModel.getStateRewardVector()[representativeState];
667  }
668  if (rewardModel.hasStateActionRewards()) {
669  stateRewards.get()[blockIndex] += rewardModel.getStateActionRewardVector()[representativeState];
670  }
671  }
672  }
673 
674  // Now check which of the blocks of the partition contain at least one initial state.
675  for (auto initialState : this->model.getInitialStates()) {
676  Block<BlockDataType> const& initialBlock = this->partition.getBlock(initialState);
677  newLabeling.addLabelToState("init", initialBlock.getId());
678  }
679 
680  // Construct the reward model mapping.
681  std::unordered_map<std::string, typename ModelType::RewardModelType> rewardModels;
682  if (this->options.getKeepRewards() && this->model.hasRewardModel()) {
683  STORM_LOG_THROW(this->model.hasUniqueRewardModel(), storm::exceptions::IllegalFunctionCallException, "Cannot preserve more than one reward model.");
684  typename std::unordered_map<std::string, typename ModelType::RewardModelType>::const_iterator nameRewardModelPair = this->model.getRewardModels().begin();
685  rewardModels.insert(std::make_pair(nameRewardModelPair->first, typename ModelType::RewardModelType(stateRewards)));
686  }
687 
688  // Finally construct the quotient model.
689  this->quotient = std::shared_ptr<ModelType>(new ModelType(builder.build(), std::move(newLabeling), std::move(rewardModels)));
690  }
