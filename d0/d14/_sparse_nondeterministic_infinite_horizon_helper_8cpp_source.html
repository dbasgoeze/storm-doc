109  template <typename ValueType>
110  std::pair<bool, ValueType> SparseNondeterministicInfiniteHorizonHelper<ValueType>::computeLraForTrivialMec(Environment const& env, ValueGetter const& stateRewardsGetter, ValueGetter const& actionRewardsGetter, storm::storage::MaximalEndComponent const& component) {
111 
112  // If the component only consists of a single state, we compute the LRA value directly
113  if (component.size() == 1) {
114  auto const& element = *component.begin();
115  uint64_t state = internal::getComponentElementState(element);
116  auto choiceIt = internal::getComponentElementChoicesBegin(element);
117  if (!this->isContinuousTime()) {
118  // This is an MDP.
119  // Find the choice with the highest/lowest reward
120  ValueType bestValue = actionRewardsGetter(*choiceIt);
121  uint64_t bestChoice = *choiceIt;
122  for (++choiceIt; choiceIt != internal::getComponentElementChoicesEnd(element); ++choiceIt) {
123  ValueType currentValue = actionRewardsGetter(*choiceIt);
124  if ((this->minimize() && currentValue < bestValue) || (this->maximize() && currentValue > bestValue)) {
125  bestValue = std::move(currentValue);
126  bestChoice = *choiceIt;
127  }
128  }
129  if (this->isProduceSchedulerSet()) {
130  this->_producedOptimalChoices.get()[state] = bestChoice - this->_transitionMatrix.getRowGroupIndices()[state];
131  }
132  bestValue += stateRewardsGetter(state);
133  return {true, bestValue};
134  } else {
135  // In a Markov Automaton, singleton components have to consist of a Markovian state because of the non-Zenoness assumption. Then, there is just one possible choice.
136  STORM_LOG_ASSERT(this->_markovianStates != nullptr, "Nondeterministic continuous time model without Markovian states... Is this a not a Markov Automaton?");
137  STORM_LOG_THROW(this->_markovianStates->get(state), storm::exceptions::InvalidOperationException, "Markov Automaton has Zeno behavior. Computation of Long Run Average values not supported.");
138  STORM_LOG_ASSERT(internal::getComponentElementChoiceCount(element) == 1, "Markovian state has Nondeterministic behavior.");
139  if (this->isProduceSchedulerSet()) {
140  this->_producedOptimalChoices.get()[state] = 0;
141  }
142  ValueType result = stateRewardsGetter(state) + (this->isContinuousTime() ? (*this->_exitRates)[state] * actionRewardsGetter(*choiceIt) : actionRewardsGetter(*choiceIt));
143  return {true, result};
144  }
145  }
146  return {false, storm::utility::zero<ValueType>()};
147  }

172  template <typename ValueType>
173  ValueType SparseNondeterministicInfiniteHorizonHelper<ValueType>::computeLraForMecLp(Environment const& env, ValueGetter const& stateRewardsGetter, ValueGetter const& actionRewardsGetter, storm::storage::MaximalEndComponent const& mec) {
174  // Create an LP solver
175  auto solver = storm::utility::solver::LpSolverFactory<ValueType>().create("LRA for MEC");
176 
177  // Now build the LP formulation as described in:
178  // Guck et al.: Modelling and Analysis of Markov Reward Automata (ATVA'14), https://doi.org/10.1007/978-3-319-11936-6_13
179  solver->setOptimizationDirection(invert(this->getOptimizationDirection()));
180 
181  // Create variables
182  // TODO: Investigate whether we can easily make the variables bounded
183  std::map<uint_fast64_t, storm::expressions::Variable> stateToVariableMap;
184  for (auto const& stateChoicesPair : mec) {
185  std::string variableName = "x" + std::to_string(stateChoicesPair.first);
186  stateToVariableMap[stateChoicesPair.first] = solver->addUnboundedContinuousVariable(variableName);
187  }
188  storm::expressions::Variable k = solver->addUnboundedContinuousVariable("k", storm::utility::one<ValueType>());
189  solver->update();
190 
191  // Add constraints.
192  for (auto const& stateChoicesPair : mec) {
193  uint_fast64_t state = stateChoicesPair.first;
194  bool stateIsMarkovian = this->_markovianStates && this->_markovianStates->get(state);
195 
196  // Now create a suitable constraint for each choice
197  // x_s {, } -k/rate(s) + sum_s' P(s,act,s') * x_s' + (value(s)/rate(s) + value(s,act))
198  for (auto choice : stateChoicesPair.second) {
199  std::vector<storm::expressions::Expression> summands;
200  auto matrixRow = this->_transitionMatrix.getRow(choice);
201  summands.reserve(matrixRow.getNumberOfEntries() + 2);
202  // add -k/rate(s) (only if s is either a Markovian state or we have an MDP)
203  if (stateIsMarkovian) {
204  summands.push_back(-(k / solver->getManager().rational((*this->_exitRates)[state])));
205  } else if (!this->isContinuousTime()) {
206  summands.push_back(-k);
207  }
208  // add sum_s' P(s,act,s') * x_s'
209  for (auto const& element : matrixRow) {
210  summands.push_back(stateToVariableMap.at(element.getColumn()) * solver->getConstant(element.getValue()));
211  }
212  // add value for state and choice
213  ValueType value;
214  if (stateIsMarkovian) {
215  // divide state reward with exit rate
216  value = stateRewardsGetter(state) / (*this->_exitRates)[state] + actionRewardsGetter(choice);
217  } else if (!this->isContinuousTime()) {
218  // in discrete time models no scaling is needed
219  value = stateRewardsGetter(state) + actionRewardsGetter(choice);
220  } else {
221  // state is a probabilistic state of a Markov automaton. The state reward is not collected
222  value = actionRewardsGetter(choice);
223  }
224  summands.push_back(solver->getConstant(value));
225  storm::expressions::Expression constraint;
226  if (this->minimize()) {
227  constraint = stateToVariableMap.at(state) <= storm::expressions::sum(summands);
228  } else {
229  constraint = stateToVariableMap.at(state) >= storm::expressions::sum(summands);
230  }
231  solver->addConstraint("s" + std::to_string(state) + "," + std::to_string(choice), constraint);
232  }
233  }
234 
235  solver->optimize();
236  STORM_LOG_THROW(!this->isProduceSchedulerSet(), storm::exceptions::NotImplementedException, "Scheduler extraction is not yet implemented for LP based LRA method.");
237  return solver->getContinuousValue(k);
238  }

245  template <typename ValueType>
246  void addSspMatrixChoice(uint64_t const& inputMatrixChoice, storm::storage::SparseMatrix<ValueType> const& inputTransitionMatrix, std::vector<uint64_t> const& inputToSspStateMap, uint64_t const& numberOfNonComponentStates, uint64_t const& currentSspChoice, storm::storage::SparseMatrixBuilder<ValueType>& sspMatrixBuilder) {
247 
248  // As there could be multiple transitions to the same MEC, we accumulate them in this map before adding them to the matrix builder.
249  std::map<uint64_t, ValueType> auxiliaryStateToProbabilityMap;
250 
251  for (auto const& transition : inputTransitionMatrix.getRow(inputMatrixChoice)) {
252  if (!storm::utility::isZero(transition.getValue())) {
253  auto const& sspTransitionTarget = inputToSspStateMap[transition.getColumn()];
254  // Since the auxiliary Component states are appended at the end of the matrix, we can use this check to
255  // decide whether the transition leads to a component state or not
256  if (sspTransitionTarget < numberOfNonComponentStates) {
257  // If the target state is not contained in a component, we can copy over the entry.
258  sspMatrixBuilder.addNextValue(currentSspChoice, sspTransitionTarget, transition.getValue());
259  } else {
260  // If the target state is contained in component i, we need to add the probability to the corresponding field in the vector
261  // so that we are able to write the cumulative probability to the component into the matrix later.
262  auto insertionRes = auxiliaryStateToProbabilityMap.emplace(sspTransitionTarget, transition.getValue());
263  if (!insertionRes.second) {
264  // sspTransitionTarget already existed in the map, i.e., there already was a transition to that component.
265  // Hence, we add up the probabilities.
266  insertionRes.first->second += transition.getValue();
267  }
268  }
269  }
270  }
271 
272  // Now insert all (cumulative) probability values that target a component.
273  for (auto const& componentToProbEntry : auxiliaryStateToProbabilityMap) {
274  sspMatrixBuilder.addNextValue(currentSspChoice, componentToProbEntry.first, componentToProbEntry.second);
275  }
276  }

279  std::pair<storm::storage::SparseMatrix<ValueType>, std::vector<ValueType>> SparseNondeterministicInfiniteHorizonHelper<ValueType>::buildSspMatrixVector(std::vector<ValueType> const& mecLraValues, std::vector<uint64_t> const& inputToSspStateMap, storm::storage::BitVector const& statesNotInComponent, uint64_t numberOfNonComponentStates, std::vector<std::pair<uint64_t, uint64_t>>* sspComponentExitChoicesToOriginalMap) {
280 
281  auto const& choiceIndices = this->_transitionMatrix.getRowGroupIndices();
282 
283  std::vector<ValueType> rhs;
284  uint64_t numberOfSspStates = numberOfNonComponentStates + this->_longRunComponentDecomposition->size();
285  storm::storage::SparseMatrixBuilder<ValueType> sspMatrixBuilder(0, numberOfSspStates , 0, true, true, numberOfSspStates);
286  // If the source state of a transition is not contained in any component, we copy its choices (and perform the necessary modifications).
287  uint64_t currentSspChoice = 0;
288  for (auto const& nonComponentState : statesNotInComponent) {
289  sspMatrixBuilder.newRowGroup(currentSspChoice);
290  for (uint64_t choice = choiceIndices[nonComponentState]; choice < choiceIndices[nonComponentState + 1]; ++choice, ++currentSspChoice) {
291  rhs.push_back(storm::utility::zero<ValueType>());
292  addSspMatrixChoice(choice, this->_transitionMatrix, inputToSspStateMap, numberOfNonComponentStates, currentSspChoice, sspMatrixBuilder);
293  }
294  }
295  // Now we construct the choices for the auxiliary states which reflect former Component states.
296  for (uint64_t componentIndex = 0; componentIndex < this->_longRunComponentDecomposition->size(); ++componentIndex) {
297  auto const& component = (*this->_longRunComponentDecomposition)[componentIndex];
298  sspMatrixBuilder.newRowGroup(currentSspChoice);
299  // For nondeterministic models it might still be that we leave the component again. This needs to be reflected in the SSP
300  // by adding the "exiting" choices of the MEC to the axiliary states
301  for (auto const& element : component) {
302  uint64_t componentState = internal::getComponentElementState(element);
303  for (uint64_t choice = choiceIndices[componentState]; choice < choiceIndices[componentState + 1]; ++choice) {
304  // If the choice is not contained in the component itself, we have to add a similar distribution to the auxiliary state.
305  if (!internal::componentElementChoicesContains(element, choice)) {
306  rhs.push_back(storm::utility::zero<ValueType>());
307  addSspMatrixChoice(choice, this->_transitionMatrix, inputToSspStateMap, numberOfNonComponentStates, currentSspChoice, sspMatrixBuilder);
308  if (sspComponentExitChoicesToOriginalMap) {
309  // Later we need to be able to map this choice back to the original input model
310  sspComponentExitChoicesToOriginalMap->emplace_back(componentState, choice - choiceIndices[componentState]);
311  }
312  ++currentSspChoice;
313  }
314  }
315  }
316  // For each auxiliary state, there is the option to achieve the reward value of the LRA associated with the component.
317  rhs.push_back(mecLraValues[componentIndex]);
318  if (sspComponentExitChoicesToOriginalMap) {
319  // Insert some invalid values so we can later detect that this choice is not an exit choice
320  sspComponentExitChoicesToOriginalMap->emplace_back(std::numeric_limits<uint_fast64_t>::max(), std::numeric_limits<uint_fast64_t>::max());
321  }
322  ++currentSspChoice;
323  }
324  return std::make_pair(sspMatrixBuilder.build(currentSspChoice, numberOfSspStates, numberOfSspStates), std::move(rhs));
325  }

327  template <typename ValueType>
328  void SparseNondeterministicInfiniteHorizonHelper<ValueType>::constructOptimalChoices(std::vector<uint64_t> const& sspChoices, storm::storage::SparseMatrix<ValueType> const& sspMatrix, std::vector<uint64_t> const& inputToSspStateMap, storm::storage::BitVector const& statesNotInComponent, uint64_t numberOfNonComponentStates, std::vector<std::pair<uint64_t, uint64_t>> const& sspComponentExitChoicesToOriginalMap) {
329  // We first take care of non-mec states
330  storm::utility::vector::setVectorValues(this->_producedOptimalChoices.get(), statesNotInComponent, sspChoices);
331  // Secondly, we consider MEC states. There are 3 cases for each MEC state:
332  // 1. The SSP choices encode that we want to stay in the MEC
333  // 2. The SSP choices encode that we want to leave the MEC and
334  // a) we take an exit (non-MEC) choice at the given state
335  // b) we have to take a MEC choice at the given state in a way that eventually an exit state of the MEC is reached
336  uint64_t exitChoiceOffset = sspMatrix.getRowGroupIndices()[numberOfNonComponentStates];
337  for (auto const& mec : *this->_longRunComponentDecomposition) {
338  // Get the sspState of this MEC (using one representative mec state)
339  auto const& sspState = inputToSspStateMap[mec.begin()->first];
340  uint64_t sspChoiceIndex = sspMatrix.getRowGroupIndices()[sspState] + sspChoices[sspState];
341  // Obtain the state and choice of the original model to which the selected choice corresponds.
342  auto const& originalStateChoice = sspComponentExitChoicesToOriginalMap[sspChoiceIndex - exitChoiceOffset];
343  // Check if we are in Case 1 or 2
344  if (originalStateChoice.first == std::numeric_limits<uint_fast64_t>::max()) {
345  // The optimal choice is to stay in this mec (Case 1)
346  // In this case, no further operations are necessary. The scheduler has already been set to the optimal choices during the call of computeLraForMec.
347  STORM_LOG_ASSERT(sspMatrix.getRow(sspState, sspChoices[sspState]).getNumberOfEntries() == 0, "Expected empty row at choice that stays in MEC.");
348  } else {
349  // The best choice is to leave this MEC via the selected state and choice. (Case 2)
350  // Set the exit choice (Case 2.a)
351  this->_producedOptimalChoices.get()[originalStateChoice.first] = originalStateChoice.second;
352  // The remaining states in this MEC need to reach the state with the exit choice with probability 1. (Case 2.b)
353  // Perform a backwards search from the exit state, only using MEC choices
354  // We start by setting an invalid choice to all remaining mec states (so that we can easily detect them as unprocessed)
355  for (auto const& stateActions : mec) {
356  if (stateActions.first != originalStateChoice.first) {
357  this->_producedOptimalChoices.get()[stateActions.first] = std::numeric_limits<uint64_t>::max();
358  }
359  }
360  // Ensure that backwards transitions are available
361  if (this->_backwardTransitions == nullptr) {
362  this->_computedBackwardTransitions = std::make_unique<storm::storage::SparseMatrix<ValueType>>(this->_transitionMatrix.transpose(true));
363  this->_backwardTransitions = this->_computedBackwardTransitions.get();
364  }
365  // Now start a backwards DFS
366  std::vector<uint64_t> stack = {originalStateChoice.first};
367  while (!stack.empty()) {
368  uint64_t currentState = stack.back();
369  stack.pop_back();
370  for (auto const& backwardsTransition : this->_backwardTransitions->getRowGroup(currentState)) {
371  uint64_t predecessorState = backwardsTransition.getColumn();
372  if (mec.containsState(predecessorState)) {
373  auto& selectedPredChoice = this->_producedOptimalChoices.get()[predecessorState];
374  if (selectedPredChoice == std::numeric_limits<uint64_t>::max()) {
375  // We don't already have a choice for this predecessor.
376  // We now need to check whether there is a *MEC* choice leading to currentState
377  for (auto const& predChoice : mec.getChoicesForState(predecessorState)) {
378  for (auto const& forwardTransition : this->_transitionMatrix.getRow(predChoice)) {
379  if (forwardTransition.getColumn() == currentState && !storm::utility::isZero(forwardTransition.getValue())) {
380  // Playing this choice (infinitely often) will lead to current state (infinitely often)!
381  selectedPredChoice = predChoice - this->_transitionMatrix.getRowGroupIndices()[predecessorState];
382  stack.push_back(predecessorState);
383  break;
384  }
385  }
386  if (selectedPredChoice != std::numeric_limits<uint64_t>::max()) {
387  break;
388  }
389  }
390  }
391  }
392  }
393  }
394  }
395  }

398  template <typename ValueType>
399  std::vector<ValueType> SparseNondeterministicInfiniteHorizonHelper<ValueType>::buildAndSolveSsp(Environment const& env, std::vector<ValueType> const& componentLraValues) {
400  STORM_LOG_ASSERT(this->_longRunComponentDecomposition != nullptr, "Decomposition not computed, yet.");
401 
402  // For fast transition rewriting, we build a mapping from the input state indices to the state indices of a new transition matrix
403  // which redirects all transitions leading to a former component state to a new auxiliary state.
404  // There will be one auxiliary state for each component. These states will be appended to the end of the matrix.
405 
406  // First gather the states that are part of a component
407  // and create a mapping from states that lie in a component to the corresponding component index.
408  storm::storage::BitVector statesInComponents(this->_transitionMatrix.getRowGroupCount());
409  std::vector<uint64_t> inputToSspStateMap(this->_transitionMatrix.getRowGroupCount(), std::numeric_limits<uint64_t>::max());
410  for (uint64_t currentComponentIndex = 0; currentComponentIndex < this->_longRunComponentDecomposition->size(); ++currentComponentIndex) {
411  for (auto const& element : (*this->_longRunComponentDecomposition)[currentComponentIndex]) {
412  uint64_t state = internal::getComponentElementState(element);
413  statesInComponents.set(state);
414  inputToSspStateMap[state] = currentComponentIndex;
415  }
416  }
417  // Now take care of the non-component states. Note that the order of these states will be preserved.
418  uint64_t numberOfNonComponentStates = 0;
419  storm::storage::BitVector statesNotInComponent = ~statesInComponents;
420  for (auto const& nonComponentState : statesNotInComponent) {
421  inputToSspStateMap[nonComponentState] = numberOfNonComponentStates;
422  ++numberOfNonComponentStates;
423  }
424  // Finalize the mapping for the component states which now still assigns component states to to their component index.
425  // To make sure that they point to the auxiliary states (located at the end of the SspMatrix), we need to shift them by the
426  // number of states that are not in a component.
427  for (auto const& mecState : statesInComponents) {
428  inputToSspStateMap[mecState] += numberOfNonComponentStates;
429  }
430 
431  // For scheduler extraction, we will need to create a mapping between choices at the auxiliary states and the
432  // corresponding choices in the original model.
433  std::vector<std::pair<uint_fast64_t, uint_fast64_t>> sspComponentExitChoicesToOriginalMap;
434 
435  // The next step is to create the SSP matrix and the right-hand side of the SSP.
436  auto sspMatrixVector = buildSspMatrixVector(componentLraValues, inputToSspStateMap, statesNotInComponent, numberOfNonComponentStates, this->isProduceSchedulerSet() ? &sspComponentExitChoicesToOriginalMap : nullptr);
437 
438  // Set-up a solver
439  storm::solver::GeneralMinMaxLinearEquationSolverFactory<ValueType> minMaxLinearEquationSolverFactory;
440  storm::solver::MinMaxLinearEquationSolverRequirements requirements = minMaxLinearEquationSolverFactory.getRequirements(env, true, true, this->getOptimizationDirection(), false, this->isProduceSchedulerSet());
441  requirements.clearBounds();
442  STORM_LOG_THROW(!requirements.hasEnabledCriticalRequirement(), storm::exceptions::UnmetRequirementException, "Solver requirements " + requirements.getEnabledRequirementsAsString() + " not checked.");
443  std::unique_ptr<storm::solver::MinMaxLinearEquationSolver<ValueType>> solver = minMaxLinearEquationSolverFactory.create(env, sspMatrixVector.first);
444  solver->setHasUniqueSolution();
445  solver->setHasNoEndComponents();
446  solver->setTrackScheduler(this->isProduceSchedulerSet());
447  auto lowerUpperBounds = std::minmax_element(componentLraValues.begin(), componentLraValues.end());
448  solver->setLowerBound(*lowerUpperBounds.first);
449  solver->setUpperBound(*lowerUpperBounds.second);
450  solver->setRequirementsChecked();
451 
452  // Solve the equation system
453  std::vector<ValueType> x(sspMatrixVector.first.getRowGroupCount());
454  solver->solveEquations(env, this->getOptimizationDirection(), x, sspMatrixVector.second);
455 
456  // Prepare scheduler (if requested)
457  if (this->isProduceSchedulerSet() && solver->hasScheduler()) {
458  // Translate result for ssp matrix to original model
459  constructOptimalChoices(solver->getSchedulerChoices(), sspMatrixVector.first, inputToSspStateMap, statesNotInComponent, numberOfNonComponentStates, sspComponentExitChoicesToOriginalMap);
460  } else {
461  STORM_LOG_ERROR_COND(!this->isProduceSchedulerSet(), "Requested to produce a scheduler, but no scheduler was generated.");
462  }
463 
464  // Prepare result vector.
465  // For efficiency reasons, we re-use the memory of our rhs for this!
466  std::vector<ValueType> result = std::move(sspMatrixVector.second);
467  result.resize(this->_transitionMatrix.getRowGroupCount());
468  result.shrink_to_fit();
469  storm::utility::vector::selectVectorValues(result, inputToSspStateMap, x);
470  return result;
471  }

474  template class SparseNondeterministicInfiniteHorizonHelper<storm::RationalNumber>;
