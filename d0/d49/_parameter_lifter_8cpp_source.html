172  std::vector<AbstractValuation> result(numOfVertices);
173 
174  for (uint_fast64_t vertexId = 0; vertexId < numOfVertices; ++vertexId) {
175  //interprete vertexId as a bit sequence
176  //the consideredVariables.size() least significant bits of vertex will always represent the next vertex
177  //(00...0 = lower boundaries for all variables, 11...1 = upper boundaries for all variables)
178  uint_fast64_t variableIndex = 0;
179  for (auto const& variable : variables) {

181  result[vertexId].addParameterLower(variable);
182  } else {
183  result[vertexId].addParameterUpper(variable);
184  }

186  }
187  }
