222  Automaton::Edges Automaton::getEdgesFromLocation(uint64_t locationIndex, uint64_t actionIndex) {
223  typedef std::vector<Edge>::iterator ForwardIt;
224 
225  // Perform binary search for start of edges with the given action index.
226  auto first = edges.begin();
227  std::advance(first, locationToStartingIndex[locationIndex]);
228  auto last = edges.begin();
229  std::advance(last, locationToStartingIndex[locationIndex + 1]);
230  typename std::iterator_traits<ForwardIt>::difference_type count, step;
231  count = std::distance(first, last);
232 
233  ForwardIt it1;
234  while (count > 0) {
235  it1 = first;
236  step = count / 2;
237  std::advance(it1, step);
238  if (it1->getActionIndex() < actionIndex) {
239  first = ++it1;
240  count -= step + 1;
241  }
242  else {
243  count = step;
244  }
245  }
246  it1 = first;
247 
248  // If there is no such edge, we can return now.
249  if (it1 != last && it1->getActionIndex() > actionIndex) {
250  return Edges(last, last);
251  }
252 
253  // Otherwise, perform a binary search for the end of the edges with the given action index.
254  count = std::distance(it1,last);
255 
256  ForwardIt it2;
257  while (count > 0) {
258  it2 = it1;
259  step = count / 2;
260  std::advance(it2, step);
261  if (!(actionIndex < it2->getActionIndex())) {
262  first = ++it2;
263  count -= step + 1;
264  } else count = step;
265  }
266  it2 = first;
267 
268  return Edges(it1, it2);
