236  uint_fast64_t numberOfSetBits = states->getNumberOfSetBits();
237  auto stateSize = states->size();
238  auto result = std::vector<uint_fast64_t>(numberOfSetBits, stateSize);
239  for (auto state : *states) {
240  if (result[0] == stateSize) {
241  result[0] = state;
242  } else {
243  uint_fast64_t i = 0;
244  bool added = false;
245  while (i < numberOfSetBits && !added) {
246  if (result[i] == stateSize) {
247  result[i] = state;
248  added = true;
249  } else {
250  auto compareRes = compare(state, result[i]);
251  if (compareRes == ABOVE) {
252  auto temp = result[i];
253  result[i] = state;
254  for (uint_fast64_t j = i + 1; j < numberOfSetBits && result[j + 1] != stateSize; j++) {
255  auto temp2 = result[j];
256  result[j] = temp;
257  temp = temp2;
258  }
259  added = true;
260  } else if (compareRes == UNKNOWN) {
261  break;
262  } else if (compareRes == SAME) {
263  ++i;
264  auto temp = result[i];
265  result[i] = state;
266  for (uint_fast64_t j = i + 1; j < numberOfSetBits && result[j + 1] != stateSize; j++) {
267  auto temp2 = result[j];
268  result[j] = temp;
269  temp = temp2;
270  }
271  added = true;
272  }
