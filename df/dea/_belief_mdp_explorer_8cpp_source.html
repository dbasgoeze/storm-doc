405  void BeliefMdpExplorer<PomdpType, BeliefValueType>::dropUnexploredStates() {
406  STORM_LOG_ASSERT(status == Status::Exploring, "Method call is invalid in current status.");
407  STORM_LOG_ASSERT(!hasUnexploredState(), "Finishing exploration not possible if there are still unexplored states.");
408 
409  STORM_LOG_ASSERT(exploredMdp, "Method called although no 'old' MDP is available.");
410  // Find the states (and corresponding choices) that were not explored.
411  // These correspond to "empty" MDP transitions
412  storm::storage::BitVector relevantMdpStates(getCurrentNumberOfMdpStates(), true), relevantMdpChoices(getCurrentNumberOfMdpChoices(), true);
413  std::vector<MdpStateType> toRelevantStateIndexMap(getCurrentNumberOfMdpStates(), noState());
414  MdpStateType nextRelevantIndex = 0;
415  for (uint64_t groupIndex = 0; groupIndex < exploredChoiceIndices.size() - 1; ++groupIndex) {
416  uint64_t rowIndex = exploredChoiceIndices[groupIndex];
417  // Check first row in group
418  if (exploredMdpTransitions[rowIndex].empty()) {
419  relevantMdpChoices.set(rowIndex, false);
420  relevantMdpStates.set(groupIndex, false);
421  } else {
422  toRelevantStateIndexMap[groupIndex] = nextRelevantIndex;
423  ++nextRelevantIndex;
424  }
425  uint64_t groupEnd = exploredChoiceIndices[groupIndex + 1];
426  // process remaining rows in group
427  for (++rowIndex; rowIndex < groupEnd; ++rowIndex) {
428  // Assert that all actions at the current state were consistently explored or unexplored.
429  STORM_LOG_ASSERT(exploredMdpTransitions[rowIndex].empty() != relevantMdpStates.get(groupIndex),
430  "Actions at 'old' MDP state " << groupIndex << " were only partly explored.");
431  if (exploredMdpTransitions[rowIndex].empty()) {
432  relevantMdpChoices.set(rowIndex, false);
433  }
434  }
435  }
436 
437  if (relevantMdpStates.full()) {
438  // All states are relevant so nothing to do
439  return;
440  }
441 
442  // Translate various components to the "new" MDP state set
443  storm::utility::vector::filterVectorInPlace(mdpStateToBeliefIdMap, relevantMdpStates);
444  { // beliefIdToMdpStateMap
445  for (auto belIdToMdpStateIt = beliefIdToMdpStateMap.begin(); belIdToMdpStateIt != beliefIdToMdpStateMap.end();) {
446  if (relevantMdpStates.get(belIdToMdpStateIt->second)) {
447  // Translate current entry and move on to the next one.
448  belIdToMdpStateIt->second = toRelevantStateIndexMap[belIdToMdpStateIt->second];
449  ++belIdToMdpStateIt;
450  } else {
451  STORM_LOG_ASSERT(!exploredBeliefIds.get(belIdToMdpStateIt->first),
452  "Inconsistent exploration information: Unexplored MDPState corresponds to explored beliefId");
453  // Delete current entry and move on to the next one.
454  // This works because std::map::erase does not invalidate other iterators within the map!
455  beliefIdToMdpStateMap.erase(belIdToMdpStateIt++);
456  }
457  }
458  }
459  { // exploredMdpTransitions
460  storm::utility::vector::filterVectorInPlace(exploredMdpTransitions, relevantMdpChoices);

668  template<typename PomdpType, typename BeliefValueType>
669  void BeliefMdpExplorer<PomdpType, BeliefValueType>::computeOptimalChoicesAndReachableMdpStates(ValueType const &ancillaryChoicesEpsilon, bool relativeDifference) {
670  STORM_LOG_ASSERT(status == Status::ModelChecked, "Method call is invalid in current status.");
671  STORM_LOG_ASSERT(exploredMdp, "Method call is invalid in if no MDP is available.");
672  STORM_LOG_ASSERT(!optimalChoices.is_initialized(), "Tried to compute optimal scheduler but this has already been done before.");
673  STORM_LOG_ASSERT(!optimalChoicesReachableMdpStates.is_initialized(),
674  "Tried to compute states that are reachable under an optimal scheduler but this has already been done before.");
675 
676  // First find the choices that are optimal
677  optimalChoices = storm::storage::BitVector(exploredMdp->getNumberOfChoices(), false);
678  auto const &choiceIndices = exploredMdp->getNondeterministicChoiceIndices();
679  auto const &transitions = exploredMdp->getTransitionMatrix();
680  auto const &targetStates = exploredMdp->getStates("target");
681  for (uint64_t mdpState = 0; mdpState < exploredMdp->getNumberOfStates(); ++mdpState) {
682  if (targetStates.get(mdpState)) {
683  // Target states can be skipped.
684  continue;
685  } else {
686  auto const &stateValue = values[mdpState];
687  for (uint64_t globalChoice = choiceIndices[mdpState]; globalChoice < choiceIndices[mdpState + 1]; ++globalChoice) {
688  ValueType choiceValue = transitions.multiplyRowWithVector(globalChoice, values);
689  if (exploredMdp->hasRewardModel()) {
690  choiceValue += exploredMdp->getUniqueRewardModel().getStateActionReward(globalChoice);
691  }
692  ValueType absDiff = storm::utility::abs<ValueType>((choiceValue - stateValue));
693  if ((relativeDifference && absDiff <= ancillaryChoicesEpsilon * stateValue) || (!relativeDifference && absDiff <= ancillaryChoicesEpsilon)) {
694  optimalChoices->set(globalChoice, true);
695  }
696  }
697  STORM_LOG_ASSERT(optimalChoices->getNextSetIndex(choiceIndices[mdpState]) < optimalChoices->size(), "Could not find an optimal choice.");
698  }
699  }
700 
701  // Then, find the states that are reachable via these choices
702  optimalChoicesReachableMdpStates = storm::utility::graph::getReachableStates(transitions, exploredMdp->getInitialStates(), ~targetStates, targetStates, false, 0,
703  optimalChoices.get());
704  }
