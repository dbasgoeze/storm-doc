76  Bdd<LibraryType> DdManager<LibraryType>::getEncoding(storm::expressions::Variable const& variable, int_fast64_t value, bool mostSignificantBitAtTop) const {
77  DdMetaVariable<LibraryType> const& metaVariable = this->getMetaVariable(variable);
78 
79  STORM_LOG_THROW(metaVariable.canRepresent(value), storm::exceptions::InvalidArgumentException, "Illegal value " << value << " for meta variable '" << variable.getName() << "'.");
80 
81  // Now compute the encoding relative to the low value of the meta variable.
82  value -= metaVariable.getLow();
83 
84  std::vector<Bdd<LibraryType>> const& ddVariables = metaVariable.getDdVariables();
85 
86  Bdd<LibraryType> result;
87  if (mostSignificantBitAtTop) {
88  if (value & (1ull << (ddVariables.size() - 1))) {
89  result = ddVariables[0];
90  } else {
91  result = !ddVariables[0];
92  }
93 
94  for (std::size_t i = 1; i < ddVariables.size(); ++i) {
95  if (value & (1ull << (ddVariables.size() - i - 1))) {
96  result &= ddVariables[i];
97  } else {
98  result &= !ddVariables[i];
99  }
100  }
101  } else {
102  if (value & 1ull) {
103  result = ddVariables[0];
104  } else {
105  result = !ddVariables[0];
106  }
107  value >>= 1;
108 
109  for (std::size_t i = 1; i < ddVariables.size(); ++i) {
110  if (value & 1ull) {
111  result &= ddVariables[i];
112  } else {
113  result &= !ddVariables[i];
114  }
115  value >>= 1;
116  }
117  }
118 
119  return result;
120  }

241  template<DdType LibraryType>
242  std::vector<storm::expressions::Variable> DdManager<LibraryType>::addMetaVariableHelper(MetaVariableType const& type, std::string const& name, uint64_t numberOfDdVariables, uint64_t numberOfLayers, boost::optional<std::pair<MetaVariablePosition, storm::expressions::Variable>> const& position, boost::optional<std::pair<int_fast64_t, int_fast64_t>> const& bounds) {
243  // Check whether number of layers is legal.
244  STORM_LOG_THROW(numberOfLayers >= 1, storm::exceptions::InvalidArgumentException, "Layers must be at least 1.");
245 
246  // Check that the number of DD variables is legal.
247  STORM_LOG_THROW(numberOfDdVariables >= 1, storm::exceptions::InvalidArgumentException, "Illegal number of DD variables.");
248 
249  // Check whether the variable name is legal.
250  STORM_LOG_THROW(name != "" && name.back() != '\'', storm::exceptions::InvalidArgumentException, "Illegal name of meta variable: '" << name << "'.");
251 
252  // Check whether a meta variable already exists.
253  STORM_LOG_THROW(!this->hasMetaVariable(name), storm::exceptions::InvalidArgumentException, "A meta variable '" << name << "' already exists.");
254 
255  // If a specific position was requested, we compute it now.
256  boost::optional<uint_fast64_t> level;
257  if (position) {
258  storm::dd::DdMetaVariable<LibraryType> beforeVariable = this->getMetaVariable(position.get().second);
259  level = position.get().first == MetaVariablePosition::Above ? std::numeric_limits<uint_fast64_t>::max() : std::numeric_limits<uint_fast64_t>::min();
260  for (auto const& ddVariable : beforeVariable.getDdVariables()) {
261  level = position.get().first == MetaVariablePosition::Above ? std::min(level.get(), ddVariable.getLevel()) : std::max(level.get(), ddVariable.getLevel());
262  }
263  if (position.get().first == MetaVariablePosition::Below) {
264  ++level.get();
265  }
266  }
267 
268  STORM_LOG_TRACE("Creating meta variable with " << numberOfDdVariables << " bit(s) and " << numberOfLayers << " layer(s).");
269 
270  std::stringstream tmp1;
271  std::vector<storm::expressions::Variable> result;
272  for (uint64_t layer = 0; layer < numberOfLayers; ++layer) {
273  if (type == MetaVariableType::Int) {
274  result.emplace_back(manager->declareIntegerVariable(name + tmp1.str()));
275  } else if (type == MetaVariableType::Bool) {
276  result.emplace_back(manager->declareBooleanVariable(name + tmp1.str()));
277  } else if (type == MetaVariableType::BitVector) {
278  result.emplace_back(manager->declareBitVectorVariable(name + tmp1.str(), numberOfDdVariables));
279  }
280  tmp1 << "'";
281  }
282 
283  std::vector<std::vector<Bdd<LibraryType>>> variables(numberOfLayers);
284  for (std::size_t i = 0; i < numberOfDdVariables; ++i) {
285  std::vector<InternalBdd<LibraryType>> ddVariables = internalDdManager.createDdVariables(numberOfLayers, level);
286  for (uint64_t layer = 0; layer < numberOfLayers; ++layer) {
287  variables[layer].emplace_back(Bdd<LibraryType>(*this, ddVariables[layer], {result[layer]}));
288  }
289 
290  // If we are inserting the variable at a specific level, we need to prepare the level for the next pair
291  // of variables.
292  if (level) {
293  level.get() += numberOfLayers;
294  }
295  }
296 
297  std::stringstream tmp2;
298  for (uint64_t layer = 0; layer < numberOfLayers; ++layer) {
299  if (bounds) {
300  metaVariableMap.emplace(result[layer], DdMetaVariable<LibraryType>(name + tmp2.str(), bounds.get().first, bounds.get().second, variables[layer]));
301  } else {
302  metaVariableMap.emplace(result[layer], DdMetaVariable<LibraryType>(type, name + tmp2.str(), variables[layer]));
303  }
304  tmp2 << "'";
305  }
306 
307  return result;
308  }

362  template<DdType LibraryType>
363  std::vector<std::string> DdManager<LibraryType>::getDdVariableNames() const {
364  // First, we initialize a list DD variables and their names.
365  std::vector<std::pair<uint_fast64_t, std::string>> variablePairs;
366  for (auto const& variablePair : this->metaVariableMap) {
367  DdMetaVariable<LibraryType> const& metaVariable = variablePair.second;
368  // If the meta variable is of type bool, we don't need to suffix it with the bit number.
369  if (metaVariable.getType() == MetaVariableType::Bool) {
370  variablePairs.emplace_back(metaVariable.getDdVariables().front().getIndex(), variablePair.first.getName());
371  } else {
372  // For integer-valued meta variables, we, however, have to add the suffix.
373  for (uint_fast64_t variableIndex = 0; variableIndex < metaVariable.getNumberOfDdVariables(); ++variableIndex) {
374  variablePairs.emplace_back(metaVariable.getDdVariables()[variableIndex].getIndex(), variablePair.first.getName() + '.' + std::to_string(variableIndex));
375  }
376  }
377  }
378 
379  // Then, we sort this list according to the indices of the ADDs.
380  std::sort(variablePairs.begin(), variablePairs.end(), [](std::pair<uint_fast64_t, std::string> const& a, std::pair<uint_fast64_t, std::string> const& b) { return a.first < b.first; });
381 
382  // Now, we project the sorted vector to its second component.
383  std::vector<std::string> result;
384  for (auto const& element : variablePairs) {
385  result.push_back(element.second);
386  }
387 
388  return result;
389  }
