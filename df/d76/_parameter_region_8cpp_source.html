92  std::vector<typename ParameterRegion<ParametricType>::Valuation> ParameterRegion<ParametricType>::getVerticesOfRegion(std::set<VariableType> const& consideredVariables) const {
93  std::size_t const numOfVariables = consideredVariables.size();
94  STORM_LOG_THROW(numOfVariables <= std::numeric_limits<std::size_t>::digits, storm::exceptions::OutOfRangeException, "Number of variables " << numOfVariables << " is too high.");
95  std::size_t const numOfVertices = std::pow(2, numOfVariables);
96  std::vector<Valuation> resultingVector(numOfVertices);
97 
98  for (uint_fast64_t vertexId = 0; vertexId < numOfVertices; ++vertexId) {
99  //interprete vertexId as a bit sequence
100  //the consideredVariables.size() least significant bits of vertex will always represent the next vertex
101  //(00...0 = lower boundaries for all variables, 11...1 = upper boundaries for all variables)
102  uint_fast64_t variableIndex = 0;
103  for (auto const& variable : consideredVariables) {
104  if ((vertexId >> variableIndex) % 2 == 0) {
105  resultingVector[vertexId].insert(std::pair<VariableType, CoefficientType>(variable, getLowerBoundary(variable)));
106  } else {
107  resultingVector[vertexId].insert(std::pair<VariableType, CoefficientType>(variable, getUpperBoundary(variable)));
108  }
109  ++variableIndex;
110  }
111  }
112  return resultingVector;
113  }

138  template<typename ParametricType>
139  void ParameterRegion<ParametricType>::split(Valuation const& splittingPoint, std::vector<ParameterRegion<ParametricType> >& regionVector) const{
140  //Check if splittingPoint is valid.
141  STORM_LOG_THROW(splittingPoint.size() == this->variables.size(), storm::exceptions::InvalidArgumentException, "Tried to split a region w.r.t. a point, but the point considers a different number of variables.");
142  for(auto const& variable : this->variables){
143  auto splittingPointEntry=splittingPoint.find(variable);
144  STORM_LOG_THROW(splittingPointEntry != splittingPoint.end(), storm::exceptions::InvalidArgumentException, "Tried to split a region but a variable of this region is not defined by the splitting point.");
145  STORM_LOG_THROW(this->getLowerBoundary(variable) <=splittingPointEntry->second, storm::exceptions::InvalidArgumentException, "Tried to split a region but the splitting point is not contained in the region.");
146  STORM_LOG_THROW(this->getUpperBoundary(variable) >=splittingPointEntry->second, storm::exceptions::InvalidArgumentException, "Tried to split a region but the splitting point is not contained in the region.");
147  }
148 
149  //Now compute the subregions.
150  std::vector<Valuation> vertices(this->getVerticesOfRegion(this->variables));
151  for(auto const& vertex : vertices){
152  //The resulting subregion is the smallest region containing vertex and splittingPoint.
153  Valuation subLower, subUpper;
154  for(auto variableBound : this->lowerBoundaries){
155  VariableType variable = variableBound.first;
156  auto vertexEntry=vertex.find(variable);
157  auto splittingPointEntry=splittingPoint.find(variable);
158  subLower.insert(typename Valuation::value_type(variable, std::min(vertexEntry->second, splittingPointEntry->second)));
159  subUpper.insert(typename Valuation::value_type(variable, std::max(vertexEntry->second, splittingPointEntry->second)));
160  }
161  ParameterRegion<ParametricType> subRegion(std::move(subLower), std::move(subUpper));
162  if(!storm::utility::isZero(subRegion.area())){
163  regionVector.push_back(std::move(subRegion));
164  }
165  }
