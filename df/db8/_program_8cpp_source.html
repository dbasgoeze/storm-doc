225  std::set<storm::expressions::Variable> undefinedConstantVariables;
226  for (auto const& constant : this->getConstants()) {
227  if (!constant.isDefined()) {
228  undefinedConstantVariables.insert(constant.getExpressionVariable());
229  }
230  }
231 
232  // Start by checking the defining expressions of all defined constants. If it contains a currently undefined
233  // constant, we need to mark the target constant as undefined as well.
234  for (auto const& constant : this->getConstants()) {
235  if (constant.isDefined()) {
236  if (constant.getExpression().containsVariable(undefinedConstantVariables)) {
237  undefinedConstantVariables.insert(constant.getExpressionVariable());
238  }
239  }
240  }
241 
242  // Now check initial value and range expressions of global variables.
243  for (auto const& booleanVariable : this->getGlobalBooleanVariables()) {
244  if (booleanVariable.hasInitialValue()) {
245  if (booleanVariable.getInitialValueExpression().containsVariable(undefinedConstantVariables)) {
246  return false;
247  }
248  }
249  }
250  for (auto const& integerVariable : this->getGlobalIntegerVariables()) {
251  if (integerVariable.hasInitialValue()) {
252  if (integerVariable.getInitialValueExpression().containsVariable(undefinedConstantVariables)) {
253  return false;
254  }
255  }
256  if (integerVariable.getLowerBoundExpression().containsVariable(undefinedConstantVariables)) {
257  return false;
258  }
259  if (integerVariable.getUpperBoundExpression().containsVariable(undefinedConstantVariables)) {
260  return false;
261  }
262  }
263 
264 
265  // Proceed by checking each of the modules.
266  for (auto const& module : this->getModules()) {
267  if (!module.containsVariablesOnlyInUpdateProbabilities(undefinedConstantVariables)) {
268  return false;
269  }
270  }
271 
272  // Check the reward models.
273  for (auto const& rewardModel : this->getRewardModels()) {
274  rewardModel.containsVariablesOnlyInRewardValueExpressions(undefinedConstantVariables);
275  }
276 
277  // Initial construct.
278  if (this->hasInitialConstruct()) {
279  if (this->getInitialConstruct().getInitialStatesExpression().containsVariable(undefinedConstantVariables)) {
280  return false;
281  }
282  }
283

285  for (auto const& label : this->getLabels()) {
286  if (label.getStatePredicateExpression().containsVariable(undefinedConstantVariables)) {

532  storm::expressions::Expression Program::getInitialStatesExpression() const {
533  // If there is an initial construct, return its expression. If not, we construct the expression from the
534  // initial values of the variables (which have to exist).
535  if (this->hasInitialConstruct()) {
536  return this->getInitialConstruct().getInitialStatesExpression();
537  } else {

539 
540  for (auto const& variable : this->getGlobalBooleanVariables()) {
541  if (result.isInitialized()) {
542  result = result && storm::expressions::iff(variable.getExpressionVariable(), variable.getInitialValueExpression());
543  } else {
544  result = storm::expressions::iff(variable.getExpressionVariable(), variable.getInitialValueExpression());
545  }
546  }
547  for (auto const& variable : this->getGlobalIntegerVariables()) {
548  if (result.isInitialized()) {
549  result = result && variable.getExpressionVariable() == variable.getInitialValueExpression();
550  } else {

552  }
553  }
554  for (auto const& module : this->getModules()) {
555  for (auto const& variable : module.getBooleanVariables()) {
556  if (result.isInitialized()) {

784  void Program::createMappings() {
785  // Build the mappings for constants, global variables, formulas, modules, reward models and labels.
786  for (uint_fast64_t constantIndex = 0; constantIndex < this->getNumberOfConstants(); ++constantIndex) {
787  this->constantToIndexMap[this->getConstants()[constantIndex].getName()] = constantIndex;
788  }
789  for (uint_fast64_t globalVariableIndex = 0; globalVariableIndex < this->getNumberOfGlobalBooleanVariables(); ++globalVariableIndex) {
790  this->globalBooleanVariableToIndexMap[this->getGlobalBooleanVariables()[globalVariableIndex].getName()] = globalVariableIndex;
791  }
792  for (uint_fast64_t globalVariableIndex = 0; globalVariableIndex < this->getNumberOfGlobalIntegerVariables(); ++globalVariableIndex) {
793  this->globalIntegerVariableToIndexMap[this->getGlobalIntegerVariables()[globalVariableIndex].getName()] = globalVariableIndex;
794  }
795  for (uint_fast64_t formulaIndex = 0; formulaIndex < this->getNumberOfFormulas(); ++formulaIndex) {
796  this->formulaToIndexMap[this->getFormulas()[formulaIndex].getName()] = formulaIndex;
797  }
798  for (uint_fast64_t labelIndex = 0; labelIndex < this->getNumberOfLabels(); ++labelIndex) {
799  this->labelToIndexMap[this->getLabels()[labelIndex].getName()] = labelIndex;
800  }
801  for (uint_fast64_t moduleIndex = 0; moduleIndex < this->getNumberOfModules(); ++moduleIndex) {
802  this->moduleToIndexMap[this->getModules()[moduleIndex].getName()] = moduleIndex;
803  }
804  for (uint_fast64_t rewardModelIndex = 0; rewardModelIndex < this->getNumberOfRewardModels(); ++rewardModelIndex) {
805  this->rewardModelToIndexMap[this->getRewardModels()[rewardModelIndex].getName()] = rewardModelIndex;
806  }
807 
808  for (auto const& actionIndexPair : this->getActionNameToIndexMapping()) {
809  this->actions.insert(actionIndexPair.first);
810  this->indexToActionMap.emplace(actionIndexPair.second, actionIndexPair.first);
811 
812  // Only let all non-zero indices be synchronizing.
813  if (actionIndexPair.second != 0) {
814  this->synchronizingActionIndices.insert(actionIndexPair.second);
815  }
816  }
817 
818  // Build the mapping from action names to module indices so that the lookup can later be performed quickly.
819  for (unsigned int moduleIndex = 0; moduleIndex < this->getNumberOfModules(); moduleIndex++) {
820  Module const& module = this->getModule(moduleIndex);
821 
822  for (auto const& actionIndex : module.getSynchronizingActionIndices()) {
823  auto const& actionModuleIndicesPair = this->actionIndicesToModuleIndexMap.find(actionIndex);
824  if (actionModuleIndicesPair == this->actionIndicesToModuleIndexMap.end()) {
825  this->actionIndicesToModuleIndexMap[actionIndex] = std::set<uint_fast64_t>();
826  }
827  this->actionIndicesToModuleIndexMap[actionIndex].insert(moduleIndex);
828  }
829 
830  // Put in the appropriate entries for the mapping from variable names to module index.
831  for (auto const& booleanVariable : module.getBooleanVariables()) {
832  this->variableToModuleIndexMap[booleanVariable.getName()] = moduleIndex;
833  }
834  for (auto const& integerVariable : module.getIntegerVariables()) {
835  this->variableToModuleIndexMap[integerVariable.getName()] = moduleIndex;
836  }
837  for (auto const& clockVariable : module.getClockVariables()) {
838  this->variableToModuleIndexMap[clockVariable.getName()] = moduleIndex;
839  }
840  }

844  Program Program::defineUndefinedConstants(std::map<storm::expressions::Variable, storm::expressions::Expression> const& constantDefinitions) const {
845  // For sanity checking, we keep track of all undefined constants that we define in the course of this procedure.
846  std::set<storm::expressions::Variable> definedUndefinedConstants;
847 
848  std::vector<Constant> newConstants;
849  newConstants.reserve(this->getNumberOfConstants());
850  for (auto const& constant : this->getConstants()) {
851  // If the constant is already defined, we need to replace the appearances of undefined constants in its
852  // defining expression
853  if (constant.isDefined()) {
854  // Make sure we are not trying to define an already defined constant.
855  STORM_LOG_THROW(constantDefinitions.find(constant.getExpressionVariable()) == constantDefinitions.end(), storm::exceptions::InvalidArgumentException, "Illegally defining already defined constant '" << constant.getName() << "'.");
856 
857  // Now replace the occurrences of undefined constants in its defining expression.
858  newConstants.emplace_back(constant.getExpressionVariable(), constant.getExpression().substitute(constantDefinitions), constant.getFilename(), constant.getLineNumber());
859  } else {
860  auto const& variableExpressionPair = constantDefinitions.find(constant.getExpressionVariable());
861 
862  // If the constant is not defined by the mapping, we leave it like it is.
863  if (variableExpressionPair == constantDefinitions.end()) {
864  newConstants.emplace_back(constant);
865  } else {
866  // Otherwise, we add it to the defined constants and assign it the appropriate expression.
867  definedUndefinedConstants.insert(constant.getExpressionVariable());
868 
869  // Make sure the type of the constant is correct.
870  STORM_LOG_THROW(variableExpressionPair->second.getType() == constant.getType(), storm::exceptions::InvalidArgumentException, "Illegal type of expression defining constant '" << constant.getName() << "'.");
871 
872  // Now create the defined constant.
873  newConstants.emplace_back(constant.getExpressionVariable(), variableExpressionPair->second, constant.getFilename(), constant.getLineNumber());
874  }
875  }
876  }
877 
878  return Program(this->manager, this->getModelType(), newConstants, this->getGlobalBooleanVariables(), this->getGlobalIntegerVariables(), this->getFormulas(), this->getModules(), this->getActionNameToIndexMapping(), this->getRewardModels(), this->getLabels(), this->getObservationLabels(), this->getOptionalInitialConstruct(), this->getOptionalSystemCompositionConstruct(), prismCompatibility);
879  }

1518  Program Program::flattenModules(std::shared_ptr<storm::utility::solver::SmtSolverFactory> const& smtSolverFactory) const {
1519  // If the current program has only one module, we can simply return a copy.
1520  if (this->getNumberOfModules() == 1) {
1521  return Program(*this);
1522  }
1523 
1524  STORM_LOG_THROW(this->getModelType() == ModelType::DTMC || this->getModelType() == ModelType::MDP, storm::exceptions::InvalidTypeException, "Unable to flatten modules for model of type '" << this->getModelType() << "'.");
1525 
1526  // Otherwise, we need to actually flatten the contained modules.
1527 
1528  // Get an SMT solver for computing the possible guard combinations.
1529  std::unique_ptr<storm::solver::SmtSolver> solver = smtSolverFactory->create(*manager);
1530 
1531  // Set up the data we need to gather to create the flat module.
1532  std::stringstream newModuleName;
1533  std::vector<storm::prism::BooleanVariable> allBooleanVariables;
1534  std::vector<storm::prism::IntegerVariable> allIntegerVariables;
1535  std::vector<storm::prism::ClockVariable> allClockVariables;
1536  std::vector<storm::prism::Command> newCommands;
1537  uint_fast64_t nextCommandIndex = 0;
1538  uint_fast64_t nextUpdateIndex = 0;
1539 
1540  // Assert the values of the constants.
1541  for (auto const& constant : this->getConstants()) {
1542  if (constant.isDefined()) {
1543  if (constant.getType().isBooleanType()) {
1544  solver->add(storm::expressions::iff(constant.getExpressionVariable(),constant.getExpression()));
1545  } else {
1546  solver->add(constant.getExpressionVariable() == constant.getExpression());
1547  }
1548  }
1549  }
1550 
1551  // Assert the bounds of the global variables.
1552  for (auto const& variable : this->getGlobalIntegerVariables()) {
1553  solver->add(variable.getExpression() >= variable.getLowerBoundExpression());
1554  solver->add(variable.getExpression() <= variable.getUpperBoundExpression());
1555  }
1556 
1557  // Make the global variables local, such that the resulting module covers all occurring variables. Note that
1558  // this is just for simplicity and is not needed.
1559  allBooleanVariables.insert(allBooleanVariables.end(), this->getGlobalBooleanVariables().begin(), this->getGlobalBooleanVariables().end());
1560  allIntegerVariables.insert(allIntegerVariables.end(), this->getGlobalIntegerVariables().begin(), this->getGlobalIntegerVariables().end());
1561  storm::expressions::Expression newInvariant;
1562 
1563  // Now go through the modules, gather the variables, construct the name of the new module and assert the
1564  // bounds of the discovered variables.
1565  for (auto const& module : this->getModules()) {
1566  newModuleName << module.getName() << "_";
1567  allBooleanVariables.insert(allBooleanVariables.end(), module.getBooleanVariables().begin(), module.getBooleanVariables().end());
1568  allIntegerVariables.insert(allIntegerVariables.end(), module.getIntegerVariables().begin(), module.getIntegerVariables().end());
1569  allClockVariables.insert(allClockVariables.end(), module.getClockVariables().begin(), module.getClockVariables().end());
1570 
1571  for (auto const& variable : module.getIntegerVariables()) {
1572  solver->add(variable.getExpression() >= variable.getLowerBoundExpression());
1573  solver->add(variable.getExpression() <= variable.getUpperBoundExpression());
1574  }
1575 
1576  if (module.hasInvariant()) {
1577  newInvariant = newInvariant.isInitialized() ? (newInvariant && module.getInvariant()) : module.getInvariant();
1578  }
1579 
1580  // The commands without a synchronizing action name, can simply be copied (plus adjusting the global
1581  // indices of the command and its updates).
1582  for (auto const& command : module.getCommands()) {
1583  if (!command.isLabeled()) {
1584  std::vector<storm::prism::Update> updates;
1585  updates.reserve(command.getUpdates().size());
1586 
1587  for (auto const& update : command.getUpdates()) {
1588  updates.push_back(storm::prism::Update(nextUpdateIndex, update.getLikelihoodExpression(), update.getAssignments(), update.getFilename(), 0));
1589  ++nextUpdateIndex;
1590  }
1591 
1592  newCommands.push_back(storm::prism::Command(nextCommandIndex, command.isMarkovian(), actionToIndexMap.find("")->second, "", command.getGuardExpression(), updates, command.getFilename(), 0));
1593  ++nextCommandIndex;
1594  }
1595  }
1596  }
1597 
1598  // Save state of solver so that we can always restore the point where we have exactly the constant values
1599  // and variables bounds on the assertion stack.
1600  solver->push();
1601 
1602  // Now we need to enumerate all possible combinations of synchronizing commands. For this, we iterate over
1603  // all actions and let the solver enumerate the possible combinations of commands that can be enabled together.
1604  for (auto const& actionIndex : this->getSynchronizingActionIndices()) {
1605  bool noCombinationsForAction = false;
1606 
1607  // Prepare the list that stores for each module the list of commands with the given action.
1608  std::vector<std::vector<std::reference_wrapper<storm::prism::Command const>>> possibleCommands;
1609 
1610  for (auto const& module : this->getModules()) {
1611  // If the module has no command with this action, we can skip it.
1612  if (!module.hasActionIndex(actionIndex)) {
1613  continue;
1614  }
1615 
1616  std::set<uint_fast64_t> const& commandIndices = module.getCommandIndicesByActionIndex(actionIndex);
1617 
1618  // If there is no command even though the module has this action, there is no valid command
1619  // combination with this action.
1620  if (commandIndices.empty()) {
1621  noCombinationsForAction = true;
1622  break;
1623  }
1624 
1625  // Prepare empty list of commands for this module.
1626  possibleCommands.push_back(std::vector<std::reference_wrapper<storm::prism::Command const>>());
1627 
1628  // Add references to the commands labeled with the current action.
1629  for (auto const& commandIndex : commandIndices) {
1630  possibleCommands.back().push_back(module.getCommand(commandIndex));
1631  }
1632  }
1633 
1634  // If there are no valid combinations for the action, we need to skip the generation of synchronizing
1635  // commands.
1636  if (!noCombinationsForAction) {
1637  // Save the solver state to be able to restore it when this action index is done.
1638  solver->push();
1639 
1640  // Start by creating a fresh auxiliary variable for each command and link it with the guard.
1641  std::vector<std::vector<storm::expressions::Variable>> commandVariables(possibleCommands.size());
1642  std::vector<storm::expressions::Variable> allCommandVariables;
1643  for (uint_fast64_t outerIndex = 0; outerIndex < possibleCommands.size(); ++outerIndex) {
1644  // Create auxiliary variables and link them with the guards.
1645  for (uint_fast64_t innerIndex = 0; innerIndex < possibleCommands[outerIndex].size(); ++innerIndex) {
1646  commandVariables[outerIndex].push_back(manager->declareFreshBooleanVariable());
1647  allCommandVariables.push_back(commandVariables[outerIndex].back());
1648  solver->add(implies(commandVariables[outerIndex].back(), possibleCommands[outerIndex][innerIndex].get().getGuardExpression()));
1649  }
1650 
1651  storm::expressions::Expression atLeastOneCommandFromModule = manager->boolean(false);
1652  for (auto const& commandVariable : commandVariables[outerIndex]) {
1653  atLeastOneCommandFromModule = atLeastOneCommandFromModule || commandVariable;
1654  }
1655  solver->add(atLeastOneCommandFromModule);
1656  }
1657 
1658  // Now we are in a position to start the enumeration over all command variables. While doing so, we
1659  // keep track of previously seen command combinations, because the AllSat procedures are not
1660  // always guaranteed to only provide distinct models.
1661  std::unordered_set<std::vector<uint_fast64_t>, storm::utility::vector::VectorHash<uint_fast64_t>> seenCommandCombinations;
1662  solver->allSat(allCommandVariables, [&] (storm::solver::SmtSolver::ModelReference& modelReference) -> bool {
1663  // Now we need to reconstruct the chosen commands from the valuation of the command variables.
1664  std::vector<std::vector<std::reference_wrapper<Command const>>> chosenCommands(possibleCommands.size());
1665 
1666  for (uint_fast64_t outerIndex = 0; outerIndex < commandVariables.size(); ++outerIndex) {
1667  for (uint_fast64_t innerIndex = 0; innerIndex < commandVariables[outerIndex].size(); ++innerIndex) {
1668  if (modelReference.getBooleanValue(commandVariables[outerIndex][innerIndex])) {
1669  chosenCommands[outerIndex].push_back(possibleCommands[outerIndex][innerIndex]);
1670  }
1671  }
1672  }
1673 
1674  // Now that we have retrieved the commands, we need to build their synchronizations and add them
1675  // to the flattened module.
1676  std::vector<std::vector<std::reference_wrapper<Command const>>::const_iterator> iterators;
1677  for (auto const& element : chosenCommands) {
1678  iterators.push_back(element.begin());
1679  }
1680 
1681  bool movedAtLeastOneIterator = false;
1682  std::vector<std::reference_wrapper<Command const>> commandCombination(chosenCommands.size(), chosenCommands.front().front());
1683  std::vector<uint_fast64_t> commandCombinationIndices(iterators.size());
1684  do {
1685  for (uint_fast64_t index = 0; index < iterators.size(); ++index) {
1686  commandCombination[index] = *iterators[index];
1687  commandCombinationIndices[index] = commandCombination[index].get().getGlobalIndex();
1688  }
1689 
1690  // Only add the command combination if it was not previously seen.
1691  auto seenIt = seenCommandCombinations.find(commandCombinationIndices);
1692  if (seenIt == seenCommandCombinations.end()) {
1693  newCommands.push_back(synchronizeCommands(nextCommandIndex, actionIndex, nextUpdateIndex, indexToActionMap.find(actionIndex)->second, commandCombination));
1694  seenCommandCombinations.insert(commandCombinationIndices);
1695 
1696  // Move the counters appropriately.
1697  ++nextCommandIndex;
1698  nextUpdateIndex += newCommands.back().getNumberOfUpdates();
1699  }
1700 
1701  movedAtLeastOneIterator = false;
1702  for (uint_fast64_t index = 0; index < iterators.size(); ++index) {
1703  ++iterators[index];
1704  if (iterators[index] != chosenCommands[index].cend()) {
1705  movedAtLeastOneIterator = true;
1706  break;
1707  } else {
1708  iterators[index] = chosenCommands[index].cbegin();
1709  }
1710  }
1711  } while (movedAtLeastOneIterator);
1712 
1713  return true;
1714  });
1715 
1716  solver->pop();
1717  }
1718  }
1719 
1720  // Finally, we can create the module and the program and return it.
1721  storm::prism::Module singleModule(newModuleName.str(), allBooleanVariables, allIntegerVariables, allClockVariables, newInvariant, newCommands, this->getFilename(), 0);
1722 
1723  return Program(manager, this->getModelType(), this->getConstants(), std::vector<storm::prism::BooleanVariable>(), std::vector<storm::prism::IntegerVariable>(), this->getFormulas(), {singleModule}, actionToIndexMap, this->getRewardModels(), this->getLabels(), this->getObservationLabels(), this->getOptionalInitialConstruct(), this->getOptionalSystemCompositionConstruct(), prismCompatibility, this->getFilename(), 0, true);
1724  }

1827  Command Program::synchronizeCommands(uint_fast64_t newCommandIndex, uint_fast64_t actionIndex, uint_fast64_t firstUpdateIndex, std::string const& actionName, std::vector<std::reference_wrapper<Command const>> const& commands) const {
1828  // To construct the synchronous product of the commands, we need to store a list of its updates.
1829  std::vector<storm::prism::Update> newUpdates;
1830  uint_fast64_t numberOfUpdates = 1;
1831  for (uint_fast64_t i = 0; i < commands.size(); ++i) {
1832  numberOfUpdates *= commands[i].get().getNumberOfUpdates();
1833  }
1834  newUpdates.reserve(numberOfUpdates);
1835 
1836  // Initialize all update iterators.
1837  std::vector<std::vector<storm::prism::Update>::const_iterator> updateIterators;
1838  for (uint_fast64_t i = 0; i < commands.size(); ++i) {
1839  updateIterators.push_back(commands[i].get().getUpdates().cbegin());
1840  }
1841 
1842  bool doneUpdates = false;
1843  do {
1844  // We create the new likelihood expression by multiplying the particapting updates' expressions.
1845  storm::expressions::Expression newLikelihoodExpression = updateIterators[0]->getLikelihoodExpression();
1846  for (uint_fast64_t i = 1; i < updateIterators.size(); ++i) {
1847  newLikelihoodExpression = newLikelihoodExpression * updateIterators[i]->getLikelihoodExpression();
1848  }
1849 
1850  // Now concatenate all assignments of all participating updates.
1851  std::vector<storm::prism::Assignment> newAssignments;
1852  for (uint_fast64_t i = 0; i < updateIterators.size(); ++i) {
1853  newAssignments.insert(newAssignments.end(), updateIterators[i]->getAssignments().begin(), updateIterators[i]->getAssignments().end());
1854  }
1855 
1856  // Then we are ready to create the new update.
1857  newUpdates.push_back(storm::prism::Update(firstUpdateIndex, newLikelihoodExpression, newAssignments, this->getFilename(), 0));
1858  ++firstUpdateIndex;
1859 
1860  // Now check whether there is some update combination we have not yet explored.
1861  bool movedIterator = false;
1862  for (int_fast64_t j = updateIterators.size() - 1; j >= 0; --j) {
1863  ++updateIterators[j];
1864  if (updateIterators[j] != commands[j].get().getUpdates().cend()) {
1865  movedIterator = true;
1866  break;
1867  } else {
1868  // Reset the iterator to the beginning of the list.
1869  updateIterators[j] = commands[j].get().getUpdates().cbegin();
1870  }
1871  }
1872 
1873  doneUpdates = !movedIterator;
1874  } while (!doneUpdates);
1875 
1876  storm::expressions::Expression newGuard = commands[0].get().getGuardExpression();
1877  for (uint_fast64_t i = 1; i < commands.size(); ++i) {
1878  newGuard = newGuard && commands[i].get().getGuardExpression();
1879  }
1880 
1881  return Command(newCommandIndex, false, actionIndex, actionName, newGuard, newUpdates, this->getFilename(), 0);
1882  }
