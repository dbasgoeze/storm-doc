180  void performSmtLoop(bool lowerBound, double boundary, PermissiveSchedulerPenalties const& penalties) {
181  storm::storage::BitVector irrelevant = this->mGoals | this->mSinks;
182  storm::storage::BitVector relevantStates = ~irrelevant;
183  createVariables(relevantStates);
184  createConstraints(lowerBound, boundary, relevantStates);
185 
186  // Find the initial solution (if possible).
187  storm::solver::SmtSolver::CheckResult result = solver.check();
188 
189  if (result == storm::solver::SmtSolver::CheckResult::Sat) {
190  // Extract the solution from the multi-strategy variables and track all state-action pairs that were
191  // not taken. Also, we assert all decided choices, so they are not altered anymore.
192  std::shared_ptr<storm::solver::SmtSolver::ModelReference> model = solver.getModel();
193 
194  std::vector<storage::StateActionPair> availableStateActionPairs;
195  for (uint_fast64_t s : relevantStates) {
196  for(uint_fast64_t a = 0; a < this->mdp.getNumberOfChoices(s); ++a) {
197  auto stateAndAction = storage::StateActionPair(s,a);
198 
199  auto multistrategyVariable = multistrategyVariables.at(stateAndAction);
200  if (model->getBooleanValue(multistrategyVariable)) {
201  multistrategyVariablesToTakenMap[multistrategyVariable] = true;
202  solver.add(multistrategyVariable);
203  } else {
204  availableStateActionPairs.push_back(stateAndAction);
205  }
206  }
207  }
208 
209  // Now we sort the available state-action pairs in decending penalty order, so we can try taking more
210  // and more actions from the back (and discard them if not).
211  std::sort(availableStateActionPairs.begin(), availableStateActionPairs.end(), [&penalties] (storage::StateActionPair const& first, storage::StateActionPair const& second) { return penalties.get(first) < penalties.get(second); } );
212 
213  do {
214  auto multistrategyVariable = multistrategyVariables.at(availableStateActionPairs.back());
215 
216  result = solver.checkWithAssumptions({ multistrategyVariable });
217 
218  if (result == storm::solver::SmtSolver::CheckResult::Sat) {
219  model = solver.getModel();
220  if (model->getBooleanValue(multistrategyVariable)) {
221  solver.add(multistrategyVariable);
222  multistrategyVariablesToTakenMap[multistrategyVariable] = true;
223  }
224  }
225  availableStateActionPairs.pop_back();
226 
227  } while (!availableStateActionPairs.empty());
228 
229  mFoundSolution = true;
230  } else {
231  mFoundSolution = false;
232  }
233  }
