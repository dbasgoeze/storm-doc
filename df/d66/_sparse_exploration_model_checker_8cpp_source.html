75  stateGeneration.computeInitialStates();

81

84

114  }
115  }
116

118  if (storm::settings::getModule<storm::settings::modules::CoreSettings>().isShowStatisticsSet()) {
119  stats.printToStream(std::cout, explorationInformation);

125  template<typename ModelType, typename StateType>
126  bool SparseExplorationModelChecker<ModelType, StateType>::samplePathFromInitialState(StateGeneration<StateType, ValueType>& stateGeneration, ExplorationInformation<StateType, ValueType>& explorationInformation, StateActionStack& stack, Bounds<StateType, ValueType>& bounds, Statistics<StateType, ValueType>& stats) const {
127  // Start the search from the initial state.
128  stack.push_back(std::make_pair(stateGeneration.getFirstInitialState(), 0));
129 
130  // As long as we didn't find a terminal (accepting or rejecting) state in the search, sample a new successor.
131  bool foundTerminalState = false;
132  while (!foundTerminalState) {
133  StateType const& currentStateId = stack.back().first;
134  STORM_LOG_TRACE("State on top of stack is: " << currentStateId << ".");
135 
136  // If the state is not yet explored, we need to retrieve its behaviors.
137  auto unexploredIt = explorationInformation.findUnexploredState(currentStateId);
138  if (unexploredIt != explorationInformation.unexploredStatesEnd()) {
139  STORM_LOG_TRACE("State was not yet explored.");
140 
141  // Explore the previously unexplored state.
142  storm::generator::CompressedState const& compressedState = unexploredIt->second;
143  foundTerminalState = exploreState(stateGeneration, currentStateId, compressedState, explorationInformation, bounds, stats);
144  if (foundTerminalState) {
145  STORM_LOG_TRACE("Aborting sampling of path, because a terminal state was reached.");
146  }
147  explorationInformation.removeUnexploredState(unexploredIt);
148  } else {
149  // If the state was already explored, we check whether it is a terminal state or not.
150  if (explorationInformation.isTerminal(currentStateId)) {
151  STORM_LOG_TRACE("Found already explored terminal state: " << currentStateId << ".");
152  foundTerminalState = true;
153  }
154  }
155 
156  // Notify the stats about the performed exploration step.
157  stats.explorationStep();
158 
159  // If the state was not a terminal state, we continue the path search and sample the next state.
160  if (!foundTerminalState) {
161  // At this point, we can be sure that the state was expanded and that we can sample according to the
162  // probabilities in the matrix.
163  uint32_t chosenAction = sampleActionOfState(currentStateId, explorationInformation, bounds);
164  stack.back().second = chosenAction;
165  STORM_LOG_TRACE("Sampled action " << chosenAction << " in state " << currentStateId << ".");
166 
167  StateType successor = sampleSuccessorFromAction(chosenAction, explorationInformation, bounds);
168  STORM_LOG_TRACE("Sampled successor " << successor << " according to action " << chosenAction << " of state " << currentStateId << ".");
169 
170  // Put the successor state and a dummy action on top of the stack.
171  stack.emplace_back(successor, 0);
172 
173  // If the number of exploration steps exceeds a certain threshold, do a precomputation.
174  if (explorationInformation.performPrecomputationExcessiveExplorationSteps(stats.explorationStepsSinceLastPrecomputation)) {
175  performPrecomputation(stack, explorationInformation, bounds, stats);
176 
177  STORM_LOG_TRACE("Aborting the search after precomputation.");
178  stack.clear();
179  break;
180  }
181  }
182  }
183 
184  return foundTerminalState;
185  }

187  template<typename ModelType, typename StateType>
188  bool SparseExplorationModelChecker<ModelType, StateType>::exploreState(StateGeneration<StateType, ValueType>& stateGeneration, StateType const& currentStateId, storm::generator::CompressedState const& currentState, ExplorationInformation<StateType, ValueType>& explorationInformation, Bounds<StateType, ValueType>& bounds, Statistics<StateType, ValueType>& stats) const {
189  bool isTerminalState = false;
190  bool isTargetState = false;
191 
192  ++stats.numberOfExploredStates;
193 
194  // Finally, map the unexplored state to the row group.
195  explorationInformation.assignStateToNextRowGroup(currentStateId);
196  STORM_LOG_TRACE("Assigning row group " << explorationInformation.getRowGroup(currentStateId) << " to state " << currentStateId << ".");
197 
198  // Initialize the bounds, because some of the following computations depend on the values to be available for
199  // all states that have been assigned to a row-group.
200  bounds.initializeBoundsForNextState();
201 
202  // Before generating the behavior of the state, we need to determine whether it's a target state that
203  // does not need to be expanded.
204  stateGeneration.load(currentState);
205  if (stateGeneration.isTargetState()) {
206  ++stats.numberOfTargetStates;
207  isTargetState = true;
208  isTerminalState = true;
209  } else if (stateGeneration.isConditionState()) {
210  STORM_LOG_TRACE("Exploring state.");
211 
212  // If it needs to be expanded, we use the generator to retrieve the behavior of the new state.
213  storm::generator::StateBehavior<ValueType, StateType> behavior = stateGeneration.expand();
214  STORM_LOG_TRACE("State has " << behavior.getNumberOfChoices() << " choices.");
215 
216  // Clumsily check whether we have found a state that forms a trivial BMEC.
217  bool otherSuccessor = false;
218  for (auto const& choice : behavior) {
219  for (auto const& entry : choice) {
220  if (entry.first != currentStateId) {
221  otherSuccessor = true;
222  break;
223  }
224  }
225  }
226  isTerminalState = !otherSuccessor;
227 
228  // If the state was neither a trivial (non-accepting) terminal state nor a target state, we
229  // need to store its behavior.
230  if (!isTerminalState) {
231  // Next, we insert the behavior into our matrix structure.
232  StateType startAction = explorationInformation.getActionCount();
233  explorationInformation.addActionsToMatrix(behavior.getNumberOfChoices());
234 
235  ActionType localAction = 0;
236 
237  // Retrieve the lowest state bounds (wrt. to the current optimization direction).
238  std::pair<ValueType, ValueType> stateBounds = getLowestBounds(explorationInformation.getOptimizationDirection());
239 
240  for (auto const& choice : behavior) {
241  for (auto const& entry : choice) {
242  explorationInformation.getRowOfMatrix(startAction + localAction).emplace_back(entry.first, entry.second);
243  STORM_LOG_TRACE("Found transition " << currentStateId << "-[" << (startAction + localAction) << ", " << entry.second << "]-> " << entry.first << ".");
244  }
245 
246  std::pair<ValueType, ValueType> actionBounds = computeBoundsOfAction(startAction + localAction, explorationInformation, bounds);
247  bounds.initializeBoundsForNextAction(actionBounds);
248  stateBounds = combineBounds(explorationInformation.getOptimizationDirection(), stateBounds, actionBounds);
249 
250  STORM_LOG_TRACE("Initializing bounds of action " << (startAction + localAction) << " to " << bounds.getLowerBoundForAction(startAction + localAction) << " and " << bounds.getUpperBoundForAction(startAction + localAction) << ".");
251 
252  ++localAction;
253  }
254 
255  // Terminate the row group.
256  explorationInformation.terminateCurrentRowGroup();
257 
258  bounds.setBoundsForState(currentStateId, explorationInformation, stateBounds);
259  STORM_LOG_TRACE("Initializing bounds of state " << currentStateId << " to " << bounds.getLowerBoundForState(currentStateId, explorationInformation) << " and " << bounds.getUpperBoundForState(currentStateId, explorationInformation) << ".");
260  }
261  } else {
262  // In this case, the state is neither a target state nor a condition state and therefore a rejecting
263  // terminal state.
264  isTerminalState = true;
265  }
266 
267  if (isTerminalState) {
268  STORM_LOG_TRACE("State does not need to be explored, because it is " << (isTargetState ? "a target state" : "a rejecting terminal state") << ".");
269  explorationInformation.addTerminalState(currentStateId);
270 
271  if (isTargetState) {
272  bounds.setBoundsForState(currentStateId, explorationInformation, std::make_pair(storm::utility::one<ValueType>(), storm::utility::one<ValueType>()));
273  bounds.initializeBoundsForNextAction(std::make_pair(storm::utility::one<ValueType>(), storm::utility::one<ValueType>()));
274  } else {
275  bounds.setBoundsForState(currentStateId, explorationInformation, std::make_pair(storm::utility::zero<ValueType>(), storm::utility::zero<ValueType>()));
276  bounds.initializeBoundsForNextAction(std::make_pair(storm::utility::zero<ValueType>(), storm::utility::zero<ValueType>()));
277  }
278 
279  // Increase the size of the matrix, but leave the row empty.
280  explorationInformation.addActionsToMatrix(1);
281 
282  // Terminate the row group.
283  explorationInformation.newRowGroup();
284  }
285 
286  return isTerminalState;
287  }

289  template<typename ModelType, typename StateType>
290  typename SparseExplorationModelChecker<ModelType, StateType>::ActionType SparseExplorationModelChecker<ModelType, StateType>::sampleActionOfState(StateType const& currentStateId, ExplorationInformation<StateType, ValueType> const& explorationInformation, Bounds<StateType, ValueType>& bounds) const {
291  // Determine the values of all available actions.
292  std::vector<std::pair<ActionType, ValueType>> actionValues;
293  StateType rowGroup = explorationInformation.getRowGroup(currentStateId);
294 
295  // Check for cases in which we do not need to perform more work.
296  if (explorationInformation.onlyOneActionAvailable(rowGroup)) {
297  return explorationInformation.getStartRowOfGroup(rowGroup);
298  }
299 
300  // If there are more choices to consider, start by gathering the values of relevant actions.
301  STORM_LOG_TRACE("Sampling from actions leaving the state.");
302 
303  for (uint32_t row = explorationInformation.getStartRowOfGroup(rowGroup); row < explorationInformation.getStartRowOfGroup(rowGroup + 1); ++row) {
304  actionValues.push_back(std::make_pair(row, bounds.getBoundForAction(explorationInformation.getOptimizationDirection(), row)));
305  }
306 
307  STORM_LOG_ASSERT(!actionValues.empty(), "Values for actions must not be empty.");
308 
309  // Sort the actions wrt. to the optimization direction.
310  if (explorationInformation.maximize()) {
311  std::sort(actionValues.begin(), actionValues.end(), [] (std::pair<ActionType, ValueType> const& a, std::pair<ActionType, ValueType> const& b) { return a.second > b.second; } );
312  } else {
313  std::sort(actionValues.begin(), actionValues.end(), [] (std::pair<ActionType, ValueType> const& a, std::pair<ActionType, ValueType> const& b) { return a.second < b.second; } );
314  }
315 
316  // Determine the first elements of the sorted range that agree on their value.
317  auto end = ++actionValues.begin();
318  while (end != actionValues.end() && comparator.isEqual(actionValues.begin()->second, end->second)) {
319  ++end;
320  }
321 
322  // Now sample from all maximizing actions.
323  std::uniform_int_distribution<ActionType> distribution(0, std::distance(actionValues.begin(), end) - 1);
324  return actionValues[distribution(randomGenerator)].first;
325  }

327  template<typename ModelType, typename StateType>
328  StateType SparseExplorationModelChecker<ModelType, StateType>::sampleSuccessorFromAction(ActionType const& chosenAction, ExplorationInformation<StateType, ValueType> const& explorationInformation, Bounds<StateType, ValueType> const& bounds) const {
329  std::vector<storm::storage::MatrixEntry<StateType, ValueType>> const& row = explorationInformation.getRowOfMatrix(chosenAction);
330  if (row.size() == 1) {
331  return row.front().getColumn();
332  }
333 
334  // Depending on the selected next-state heuristic, we give the states other likelihoods of getting chosen.
335  if (explorationInformation.useDifferenceProbabilitySumHeuristic() || explorationInformation.useProbabilityHeuristic()) {
336  std::vector<ValueType> probabilities(row.size());
337  if (explorationInformation.useDifferenceProbabilitySumHeuristic()) {
338  std::transform(row.begin(), row.end(), probabilities.begin(),
339  [&bounds, &explorationInformation] (storm::storage::MatrixEntry<StateType, ValueType> const& entry) {
340  return entry.getValue() + bounds.getDifferenceOfStateBounds(entry.getColumn(), explorationInformation);
341  });
342  } else if (explorationInformation.useProbabilityHeuristic()) {
343  std::transform(row.begin(), row.end(), probabilities.begin(),
344  [] (storm::storage::MatrixEntry<StateType, ValueType> const& entry) {
345  return entry.getValue();
346  });
347  }
348 
349  // Now sample according to the probabilities.
350  std::discrete_distribution<StateType> distribution(probabilities.begin(), probabilities.end());
351  return row[distribution(randomGenerator)].getColumn();
352  } else {
353  STORM_LOG_ASSERT(explorationInformation.useUniformHeuristic(), "Illegal next-state heuristic.");
354  std::uniform_int_distribution<ActionType> distribution(0, row.size() - 1);
355  return row[distribution(randomGenerator)].getColumn();
356  }

359  template<typename ModelType, typename StateType>
360  bool SparseExplorationModelChecker<ModelType, StateType>::performPrecomputation(StateActionStack const& stack, ExplorationInformation<StateType, ValueType>& explorationInformation, Bounds<StateType, ValueType>& bounds, Statistics<StateType, ValueType>& stats) const {
361  ++stats.numberOfPrecomputations;
362 
363  // Outline:
364  // 1. construct a sparse transition matrix of the relevant part of the state space.
365  // 2. use this matrix to compute states with probability 0/1 and an MEC decomposition (in the max case).
366  // 3. use MEC decomposition to collapse MECs.
367  STORM_LOG_TRACE("Starting " << (explorationInformation.useLocalPrecomputation() ? "local" : "global") << " precomputation.");
368 
369  // Construct the matrix that represents the fragment of the system contained in the currently sampled path.
370  storm::storage::SparseMatrixBuilder<ValueType> builder(0, 0, 0, false, true, 0);
371 
372  // Determine the set of states that was expanded.
373  std::vector<StateType> relevantStates;
374  if (explorationInformation.useLocalPrecomputation()) {
375  for (auto const& stateActionPair : stack) {
376  if (explorationInformation.maximize() || !storm::utility::isOne(bounds.getLowerBoundForState(stateActionPair.first, explorationInformation))) {
377  relevantStates.push_back(stateActionPair.first);
378  }
379  }
380  std::sort(relevantStates.begin(), relevantStates.end());
381  auto newEnd = std::unique(relevantStates.begin(), relevantStates.end());
382  relevantStates.resize(std::distance(relevantStates.begin(), newEnd));
383  } else {
384  for (StateType state = 0; state < explorationInformation.getNumberOfDiscoveredStates(); ++state) {
385  // Add the state to the relevant states if they are not unexplored.
386  if (!explorationInformation.isUnexplored(state)) {
387  relevantStates.push_back(state);
388  }
389  }
390  }
391  StateType sink = relevantStates.size();
392 
393  // Create a mapping for faster look-up during the translation of flexible matrix to the real sparse matrix.
394  // While doing so, record all target states.
395  std::unordered_map<StateType, StateType> relevantStateToNewRowGroupMapping;
396  storm::storage::BitVector targetStates(sink + 1);
397  for (StateType index = 0; index < relevantStates.size(); ++index) {
398  relevantStateToNewRowGroupMapping.emplace(relevantStates[index], index);
399  if (storm::utility::isOne(bounds.getLowerBoundForState(relevantStates[index], explorationInformation))) {
400  targetStates.set(index);
401  }
402  }
403 
404  // Do the actual translation.
405  StateType currentRow = 0;
406  for (auto const& state : relevantStates) {
407  builder.newRowGroup(currentRow);
408  StateType rowGroup = explorationInformation.getRowGroup(state);
409  for (auto row = explorationInformation.getStartRowOfGroup(rowGroup); row < explorationInformation.getStartRowOfGroup(rowGroup + 1); ++row) {
410  ValueType unexpandedProbability = storm::utility::zero<ValueType>();
411  for (auto const& entry : explorationInformation.getRowOfMatrix(row)) {
412  auto it = relevantStateToNewRowGroupMapping.find(entry.getColumn());
413  if (it != relevantStateToNewRowGroupMapping.end()) {
414  // If the entry is a relevant state, we copy it over (and compensate for the offset change).
415  builder.addNextValue(currentRow, it->second, entry.getValue());
416  } else {
417  // If the entry is an unexpanded state, we gather the probability to later redirect it to an unexpanded sink.
418  unexpandedProbability += entry.getValue();
419  }
420  }
421  if (unexpandedProbability != storm::utility::zero<ValueType>()) {
422  builder.addNextValue(currentRow, sink, unexpandedProbability);
423  }
424  ++currentRow;
425  }
426  }
427  // Then, make the unexpanded state absorbing.
428  builder.newRowGroup(currentRow);
429  builder.addNextValue(currentRow, sink, storm::utility::one<ValueType>());
430  storm::storage::SparseMatrix<ValueType> relevantStatesMatrix = builder.build();
431  storm::storage::SparseMatrix<ValueType> transposedMatrix = relevantStatesMatrix.transpose(true);
432  STORM_LOG_TRACE("Successfully built matrix for precomputation.");
433 
434  storm::storage::BitVector allStates(sink + 1, true);
435  storm::storage::BitVector statesWithProbability0;
436  storm::storage::BitVector statesWithProbability1;
437  if (explorationInformation.maximize()) {
438  // If we are computing maximal probabilities, we first perform a detection of states that have
439  // probability 01 and then additionally perform an MEC decomposition. The reason for this somewhat
440  // duplicate work is the following. Optimally, we would only do the MEC decomposition, because we need
441  // it anyway. However, when only detecting (accepting) MECs, we do not infer which of the other states
442  // (not contained in MECs) also have probability 0/1.
443  targetStates.set(sink, true);
444  statesWithProbability0 = storm::utility::graph::performProb0A(transposedMatrix, allStates, targetStates);
445  targetStates.set(sink, false);
446  statesWithProbability1 = storm::utility::graph::performProb1E(relevantStatesMatrix, relevantStatesMatrix.getRowGroupIndices(), transposedMatrix, allStates, targetStates);
447 
448  storm::storage::MaximalEndComponentDecomposition<ValueType> mecDecomposition(relevantStatesMatrix, relevantStatesMatrix.transpose(true));
449  ++stats.ecDetections;
450  STORM_LOG_TRACE("Successfully computed MEC decomposition. Found " << (mecDecomposition.size() > 1 ? (mecDecomposition.size() - 1) : 0) << " MEC(s).");
451 
452  // If the decomposition contains only the MEC consisting of the sink state, we count it as 'failed'.
453  STORM_LOG_ASSERT(mecDecomposition.size() > 0, "Expected at least one MEC (the trivial sink MEC).");
454  if (mecDecomposition.size() == 1) {
455  ++stats.failedEcDetections;
456  } else {
457  stats.totalNumberOfEcDetected += mecDecomposition.size() - 1;
458 
459  // 3. Analyze the MEC decomposition.
460  for (auto const& mec : mecDecomposition) {
461  // Ignore the (expected) MEC of the sink state.
462  if (mec.containsState(sink)) {
463  continue;
464  }
465 
466  collapseMec(mec, relevantStates, relevantStatesMatrix, explorationInformation, bounds);
467  }
468  }
469  } else {
470  // If we are computing minimal probabilities, we do not need to perform an EC-detection. We rather
471  // compute all states (of the considered fragment) that have probability 0/1. For states with
472  // probability 0, we have to mark the sink as being a target. For states with probability 1, however,
473  // we must treat the sink as being rejecting.
474  targetStates.set(sink, true);
475  statesWithProbability0 = storm::utility::graph::performProb0E(relevantStatesMatrix, relevantStatesMatrix.getRowGroupIndices(), transposedMatrix, allStates, targetStates);
476  targetStates.set(sink, false);
477  statesWithProbability1 = storm::utility::graph::performProb1A(relevantStatesMatrix, relevantStatesMatrix.getRowGroupIndices(), transposedMatrix, allStates, targetStates);
478  }
479 
480  // Set the bounds of the identified states.
481  STORM_LOG_ASSERT((statesWithProbability0 & statesWithProbability1).empty(), "States with probability 0 and 1 overlap.");
482  for (auto state : statesWithProbability0) {
483  // Skip the sink state as it is not contained in the original system.
484  if (state == sink) {
485  continue;
486  }
487 
488  StateType originalState = relevantStates[state];
489  bounds.setUpperBoundForState(originalState, explorationInformation, storm::utility::zero<ValueType>());
490  explorationInformation.addTerminalState(originalState);
491  }
492  for (auto state : statesWithProbability1) {
493  // Skip the sink state as it is not contained in the original system.
494  if (state == sink) {
495  continue;
496  }
497 
498  StateType originalState = relevantStates[state];
499  bounds.setLowerBoundForState(originalState, explorationInformation, storm::utility::one<ValueType>());
500  explorationInformation.addTerminalState(originalState);
501  }

505  template<typename ModelType, typename StateType>
506  void SparseExplorationModelChecker<ModelType, StateType>::collapseMec(storm::storage::MaximalEndComponent const& mec, std::vector<StateType> const& relevantStates, storm::storage::SparseMatrix<ValueType> const& relevantStatesMatrix, ExplorationInformation<StateType, ValueType>& explorationInformation, Bounds<StateType, ValueType>& bounds) const {
507  bool containsTargetState = false;
508 
509  // Now we record all actions leaving the EC.
510  std::vector<ActionType> leavingActions;
511  for (auto const& stateAndChoices : mec) {
512  // Compute the state of the original model that corresponds to the current state.
513  StateType originalState = relevantStates[stateAndChoices.first];
514  StateType originalRowGroup = explorationInformation.getRowGroup(originalState);
515 
516  // Check whether a target state is contained in the MEC.
517  if (!containsTargetState && storm::utility::isOne(bounds.getLowerBoundForRowGroup(originalRowGroup))) {
518  containsTargetState = true;
519  }
520 
521  // For each state, compute the actions that leave the MEC.
522  auto includedChoicesIt = stateAndChoices.second.begin();
523  auto includedChoicesIte = stateAndChoices.second.end();
524  for (auto action = explorationInformation.getStartRowOfGroup(originalRowGroup); action < explorationInformation.getStartRowOfGroup(originalRowGroup + 1); ++action) {
525  if (includedChoicesIt != includedChoicesIte) {
526  STORM_LOG_TRACE("Next (local) choice contained in MEC is " << (*includedChoicesIt - relevantStatesMatrix.getRowGroupIndices()[stateAndChoices.first]));
527  STORM_LOG_TRACE("Current (local) choice iterated is " << (action - explorationInformation.getStartRowOfGroup(originalRowGroup)));
528  if (action - explorationInformation.getStartRowOfGroup(originalRowGroup) != *includedChoicesIt - relevantStatesMatrix.getRowGroupIndices()[stateAndChoices.first]) {
529  STORM_LOG_TRACE("Choice leaves the EC.");
530  leavingActions.push_back(action);
531  } else {
532  STORM_LOG_TRACE("Choice stays in the EC.");
533  ++includedChoicesIt;
534  }
535  } else {
536  STORM_LOG_TRACE("Choice leaves the EC, because there is no more choice staying in the EC.");
537  leavingActions.push_back(action);
538  }
539  }
540  }
541 
542  // Now, we need to collapse the EC only if it does not contain a target state and the leaving actions are
543  // non-empty, because only then have the states a (potentially) non-zero, non-one probability.
544  if (!containsTargetState && !leavingActions.empty()) {
545  // In this case, no target state is contained in the MEC, but there are actions leaving the MEC. To
546  // prevent the simulation getting stuck in this MEC again, we replace all states in the MEC by a new
547  // state whose outgoing actions are the ones leaving the MEC. We do this, by assigning all states in the
548  // MEC to a new row group, which will then hold all the outgoing choices.
549 
550  // Remap all contained states to the new row group.
551  StateType nextRowGroup = explorationInformation.getNextRowGroup();
552  for (auto const& stateAndChoices : mec) {
553  explorationInformation.assignStateToRowGroup(stateAndChoices.first, nextRowGroup);
554  }
555 
556  bounds.initializeBoundsForNextState();
557 
558  // Add to the new row group all leaving actions of contained states and set the appropriate bounds for
559  // the actions and the new state.
560  std::pair<ValueType, ValueType> stateBounds = getLowestBounds(explorationInformation.getOptimizationDirection());
561  for (auto const& action : leavingActions) {
562  explorationInformation.moveActionToBackOfMatrix(action);
563  std::pair<ValueType, ValueType> const& actionBounds = bounds.getBoundsForAction(action);
564  bounds.initializeBoundsForNextAction(actionBounds);
565  stateBounds = combineBounds(explorationInformation.getOptimizationDirection(), stateBounds, actionBounds);
566  }
567  bounds.setBoundsForRowGroup(nextRowGroup, stateBounds);
568 
569  // Terminate the row group of the newly introduced state.
570  explorationInformation.terminateCurrentRowGroup();
571  }
572  }

623  template<typename ModelType, typename StateType>
624  void SparseExplorationModelChecker<ModelType, StateType>::updateProbabilityOfAction(StateType const& state, ActionType const& action, ExplorationInformation<StateType, ValueType> const& explorationInformation, Bounds<StateType, ValueType>& bounds) const {
625  // Compute the new lower/upper values of the action.
626  std::pair<ValueType, ValueType> newBoundsForAction = computeBoundsOfAction(action, explorationInformation, bounds);
627 
628  // And set them as the current value.
629  bounds.setBoundsForAction(action, newBoundsForAction);
630 
631  // Check if we need to update the values for the states.
632  if (explorationInformation.maximize()) {
633  bounds.setLowerBoundOfStateIfGreaterThanOld(state, explorationInformation, newBoundsForAction.first);
634 
635  StateType rowGroup = explorationInformation.getRowGroup(state);
636  if (newBoundsForAction.second < bounds.getUpperBoundForRowGroup(rowGroup)) {
637  if (explorationInformation.getRowGroupSize(rowGroup) > 1) {
638  newBoundsForAction.second = std::max(newBoundsForAction.second, computeBoundOverAllOtherActions(storm::OptimizationDirection::Maximize, state, action, explorationInformation, bounds));
639  }
640 
641  bounds.setUpperBoundForRowGroup(rowGroup, newBoundsForAction.second);
642  }
643  } else {
644  bounds.setUpperBoundOfStateIfLessThanOld(state, explorationInformation, newBoundsForAction.second);
645 
646  StateType rowGroup = explorationInformation.getRowGroup(state);
647  if (bounds.getLowerBoundForRowGroup(rowGroup) < newBoundsForAction.first) {
648  if (explorationInformation.getRowGroupSize(rowGroup) > 1) {
649  ValueType min = computeBoundOverAllOtherActions(storm::OptimizationDirection::Minimize, state, action, explorationInformation, bounds);
650  newBoundsForAction.first = std::min(newBoundsForAction.first, min);
651  }
652 
653  bounds.setLowerBoundForRowGroup(rowGroup, newBoundsForAction.first);
654  }
655  }
656  }

658  template<typename ModelType, typename StateType>
659  typename ModelType::ValueType SparseExplorationModelChecker<ModelType, StateType>::computeBoundOverAllOtherActions(storm::OptimizationDirection const& direction, StateType const& state, ActionType const& action, ExplorationInformation<StateType, ValueType> const& explorationInformation, Bounds<StateType, ValueType> const& bounds) const {
660  ValueType bound = getLowestBound(explorationInformation.getOptimizationDirection());
661 
662  ActionType group = explorationInformation.getRowGroup(state);
663  for (auto currentAction = explorationInformation.getStartRowOfGroup(group); currentAction < explorationInformation.getStartRowOfGroup(group + 1); ++currentAction) {
664  if (currentAction == action) {
665  continue;
666  }
667 
668  if (direction == storm::OptimizationDirection::Maximize) {
669  bound = std::max(bound, computeUpperBoundOfAction(currentAction, explorationInformation, bounds));
670  } else {
671  bound = std::min(bound, computeLowerBoundOfAction(currentAction, explorationInformation, bounds));
672  }
673  }
674  return bound;
675  }
