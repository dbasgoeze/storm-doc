59  template<storm::dd::DdType Type, typename ValueType>
60  boost::any AddExpressionAdapter<Type, ValueType>::visit(storm::expressions::BinaryBooleanFunctionExpression const& expression, boost::any const& data) {
61  storm::dd::Bdd<Type> leftResult = boost::any_cast<storm::dd::Bdd<Type>>(expression.getFirstOperand()->accept(*this, data));
62  storm::dd::Bdd<Type> rightResult = boost::any_cast<storm::dd::Bdd<Type>>(expression.getSecondOperand()->accept(*this, data));
63 
64  storm::dd::Bdd<Type> result;
65  switch (expression.getOperatorType()) {
66  case storm::expressions::BinaryBooleanFunctionExpression::OperatorType::And:
67  result = (leftResult && rightResult);
68  break;
69  case storm::expressions::BinaryBooleanFunctionExpression::OperatorType::Or:
70  result = (leftResult || rightResult);
71  break;
72  case storm::expressions::BinaryBooleanFunctionExpression::OperatorType::Iff:
73  result = (leftResult.iff(rightResult));
74  break;
75  case storm::expressions::BinaryBooleanFunctionExpression::OperatorType::Implies:
76  result = (!leftResult || rightResult);
77  break;
78  case storm::expressions::BinaryBooleanFunctionExpression::OperatorType::Xor:
79  result = (leftResult.exclusiveOr(rightResult));
80  break;
81  }
82 
83  return result;
