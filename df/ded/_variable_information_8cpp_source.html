117  void VariableInformation::registerArrayVariableReplacements(storm::jani::ArrayEliminatorData const& arrayEliminatorData) {
118  arrayVariableToElementInformations.clear();
119  // Find for each replaced array variable the corresponding references in this variable information
120  for (auto const& arrayVariable : arrayEliminatorData.eliminatedArrayVariables) {
121  if (!arrayVariable->isTransient()) {
122  STORM_LOG_ASSERT(arrayEliminatorData.replacements.count(arrayVariable->getExpressionVariable()) > 0, "No replacement for array variable.");
123  auto const& replacements = arrayEliminatorData.replacements.find(arrayVariable->getExpressionVariable())->second;
124  std::vector<uint64_t> varInfoIndices;
125  for (auto const& replacedVar : replacements) {
126  if (replacedVar->getExpressionVariable().hasIntegerType()) {
127  uint64_t index = 0;
128  for (auto const& intInfo : integerVariables) {
129  if (intInfo.variable == replacedVar->getExpressionVariable()) {
130  varInfoIndices.push_back(index);
131  break;
132  }
133  ++index;
134  }
135  STORM_LOG_ASSERT(!varInfoIndices.empty() && varInfoIndices.back() == index, "Could not find a basic variable for replacement of array variable " << replacedVar->getExpressionVariable().getName() << " .");
136  } else if (replacedVar->getExpressionVariable().hasBooleanType()) {
137  uint64_t index = 0;
138  for (auto const& boolInfo : booleanVariables) {
139  if (boolInfo.variable == replacedVar->getExpressionVariable()) {
140  varInfoIndices.push_back(index);
141  break;
142  }
143  ++index;
144  }
145  STORM_LOG_ASSERT(!varInfoIndices.empty() && varInfoIndices.back() == index, "Could not find a basic variable for replacement of array variable " << replacedVar->getExpressionVariable().getName() << " .");
146  } else {
147  STORM_LOG_ASSERT(false, "Unhandled type of base variable.");
148  }
149  }
150  this->arrayVariableToElementInformations.emplace(arrayVariable->getExpressionVariable(), std::move(varInfoIndices));
151  }

175  void VariableInformation::createVariablesForVariableSet(storm::jani::VariableSet const& variableSet, uint64_t reservedBitsForUnboundedVariables, bool global) {
176  for (auto const& variable : variableSet.getBooleanVariables()) {
177  if (!variable.isTransient()) {
178  booleanVariables.emplace_back(variable.getExpressionVariable(), totalBitOffset, global, true);
179  ++totalBitOffset;
180  }
181  }
182  for (auto const& variable : variableSet.getBoundedIntegerVariables()) {
183  if (!variable.isTransient()) {
184  int64_t lowerBound;
185  int64_t upperBound;
186  if (variable.hasLowerBound()) {
187  lowerBound = variable.getLowerBound().evaluateAsInt();
188  if (variable.hasUpperBound()) {
189  upperBound = variable.getUpperBound().evaluateAsInt();
190  } else {
191  upperBound = lowerBound + ((1ll << reservedBitsForUnboundedVariables) - 1);
192  }
193  } else {
194  STORM_LOG_THROW(variable.hasUpperBound(), storm::exceptions::WrongFormatException, "Bounded integer variable has neither a lower nor an upper bound.");
195  upperBound = variable.getUpperBound().evaluateAsInt();
196  lowerBound = upperBound - ((1ll << reservedBitsForUnboundedVariables) - 1);
197  }
198  uint_fast64_t bitwidth = static_cast<uint_fast64_t>(std::ceil(std::log2(upperBound - lowerBound + 1)));
199  integerVariables.emplace_back(variable.getExpressionVariable(), lowerBound, upperBound, totalBitOffset, bitwidth, global, true, !variable.hasLowerBound() || !variable.hasUpperBound());
200  totalBitOffset += bitwidth;
201  }
202  }
203  for (auto const& variable : variableSet.getUnboundedIntegerVariables()) {
204  if (!variable.isTransient()) {
205  int64_t lowerBound = -(1ll << (reservedBitsForUnboundedVariables - 1));
206  int64_t upperBound = (1ll << (reservedBitsForUnboundedVariables - 1)) - 1;
207  integerVariables.emplace_back(variable.getExpressionVariable(), lowerBound, upperBound, totalBitOffset, reservedBitsForUnboundedVariables, global, true, true);
208  totalBitOffset += reservedBitsForUnboundedVariables;
209  }
210  }
211  }
