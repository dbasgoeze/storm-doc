127  template <typename ValueType, typename RewardModelType>
128  void SparseModelMemoryProduct<ValueType, RewardModelType>::computeReachableStates(storm::storage::BitVector const& initialStates) {
129  // Explore the reachable states via DFS.
130  // A state s on the stack corresponds to the model state (s / memoryStateCount) and memory state (s % memoryStateCount)
131  reachableStates |= initialStates;
132  if (!reachableStates.full()) {
133  std::vector<uint64_t> stack(reachableStates.begin(), reachableStates.end());
134  while (!stack.empty()) {
135  uint64_t stateIndex = stack.back();
136  stack.pop_back();
137  uint64_t modelState = stateIndex / memoryStateCount;
138  uint64_t memoryState = stateIndex % memoryStateCount;
139 
140  if (scheduler) {
141  auto choices = scheduler->getChoice(modelState, memoryState).getChoiceAsDistribution();
142  uint64_t groupStart = model.getTransitionMatrix().getRowGroupIndices()[modelState];
143  for (auto const& choice : choices) {
144  STORM_LOG_ASSERT(groupStart + choice.first < model.getTransitionMatrix().getRowGroupIndices()[modelState + 1], "Invalid choice " << choice.first << " at model state " << modelState << ".");
145  auto const& row = model.getTransitionMatrix().getRow(groupStart + choice.first);
146  for (auto modelTransitionIt = row.begin(); modelTransitionIt != row.end(); ++modelTransitionIt) {
147  if (!storm::utility::isZero(modelTransitionIt->getValue())) {
148  uint64_t successorModelState = modelTransitionIt->getColumn();
149  uint64_t modelTransitionId = modelTransitionIt - model.getTransitionMatrix().begin();
150  uint64_t successorMemoryState = memorySuccessors[modelTransitionId * memoryStateCount + memoryState];
151  uint64_t successorStateIndex = successorModelState * memoryStateCount + successorMemoryState;
152  if (!reachableStates.get(successorStateIndex)) {
153  reachableStates.set(successorStateIndex, true);
154  stack.push_back(successorStateIndex);
155  }
156  }
157  }
158  }
159  } else {
160  auto const& rowGroup = model.getTransitionMatrix().getRowGroup(modelState);
161  for (auto modelTransitionIt = rowGroup.begin(); modelTransitionIt != rowGroup.end(); ++modelTransitionIt) {
162  if (!storm::utility::isZero(modelTransitionIt->getValue())) {
163  uint64_t successorModelState = modelTransitionIt->getColumn();
164  uint64_t modelTransitionId = modelTransitionIt - model.getTransitionMatrix().begin();
165  uint64_t successorMemoryState = memorySuccessors[modelTransitionId * memoryStateCount + memoryState];
166  uint64_t successorStateIndex = successorModelState * memoryStateCount + successorMemoryState;
167  if (!reachableStates.get(successorStateIndex)) {
168  reachableStates.set(successorStateIndex, true);
169  stack.push_back(successorStateIndex);
170  }
171  }
172  }
173  }
174  }
175  }

237  storm::storage::SparseMatrix<ValueType> SparseModelMemoryProduct<ValueType, RewardModelType>::buildTransitionMatrixForScheduler() {
238  uint64_t numResStates = reachableStates.getNumberOfSetBits();
239  uint64_t numResChoices = 0;
240  uint64_t numResTransitions = 0;
241  bool hasTrivialNondeterminism = true;
242  for (auto const& stateIndex : reachableStates) {
243  uint64_t modelState = stateIndex / memoryStateCount;
244  uint64_t memoryState = stateIndex % memoryStateCount;
245  storm::storage::SchedulerChoice<ValueType> choice = scheduler->getChoice(modelState, memoryState);
246  if (choice.isDefined()) {
247  ++numResChoices;
248  if (choice.isDeterministic()) {
249  uint64_t modelRow = model.getTransitionMatrix().getRowGroupIndices()[modelState] + choice.getDeterministicChoice();
250  numResTransitions += model.getTransitionMatrix().getRow(modelRow).getNumberOfEntries();
251  } else {
252  std::set<uint64_t> successors;
253  for (auto const& choiceIndex : choice.getChoiceAsDistribution()) {
254  if (!storm::utility::isZero(choiceIndex.second)) {
255  uint64_t modelRow = model.getTransitionMatrix().getRowGroupIndices()[modelState] + choiceIndex.first;
256  for (auto const& entry : model.getTransitionMatrix().getRow(modelRow)) {
257  successors.insert(entry.getColumn());
258  }
259  }
260  }
261  numResTransitions += successors.size();
262  }
263  } else {
264  uint64_t modelRow = model.getTransitionMatrix().getRowGroupIndices()[modelState];
265  uint64_t groupEnd = model.getTransitionMatrix().getRowGroupIndices()[modelState + 1];
266  hasTrivialNondeterminism = hasTrivialNondeterminism && (groupEnd == modelRow + 1);
267  for (; modelRow < groupEnd; ++modelRow) {
268  ++numResChoices;
269  numResTransitions += model.getTransitionMatrix().getRow(modelRow).getNumberOfEntries();
270  }
271  }
272  }
273 
274  storm::storage::SparseMatrixBuilder<ValueType> builder(numResChoices, numResStates, numResTransitions, true, !hasTrivialNondeterminism, hasTrivialNondeterminism ? 0 : numResStates);
275  uint64_t currentRow = 0;
276  for (auto const& stateIndex : reachableStates) {
277  uint64_t modelState = stateIndex / memoryStateCount;
278  uint64_t memoryState = stateIndex % memoryStateCount;
279  if (!hasTrivialNondeterminism) {
280  builder.newRowGroup(currentRow);
281  }
282  storm::storage::SchedulerChoice<ValueType> choice = scheduler->getChoice(modelState, memoryState);
283  if (choice.isDefined()) {
284  if (choice.isDeterministic()) {
285  uint64_t modelRowIndex = model.getTransitionMatrix().getRowGroupIndices()[modelState] + choice.getDeterministicChoice();
286  auto const& modelRow = model.getTransitionMatrix().getRow(modelRowIndex);
287  for (auto entryIt = modelRow.begin(); entryIt != modelRow.end(); ++entryIt) {
288  uint64_t transitionId = entryIt - model.getTransitionMatrix().begin();
289  uint64_t successorMemoryState = memorySuccessors[transitionId * memoryStateCount + memoryState];
290  builder.addNextValue(currentRow, getResultState(entryIt->getColumn(), successorMemoryState), entryIt->getValue());
291  }
292  } else {
293  std::map<uint64_t, ValueType> transitions;
294  for (auto const& choiceIndex : choice.getChoiceAsDistribution()) {
295  if (!storm::utility::isZero(choiceIndex.second)) {
296  uint64_t modelRowIndex = model.getTransitionMatrix().getRowGroupIndices()[modelState] + choiceIndex.first;
297  auto const& modelRow = model.getTransitionMatrix().getRow(modelRowIndex);
298  for (auto entryIt = modelRow.begin(); entryIt != modelRow.end(); ++entryIt) {
299  uint64_t transitionId = entryIt - model.getTransitionMatrix().begin();
300  uint64_t successorMemoryState = memorySuccessors[transitionId * memoryStateCount + memoryState];
301  ValueType transitionValue = choiceIndex.second * entryIt->getValue();
302  auto insertionRes = transitions.insert(std::make_pair(getResultState(entryIt->getColumn(), successorMemoryState), transitionValue));
303  if (!insertionRes.second) {
304  insertionRes.first->second += transitionValue;
305  }
306  }
307  }
308  }
309  for (auto const& transition : transitions) {
310  builder.addNextValue(currentRow, transition.first, transition.second);
311  }
312  }
313  ++currentRow;
314  } else {
315  for (uint64_t modelRowIndex = model.getTransitionMatrix().getRowGroupIndices()[modelState]; modelRowIndex < model.getTransitionMatrix().getRowGroupIndices()[modelState + 1]; ++modelRowIndex) {
316  auto const& modelRow = model.getTransitionMatrix().getRow(modelRowIndex);
317  for (auto entryIt = modelRow.begin(); entryIt != modelRow.end(); ++entryIt) {
318  uint64_t transitionId = entryIt - model.getTransitionMatrix().begin();
319  uint64_t successorMemoryState = memorySuccessors[transitionId * memoryStateCount + memoryState];
320  builder.addNextValue(currentRow, getResultState(entryIt->getColumn(), successorMemoryState), entryIt->getValue());
321  }
322  ++currentRow;
323  }
324  }
325  }
326 
327  return builder.build();
328  }

374  template <typename ValueType, typename RewardModelType>
375  std::unordered_map<std::string, RewardModelType> SparseModelMemoryProduct<ValueType, RewardModelType>::buildRewardModels(storm::storage::SparseMatrix<ValueType> const& resultTransitionMatrix) {
376 
377  typedef typename RewardModelType::ValueType RewardValueType;
378 
379  std::unordered_map<std::string, RewardModelType> result;
380  uint64_t numResStates = resultTransitionMatrix.getRowGroupCount();
381 
382  for (auto const& rewardModel : model.getRewardModels()) {
383  boost::optional<std::vector<RewardValueType>> stateRewards;
384  if (rewardModel.second.hasStateRewards()) {
385  stateRewards = std::vector<RewardValueType>(numResStates, storm::utility::zero<RewardValueType>());
386  uint64_t modelState = 0;
387  for (auto const& modelStateReward : rewardModel.second.getStateRewardVector()) {
388  if (!storm::utility::isZero(modelStateReward)) {
389  for (uint64_t memoryState = 0; memoryState < memoryStateCount; ++memoryState) {
390  if (isStateReachable(modelState, memoryState)) {
391  stateRewards.get()[getResultState(modelState, memoryState)] = modelStateReward;
392  }
393  }
394  }
395  ++modelState;
396  }
397  }
398  boost::optional<std::vector<RewardValueType>> stateActionRewards;
399  if (rewardModel.second.hasStateActionRewards()) {
400  stateActionRewards = std::vector<RewardValueType>(resultTransitionMatrix.getRowCount(), storm::utility::zero<RewardValueType>());
401  uint64_t modelState = 0;
402  uint64_t modelRow = 0;
403  for (auto const& modelStateActionReward : rewardModel.second.getStateActionRewardVector()) {
404  if (!storm::utility::isZero(modelStateActionReward)) {
405  while (modelRow >= model.getTransitionMatrix().getRowGroupIndices()[modelState + 1]) {
406  ++modelState;
407  }
408  uint64_t rowOffset = modelRow - model.getTransitionMatrix().getRowGroupIndices()[modelState];
409  for (uint64_t memoryState = 0; memoryState < memoryStateCount; ++memoryState) {
410  if (isStateReachable(modelState, memoryState)) {
411  if (scheduler && scheduler->getChoice(modelState, memoryState).isDefined()) {
412  ValueType factor = scheduler->getChoice(modelState, memoryState).getChoiceAsDistribution().getProbability(rowOffset);
413  stateActionRewards.get()[resultTransitionMatrix.getRowGroupIndices()[getResultState(modelState, memoryState)]] += factor * modelStateActionReward;
414  } else {
415  stateActionRewards.get()[resultTransitionMatrix.getRowGroupIndices()[getResultState(modelState, memoryState)] + rowOffset] = modelStateActionReward;
416  }
417  }
418  }
419  }
420  ++modelRow;
421  }
422  }
423  boost::optional<storm::storage::SparseMatrix<RewardValueType>> transitionRewards;
424  if (rewardModel.second.hasTransitionRewards()) {
425  storm::storage::SparseMatrixBuilder<RewardValueType> builder(resultTransitionMatrix.getRowCount(), resultTransitionMatrix.getColumnCount());
426  uint64_t stateIndex = 0;
427  for (auto const& resState : toResultStateMapping) {
428  if (resState < numResStates) {
429  uint64_t modelState = stateIndex / memoryStateCount;
430  uint64_t memoryState = stateIndex % memoryStateCount;
431  uint64_t rowGroupSize = resultTransitionMatrix.getRowGroupSize(resState);
432  if (scheduler && scheduler->getChoice(modelState, memoryState).isDefined()) {
433  std::map<uint64_t, RewardValueType> rewards;
434  for (uint64_t rowOffset = 0; rowOffset < rowGroupSize; ++rowOffset) {
435  uint64_t modelRowIndex = model.getTransitionMatrix().getRowGroupIndices()[modelState] + rowOffset;
436  auto transitionEntryIt = model.getTransitionMatrix().getRow(modelRowIndex).begin();
437  for (auto const& rewardEntry : rewardModel.second.getTransitionRewardMatrix().getRow(modelRowIndex)) {
438  while (transitionEntryIt->getColumn() != rewardEntry.getColumn()) {
439  STORM_LOG_ASSERT(transitionEntryIt != model.getTransitionMatrix().getRow(modelRowIndex).end(), "The reward transition matrix is not a submatrix of the model transition matrix.");
440  ++transitionEntryIt;
441  }
442  uint64_t transitionId = transitionEntryIt - model.getTransitionMatrix().begin();
443  uint64_t successorMemoryState = memorySuccessors[transitionId * memoryStateCount + memoryState];
444  auto insertionRes = rewards.insert(std::make_pair(getResultState(rewardEntry.getColumn(), successorMemoryState), rewardEntry.getValue()));
445  if (!insertionRes.second) {
446  insertionRes.first->second += rewardEntry.getValue();
447  }
448  }
449  }
450  uint64_t resRowIndex = resultTransitionMatrix.getRowGroupIndices()[resState];
451  for (auto& reward : rewards) {
452  builder.addNextValue(resRowIndex, reward.first, reward.second);
453  }
454  } else {
455  for (uint64_t rowOffset = 0; rowOffset < rowGroupSize; ++rowOffset) {
456  uint64_t resRowIndex = resultTransitionMatrix.getRowGroupIndices()[resState] + rowOffset;
457  uint64_t modelRowIndex = model.getTransitionMatrix().getRowGroupIndices()[modelState] + rowOffset;
458  auto transitionEntryIt = model.getTransitionMatrix().getRow(modelRowIndex).begin();
459  for (auto const& rewardEntry : rewardModel.second.getTransitionRewardMatrix().getRow(modelRowIndex)) {
460  while (transitionEntryIt->getColumn() != rewardEntry.getColumn()) {
461  STORM_LOG_ASSERT(transitionEntryIt != model.getTransitionMatrix().getRow(modelRowIndex).end(), "The reward transition matrix is not a submatrix of the model transition matrix.");
462  ++transitionEntryIt;
463  }
464  uint64_t transitionId = transitionEntryIt - model.getTransitionMatrix().begin();
465  uint64_t successorMemoryState = memorySuccessors[transitionId * memoryStateCount + memoryState];
466  builder.addNextValue(resRowIndex, getResultState(rewardEntry.getColumn(), successorMemoryState), rewardEntry.getValue());
467  }
468  }
469  }
470  }
471  ++stateIndex;
472  }

483 
484  if (model.isOfType(storm::models::ModelType::Ctmc)) {
485  components.rateTransitions = true;
486  } else if (model.isOfType(storm::models::ModelType::MarkovAutomaton)) {
487  // We also need to translate the exit rates and the Markovian states
488  uint64_t numResStates = components.transitionMatrix.getRowGroupCount();
489  std::vector<ValueType> resultExitRates;
490  resultExitRates.reserve(components.transitionMatrix.getRowGroupCount());

494 
495  uint64_t stateIndex = 0;
496  for (auto const& resState : toResultStateMapping) {
497  if (resState < numResStates) {
498  assert(resState == resultExitRates.size());
499  uint64_t modelState = stateIndex / memoryStateCount;
500  resultExitRates.push_back(modelExitRates[modelState]);
501  if (modelMarkovianStates.get(modelState)) {
502  resultMarkovianStates.set(resState, true);
503  }
504  }
505  ++stateIndex;
506  }
507  components.markovianStates = std::move(resultMarkovianStates);
508  components.exitRates = std::move(resultExitRates);
509  }
510 
511  storm::models::ModelType resultType = model.getType();
512  if (scheduler && !scheduler->isPartialScheduler()) {
513  if (model.isOfType(storm::models::ModelType::Mdp)) {
514  resultType = storm::models::ModelType::Dtmc;
515  }
516  // Note that converting deterministic MAs to CTMCs via state elimination does not preserve all properties (e.g. step bounded)
517  }
518
