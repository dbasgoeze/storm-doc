40  // Check if the check is necessary for the provided settings
41  if (!env.solver().isForceSoundness() || !env.solver().timeBounded().getRelativeTerminationCriterion()) {
42  // No need to update epsilon
43  return false;
44  }
45 
46  ValueType precision = storm::utility::convertNumber<ValueType>(env.solver().timeBounded().getPrecision());
47  // If we need to compute values with relative precision, it might be necessary to increase the precision requirements (epsilon)
48  ValueType newEpsilon = epsilon;
49  // Only consider positions that are relevant for the solve goal (e.g. initial states of the model) and are supposed to have a non-zero value
50  for (auto const& state : relevantPositions) {
51  if (storm::utility::isZero(resultVector[state])) {

57  }
58  }
59  }

61  STORM_LOG_INFO("Re-computing transient probabilities with new truncation error " << newEpsilon << " to guarantee sound results with relative precision.");

63  return true;
64  } else {

66  }

557  template <typename ValueType, typename std::enable_if<storm::NumberTraits<ValueType>::SupportsExponential, int>::type>
558  storm::storage::SparseMatrix<ValueType> SparseCtmcCslHelper::computeUniformizedMatrix(storm::storage::SparseMatrix<ValueType> const& rateMatrix, storm::storage::BitVector const& maybeStates, ValueType uniformizationRate, std::vector<ValueType> const& exitRates) {
559  STORM_LOG_DEBUG("Computing uniformized matrix using uniformization rate " << uniformizationRate << ".");
560  STORM_LOG_DEBUG("Keeping " << maybeStates.getNumberOfSetBits() << " rows.");
561 
562  // Create the submatrix that only contains the states with a positive probability (including the
563  // psi states) and reserve space for elements on the diagonal.
564  storm::storage::SparseMatrix<ValueType> uniformizedMatrix = rateMatrix.getSubmatrix(false, maybeStates, maybeStates, true);
565 
566  // Now we need to perform the actual uniformization. That is, all entries need to be divided by
567  // the uniformization rate, and the diagonal needs to be set to the negative exit rate of the
568  // state plus the self-loop rate and then increased by one.
569  uint_fast64_t currentRow = 0;
570  for (auto const& state : maybeStates) {
571  for (auto& element : uniformizedMatrix.getRow(currentRow)) {
572  if (element.getColumn() == currentRow) {
573  element.setValue((element.getValue() - exitRates[state]) / uniformizationRate + storm::utility::one<ValueType>());
574  } else {
575  element.setValue(element.getValue() / uniformizationRate);
576  }
577  }
578  ++currentRow;
579  }
580 
581  return uniformizedMatrix;
582  }

584  template<typename ValueType, bool useMixedPoissonProbabilities, typename std::enable_if<storm::NumberTraits<ValueType>::SupportsExponential, int>::type>
585  std::vector<ValueType> SparseCtmcCslHelper::computeTransientProbabilities(Environment const& env, storm::storage::SparseMatrix<ValueType> const& uniformizedMatrix, std::vector<ValueType> const* addVector, ValueType timeBound, ValueType uniformizationRate, std::vector<ValueType> values, ValueType epsilon) {
586  STORM_LOG_WARN_COND(epsilon > storm::utility::convertNumber<ValueType>(1e-20), "Very low truncation error " << epsilon << " requested. Numerical inaccuracies are possible.");
587  ValueType lambda = timeBound * uniformizationRate;
588 
589  // If no time can pass, the current values are the result.
590  if (storm::utility::isZero(lambda)) {
591  return values;
592  }
593 
594  // Use Fox-Glynn to get the truncation points and the weights.
595  storm::utility::numerical::FoxGlynnResult<ValueType> foxGlynnResult = storm::utility::numerical::foxGlynn(lambda, epsilon);
596  STORM_LOG_DEBUG("Fox-Glynn cutoff points: left=" << foxGlynnResult.left << ", right=" << foxGlynnResult.right);
597  // foxGlynnResult.weights do not sum up to one. This is to enhance numerical stability.
598 
599  // If the cumulative reward is to be computed, we need to adjust the weights.
600  if (useMixedPoissonProbabilities) {
601  ValueType sum = storm::utility::zero<ValueType>();
602 
603  for (auto& element : foxGlynnResult.weights) {
604  sum += element;
605  element = (foxGlynnResult.totalWeight - sum) / uniformizationRate;
606  }
607  }
608 
609  STORM_LOG_DEBUG("Starting iterations with " << uniformizedMatrix.getRowCount() << " x " << uniformizedMatrix.getColumnCount() << " matrix.");
610 
611  // Initialize result.
612  std::vector<ValueType> result;
613  uint_fast64_t startingIteration = foxGlynnResult.left;
614  if (startingIteration == 0) {
615  result = values;
616  storm::utility::vector::scaleVectorInPlace(result, foxGlynnResult.weights.front());
617  ++startingIteration;
618  } else {
619  if (useMixedPoissonProbabilities) {
620  result = std::vector<ValueType>(values.size());
621  std::function<ValueType (ValueType const&)> scaleWithUniformizationRate = [&uniformizationRate] (ValueType const& a) -> ValueType { return a / uniformizationRate; };
622  storm::utility::vector::applyPointwise(values, result, scaleWithUniformizationRate);
623  } else {
624  result = std::vector<ValueType>(values.size());
625  }
626  }
627 
628  auto multiplier = storm::solver::MultiplierFactory<ValueType>().create(env, uniformizedMatrix);
629  if (!useMixedPoissonProbabilities && foxGlynnResult.left > 1) {
630  // Perform the matrix-vector multiplications (without adding).
631  multiplier->repeatedMultiply(env, values, addVector, foxGlynnResult.left - 1);
632  } else if (useMixedPoissonProbabilities) {
633  std::function<ValueType(ValueType const&, ValueType const&)> addAndScale = [&uniformizationRate] (ValueType const& a, ValueType const& b) { return a + b / uniformizationRate; };
634 
635  // For the iterations below the left truncation point, we need to add and scale the result with the uniformization rate.
636  for (uint_fast64_t index = 1; index < startingIteration; ++index) {
637  multiplier->multiply(env, values, nullptr, values);
638  storm::utility::vector::applyPointwise(result, values, result, addAndScale);
639  }
640  // To make sure that the values obtained before the left truncation point have the same 'impact' on the total result as the values obtained
641  // between the left and right truncation point, we scale them here with the total sum of the weights.
642  // Note that we divide with this value afterwards. This is to improve numerical stability.
643  storm::utility::vector::scaleVectorInPlace<ValueType, ValueType>(result, foxGlynnResult.totalWeight);
644  }
645 
646  // For the indices that fall in between the truncation points, we need to perform the matrix-vector
647  // multiplication, scale and add the result.
648  ValueType weight = 0;
649  std::function<ValueType(ValueType const&, ValueType const&)> addAndScale = [&weight] (ValueType const& a, ValueType const& b) { return a + weight * b; };
650  for (uint_fast64_t index = startingIteration; index <= foxGlynnResult.right; ++index) {
651  multiplier->multiply(env, values, addVector, values);
652 
653  weight = foxGlynnResult.weights[index - foxGlynnResult.left];
654  storm::utility::vector::applyPointwise(result, values, result, addAndScale);
655  }
656 
657  // Finally, divide the result by the total weight
658  storm::utility::vector::scaleVectorInPlace<ValueType, ValueType>(result, storm::utility::one<ValueType>() / foxGlynnResult.totalWeight);
659  return result;
660  }
