90  void StandardMaPcaaWeightVectorChecker<SparseMaModelType>::boundedPhase(Environment const& env, std::vector<ValueType> const& weightVector, std::vector<ValueType>& weightedRewardVector) {
91 
92  // Split the preprocessed model into transitions from/to probabilistic/Markovian states.
93  SubModel MS = createSubModel(true, weightedRewardVector);
94  SubModel PS = createSubModel(false, weightedRewardVector);
95 
96  // Apply digitization to Markovian transitions
97  ValueType digitizationConstant = getDigitizationConstant(weightVector);
98  digitize(MS, digitizationConstant);
99 
100  // Get for each occurring (digitized) timeBound the indices of the objectives with that bound.
101  TimeBoundMap upperTimeBounds;
102  digitizeTimeBounds(upperTimeBounds, digitizationConstant);
103 
104  // Check whether there is a cycle in of probabilistic states
105  bool acyclic = !storm::utility::graph::hasCycle(PS.toPS);
106 
107  // Initialize a minMaxSolver to compute an optimal scheduler (w.r.t. PS) for each epoch
108  // No EC elimination is necessary as we assume non-zenoness
109  std::unique_ptr<MinMaxSolverData> minMax = initMinMaxSolver(env, PS, acyclic, weightVector);
110 
111  // create a linear equation solver for the model induced by the optimal choice vector.
112  // the solver will be updated whenever the optimal choice vector has changed.
113  std::unique_ptr<LinEqSolverData> linEq = initLinEqSolver(env, PS, acyclic);
114 
115  // Store the optimal choices of PS as computed by the minMax solver.
116  std::vector<uint_fast64_t> optimalChoicesAtCurrentEpoch(PS.getNumberOfStates(), std::numeric_limits<uint_fast64_t>::max());
117 
118  // Stores the objectives for which we need to compute values in the current time epoch.
119  storm::storage::BitVector consideredObjectives = this->objectivesWithNoUpperTimeBound & ~this->lraObjectives;
120 
121  auto upperTimeBoundIt = upperTimeBounds.begin();
122  uint_fast64_t currentEpoch = upperTimeBounds.empty() ? 0 : upperTimeBoundIt->first;
123  while (!storm::utility::resources::isTerminate()) {
124  // Update the objectives that are considered at the current time epoch as well as the (weighted) reward vectors.
125  updateDataToCurrentEpoch(MS, PS, *minMax, consideredObjectives, currentEpoch, weightVector, upperTimeBoundIt, upperTimeBounds);
126 
127  // Compute the values that can be obtained at probabilistic states in the current time epoch
128  performPSStep(env, PS, MS, *minMax, *linEq, optimalChoicesAtCurrentEpoch, consideredObjectives, weightVector);
129 
130  // Compute values that can be obtained at Markovian states after letting one (digitized) time unit pass.
131  // Only perform such a step if there is time left.
132  if (currentEpoch>0) {
133  performMSStep(env, MS, PS, consideredObjectives, weightVector);
134  --currentEpoch;
135  } else {
136  break;
137  }
138  }
139  STORM_LOG_WARN_COND(!storm::utility::resources::isTerminate(), "Time-bounded reachability computation aborted.");
140 
141  // compose the results from MS and PS
142  storm::utility::vector::setVectorValues(this->weightedResult, MS.states, MS.weightedSolutionVector);
143  storm::utility::vector::setVectorValues(this->weightedResult, PS.states, PS.weightedSolutionVector);
144  for (uint_fast64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
145  storm::utility::vector::setVectorValues(this->objectiveResults[objIndex], MS.states, MS.objectiveSolutionVectors[objIndex]);
146  storm::utility::vector::setVectorValues(this->objectiveResults[objIndex], PS.states, PS.objectiveSolutionVectors[objIndex]);
147  }
148  }

151  typename StandardMaPcaaWeightVectorChecker<SparseMaModelType>::SubModel StandardMaPcaaWeightVectorChecker<SparseMaModelType>::createSubModel(bool createMS, std::vector<ValueType> const& weightedRewardVector) const {
152  SubModel result;
153 
154  storm::storage::BitVector probabilisticStates = ~markovianStates;
155  result.states = createMS ? markovianStates : probabilisticStates;
156  result.choices = this->transitionMatrix.getRowFilter(result.states);
157  STORM_LOG_ASSERT(!createMS || result.states.getNumberOfSetBits() == result.choices.getNumberOfSetBits(), "row groups for Markovian states should consist of exactly one row");
158 
159  //We need to add diagonal entries for selfloops on Markovian states.
160  result.toMS = this->transitionMatrix.getSubmatrix(true, result.states, markovianStates, createMS);
161  result.toPS = this->transitionMatrix.getSubmatrix(true, result.states, probabilisticStates, false);
162  STORM_LOG_ASSERT(result.getNumberOfStates() == result.states.getNumberOfSetBits() && result.getNumberOfStates() == result.toMS.getRowGroupCount() && result.getNumberOfStates() == result.toPS.getRowGroupCount(), "Invalid state count for subsystem");
163  STORM_LOG_ASSERT(result.getNumberOfChoices() == result.choices.getNumberOfSetBits() && result.getNumberOfChoices() == result.toMS.getRowCount() && result.getNumberOfChoices() == result.toPS.getRowCount(), "Invalid choice count for subsystem");
164 
165  result.weightedRewardVector.resize(result.getNumberOfChoices());
166  storm::utility::vector::selectVectorValues(result.weightedRewardVector, result.choices, weightedRewardVector);
167  for (uint_fast64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
168  std::vector<ValueType> const& objRewards = this->actionRewards[objIndex];
169  std::vector<ValueType> subModelObjRewards;
170  subModelObjRewards.reserve(result.getNumberOfChoices());
171  for (auto choice : result.choices) {
172  subModelObjRewards.push_back(objRewards[choice]);
173  }
174  result.objectiveRewardVectors.push_back(std::move(subModelObjRewards));
175  }
176 
177  result.weightedSolutionVector.resize(result.getNumberOfStates());
178  storm::utility::vector::selectVectorValues(result.weightedSolutionVector, result.states, this->weightedResult);
179  result.objectiveSolutionVectors.resize(this->objectives.size());
180  for (uint_fast64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
181  result.objectiveSolutionVectors[objIndex].resize(result.weightedSolutionVector.size());
182  storm::utility::vector::selectVectorValues(result.objectiveSolutionVectors[objIndex], result.states, this->objectiveResults[objIndex]);
183  }
184 
185  result.auxChoiceValues.resize(result.getNumberOfChoices());
186 
187  return result;

191  template <typename VT, typename std::enable_if<storm::NumberTraits<VT>::SupportsExponential, int>::type>
192  VT StandardMaPcaaWeightVectorChecker<SparseMaModelType>::getDigitizationConstant(std::vector<ValueType> const& weightVector) const {
193  STORM_LOG_DEBUG("Retrieving digitization constant");
194  // We need to find a delta such that for each objective it holds that lowerbound/delta , upperbound/delta are natural numbers and
195  // sum_{obj_i} (
196  // If obj_i has a lower and an upper bound:
197  // weightVector_i * (1 - e^(-maxRate lowerbound) * (1 + maxRate delta) ^ (lowerbound / delta) + 1-e^(-maxRate upperbound) * (1 + maxRate delta) ^ (upperbound / delta) + (1-e^(-maxRate delta)))
198  // If there is only an upper bound:
199  // weightVector_i * ( 1-e^(-maxRate upperbound) * (1 + maxRate delta) ^ (upperbound / delta))
200  // ) <= this->maximumLowerUpperDistance
201 
202  // Initialize some data for fast and easy access
203  VT const maxRate = storm::utility::vector::max_if(exitRates, markovianStates);
204  std::vector<VT> timeBounds;
205  std::vector<VT> eToPowerOfMinusMaxRateTimesBound;
206  VT smallestNonZeroBound = storm::utility::zero<VT>();
207  for (auto const& obj : this->objectives) {
208  if (obj.formula->getSubformula().isCumulativeRewardFormula()) {
209  timeBounds.push_back(obj.formula->getSubformula().asCumulativeRewardFormula().template getBound<VT>());
210  STORM_LOG_THROW(!storm::utility::isZero(timeBounds.back()), storm::exceptions::InvalidPropertyException, "Got zero-valued upper time bound. This is not suppoted.");
211  eToPowerOfMinusMaxRateTimesBound.push_back(std::exp(-maxRate * timeBounds.back()));
212  smallestNonZeroBound = storm::utility::isZero(smallestNonZeroBound) ? timeBounds.back() : std::min(smallestNonZeroBound, timeBounds.back());
213  } else {
214  timeBounds.push_back(storm::utility::zero<VT>());
215  eToPowerOfMinusMaxRateTimesBound.push_back(storm::utility::zero<VT>());
216  }
217  }
218  if (storm::utility::isZero(smallestNonZeroBound)) {
219  // There are no time bounds. In this case, one is a valid digitization constant.
220  return storm::utility::one<VT>();
221  }
222  VT goalPrecisionTimesNorm = this->weightedPrecision * storm::utility::sqrt(storm::utility::vector::dotProduct(weightVector, weightVector));
223 
224  // We brute-force a delta, since a direct computation is apparently not easy.
225  // Also note that the number of times this loop runs is a lower bound for the number of minMaxSolver invocations.
226  // Hence, this brute-force approach will most likely not be a bottleneck.
227  storm::storage::BitVector objectivesWithTimeBound = ~this->objectivesWithNoUpperTimeBound;
228  uint_fast64_t smallestStepBound = 1;
229  VT delta = smallestNonZeroBound / smallestStepBound;
230  while(true) {
231  bool deltaValid = true;
232  for (auto objIndex : objectivesWithTimeBound) {
233  auto const& timeBound = timeBounds[objIndex];
234  if (timeBound/delta != std::floor(timeBound/delta)) {
235  deltaValid = false;
236  break;
237  }
238  }
239  if (deltaValid) {
240  VT weightedPrecisionForCurrentDelta = storm::utility::zero<VT>();
241  for (uint_fast64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
242  VT precisionOfObj = storm::utility::zero<VT>();
243  if (objectivesWithTimeBound.get(objIndex)) {
244  precisionOfObj += storm::utility::one<VT>() - (eToPowerOfMinusMaxRateTimesBound[objIndex] * storm::utility::pow(storm::utility::one<VT>() + maxRate * delta, timeBounds[objIndex] / delta) );
245  }
246  weightedPrecisionForCurrentDelta += weightVector[objIndex] * precisionOfObj;
247  }
248  deltaValid &= weightedPrecisionForCurrentDelta <= goalPrecisionTimesNorm;
249  }
250  if (deltaValid) {
251  break;
252  }
253  ++smallestStepBound;
254  STORM_LOG_ASSERT(delta>smallestNonZeroBound / smallestStepBound, "Digitization constant is expected to become smaller in every iteration");
255  delta = smallestNonZeroBound / smallestStepBound;
256  }
257  STORM_LOG_DEBUG("Found digitization constant: " << delta << ". At least " << smallestStepBound << " digitization steps will be necessarry");
258  return delta;
259  }

268  template <typename VT, typename std::enable_if<storm::NumberTraits<VT>::SupportsExponential, int>::type>
269  void StandardMaPcaaWeightVectorChecker<SparseMaModelType>::digitize(SubModel& MS, VT const& digitizationConstant) const {
270  std::vector<VT> rateVector(MS.getNumberOfChoices());
271  storm::utility::vector::selectVectorValues(rateVector, MS.states, exitRates);
272  for (uint_fast64_t row = 0; row < rateVector.size(); ++row) {
273  VT const eToMinusRateTimesDelta = std::exp(-rateVector[row] * digitizationConstant);
274  for (auto& entry : MS.toMS.getRow(row)) {
275  entry.setValue((storm::utility::one<VT>() - eToMinusRateTimesDelta) * entry.getValue());
276  if (entry.getColumn() == row) {
277  entry.setValue(entry.getValue() + eToMinusRateTimesDelta);
278  }
279  }
280  for (auto& entry : MS.toPS.getRow(row)) {
281  entry.setValue((storm::utility::one<VT>() - eToMinusRateTimesDelta) * entry.getValue());
282  }
283  MS.weightedRewardVector[row] *= storm::utility::one<VT>() - eToMinusRateTimesDelta;
284  for (auto& objVector : MS.objectiveRewardVectors) {
285  objVector[row] *= storm::utility::one<VT>() - eToMinusRateTimesDelta;
286  }
287  }
288  }

298  void StandardMaPcaaWeightVectorChecker<SparseMaModelType>::digitizeTimeBounds(TimeBoundMap& upperTimeBounds, VT const& digitizationConstant) {
299 
300  VT const maxRate = storm::utility::vector::max_if(exitRates, markovianStates);
301  for (uint_fast64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
302  auto const& obj = this->objectives[objIndex];
303  VT errorTowardsZero = storm::utility::zero<VT>();
304  VT errorAwayFromZero = storm::utility::zero<VT>();
305  if (obj.formula->getSubformula().isCumulativeRewardFormula()) {
306  VT timeBound = obj.formula->getSubformula().asCumulativeRewardFormula().template getBound<VT>();
307  uint_fast64_t digitizedBound = storm::utility::convertNumber<uint_fast64_t>(timeBound/digitizationConstant);
308  auto timeBoundIt = upperTimeBounds.insert(std::make_pair(digitizedBound, storm::storage::BitVector(this->objectives.size(), false))).first;
309  timeBoundIt->second.set(objIndex);
310  VT digitizationError = storm::utility::one<VT>();
311  digitizationError -= std::exp(-maxRate * timeBound) * storm::utility::pow(storm::utility::one<VT>() + maxRate * digitizationConstant, digitizedBound);
312  errorAwayFromZero += digitizationError;
313  }
314  if (storm::solver::maximize(obj.formula->getOptimalityType())) {
315  this->offsetsToUnderApproximation[objIndex] = -errorTowardsZero;
316  this->offsetsToOverApproximation[objIndex] = errorAwayFromZero;
317  } else {
318  this->offsetsToUnderApproximation[objIndex] = errorAwayFromZero;
319  this->offsetsToOverApproximation[objIndex] = -errorTowardsZero;
320  }
321  }
322  }

400  }
401

407  template <class SparseMaModelType>
408  void StandardMaPcaaWeightVectorChecker<SparseMaModelType>::performPSStep(Environment const& env, SubModel& PS, SubModel const& MS, MinMaxSolverData& minMax, LinEqSolverData& linEq, std::vector<uint_fast64_t>& optimalChoicesAtCurrentEpoch, storm::storage::BitVector const& consideredObjectives, std::vector<ValueType> const& weightVector) const {
409  // compute a choice vector for the probabilistic states that is optimal w.r.t. the weighted reward vector
410  minMax.solver->solveEquations(*minMax.env, PS.weightedSolutionVector, minMax.b);
411  auto const& newChoices = minMax.solver->getSchedulerChoices();
412  if (consideredObjectives.getNumberOfSetBits() == 1 && storm::utility::isOne(weightVector[*consideredObjectives.begin()])) {
413  // In this case there is no need to perform the computation on the individual objectives
414  optimalChoicesAtCurrentEpoch = newChoices;
415  PS.objectiveSolutionVectors[*consideredObjectives.begin()] = PS.weightedSolutionVector;
416  if (storm::solver::minimize(this->objectives[*consideredObjectives.begin()].formula->getOptimalityType())) {
417  storm::utility::vector::scaleVectorInPlace(PS.objectiveSolutionVectors[*consideredObjectives.begin()], -storm::utility::one<ValueType>());
418  }
419  } else {
420  // check whether the linEqSolver needs to be updated, i.e., whether the scheduler has changed
421  if (linEq.solver == nullptr || newChoices != optimalChoicesAtCurrentEpoch) {
422  optimalChoicesAtCurrentEpoch = newChoices;
423  linEq.solver = nullptr;
424  bool needEquationSystem = linEq.factory->getEquationProblemFormat(*linEq.env) == storm::solver::LinearEquationSolverProblemFormat::EquationSystem;
425  storm::storage::SparseMatrix<ValueType> linEqMatrix = PS.toPS.selectRowsFromRowGroups(optimalChoicesAtCurrentEpoch, needEquationSystem);
426  if (needEquationSystem) {
427  linEqMatrix.convertToEquationSystem();
428  }
429  linEq.solver = linEq.factory->create(*linEq.env, std::move(linEqMatrix));
430  linEq.solver->setCachingEnabled(true);
431  auto req = linEq.solver->getRequirements(*linEq.env);
432  if (linEq.acyclic) {
433  req.clearAcyclic();
434  }
435  STORM_LOG_THROW(!req.hasEnabledCriticalRequirement(), storm::exceptions::UncheckedRequirementException, "Solver requirements " + req.getEnabledRequirementsAsString() + " not checked.");
436  }
437 
438  // Get the results for the individual objectives.
439  // Note that we do not consider an estimate for each objective (as done in the unbounded phase) since the results from the previous epoch are already pretty close
440  for (auto objIndex : consideredObjectives) {
441  auto const& objectiveRewardVectorPS = PS.objectiveRewardVectors[objIndex];
442  auto const& objectiveSolutionVectorMS = MS.objectiveSolutionVectors[objIndex];
443  // compute rhs of equation system, i.e., PS.toMS * x + Rewards
444  // To safe some time, only do this for the obtained optimal choices
445  auto itGroupIndex = PS.toPS.getRowGroupIndices().begin();
446  auto itChoiceOffset = optimalChoicesAtCurrentEpoch.begin();
447  for (auto& bValue : linEq.b) {
448  uint_fast64_t row = (*itGroupIndex) + (*itChoiceOffset);
449  bValue = objectiveRewardVectorPS[row];
450  for (auto const& entry : PS.toMS.getRow(row)){
451  bValue += entry.getValue() * objectiveSolutionVectorMS[entry.getColumn()];
452  }
453  ++itGroupIndex;
454  ++itChoiceOffset;
455  }
456  linEq.solver->solveEquations(*linEq.env, PS.objectiveSolutionVectors[objIndex], linEq.b);
457  }
458  }
459  }
