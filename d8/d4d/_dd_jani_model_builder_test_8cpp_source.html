306  // Now we tweak it's system composition to check whether synchronization vectors work.

308  automataCompositions.push_back(std::make_shared<storm::jani::AutomatonComposition>("one"));
309  automataCompositions.push_back(std::make_shared<storm::jani::AutomatonComposition>("two"));
310  automataCompositions.push_back(std::make_shared<storm::jani::AutomatonComposition>("three"));
311 
312  // First, make all actions non-synchronizing.
313  std::vector<storm::jani::SynchronizationVector> synchronizationVectors;
314
