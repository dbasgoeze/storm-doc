44  return index;
45  }

47

49  predicateToIndexMap[predicate] = predicateIndex;
50 
51  // Add the new predicate to the list of known predicates.
52  predicates.push_back(predicate);
53 
54  // Add DD variables for the new predicate.
55  std::stringstream stream;
56  stream << predicate;
57  std::pair<storm::expressions::Variable, storm::expressions::Variable> newMetaVariable = ddManager->addMetaVariable(stream.str());
58

60  extendedPredicateDdVariables.push_back(newMetaVariable);
61  predicateBdds.emplace_back(ddManager->getEncoding(newMetaVariable.first, 1), ddManager->getEncoding(newMetaVariable.second, 1));
62  predicateIdentities.push_back(ddManager->getEncoding(newMetaVariable.first, 1).iff(ddManager->getEncoding(newMetaVariable.second, 1)));
63  allPredicateIdentities &= predicateIdentities.back();

65  successorVariables.insert(newMetaVariable.second);
66  sourcePredicateVariables.insert(newMetaVariable.first);
