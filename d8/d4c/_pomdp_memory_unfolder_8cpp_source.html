118  std::vector<uint32_t> PomdpMemoryUnfolder<ValueType>::transformObservabilityClasses(storm::storage::BitVector const& reachableStates) const {
119  std::vector<uint32_t> observations;
120  observations.reserve(pomdp.getNumberOfStates() * memory.getNumberOfStates());
121  for (uint64_t modelState = 0; modelState < pomdp.getNumberOfStates(); ++modelState) {
122  for (uint64_t memState = 0; memState < memory.getNumberOfStates(); ++memState) {
123  if (reachableStates.get(getUnfoldingState(modelState, memState))) {
124  observations.push_back(getUnfoldingObersvation(pomdp.getObservation(modelState), memState));
125  }
126  }
127  }
128 
129  // Eliminate observations that are not in use (as they are not reachable).
130  std::set<uint32_t> occuringObservations(observations.begin(), observations.end());
131  uint32_t highestObservation = *occuringObservations.rbegin();
132  std::vector<uint32_t> oldToNewObservationMapping(highestObservation + 1, std::numeric_limits<uint32_t>::max());
133  uint32_t newObs = 0;
134  for (auto const& oldObs : occuringObservations) {
135  oldToNewObservationMapping[oldObs] = newObs;
136  ++newObs;
137  }
138  for (auto& obs : observations) {
139  obs = oldToNewObservationMapping[obs];
140  }
141 
142  return observations;
143  }
