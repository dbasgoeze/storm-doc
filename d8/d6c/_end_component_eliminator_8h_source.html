31  /*
32  * Substitutes the given end components by a single state.
33  *
34  *
35  * Only states in the given subsystem are kept. Transitions leading to a state outside of the subsystem will be
36  * removed (but the corresponding row is kept, possibly yielding empty rows).
37  * The given ECs all have to lie within the provided subsystem.
38  *
39  * For each such EC (that is not contained in another EC), we add a new state and redirect all incoming and outgoing
40  * transitions of the EC to (and from) this state.
41  * If addSinkRowStates is true for at least one state of an eliminated EC, a row is added to the new state (representing the choice to stay at the EC forever).
42  * If addSelfLoopAtSinkStates is true, such rows get a selfloop (with value 1). Otherwise, the row remains empty.
43  */

116  static storm::storage::MaximalEndComponentDecomposition<ValueType> computeECs(storm::storage::SparseMatrix<ValueType> const& originalMatrix, storm::storage::BitVector const& possibleECRows, storm::storage::BitVector const& subsystemStates) {
117  // Get an auxiliary matrix to identify the correct end components w.r.t. the possible EC rows and the subsystem.
118  // This is done by redirecting choices that can never be part of an EC to a sink state.
119  // Such choices are either non-possible EC rows or rows that lead to a state that is not in the subsystem.
120  uint_fast64_t sinkState = originalMatrix.getRowGroupCount();
121  storm::storage::SparseMatrixBuilder<ValueType> builder(originalMatrix.getRowCount() + 1, originalMatrix.getColumnCount() + 1, originalMatrix.getEntryCount() + 1, false, true, originalMatrix.getRowGroupCount()+1);
122  uint_fast64_t row = 0;
123  for(uint_fast64_t rowGroup = 0; rowGroup < originalMatrix.getRowGroupCount(); ++rowGroup) {
124  builder.newRowGroup(row);
125  for (; row < originalMatrix.getRowGroupIndices()[rowGroup+1]; ++row ){
126  bool keepRow = possibleECRows.get(row);
127  if(keepRow) { //Also check whether all successors are in the subsystem
128  for(auto const& entry : originalMatrix.getRow(row)){
129  keepRow &= subsystemStates.get(entry.getColumn());
130  }
131  }
132  if(keepRow) {
133  for(auto const& entry : originalMatrix.getRow(row)){
134  builder.addNextValue(row, entry.getColumn(), entry.getValue());
135  }
136  } else {
137  builder.addNextValue(row, sinkState, storm::utility::one<ValueType>());
138  }
139  }
140  }
141  builder.newRowGroup(row);
142  builder.addNextValue(row, sinkState, storm::utility::one<ValueType>());
143  storm::storage::SparseMatrix<ValueType> auxiliaryMatrix = builder.build(originalMatrix.getRowCount() + 1, originalMatrix.getColumnCount()+1, originalMatrix.getRowGroupCount() +1);
144  storm::storage::SparseMatrix<ValueType> backwardsTransitions = auxiliaryMatrix.transpose(true);
145  storm::storage::BitVector sinkStateAsBitVector(auxiliaryMatrix.getRowGroupCount(), false);
146  sinkStateAsBitVector.set(sinkState);
147  storm::storage::BitVector auxSubsystemStates = subsystemStates;
148  auxSubsystemStates.resize(subsystemStates.size() + 1, true);
149  // The states for which sinkState is reachable under every scheduler can not be part of an EC
150  auxSubsystemStates &= ~(storm::utility::graph::performProbGreater0A(auxiliaryMatrix, auxiliaryMatrix.getRowGroupIndices(), backwardsTransitions, auxSubsystemStates, sinkStateAsBitVector));
151  return storm::storage::MaximalEndComponentDecomposition<ValueType>(auxiliaryMatrix, backwardsTransitions, auxSubsystemStates);
152  }

154  static storm::storage::SparseMatrix<ValueType> buildTransformedMatrix(storm::storage::SparseMatrix<ValueType> const& originalMatrix,
155  std::vector<uint_fast64_t> const& newRowGroupIndices,
156  std::vector<uint_fast64_t> const& newToOldRowMapping,
157  std::vector<uint_fast64_t> const& oldToNewStateMapping,
158  storm::storage::BitVector const& sinkRows,
159  bool addSelfLoopAtSinkStates) {
160 
161  uint_fast64_t numRowGroups = newRowGroupIndices.size()-1;
162  uint_fast64_t newRow = 0;
163  storm::storage::SparseMatrixBuilder<ValueType> builder(newToOldRowMapping.size(), numRowGroups, originalMatrix.getEntryCount(), false, true, numRowGroups);
164  for (uint_fast64_t newRowGroup = 0; newRowGroup < numRowGroups; ++newRowGroup) {
165  builder.newRowGroup(newRow);
166  for (; newRow < newRowGroupIndices[newRowGroup + 1]; ++newRow) {
167  if (sinkRows.get(newRow)) {
168  if (addSelfLoopAtSinkStates) {
169  builder.addNextValue(newRow, newRowGroup, storm::utility::one<ValueType>());
170  }
171  } else {
172  // Make sure that the entries for this row are inserted in the right order.
173  // Also, transitions to the same EC need to be merged and transitions to states that are erased need to be ignored
174  std::map<uint_fast64_t, ValueType> sortedEntries;
175  for(auto const& entry : originalMatrix.getRow(newToOldRowMapping[newRow])){
176  uint_fast64_t newColumn = oldToNewStateMapping[entry.getColumn()];
177  if(newColumn < numRowGroups) {
178  auto insertResult = sortedEntries.insert(std::make_pair(newColumn, entry.getValue()));
179  if(!insertResult.second) {
180  // We have already seen an entry with this column. ==> merge transitions
181  insertResult.first->second += entry.getValue();
182  }
183  }
184  }
185  for(auto const& sortedEntry : sortedEntries) {
186  builder.addNextValue(newRow, sortedEntry.first, sortedEntry.second);
187  }
188  }
189  }
190  }
191  return builder.build(newToOldRowMapping.size(), numRowGroups, numRowGroups);
192  }
