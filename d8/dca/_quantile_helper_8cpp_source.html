74  std::shared_ptr<storm::logic::ProbabilityOperatorFormula> transformBoundedUntilOperator(storm::logic::ProbabilityOperatorFormula const& boundedUntilOperator, std::vector<BoundTransformation> const& transformations, bool complementQuery = false) {
75  auto const& origBoundedUntil = boundedUntilOperator.getSubformula().asBoundedUntilFormula();
76  STORM_LOG_ASSERT(transformations.size() == origBoundedUntil.getDimension(), "Tried to replace the bound of a dimension that is higher than the number of dimensions of the formula.");
77  std::vector<std::shared_ptr<storm::logic::Formula const>> leftSubformulas, rightSubformulas;
78  std::vector<boost::optional<storm::logic::TimeBound>> lowerBounds, upperBounds;
79  std::vector<storm::logic::TimeBoundReference> timeBoundReferences;
80 
81  for (uint64_t dim = 0; dim < origBoundedUntil.getDimension(); ++dim) {
82  if (origBoundedUntil.hasMultiDimensionalSubformulas()) {
83  leftSubformulas.push_back(origBoundedUntil.getLeftSubformula(dim).asSharedPointer());
84  rightSubformulas.push_back(origBoundedUntil.getRightSubformula(dim).asSharedPointer());
85  }
86  timeBoundReferences.push_back(origBoundedUntil.getTimeBoundReference(dim));
87  if (transformations[dim] == BoundTransformation::None) {
88  if (origBoundedUntil.hasLowerBound(dim)) {
89  lowerBounds.push_back(storm::logic::TimeBound(origBoundedUntil.isLowerBoundStrict(dim), origBoundedUntil.getLowerBound(dim)));
90  } else {
91  lowerBounds.push_back(boost::none);
92  }
93  if (origBoundedUntil.hasUpperBound(dim)) {
94  upperBounds.push_back(storm::logic::TimeBound(origBoundedUntil.isUpperBoundStrict(dim), origBoundedUntil.getUpperBound(dim)));
95  } else {
96  upperBounds.push_back(boost::none);
97  }
98  } else {
99  // We need a zero expression in all other cases
100  storm::expressions::Expression zero;
101  if (origBoundedUntil.hasLowerBound(dim)) {
102  zero = origBoundedUntil.getLowerBound(dim).getManager().rational(0.0);
103  } else {
104  STORM_LOG_THROW(origBoundedUntil.hasUpperBound(dim), storm::exceptions::InvalidOperationException, "The given bounded until formula has no cost-bound for one dimension.");
105  zero = origBoundedUntil.getUpperBound(dim).getManager().rational(0.0);
106  }
107  if (transformations[dim] == BoundTransformation::LessEqualZero) {
108  lowerBounds.push_back(boost::none);
109  upperBounds.push_back(storm::logic::TimeBound(false, zero));
110  } else {
111  STORM_LOG_ASSERT(transformations[dim] == BoundTransformation::GreaterZero || transformations[dim] == BoundTransformation::GreaterEqualZero, "Unhandled bound transformation.");
112  lowerBounds.push_back(storm::logic::TimeBound(transformations[dim] == BoundTransformation::GreaterZero, zero));
113  upperBounds.push_back(boost::none);
114  }
115  }
116  }
117  std::shared_ptr<storm::logic::Formula> newBoundedUntil;
118  if (origBoundedUntil.hasMultiDimensionalSubformulas()) {
119  newBoundedUntil = std::make_shared<storm::logic::BoundedUntilFormula>(leftSubformulas, rightSubformulas, lowerBounds, upperBounds, timeBoundReferences);
120  } else {
121  newBoundedUntil = std::make_shared<storm::logic::BoundedUntilFormula>(origBoundedUntil.getLeftSubformula().asSharedPointer(), origBoundedUntil.getRightSubformula().asSharedPointer(), lowerBounds, upperBounds, timeBoundReferences);
122  }
123  storm::logic::OperatorInformation newOpInfo(boundedUntilOperator.getOperatorInformation().optimalityType, boundedUntilOperator.getBound());
124  if (complementQuery) {
125  newOpInfo.bound->comparisonType = storm::logic::invert(newOpInfo.bound->comparisonType);
126  }
127  return std::make_shared<storm::logic::ProbabilityOperatorFormula>(newBoundedUntil, newOpInfo);
128  }

144  ValueType prec;
145  bool relative;

149  } else {

156  ValueType lower = value * (one / (prec + one));
157  ValueType upper = value * (one + prec / (prec + one));
158  return std::make_pair(lower, upper);
159  } else {
160  return std::pair<ValueType, ValueType>(value - prec, value + prec);

232  std::pair<CostLimitClosure, std::vector<typename QuantileHelper<ModelType>::ValueType>> QuantileHelper<ModelType>::computeQuantile(Environment& env, storm::storage::BitVector const& consideredDimensions, bool complementaryQuery) {
233  STORM_LOG_ASSERT(consideredDimensions.isSubsetOf(getOpenDimensions()), "Considered dimensions for a quantile query should be a subset of the set of dimensions without a fixed bound.");
234 
235  storm::storage::BitVector cacheKey = consideredDimensions;
236  cacheKey.resize(cacheKey.size() + 1, complementaryQuery);
237  auto cacheIt = cachedSubQueryResults.find(cacheKey);
238  if (cacheIt != cachedSubQueryResults.end()) {
239  return cacheIt->second;
240  }
241 
242  auto boundedUntilOp = transformBoundedUntilOperator(quantileFormula.getSubformula().asProbabilityOperatorFormula(), std::vector<BoundTransformation>(getDimension(), BoundTransformation::None), complementaryQuery);
243  std::set<storm::expressions::Variable> infinityVariables;
244  storm::storage::BitVector lowerBoundedDimensions(getDimension());
245  storm::storage::BitVector downwardClosedDimensions(getDimension());
246  bool hasLowerValueBound = storm::logic::isLowerBound(boundedUntilOp->getComparisonType());
247  for (auto const& d : getOpenDimensions()) {
248  if (consideredDimensions.get(d)) {
249  bool hasLowerCostBound = boundedUntilOp->getSubformula().asBoundedUntilFormula().hasLowerBound(d);
250  lowerBoundedDimensions.set(d, hasLowerCostBound);
251  downwardClosedDimensions.set(d, hasLowerCostBound == hasLowerValueBound);
252  } else {
253  infinityVariables.insert(getVariableForDimension(d));
254  }
255  }
256  downwardClosedDimensions = downwardClosedDimensions % consideredDimensions;
257  CostLimitClosure satCostLimits(downwardClosedDimensions), unsatCostLimits(~downwardClosedDimensions);
258 
259  // Initialize the (un)sat cost limits to guarantee termination
260  bool onlyUpperCostBounds = lowerBoundedDimensions.empty();
261  bool onlyLowerCostBounds = lowerBoundedDimensions == consideredDimensions;
262  if (onlyUpperCostBounds || onlyLowerCostBounds) {
263  for (auto const& k : consideredDimensions) {
264  storm::storage::BitVector subQueryDimensions = consideredDimensions;
265  subQueryDimensions.set(k, false);
266  bool subQueryComplement = complementaryQuery != ((onlyUpperCostBounds && hasLowerValueBound) || (onlyLowerCostBounds && !hasLowerValueBound));
267  auto subQueryResult = computeQuantile(env, subQueryDimensions, subQueryComplement);
268  for (auto const& subQueryCostLimit : subQueryResult.first.getGenerator()) {
269  CostLimits initPoint;
270  uint64_t i = 0;
271  for (auto const& dim : consideredDimensions) {
272  if (dim == k) {
273  initPoint.push_back(CostLimit::infinity());
274  } else {
275  initPoint.push_back(subQueryCostLimit[i]);
276  ++i;
277  }
278  }
279  if (subQueryComplement == complementaryQuery) {
280  satCostLimits.insert(initPoint);
281  } else {
282  unsatCostLimits.insert(initPoint);
283  }
284  }
285  }
286  } else {
287  STORM_LOG_WARN("Quantile formula considers mixtures of upper and lower reward-bounds. Termination is not guaranteed.");
288  }
289 
290  // Loop until the goal precision is reached.
291  STORM_LOG_DEBUG("Computing quantile for dimensions: " << consideredDimensions);
292  while (true) {
293  // initialize reward unfolding and data that will be needed for each epoch
294  MultiDimensionalRewardUnfolding<ValueType, true> rewardUnfolding(model, boundedUntilOp, infinityVariables);
295  if (computeQuantile(env, consideredDimensions, *boundedUntilOp, lowerBoundedDimensions, satCostLimits, unsatCostLimits, rewardUnfolding)) {
296  std::vector<ValueType> scalingFactors;
297  for (auto const& dim : consideredDimensions) {
298  scalingFactors.push_back(rewardUnfolding.getDimension(dim).scalingFactor);
299  }
300  std::pair<CostLimitClosure, std::vector<ValueType>> result(satCostLimits, scalingFactors);
301  cachedSubQueryResults.emplace(cacheKey, result);
302  return result;
303  }
304  STORM_LOG_WARN("Restarting quantile computation after " << swExploration << " seconds due to insufficient precision.");
305  ++numPrecisionRefinements;
306  increasePrecision(env);
307  }
308  }

310  bool getNextCandidateCostLimit(CostLimit const& candidateCostLimitSum, CostLimits& current) {
311  if (current.size() == 0) {
312  return false;
313  }
314  uint64_t iSum = current.front().get();
315  if (iSum == candidateCostLimitSum.get()) {
316  return false;
317  }
318  for (uint64_t i = 1; i < current.size(); ++i) {
319  iSum += current[i].get();
320  if (iSum == candidateCostLimitSum.get()) {
321  ++current[i-1].get();
322  uint64_t newVal = current[i].get() - 1;
323  current[i].get() = 0;
324  current.back().get() = newVal;
325  return true;
326  }
327  }
328  STORM_LOG_THROW(false, storm::exceptions::UnexpectedException, "The entries of the current cost limit candidate do not sum up to the current candidate sum.");
329  return false;

400  }

407  if (model.isNondeterministicModel()) {

409  } else {

417  bool propertySatisfied;
418  if (env.solver().isForceSoundness()) {
419  ValueType sumOfEpochDimensions = storm::utility::convertNumber<ValueType>(rewardUnfolding.getEpochManager().getSumOfDimensions(epoch) + 1);
420  auto lowerUpperValue = getLowerUpperBound(env, sumOfEpochDimensions, currValue);
421  propertySatisfied = boundedUntilOperator.getBound().isSatisfied(lowerUpperValue.first);
422  if (propertySatisfied != boundedUntilOperator.getBound().isSatisfied(lowerUpperValue.second)) {
423  // unclear result due to insufficient precision.
424  swExploration.stop();
425  return false;
426  }
427  } else {
428  propertySatisfied = boundedUntilOperator.getBound().isSatisfied(currValue);
429  }
430  if (propertySatisfied) {
431  satCostLimits.insert(epochAsCostLimits);
432  } else {
433  unsatCostLimits.insert(epochAsCostLimits);
434  }
435  }
436  }
437  }
438  } while (getNextCandidateCostLimit(candidateCostLimitSum, currentCandidate));
439  if (!progress) {
440  progress = !CostLimitClosure::unionFull(satCostLimits, unsatCostLimits);
441  }
442  }
443  swExploration.stop();
444  return true;
445  }
