219  template<typename ValueType, bool SingleObjectiveMode>
220  void MultiDimensionalRewardUnfolding<ValueType, SingleObjectiveMode>::computeMaxDimensionValues() {
221  for (uint64_t dim = 0; dim < epochManager.getDimensionCount(); ++dim) {
222  storm::expressions::Expression bound;
223  bool isStrict = false;
224  storm::logic::Formula const& dimFormula = *dimensions[dim].formula;
225  if (dimFormula.isBoundedUntilFormula()) {
226  assert(!dimFormula.asBoundedUntilFormula().hasMultiDimensionalSubformulas());
227  if (dimFormula.asBoundedUntilFormula().hasUpperBound()) {
228  STORM_LOG_ASSERT(!dimFormula.asBoundedUntilFormula().hasLowerBound(), "Bounded until formulas with interval bounds are not supported.");
229  bound = dimFormula.asBoundedUntilFormula().getUpperBound();
230  isStrict = dimFormula.asBoundedUntilFormula().isUpperBoundStrict();
231  } else {
232  STORM_LOG_ASSERT(dimFormula.asBoundedUntilFormula().hasLowerBound(), "Bounded until formulas without any bounds are not supported.");
233  bound = dimFormula.asBoundedUntilFormula().getLowerBound();
234  isStrict = dimFormula.asBoundedUntilFormula().isLowerBoundStrict();
235  }
236  } else if (dimFormula.isCumulativeRewardFormula()) {
237  assert(!dimFormula.asCumulativeRewardFormula().isMultiDimensional());
238  bound = dimFormula.asCumulativeRewardFormula().getBound();
239  isStrict = dimFormula.asCumulativeRewardFormula().isBoundStrict();
240  }
241 
242  if (!bound.containsVariables()) {
243  // We always consider upper bounds to be non-strict and lower bounds to be strict.
244  // Thus, >=N would become >N-1. However, note that the case N=0 is treated separately.
245  if (dimensions[dim].boundType == DimensionBoundType::LowerBound || dimensions[dim].boundType == DimensionBoundType::UpperBound) {
246  ValueType discretizedBound = storm::utility::convertNumber<ValueType>(bound.evaluateAsRational());
247  discretizedBound /= dimensions[dim].scalingFactor;
248  if (storm::utility::isInteger(discretizedBound)) {
249  if (isStrict == (dimensions[dim].boundType == DimensionBoundType::UpperBound)) {
250  discretizedBound -= storm::utility::one<ValueType>();
251  }
252  } else {
253  discretizedBound = storm::utility::floor(discretizedBound);
254  }
255  uint64_t dimensionValue = storm::utility::convertNumber<uint64_t>(discretizedBound);
256  STORM_LOG_THROW(epochManager.isValidDimensionValue(dimensionValue), storm::exceptions::NotSupportedException, "The bound " << bound << " is too high for the considered number of dimensions.");
257  dimensions[dim].maxValue = dimensionValue;
258  }
259  }
260  }

263  template<typename ValueType, bool SingleObjectiveMode>
264  void MultiDimensionalRewardUnfolding<ValueType, SingleObjectiveMode>::translateLowerBoundInfinityDimensions(std::vector<Epoch>& epochSteps) {
265  // Translate lowerBoundedByInfinity dimensions to finite bounds
266  storm::storage::BitVector infLowerBoundedDimensions(dimensions.size(), false);
267  storm::storage::BitVector upperBoundedDimensions(dimensions.size(), false);
268  for (uint64_t dim = 0; dim < dimensions.size(); ++dim) {
269  infLowerBoundedDimensions.set(dim, dimensions[dim].boundType == DimensionBoundType::LowerBoundInfinity);
270  upperBoundedDimensions.set(dim, dimensions[dim].boundType == DimensionBoundType::UpperBound);
271  }
272  if (!infLowerBoundedDimensions.empty()) {
273  // We can currently only handle this case for single maximizing bounded until probability objectives.
274  // The approach is to erase all epochSteps that are not part of an end component and then change the reward bound to '> 0'.
275  // Then, reaching a reward means reaching an end component where arbitrarily many reward can be collected.
276  STORM_LOG_THROW(SingleObjectiveMode, storm::exceptions::NotSupportedException, "Letting lower bounds approach infinity is only supported in single objective mode."); // It most likely also works with multiple objectives with the same shape. However, we haven't checked this yet.
277  STORM_LOG_THROW(objectives.front().formula->isProbabilityOperatorFormula(), storm::exceptions::NotSupportedException, "Letting lower bounds approach infinity is only supported for probability operator formulas");
278  auto const& probabilityOperatorFormula = objectives.front().formula->asProbabilityOperatorFormula();
279  STORM_LOG_THROW(probabilityOperatorFormula.getSubformula().isBoundedUntilFormula(), storm::exceptions::NotSupportedException, "Letting lower bounds approach infinity is only supported for bounded until probabilities.");
280  STORM_LOG_THROW(!model.isNondeterministicModel() || (probabilityOperatorFormula.hasOptimalityType() && storm::solver::maximize(probabilityOperatorFormula.getOptimalityType())), storm::exceptions::NotSupportedException, "Letting lower bounds approach infinity is only supported for maximizing bounded until probabilities.");
281 
282  STORM_LOG_THROW(upperBoundedDimensions.empty() || !probabilityOperatorFormula.getSubformula().asBoundedUntilFormula().hasMultiDimensionalSubformulas(), storm::exceptions::NotSupportedException, "Letting lower bounds approach infinity is only supported if the formula has either only lower bounds or if it has a single goal state."); // This would fail because the upper bounded dimension(s) might be satisfied already. One should erase epoch steps in the epoch model (after applying the goal-unfolding).
283  storm::storage::BitVector choicesWithoutUpperBoundedStep(model.getNumberOfChoices(), true);
284  if (!upperBoundedDimensions.empty()) {
285  // To not invalidate upper-bounded dimensions, one needs to consider MECS where no reward for such a dimension is collected.
286  for (uint64_t choiceIndex = 0; choiceIndex < model.getNumberOfChoices(); ++choiceIndex) {
287  for (auto const& dim : upperBoundedDimensions) {
288  if (epochManager.getDimensionOfEpoch(epochSteps[choiceIndex], dim) != 0) {
289  choicesWithoutUpperBoundedStep.set(choiceIndex, false);
290  break;
291  }
292  }
293  }
294  }
295  storm::storage::MaximalEndComponentDecomposition<ValueType> mecDecomposition(model.getTransitionMatrix(), model.getBackwardTransitions(), storm::storage::BitVector(model.getNumberOfStates(), true), choicesWithoutUpperBoundedStep);
296  storm::storage::BitVector nonMecChoices(model.getNumberOfChoices(), true);
297  for (auto const& mec : mecDecomposition) {
298  for (auto const& stateChoicesPair : mec) {
299  for (auto const& choice : stateChoicesPair.second) {
300  nonMecChoices.set(choice, false);
301  }
302  }
303  }
304  for (auto const& choice : nonMecChoices) {
305  for (auto const& dim : infLowerBoundedDimensions) {
306  epochManager.setDimensionOfEpoch(epochSteps[choice], dim, 0);
307  }
308  }
309 
310  // Translate the dimension to '>0'
311  for (auto const& dim : infLowerBoundedDimensions) {
312  dimensions[dim].boundType = DimensionBoundType::LowerBound;
313  dimensions[dim].maxValue = 0;
314  }
315  }
316  }

334  std::vector<typename MultiDimensionalRewardUnfolding<ValueType, SingleObjectiveMode>::Epoch> MultiDimensionalRewardUnfolding<ValueType, SingleObjectiveMode>::getEpochComputationOrder(Epoch const& startEpoch, bool stopAtComputedEpochs) {

336  std::vector<Epoch> dfsStack;
337  std::set<Epoch, std::function<bool(Epoch const&, Epoch const&)>> collectedEpochs(std::bind(&EpochManager::epochClassZigZagOrder, &epochManager, std::placeholders::_1, std::placeholders::_2));
338

340  collectedEpochs.insert(startEpoch);

342  }
343  while (!dfsStack.empty()) {

345  dfsStack.pop_back();
346  for (auto const& step : possibleEpochSteps) {
347  Epoch successorEpoch = epochManager.getSuccessorEpoch(currentEpoch, step);
348  if (!stopAtComputedEpochs || epochSolutions.count(successorEpoch) == 0) {
349  if (collectedEpochs.insert(successorEpoch).second) {
350  dfsStack.push_back(std::move(successorEpoch));
351  }
352  }
353  }
354  }
355  return std::vector<Epoch>(collectedEpochs.begin(), collectedEpochs.end());

367  if (storm::utility::graph::hasCycle(epochModel.epochMatrix)) {

369  }

373  }
374 
375  bool containsLowerBoundedObjective = false;
376  for (auto const& dimension : dimensions) {
377  if (dimension.boundType == DimensionBoundType::LowerBound) {
378  containsLowerBoundedObjective = true;
379  break;
380  }
381  }
382  std::map<Epoch, EpochSolution const*> subSolutions;
383  for (auto const& step : possibleEpochSteps) {
384  Epoch successorEpoch = epochManager.getSuccessorEpoch(epoch, step);
385  if (successorEpoch != epoch) {
386  auto successorSolIt = epochSolutions.find(successorEpoch);
387  STORM_LOG_ASSERT(successorSolIt != epochSolutions.end(), "Solution for successor epoch does not exist (anymore).");
388  subSolutions.emplace(successorEpoch, &successorSolIt->second);
389  }
390  }
391  epochModel.stepSolutions.resize(epochModel.stepChoices.getNumberOfSetBits());
392  auto stepSolIt = epochModel.stepSolutions.begin();
393  for (auto const& reducedChoice : epochModel.stepChoices) {
394  uint64_t productChoice = epochModelToProductChoiceMap[reducedChoice];
395  uint64_t productState = productModel->getProductStateFromChoice(productChoice);
396  auto const& memoryState = productModel->getMemoryState(productState);
397  Epoch successorEpoch = epochManager.getSuccessorEpoch(epoch, productModel->getSteps()[productChoice]);
398  EpochClass successorEpochClass = epochManager.getEpochClass(successorEpoch);
399  // Find out whether objective reward is earned for the current choice
400  // Objective reward is not earned if
401  // a) there is an upper bounded subObjective that is __still_relevant__ but the corresponding reward bound is passed after taking the choice
402  // b) there is a lower bounded subObjective and the corresponding reward bound is not passed yet.
403  for (uint64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
404  bool rewardEarned = !storm::utility::isZero(epochModel.objectiveRewards[objIndex][reducedChoice]);
405  if (rewardEarned) {
406  for (auto const& dim : objectiveDimensions[objIndex]) {
407  if ((dimensions[dim].boundType == DimensionBoundType::UpperBound) == epochManager.isBottomDimension(successorEpoch, dim) && productModel->getMemoryStateManager().isRelevantDimension(memoryState, dim)) {
408  rewardEarned = false;
409  break;
410  }
411  }
412  }
413  epochModel.objectiveRewardFilter[objIndex].set(reducedChoice, rewardEarned);
414  }
415  // compute the solution for the stepChoices
416  // For optimization purposes, we distinguish the case where the memory state does not have to be transformed
417  EpochSolution const& successorEpochSolution = getEpochSolution(subSolutions, successorEpoch);
418  SolutionType choiceSolution;
419  bool firstSuccessor = true;
420  if (!containsLowerBoundedObjective && epochManager.compareEpochClass(epoch, successorEpoch)) {
421  for (auto const& successor : productModel->getProduct().getTransitionMatrix().getRow(productChoice)) {
422  if (firstSuccessor) {
423  choiceSolution = getScaledSolution(getStateSolution(successorEpochSolution, successor.getColumn()), successor.getValue());
424  firstSuccessor = false;
425  } else {
426  addScaledSolution(choiceSolution, getStateSolution(successorEpochSolution, successor.getColumn()), successor.getValue());
427  }
428  }
429  } else {
430  for (auto const& successor : productModel->getProduct().getTransitionMatrix().getRow(productChoice)) {
431  uint64_t successorProductState = productModel->transformProductState(successor.getColumn(), successorEpochClass, memoryState);
432  SolutionType const& successorSolution = getStateSolution(successorEpochSolution, successorProductState);
433  if (firstSuccessor) {
434  choiceSolution = getScaledSolution(successorSolution, successor.getValue());
435  firstSuccessor = false;
436  } else {
437  addScaledSolution(choiceSolution, successorSolution, successor.getValue());
438  }
439  }
440  }
441  assert(!firstSuccessor);
442  *stepSolIt = std::move(choiceSolution);
443  ++stepSolIt;
444  }
445 
446  assert(epochModel.objectiveRewards.size() == objectives.size());
447  assert(epochModel.objectiveRewardFilter.size() == objectives.size());
448  assert(epochModel.epochMatrix.getRowCount() == epochModel.stepChoices.size());
449  assert(epochModel.stepChoices.size() == epochModel.objectiveRewards.front().size());
450  assert(epochModel.objectiveRewards.front().size() == epochModel.objectiveRewards.back().size());
451  assert(epochModel.objectiveRewards.front().size() == epochModel.objectiveRewardFilter.front().size());
452  assert(epochModel.objectiveRewards.back().size() == epochModel.objectiveRewardFilter.back().size());
453  assert(epochModel.stepChoices.getNumberOfSetBits() == epochModel.stepSolutions.size());
454 
455  currentEpoch = epoch;
456  /*
457  std::cout << "Epoch model for epoch " << storm::utility::vector::toString(epoch) << std::endl;
458  std::cout << "Matrix: " << std::endl << epochModel.epochMatrix << std::endl;
459  std::cout << "ObjectiveRewards: " << storm::utility::vector::toString(epochModel.objectiveRewards[0]) << std::endl;
460  std::cout << "steps: " << epochModel.stepChoices << std::endl;
461  std::cout << "step solutions: ";
462  for (int i = 0; i < epochModel.stepSolutions.size(); ++i) {
463  std::cout << " " << epochModel.stepSolutions[i].weightedValue;
464  }
465  std::cout << std::endl;
466  */
467  return epochModel;

471  template<typename ValueType, bool SingleObjectiveMode>
472  void MultiDimensionalRewardUnfolding<ValueType, SingleObjectiveMode>::setCurrentEpochClass(Epoch const& epoch) {
473  EpochClass epochClass = epochManager.getEpochClass(epoch);
474  // std::cout << "Setting epoch class for epoch " << epochManager.toString(epoch) << std::endl;
475  auto productObjectiveRewards = productModel->computeObjectiveRewards(epochClass, objectives);
476 
477  storm::storage::BitVector stepChoices(productModel->getProduct().getTransitionMatrix().getRowCount(), false);
478  uint64_t choice = 0;
479  for (auto const& step : productModel->getSteps()) {
480  if (!epochManager.isZeroEpoch(step) && epochManager.getSuccessorEpoch(epoch, step) != epoch) {
481  stepChoices.set(choice, true);
482  }
483  ++choice;
484  }
485  epochModel.epochMatrix = productModel->getProduct().getTransitionMatrix().filterEntries(~stepChoices);
486  // redirect transitions for the case where the lower reward bounds are not met yet
487  storm::storage::BitVector violatedLowerBoundedDimensions(dimensions.size(), false);
488  for (uint64_t dim = 0; dim < dimensions.size(); ++dim) {
489  if (dimensions[dim].boundType == DimensionBoundType::LowerBound && !epochManager.isBottomDimensionEpochClass(epochClass, dim)) {
490  violatedLowerBoundedDimensions.set(dim);
491  }
492  }
493  if (!violatedLowerBoundedDimensions.empty()) {
494  for (uint64_t state = 0; state < epochModel.epochMatrix.getRowGroupCount(); ++state) {
495  auto const& memoryState = productModel->getMemoryState(state);
496  for (auto& entry : epochModel.epochMatrix.getRowGroup(state)) {
497  entry.setColumn(productModel->transformProductState(entry.getColumn(), epochClass, memoryState));
498  }
499  }
500  }
501 
502  storm::storage::BitVector zeroObjRewardChoices(productModel->getProduct().getTransitionMatrix().getRowCount(), true);
503  for (uint64_t objIndex = 0; objIndex < objectives.size(); ++objIndex) {
504  if (violatedLowerBoundedDimensions.isDisjointFrom(objectiveDimensions[objIndex])) {
505  zeroObjRewardChoices &= storm::utility::vector::filterZero(productObjectiveRewards[objIndex]);
506  }
507  }
508  storm::storage::BitVector allProductStates(productModel->getProduct().getNumberOfStates(), true);
509 
510  // Get the relevant states for this epoch.
511  storm::storage::BitVector productInStates = productModel->getInStates(epochClass);
512  // The epoch model only needs to consider the states that are reachable from a relevant state
513  storm::storage::BitVector consideredStates = storm::utility::graph::getReachableStates(epochModel.epochMatrix, productInStates, allProductStates, ~allProductStates);
514 
515  // We assume that there is no end component in which objective reward is earned
516  STORM_LOG_ASSERT(!storm::utility::graph::checkIfECWithChoiceExists(epochModel.epochMatrix, epochModel.epochMatrix.transpose(true), allProductStates, ~zeroObjRewardChoices & ~stepChoices), "There is a scheduler that yields infinite reward for one objective. This case should be excluded");
517 
518  // Create the epoch model matrix
519  std::vector<uint64_t> productToEpochModelStateMapping;
520  if (model.isOfType(storm::models::ModelType::Dtmc)) {
521  assert(zeroObjRewardChoices.size() == productModel->getProduct().getNumberOfStates());
522  assert(stepChoices.size() == productModel->getProduct().getNumberOfStates());
523  STORM_LOG_ASSERT(epochModel.equationSolverProblemFormat.is_initialized(), "Linear equation problem format was not set.");
524  bool convertToEquationSystem = epochModel.equationSolverProblemFormat.get() == storm::solver::LinearEquationSolverProblemFormat::EquationSystem;
525  // For DTMCs we consider the subsystem induced by the considered states.
526  // The transitions for states with zero reward are filtered out to guarantee a unique solution of the eq-system.
527  auto backwardTransitions = epochModel.epochMatrix.transpose(true);
528  storm::storage::BitVector nonZeroRewardStates = storm::utility::graph::performProbGreater0(backwardTransitions, consideredStates, consideredStates & (~zeroObjRewardChoices | stepChoices));
529  // If there is at least one considered state with reward zero, we have to add a 'zero-reward-state' to the epoch model.
530  bool requiresZeroRewardState = nonZeroRewardStates != consideredStates;
531  uint64_t numEpochModelStates = nonZeroRewardStates.getNumberOfSetBits();
532  uint64_t zeroRewardInState = numEpochModelStates;
533  if (requiresZeroRewardState) {
534  ++numEpochModelStates;
535  }
536  storm::storage::SparseMatrixBuilder<ValueType> builder;
537  if (!nonZeroRewardStates.empty()) {
538  builder = storm::storage::SparseMatrixBuilder<ValueType>(epochModel.epochMatrix.getSubmatrix(true, nonZeroRewardStates, nonZeroRewardStates, convertToEquationSystem));
539  }
540  if (requiresZeroRewardState) {
541  if (convertToEquationSystem) {
542  // add a diagonal entry
543  builder.addNextValue(zeroRewardInState, zeroRewardInState, storm::utility::zero<ValueType>());
544  }
545  epochModel.epochMatrix = builder.build(numEpochModelStates, numEpochModelStates);
546  } else {
547  assert (!nonZeroRewardStates.empty());
548  epochModel.epochMatrix = builder.build();
549  }
550  if (convertToEquationSystem) {
551  epochModel.epochMatrix.convertToEquationSystem();
552  }
553 
554  epochModelToProductChoiceMap.clear();
555  epochModelToProductChoiceMap.reserve(numEpochModelStates);
556  productToEpochModelStateMapping.assign(nonZeroRewardStates.size(), zeroRewardInState);
557  for (auto const& productState : nonZeroRewardStates) {
558  productToEpochModelStateMapping[productState] = epochModelToProductChoiceMap.size();
559  epochModelToProductChoiceMap.push_back(productState);
560  }
561  if (requiresZeroRewardState) {
562  uint64_t zeroRewardProductState = (consideredStates & ~nonZeroRewardStates).getNextSetIndex(0);
563  assert(zeroRewardProductState < consideredStates.size());
564  epochModelToProductChoiceMap.push_back(zeroRewardProductState);
565  }
566  } else if (model.isOfType(storm::models::ModelType::Mdp)) {
567  // Eliminate zero-reward end components
568  auto ecElimResult = storm::transformer::EndComponentEliminator<ValueType>::transform(epochModel.epochMatrix, consideredStates, zeroObjRewardChoices & ~stepChoices, consideredStates);
569  epochModel.epochMatrix = std::move(ecElimResult.matrix);
570  epochModelToProductChoiceMap = std::move(ecElimResult.newToOldRowMapping);
571  productToEpochModelStateMapping = std::move(ecElimResult.oldToNewStateMapping);
572  } else {
573  STORM_LOG_THROW(false, storm::exceptions::UnexpectedException, "Unsupported model type.");
574  }
575 
576  epochModel.stepChoices = storm::storage::BitVector(epochModel.epochMatrix.getRowCount(), false);
577  for (uint64_t choice = 0; choice < epochModel.epochMatrix.getRowCount(); ++choice) {
578  if (stepChoices.get(epochModelToProductChoiceMap[choice])) {
579  epochModel.stepChoices.set(choice, true);
580  }
581  }
582 
583  epochModel.objectiveRewards.clear();
584  for (uint64_t objIndex = 0; objIndex < objectives.size(); ++objIndex) {
585  std::vector<ValueType> const& productObjRew = productObjectiveRewards[objIndex];
586  std::vector<ValueType> reducedModelObjRewards;
587  reducedModelObjRewards.reserve(epochModel.epochMatrix.getRowCount());
588  for (auto const& productChoice : epochModelToProductChoiceMap) {
589  reducedModelObjRewards.push_back(productObjRew[productChoice]);
590  }
591  // Check if the objective is violated in the current epoch
592  if (!violatedLowerBoundedDimensions.isDisjointFrom(objectiveDimensions[objIndex])) {
593  storm::utility::vector::setVectorValues(reducedModelObjRewards, ~epochModel.stepChoices, storm::utility::zero<ValueType>());
594  }
595  epochModel.objectiveRewards.push_back(std::move(reducedModelObjRewards));
596  }
597 
598  epochModel.epochInStates = storm::storage::BitVector(epochModel.epochMatrix.getRowGroupCount(), false);
599  for (auto const& productState : productInStates) {
600  STORM_LOG_ASSERT(productToEpochModelStateMapping[productState] < epochModel.epochMatrix.getRowGroupCount(), "Selected product state does not exist in the epoch model.");
601  epochModel.epochInStates.set(productToEpochModelStateMapping[productState], true);
602  }
603 
604  std::vector<uint64_t> toEpochModelInStatesMap(productModel->getProduct().getNumberOfStates(), std::numeric_limits<uint64_t>::max());
605  std::vector<uint64_t> epochModelStateToInStateMap = epochModel.epochInStates.getNumberOfSetBitsBeforeIndices();
606  for (auto const& productState : productInStates) {
607  toEpochModelInStatesMap[productState] = epochModelStateToInStateMap[productToEpochModelStateMapping[productState]];
608  }
609  productStateToEpochModelInStateMap = std::make_shared<std::vector<uint64_t> const>(std::move(toEpochModelInStatesMap));
610 
611  epochModel.objectiveRewardFilter.clear();
612  for (auto const& objRewards : epochModel.objectiveRewards) {
613  epochModel.objectiveRewardFilter.push_back(storm::utility::vector::filterZero(objRewards));
614  epochModel.objectiveRewardFilter.back().complement();
615  }
616  }

699  template<typename ValueType, bool SingleObjectiveMode>
700  boost::optional<ValueType> MultiDimensionalRewardUnfolding<ValueType, SingleObjectiveMode>::getUpperObjectiveBound(uint64_t objectiveIndex) {
701  auto& objective = this->objectives[objectiveIndex];
702  if (!objective.upperResultBound) {
703  if (objective.formula->isProbabilityOperatorFormula()) {
704  objective.upperResultBound = storm::utility::one<ValueType>();
705  } else if (objective.formula->isRewardOperatorFormula()) {
706  auto const& rewModel = this->model.getRewardModel(objective.formula->asRewardOperatorFormula().getRewardModelName());
707  auto actionRewards = rewModel.getTotalRewardVector(this->model.getTransitionMatrix());
708  if (objective.formula->getSubformula().isCumulativeRewardFormula()) {
709  // Try to get an upper bound by computing the maximal reward achievable within one epoch step
710  auto const& cumulativeRewardFormula = objective.formula->getSubformula().asCumulativeRewardFormula();
711  for (uint64_t objDim = 0; objDim < cumulativeRewardFormula.getDimension(); ++objDim) {
712  boost::optional<ValueType> resBound;
713  ValueType rewardBound = cumulativeRewardFormula.template getBound<ValueType>(objDim);
714  if (cumulativeRewardFormula.getTimeBoundReference(objDim).isRewardBound()) {
715  auto const& costModel = this->model.getRewardModel(cumulativeRewardFormula.getTimeBoundReference(objDim).getRewardName());
716  if (!costModel.hasTransitionRewards()) {
717  auto actionCosts = costModel.getTotalRewardVector(this->model.getTransitionMatrix());
718  ValueType largestRewardPerCost = storm::utility::zero<ValueType>();
719  bool isFinite = true;
720  for (auto rewIt = actionRewards.begin(), costIt = actionCosts.begin(); rewIt != actionRewards.end(); ++rewIt, ++costIt) {
721  if (!storm::utility::isZero(*rewIt)) {
722  if (storm::utility::isZero(*costIt)) {
723  isFinite = false;
724  break;
725  }
726  ValueType rewardPerCost = *rewIt / *costIt;
727  largestRewardPerCost = std::max(largestRewardPerCost, rewardPerCost);
728  }
729  }
730  if (isFinite) {
731  resBound = largestRewardPerCost * rewardBound;
732  }
733  }
734  } else {
735  resBound = (*std::max_element(actionRewards.begin(), actionRewards.end())) * rewardBound;
736  }
737  if (resBound && (!objective.upperResultBound || objective.upperResultBound.get() > resBound.get())) {
738  objective.upperResultBound = resBound;
739  }
740  }
741 
742  // If we could not find an upper bound, try to get an upper bound for the unbounded case
743  if (!objective.upperResultBound) {
744  storm::storage::BitVector allStates(model.getNumberOfStates(), true);
745  // Get the set of states from which reward is reachable
746  auto nonZeroRewardStates = rewModel.getStatesWithZeroReward(model.getTransitionMatrix());
747  nonZeroRewardStates.complement();
748  auto expRewGreater0EStates = storm::utility::graph::performProbGreater0E(model.getBackwardTransitions(), allStates, nonZeroRewardStates);
749  // Eliminate zero-reward ECs
750  auto zeroRewardChoices = rewModel.getChoicesWithZeroReward(model.getTransitionMatrix());
751  auto ecElimRes = storm::transformer::EndComponentEliminator<ValueType>::transform(model.getTransitionMatrix(), expRewGreater0EStates, zeroRewardChoices, ~allStates);
752  allStates.resize(ecElimRes.matrix.getRowGroupCount());
753  storm::storage::BitVector outStates(allStates.size(), false);
754  std::vector<ValueType> rew0StateProbs;
755  rew0StateProbs.reserve(ecElimRes.matrix.getRowCount());
756  for (uint64_t state = 0; state < allStates.size(); ++ state) {
757  for (uint64_t choice = ecElimRes.matrix.getRowGroupIndices()[state]; choice < ecElimRes.matrix.getRowGroupIndices()[state + 1]; ++choice) {
758  // Check whether the choice lead to a state with expRew 0 in the original model
759  bool isOutChoice = false;
760  uint64_t originalModelChoice = ecElimRes.newToOldRowMapping[choice];
761  for (auto const& entry : model.getTransitionMatrix().getRow(originalModelChoice)) {
762  if (!expRewGreater0EStates.get(entry.getColumn())) {
763  isOutChoice = true;
764  outStates.set(state, true);
765  rew0StateProbs.push_back(storm::utility::one<ValueType>() - ecElimRes.matrix.getRowSum(choice));
766  assert (!storm::utility::isZero(rew0StateProbs.back()));
767  break;
768  }
769  }
770  if (!isOutChoice) {
771  rew0StateProbs.push_back(storm::utility::zero<ValueType>());
772  }
773  }
774  }
775  // An upper reward bound can only be computed if it is below infinity
776  if (storm::utility::graph::performProb1A(ecElimRes.matrix, ecElimRes.matrix.getRowGroupIndices(), ecElimRes.matrix.transpose(true), allStates, outStates).full()) {
777  std::vector<ValueType> rewards;
778  rewards.reserve(ecElimRes.matrix.getRowCount());
779  for (auto row : ecElimRes.newToOldRowMapping) {
780  rewards.push_back(actionRewards[row]);
781  }
782  storm::modelchecker::helper::BaierUpperRewardBoundsComputer<ValueType> baier(ecElimRes.matrix, rewards, rew0StateProbs);
783  objective.upperResultBound = baier.computeUpperBound();
784  }
785  }
786  }
787  }
788  }
789  return objective.upperResultBound;
790  }
