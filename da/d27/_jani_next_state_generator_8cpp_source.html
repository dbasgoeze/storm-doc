207  std::vector<StateType> JaniNextStateGenerator<ValueType, StateType>::getInitialStates(StateToIdCallback const& stateToIdCallback) {
208  std::vector<StateType> initialStateIndices;
209 
210  if (this->model.hasNonTrivialInitialStates()) {
211  // Prepare an SMT solver to enumerate all initial states.
212  storm::utility::solver::SmtSolverFactory factory;
213  std::unique_ptr<storm::solver::SmtSolver> solver = factory.create(model.getExpressionManager());
214 
215  std::vector<storm::expressions::Expression> rangeExpressions = model.getAllRangeExpressions(this->parallelAutomata);
216  for (auto const& expression : rangeExpressions) {
217  solver->add(expression);
218  }
219  solver->add(model.getInitialStatesExpression(this->parallelAutomata));
220 
221  // Proceed as long as the solver can still enumerate initial states.
222  while (solver->check() == storm::solver::SmtSolver::CheckResult::Sat) {
223  // Create fresh state.
224  CompressedState initialState(this->variableInformation.getTotalBitOffset(true));
225 
226  // Read variable assignment from the solution of the solver. Also, create an expression we can use to
227  // prevent the variable assignment from being enumerated again.
228  storm::expressions::Expression blockingExpression;
229  std::shared_ptr<storm::solver::SmtSolver::ModelReference> model = solver->getModel();
230  for (auto const& booleanVariable : this->variableInformation.booleanVariables) {
231  bool variableValue = model->getBooleanValue(booleanVariable.variable);
232  storm::expressions::Expression localBlockingExpression = variableValue ? !booleanVariable.variable : booleanVariable.variable;
233  blockingExpression = blockingExpression.isInitialized() ? blockingExpression || localBlockingExpression : localBlockingExpression;
234  initialState.set(booleanVariable.bitOffset, variableValue);
235  }
236  for (auto const& integerVariable : this->variableInformation.integerVariables) {
237  int_fast64_t variableValue = model->getIntegerValue(integerVariable.variable);
238  if (integerVariable.forceOutOfBoundsCheck || this->getOptions().isExplorationChecksSet()) {
239  STORM_LOG_THROW(variableValue >= integerVariable.lowerBound, storm::exceptions::WrongFormatException, "The initial value for variable " << integerVariable.variable.getName() << " is lower than the lower bound.");
240  STORM_LOG_THROW(variableValue <= integerVariable.upperBound, storm::exceptions::WrongFormatException, "The initial value for variable " << integerVariable.variable.getName() << " is higher than the upper bound");
241  }
242  storm::expressions::Expression localBlockingExpression = integerVariable.variable != model->getManager().integer(variableValue);
243  blockingExpression = blockingExpression.isInitialized() ? blockingExpression || localBlockingExpression : localBlockingExpression;
244  initialState.setFromInt(integerVariable.bitOffset, integerVariable.bitWidth, static_cast<uint_fast64_t>(variableValue - integerVariable.lowerBound));
245  }
246 
247  // Gather iterators to the initial locations of all the automata.
248  std::vector<std::set<uint64_t>::const_iterator> initialLocationsIts;
249  std::vector<std::set<uint64_t>::const_iterator> initialLocationsItes;
250  for (auto const& automatonRef : this->parallelAutomata) {
251  auto const& automaton = automatonRef.get();
252  initialLocationsIts.push_back(automaton.getInitialLocationIndices().cbegin());
253  initialLocationsItes.push_back(automaton.getInitialLocationIndices().cend());
254  }
255  storm::utility::combinatorics::forEach(initialLocationsIts, initialLocationsItes, [this,&initialState] (uint64_t index, uint64_t value) { setLocation(initialState, this->variableInformation.locationVariables[index], value); }, [&stateToIdCallback,&initialStateIndices,&initialState] () {
256  // Register initial state.
257  StateType id = stateToIdCallback(initialState);
258  initialStateIndices.push_back(id);
259  return true;
260  });
261 
262  // Block the current initial state to search for the next one.
263  if (!blockingExpression.isInitialized()) {
264  break;
265  }
266  solver->add(blockingExpression);
267  }
268 
269  STORM_LOG_DEBUG("Enumerated " << initialStateIndices.size() << " initial states using SMT solving.");
270  } else {
271  CompressedState initialState(this->variableInformation.getTotalBitOffset(true));
272 
273  std::vector<int_fast64_t> currentIntegerValues;
274  currentIntegerValues.reserve(this->variableInformation.integerVariables.size());
275  for (auto const& variable : this->variableInformation.integerVariables) {
276  STORM_LOG_THROW(variable.lowerBound <= variable.upperBound, storm::exceptions::InvalidArgumentException, "Expecting variable with non-empty set of possible values.");
277  currentIntegerValues.emplace_back(0);
278  initialState.setFromInt(variable.bitOffset, variable.bitWidth, 0);
279  }
280 
281  initialStateIndices.emplace_back(stateToIdCallback(initialState));
282 
283  bool done = false;
284  while (!done) {
285  bool changedBooleanVariable = false;
286  for (auto const& booleanVariable : this->variableInformation.booleanVariables) {
287  if (initialState.get(booleanVariable.bitOffset)) {
288  initialState.set(booleanVariable.bitOffset);
289  changedBooleanVariable = true;
290  break;
291  } else {
292  initialState.set(booleanVariable.bitOffset, false);
293  }
294  }
295 
296  bool changedIntegerVariable = false;
297  if (changedBooleanVariable) {
298  initialStateIndices.emplace_back(stateToIdCallback(initialState));
299  } else {
300  for (uint64_t integerVariableIndex = 0; integerVariableIndex < this->variableInformation.integerVariables.size(); ++integerVariableIndex) {
301  auto const& integerVariable = this->variableInformation.integerVariables[integerVariableIndex];
302  if (currentIntegerValues[integerVariableIndex] < integerVariable.upperBound - integerVariable.lowerBound) {
303  ++currentIntegerValues[integerVariableIndex];
304  changedIntegerVariable = true;
305  } else {
306  currentIntegerValues[integerVariableIndex] = integerVariable.lowerBound;
307  }
308  initialState.setFromInt(integerVariable.bitOffset, integerVariable.bitWidth, currentIntegerValues[integerVariableIndex]);
309 
310  if (changedIntegerVariable) {
311  break;
312  }
313  }
314  }
315 
316  if (changedIntegerVariable) {
317  initialStateIndices.emplace_back(stateToIdCallback(initialState));
318  }
319 
320  done = !changedBooleanVariable && !changedIntegerVariable;
321  }
322 
323  STORM_LOG_DEBUG("Enumerated " << initialStateIndices.size() << " initial states using brute force enumeration.");
324  }
325 
326  return initialStateIndices;
327  }

330  void JaniNextStateGenerator<ValueType, StateType>::applyUpdate(CompressedState& state, storm::jani::EdgeDestination const& destination, storm::generator::LocationVariableInformation const& locationVariable, int64_t assignmentLevel, storm::expressions::ExpressionEvaluator<ValueType> const& expressionEvaluator) {

332  // Update the location of the state.
333  setLocation(state, locationVariable, destination.getLocationIndex());
334 
335  // Then perform the assignments.
336  auto const& assignments = destination.getOrderedAssignments().getNonTransientAssignments(assignmentLevel);
337  auto assignmentIt = assignments.begin();
338  auto assignmentIte = assignments.end();
339 
340  // Iterate over all boolean assignments and carry them out.
341  auto boolIt = this->variableInformation.booleanVariables.begin();
342  for (; assignmentIt != assignmentIte && assignmentIt->lValueIsVariable() && assignmentIt->getExpressionVariable().hasBooleanType(); ++assignmentIt) {
343  while (assignmentIt->getExpressionVariable() != boolIt->variable) {
344  ++boolIt;
345  }
346  state.set(boolIt->bitOffset, expressionEvaluator.asBool(assignmentIt->getAssignedExpression()));
347  }
348 
349  // Iterate over all integer assignments and carry them out.
350  auto integerIt = this->variableInformation.integerVariables.begin();
351  for (; assignmentIt != assignmentIte && assignmentIt->lValueIsVariable() && assignmentIt->getExpressionVariable().hasIntegerType(); ++assignmentIt) {
352  while (assignmentIt->getExpressionVariable() != integerIt->variable) {
353  ++integerIt;
354  }
355  int_fast64_t assignedValue = expressionEvaluator.asInt(assignmentIt->getAssignedExpression());
356  if (this->options.isAddOutOfBoundsStateSet()) {
357  if (assignedValue < integerIt->lowerBound || assignedValue > integerIt->upperBound) {
358  state = this->outOfBoundsState;
359  }
360  } else if (integerIt->forceOutOfBoundsCheck || this->options.isExplorationChecksSet()) {
361  STORM_LOG_THROW(assignedValue >= integerIt->lowerBound, storm::exceptions::WrongFormatException, "The update " << assignmentIt->getExpressionVariable().getName() << " := " << assignmentIt->getAssignedExpression() << " leads to an out-of-bounds value (" << assignedValue << ") for the variable '" << assignmentIt->getExpressionVariable().getName() << "'.");
362  STORM_LOG_THROW(assignedValue <= integerIt->upperBound, storm::exceptions::WrongFormatException, "The update " << assignmentIt->getExpressionVariable().getName() << " := " << assignmentIt->getAssignedExpression() << " leads to an out-of-bounds value (" << assignedValue << ") for the variable '" << assignmentIt->getExpressionVariable().getName() << "'.");
363  }
364  state.setFromInt(integerIt->bitOffset, integerIt->bitWidth, assignedValue - integerIt->lowerBound);
365  STORM_LOG_ASSERT(static_cast<int_fast64_t>(state.getAsInt(integerIt->bitOffset, integerIt->bitWidth)) + integerIt->lowerBound == assignedValue, "Writing to the bit vector bucket failed (read " << state.getAsInt(integerIt->bitOffset, integerIt->bitWidth) << " but wrote " << assignedValue << ").");
366  }
367  // Iterate over all array access assignments and carry them out.
368  for (; assignmentIt != assignmentIte && assignmentIt->lValueIsArrayAccess(); ++assignmentIt) {
369  int_fast64_t arrayIndex = expressionEvaluator.asInt(assignmentIt->getLValue().getArrayIndex());
370  if (assignmentIt->getAssignedExpression().hasIntegerType()) {
371  IntegerVariableInformation const& intInfo = this->variableInformation.getIntegerArrayVariableReplacement(assignmentIt->getLValue().getArray().getExpressionVariable(), arrayIndex);
372  int_fast64_t assignedValue = expressionEvaluator.asInt(assignmentIt->getAssignedExpression());
373 
374  if (this->options.isAddOutOfBoundsStateSet()) {
375  if (assignedValue < intInfo.lowerBound || assignedValue > intInfo.upperBound) {
376  state = this->outOfBoundsState;
377  }
378  } else if (this->options.isExplorationChecksSet()) {
379  STORM_LOG_THROW(assignedValue >= intInfo.lowerBound, storm::exceptions::WrongFormatException, "The update " << assignmentIt->getLValue() << " := " << assignmentIt->getAssignedExpression() << " leads to an out-of-bounds value (" << assignedValue << ") for the variable '" << assignmentIt->getExpressionVariable().getName() << "'.");
380  STORM_LOG_THROW(assignedValue <= intInfo.upperBound, storm::exceptions::WrongFormatException, "The update " << assignmentIt->getLValue() << " := " << assignmentIt->getAssignedExpression() << " leads to an out-of-bounds value (" << assignedValue << ") for the variable '" << assignmentIt->getExpressionVariable().getName() << "'.");
381  }
382  state.setFromInt(intInfo.bitOffset, intInfo.bitWidth, assignedValue - intInfo.lowerBound);
383  STORM_LOG_ASSERT(static_cast<int_fast64_t>(state.getAsInt(intInfo.bitOffset, intInfo.bitWidth)) + intInfo.lowerBound == assignedValue, "Writing to the bit vector bucket failed (read " << state.getAsInt(intInfo.bitOffset, intInfo.bitWidth) << " but wrote " << assignedValue << ").");
384  } else if (assignmentIt->getAssignedExpression().hasBooleanType()) {
385  BooleanVariableInformation const& boolInfo = this->variableInformation.getBooleanArrayVariableReplacement(assignmentIt->getLValue().getArray().getExpressionVariable(), arrayIndex);
386  state.set(boolInfo.bitOffset, expressionEvaluator.asBool(assignmentIt->getAssignedExpression()));
387  } else {
388  STORM_LOG_THROW(false, storm::exceptions::UnexpectedException, "Unhandled type of base variable.");
389  }
390  }
391 
392  // Check that we processed all assignments.
393  STORM_LOG_ASSERT(assignmentIt == assignmentIte, "Not all assignments were consumed.");
394  }

440  STORM_LOG_THROW(assignedValue >= intInfo.lowerBound, storm::exceptions::WrongFormatException, "The update " << assignmentIt->getLValue() << " := " << assignmentIt->getAssignedExpression() << " leads to an out-of-bounds value (" << assignedValue << ") for the variable '" << assignmentIt->getExpressionVariable().getName() << "'.");
441  STORM_LOG_THROW(assignedValue <= intInfo.upperBound, storm::exceptions::WrongFormatException, "The update " << assignmentIt->getLValue() << " := " << assignmentIt->getAssignedExpression() << " leads to an out-of-bounds value (" << assignedValue << ") for the variable '" << assignmentIt->getExpressionVariable().getName() << "'.");
442  }
443  transientValuation.integerValues.emplace_back(&intInfo, assignedValue);
444  } else if (baseType.isBooleanType()) {
445  auto const& boolInfo = this->transientVariableInformation.getBooleanArrayVariableReplacement(assignmentIt->getLValue().getArray().getExpressionVariable(), arrayIndex);
446  transientValuation.booleanValues.emplace_back(&boolInfo, expressionEvaluator.asBool(assignmentIt->getAssignedExpression()));
447  } else if (baseType.isRationalType()) {
448  auto const& rationalInfo = this->transientVariableInformation.getRationalArrayVariableReplacement(assignmentIt->getLValue().getArray().getExpressionVariable(), arrayIndex);
449  transientValuation.rationalValues.emplace_back(&rationalInfo, expressionEvaluator.asRational(assignmentIt->getAssignedExpression()));
450  } else {
451  STORM_LOG_THROW(false, storm::exceptions::UnexpectedException, "Unhandled type of base variable.");
452  }
453  }
454 
455  // Check that we processed all assignments.
456  STORM_LOG_ASSERT(assignmentIt == assignmentIte, "Not all assignments were consumed.");
457  }

551  template<typename ValueType, typename StateType>
552  StateBehavior<ValueType, StateType> JaniNextStateGenerator<ValueType, StateType>::expand(StateToIdCallback const& stateToIdCallback) {
553  // Prepare the result, in case we return early.
554  StateBehavior<ValueType, StateType> result;
555 
556  // Retrieve the locations from the state.
557  std::vector<uint64_t> locations = getLocations(*this->state);
558 
559  // First, construct the state rewards, as we may return early if there are no choices later and we already
560  // need the state rewards then.
561  auto transientVariableValuation = getTransientVariableValuationAtLocations(locations, *this->evaluator);
562  transientVariableValuation.setInEvaluator(*this->evaluator, this->getOptions().isExplorationChecksSet());
563  result.addStateRewards(evaluateRewardExpressions());
564  this->transientVariableInformation.setDefaultValuesInEvaluator(*this->evaluator);
565 
566 
567  // If a terminal expression was set and we must not expand this state, return now.
568  if (!this->terminalStates.empty()) {
569  for (auto const& expressionBool : this->terminalStates) {
570  if (this->evaluator->asBool(expressionBool.first) == expressionBool.second) {
571  return result;
572  }
573  }
574  }
575 
576  // Get all choices for the state.
577  result.setExpanded();
578  std::vector<Choice<ValueType>> allChoices;
579  if (this->getOptions().isApplyMaximalProgressAssumptionSet()) {
580  // First explore only edges without a rate
581  allChoices = getActionChoices(locations, *this->state, stateToIdCallback, EdgeFilter::WithoutRate);
582  if (allChoices.empty()) {
583  // Expand the Markovian edges if there are no probabilistic ones.
584  allChoices = getActionChoices(locations, *this->state, stateToIdCallback, EdgeFilter::WithRate);
585  }
586  } else {
587  allChoices = getActionChoices(locations, *this->state, stateToIdCallback);
588  }
589  std::size_t totalNumberOfChoices = allChoices.size();
590 
591  // If there is not a single choice, we return immediately, because the state has no behavior (other than
592  // the state reward).
593  if (totalNumberOfChoices == 0) {
594  return result;
595  }
596 
597  // If the model is a deterministic model, we need to fuse the choices into one.
598  if (this->isDeterministicModel() && totalNumberOfChoices > 1) {
599  Choice<ValueType> globalChoice;
600 
601  if (this->options.isAddOverlappingGuardLabelSet()) {
602  this->overlappingGuardStates->push_back(stateToIdCallback(*this->state));
603  }
604 
605  // For CTMCs, we need to keep track of the total exit rate to scale the action rewards later. For DTMCs
606  // this is equal to the number of choices, which is why we initialize it like this here.
607  ValueType totalExitRate = this->isDiscreteTimeModel() ? static_cast<ValueType>(totalNumberOfChoices) : storm::utility::zero<ValueType>();
608 
609  // Iterate over all choices and combine the probabilities/rates into one choice.
610  for (auto const& choice : allChoices) {
611  for (auto const& stateProbabilityPair : choice) {
612  if (this->isDiscreteTimeModel()) {
613  globalChoice.addProbability(stateProbabilityPair.first, stateProbabilityPair.second / totalNumberOfChoices);
614  } else {
615  globalChoice.addProbability(stateProbabilityPair.first, stateProbabilityPair.second);
616  }
617  }
618 
619  if (hasStateActionRewards && !this->isDiscreteTimeModel()) {
620  totalExitRate += choice.getTotalMass();
621  }
622  }
623 
624  std::vector<ValueType> stateActionRewards(rewardExpressions.size(), storm::utility::zero<ValueType>());
625  for (auto const& choice : allChoices) {
626  if (hasStateActionRewards) {
627  for (uint_fast64_t rewardVariableIndex = 0; rewardVariableIndex < rewardExpressions.size(); ++rewardVariableIndex) {
628  stateActionRewards[rewardVariableIndex] += choice.getRewards()[rewardVariableIndex] * choice.getTotalMass() / totalExitRate;
629  }
630  }
631 
632  if (this->options.isBuildChoiceOriginsSet() && choice.hasOriginData()) {
633  globalChoice.addOriginData(choice.getOriginData());
634  }
635  }
636  globalChoice.addRewards(std::move(stateActionRewards));
637 
638  // Move the newly fused choice in place.
639  allChoices.clear();
640  allChoices.push_back(std::move(globalChoice));
641  }
642 
643  // Move all remaining choices in place.
644  for (auto& choice : allChoices) {
645  result.addChoice(std::move(choice));
646  }
647 
648  this->postprocess(result);
649 
650  return result;
651  }

653  template<typename ValueType, typename StateType>
654  Choice<ValueType> JaniNextStateGenerator<ValueType, StateType>::expandNonSynchronizingEdge(storm::jani::Edge const& edge, uint64_t outputActionIndex, uint64_t automatonIndex, CompressedState const& state, StateToIdCallback stateToIdCallback) {
655  // Determine the exit rate if it's a Markovian edge.
656  boost::optional<ValueType> exitRate = boost::none;
657  if (edge.hasRate()) {
658  exitRate = this->evaluator->asRational(edge.getRate());
659  }
660 
661  Choice<ValueType> choice(edge.getActionIndex(), static_cast<bool>(exitRate));
662  std::vector<ValueType> stateActionRewards;
663 
664  // Perform the transient edge assignments and create the state action rewards
665  TransientVariableValuation<ValueType> transientVariableValuation;
666  if (!evaluateRewardExpressionsAtEdges || edge.getAssignments().empty()) {
667  stateActionRewards.resize(rewardModelInformation.size(), storm::utility::zero<ValueType>());
668  } else {
669  for (int64_t assignmentLevel = edge.getAssignments().getLowestLevel(true); assignmentLevel <= edge.getAssignments().getHighestLevel(true); ++assignmentLevel) {
670  transientVariableValuation.clear();
671  applyTransientUpdate(transientVariableValuation, edge.getAssignments().getTransientAssignments(assignmentLevel), *this->evaluator);
672  transientVariableValuation.setInEvaluator(*this->evaluator, this->getOptions().isExplorationChecksSet());
673  }
674  stateActionRewards = evaluateRewardExpressions();
675  transientVariableInformation.setDefaultValuesInEvaluator(*this->evaluator);
676  }
677 
678  // Iterate over all updates of the current command.
679  ValueType probabilitySum = storm::utility::zero<ValueType>();
680  for (auto const& destination : edge.getDestinations()) {
681  ValueType probability = this->evaluator->asRational(destination.getProbability());
682 
683  if (probability != storm::utility::zero<ValueType>()) {
684  bool evaluatorChanged = false;
685  // Obtain target state index and add it to the list of known states. If it has not yet been
686  // seen, we also add it to the set of states that have yet to be explored.
687  int64_t assignmentLevel = edge.getLowestAssignmentLevel(); // Might be the largest possible integer, if there is no assignment
688  int64_t const& highestLevel = edge.getHighestAssignmentLevel();
689  bool hasTransientAssignments = destination.hasTransientAssignment();
690  CompressedState newState = state;
691  applyUpdate(newState, destination, this->variableInformation.locationVariables[automatonIndex], assignmentLevel, *this->evaluator);
692  if (hasTransientAssignments) {
693  STORM_LOG_ASSERT(this->options.isScaleAndLiftTransitionRewardsSet(), "Transition rewards are not supported and scaling to action rewards is disabled.");
694  transientVariableValuation.clear();
695  applyTransientUpdate(transientVariableValuation, destination.getOrderedAssignments().getTransientAssignments(assignmentLevel), *this->evaluator);
696  transientVariableValuation.setInEvaluator(*this->evaluator, this->getOptions().isExplorationChecksSet());
697  evaluatorChanged = true;
698  }
699  if (assignmentLevel < highestLevel) {
700  while (assignmentLevel < highestLevel) {
701  ++assignmentLevel;
702  unpackStateIntoEvaluator(newState, this->variableInformation, *this->evaluator);
703  evaluatorChanged = true;
704  applyUpdate(newState, destination, this->variableInformation.locationVariables[automatonIndex], assignmentLevel, *this->evaluator);
705  if (hasTransientAssignments) {
706  transientVariableValuation.clear();
707  applyTransientUpdate(transientVariableValuation, destination.getOrderedAssignments().getTransientAssignments(assignmentLevel), *this->evaluator);
708  transientVariableValuation.setInEvaluator(*this->evaluator, this->getOptions().isExplorationChecksSet());
709  evaluatorChanged = true;
710  }
711  }
712  }
713  if (evaluateRewardExpressionsAtDestinations) {
714  unpackStateIntoEvaluator(newState, this->variableInformation, *this->evaluator);
715  evaluatorChanged = true;
716  addEvaluatedRewardExpressions(stateActionRewards, probability);
717  }
718 
719  if (evaluatorChanged) {
720  // Restore the old variable valuation
721  unpackStateIntoEvaluator(state, this->variableInformation, *this->evaluator);
722  if (hasTransientAssignments) {
723  this->transientVariableInformation.setDefaultValuesInEvaluator(*this->evaluator);
724  }
725  }
726 
727  StateType stateIndex = stateToIdCallback(newState);
728 
729  // Update the choice by adding the probability/target state to it.
730  probability = exitRate ? exitRate.get() * probability : probability;
731  choice.addProbability(stateIndex, probability);
732 
733  if (this->options.isExplorationChecksSet()) {
734  probabilitySum += probability;
735  }
736  }
737  }
738 
739  // Add the state action rewards
740  choice.addRewards(std::move(stateActionRewards));
741 
742  if (this->options.isExplorationChecksSet()) {
743  // Check that the resulting distribution is in fact a distribution.
744  STORM_LOG_THROW(!this->isDiscreteTimeModel() || this->comparator.isOne(probabilitySum), storm::exceptions::WrongFormatException, "Probabilities do not sum to one for edge (actually sum to " << probabilitySum << ").");
745  }
746 
747  return choice;
748  }

750  template<typename ValueType, typename StateType>
751  void JaniNextStateGenerator<ValueType, StateType>::generateSynchronizedDistribution(storm::storage::BitVector const& state, AutomataEdgeSets const& edgeCombination, std::vector<EdgeSetWithIndices::const_iterator> const& iteratorList, storm::builder::jit::Distribution<StateType, ValueType>& distribution, std::vector<ValueType>& stateActionRewards, EdgeIndexSet& edgeIndices, StateToIdCallback stateToIdCallback) {
752 
753  // Collect some information of the edges.
754  int64_t lowestDestinationAssignmentLevel = std::numeric_limits<int64_t>::max();
755  int64_t highestDestinationAssignmentLevel = std::numeric_limits<int64_t>::min();
756  int64_t lowestEdgeAssignmentLevel = std::numeric_limits<int64_t>::max();
757  int64_t highestEdgeAssignmentLevel = std::numeric_limits<int64_t>::min();
758  uint64_t numDestinations = 1;
759  for (uint_fast64_t i = 0; i < iteratorList.size(); ++i) {
760  if (this->getOptions().isBuildChoiceOriginsSet()) {
761  edgeIndices.insert(model.encodeAutomatonAndEdgeIndices(edgeCombination[i].first, iteratorList[i]->first));
762  }
763  storm::jani::Edge const& edge = *iteratorList[i]->second;
764  lowestDestinationAssignmentLevel = std::min(lowestDestinationAssignmentLevel, edge.getLowestAssignmentLevel());
765  highestDestinationAssignmentLevel = std::max(highestDestinationAssignmentLevel, edge.getHighestAssignmentLevel());
766  if (!edge.getAssignments().empty()) {
767  lowestEdgeAssignmentLevel = std::min(lowestEdgeAssignmentLevel, edge.getAssignments().getLowestLevel(true));
768  highestEdgeAssignmentLevel = std::max(highestEdgeAssignmentLevel, edge.getAssignments().getHighestLevel(true));
769  }
770  numDestinations *= edge.getNumberOfDestinations();
771  }
772 
773  // Perform the edge assignments (if there are any)
774  TransientVariableValuation<ValueType> transientVariableValuation;
775  if (evaluateRewardExpressionsAtEdges && lowestEdgeAssignmentLevel <= highestEdgeAssignmentLevel) {
776  for (int64_t assignmentLevel = lowestEdgeAssignmentLevel; assignmentLevel <= highestEdgeAssignmentLevel; ++assignmentLevel) {
777  transientVariableValuation.clear();
778  for (uint_fast64_t i = 0; i < iteratorList.size(); ++i) {
779  storm::jani::Edge const& edge = *iteratorList[i]->second;
780  applyTransientUpdate(transientVariableValuation, edge.getAssignments().getTransientAssignments(assignmentLevel), *this->evaluator);
781  }
782  transientVariableValuation.setInEvaluator(*this->evaluator, this->getOptions().isExplorationChecksSet());
783  }
784  addEvaluatedRewardExpressions(stateActionRewards, storm::utility::one<ValueType>());
785  transientVariableInformation.setDefaultValuesInEvaluator(*this->evaluator);
786  }
787 
788  std::vector<storm::jani::EdgeDestination const*> destinations;
789  std::vector<LocationVariableInformation const*> locationVars;
790  destinations.reserve(iteratorList.size());
791  locationVars.reserve(iteratorList.size());
792 
793  for (uint64_t destinationId = 0; destinationId < numDestinations; ++destinationId) {
794  // First assignment level
795  destinations.clear();
796  locationVars.clear();
797  transientVariableValuation.clear();
798  CompressedState successorState = state;
799  ValueType successorProbability = storm::utility::one<ValueType>();
800 
801  uint64_t destinationIndex = destinationId;
802  for (uint64_t i = 0; i < iteratorList.size(); ++i) {
803  storm::jani::Edge const& edge = *iteratorList[i]->second;
804  STORM_LOG_ASSERT(edge.getNumberOfDestinations() > 0, "Found an edge with zero destinations. This is not expected.");
805  uint64_t localDestinationIndex = destinationIndex % edge.getNumberOfDestinations();
806  destinations.push_back(&edge.getDestination(localDestinationIndex));
807  locationVars.push_back(&this->variableInformation.locationVariables[edgeCombination[i].first]);
808  destinationIndex /= edge.getNumberOfDestinations();
809  ValueType probability = this->evaluator->asRational(destinations.back()->getProbability());
810  if (edge.hasRate()) {
811  successorProbability *= probability * this->evaluator->asRational(edge.getRate());
812  } else {
813  successorProbability *= probability;
814  }
815  if (storm::utility::isZero(successorProbability)) {
816  break;
817  }
818 
819  applyUpdate(successorState, *destinations.back(), *locationVars.back(), lowestDestinationAssignmentLevel, *this->evaluator);
820  applyTransientUpdate(transientVariableValuation, destinations.back()->getOrderedAssignments().getTransientAssignments(lowestDestinationAssignmentLevel), *this->evaluator);
821  }
822 
823 
824  if (!storm::utility::isZero(successorProbability)) {
825  bool evaluatorChanged = false;
826  // remaining assignment levels (if there are any)
827  for (int64_t assignmentLevel = lowestDestinationAssignmentLevel + 1; assignmentLevel <= highestDestinationAssignmentLevel; ++assignmentLevel) {
828  unpackStateIntoEvaluator(successorState, this->variableInformation, *this->evaluator);
829  transientVariableValuation.setInEvaluator(*this->evaluator, this->getOptions().isExplorationChecksSet());
830  transientVariableValuation.clear();
831  evaluatorChanged = true;
832  auto locationVarIt = locationVars.begin();
833  for (auto const& destPtr : destinations) {
834  applyUpdate(successorState, *destPtr, **locationVarIt, assignmentLevel, *this->evaluator);
835  applyTransientUpdate(transientVariableValuation, destinations.back()->getOrderedAssignments().getTransientAssignments(assignmentLevel), *this->evaluator);
836  ++locationVarIt;
837  }
838  }
839  if (!transientVariableValuation.empty()) {
840  evaluatorChanged = true;
841  transientVariableValuation.setInEvaluator(*this->evaluator, this->getOptions().isExplorationChecksSet());
842  }
843  if (evaluateRewardExpressionsAtDestinations) {
844  unpackStateIntoEvaluator(successorState, this->variableInformation, *this->evaluator);
845  evaluatorChanged = true;
846  addEvaluatedRewardExpressions(stateActionRewards, successorProbability);
847  }
848  if (evaluatorChanged) {
849  // Restore the old state information
850  unpackStateIntoEvaluator(state, this->variableInformation, *this->evaluator);
851  this->transientVariableInformation.setDefaultValuesInEvaluator(*this->evaluator);
852  }
853 
854  StateType id = stateToIdCallback(successorState);
855  distribution.add(id, successorProbability);
856  }
857  }
858  }

860  template<typename ValueType, typename StateType>
861  void JaniNextStateGenerator<ValueType, StateType>::expandSynchronizingEdgeCombination(AutomataEdgeSets const& edgeCombination, uint64_t outputActionIndex, CompressedState const& state, StateToIdCallback stateToIdCallback, std::vector<Choice<ValueType>>& newChoices) {
862 
863  if (this->options.isExplorationChecksSet()) {
864  // Check whether a global variable is written multiple times in any combination.
865  checkGlobalVariableWritesValid(edgeCombination);
866  }
867 
868  std::vector<EdgeSetWithIndices::const_iterator> iteratorList(edgeCombination.size());
869 
870  // Initialize the list of iterators.
871  for (size_t i = 0; i < edgeCombination.size(); ++i) {
872  iteratorList[i] = edgeCombination[i].second.cbegin();
873  }
874 
875  storm::builder::jit::Distribution<StateType, ValueType> distribution;
876 
877  // As long as there is one feasible combination of commands, keep on expanding it.
878  bool done = false;
879  while (!done) {
880  distribution.clear();
881 
882  EdgeIndexSet edgeIndices;
883  std::vector<ValueType> stateActionRewards(rewardExpressions.size(), storm::utility::zero<ValueType>());
884  // old version without assignment levels generateSynchronizedDistribution(state, storm::utility::one<ValueType>(), 0, edgeCombination, iteratorList, distribution, stateActionRewards, edgeIndices, stateToIdCallback);
885  generateSynchronizedDistribution(state, edgeCombination, iteratorList, distribution, stateActionRewards, edgeIndices, stateToIdCallback);
886  distribution.compress();
887 
888  // At this point, we applied all commands of the current command combination and newTargetStates
889  // contains all target states and their respective probabilities. That means we are now ready to
890  // add the choice to the list of transitions.
891  newChoices.emplace_back(outputActionIndex);
892 
893  // Now create the actual distribution.
894  Choice<ValueType>& choice = newChoices.back();
895 
896  // Add the edge indices if requested.
897  if (this->getOptions().isBuildChoiceOriginsSet()) {
898  choice.addOriginData(boost::any(std::move(edgeIndices)));
899  }
900 
901  // Add the rewards to the choice.
902  choice.addRewards(std::move(stateActionRewards));
903 
904  // Add the probabilities/rates to the newly created choice.
905  ValueType probabilitySum = storm::utility::zero<ValueType>();
906  choice.reserve(std::distance(distribution.begin(), distribution.end()));
907  for (auto const& stateProbability : distribution) {
908  choice.addProbability(stateProbability.getState(), stateProbability.getValue());
909 
910  if (this->options.isExplorationChecksSet()) {
911  probabilitySum += stateProbability.getValue();
912  }
913  }
914 
915  if (this->options.isExplorationChecksSet()) {
916  // Check that the resulting distribution is in fact a distribution.
917  STORM_LOG_THROW(!this->isDiscreteTimeModel() || !this->comparator.isConstant(probabilitySum) || this->comparator.isOne(probabilitySum), storm::exceptions::WrongFormatException, "Sum of update probabilities do not sum to one for some edge (actually sum to " << probabilitySum << ").");
918  }
919 
920  // Now, check whether there is one more command combination to consider.
921  bool movedIterator = false;
922  for (uint64_t j = 0; !movedIterator && j < iteratorList.size(); ++j) {
923  ++iteratorList[j];
924  if (iteratorList[j] != edgeCombination[j].second.end()) {
925  movedIterator = true;
