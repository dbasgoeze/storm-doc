471  Expression applyAssociative(std::vector<storm::expressions::Expression> const& expressions, std::function<Expression (Expression const&, Expression const&)> const& function) {
472  STORM_LOG_THROW(!expressions.empty(), storm::exceptions::InvalidArgumentException, "Cannot build function application of empty expression list.");
473 
474  // Balance the syntax tree if there are enough operands
475  if (expressions.size() >= 4) {
476  // we treat operands and literals seperately so that a subsequent call to simplify() is more successful (see, e.g., (a + 1) + (b + 1))
477  std::vector<std::vector<storm::expressions::Expression>> operandTypes(2);
478  auto& nonliterals = operandTypes[0];
479  auto& literals = operandTypes[1];
480  for (auto const& expression : expressions) {
481  if (expression.isLiteral()) {
482  literals.push_back(expression);
483  } else {
484  nonliterals.push_back(expression);
485  }
486  }
487 
488  for (auto& operands : operandTypes) {
489  auto opIt = operands.begin();
490  while (operands.size() > 1) {
491  if (opIt == operands.end() || opIt == operands.end() - 1) {
492  opIt = operands.begin();
493  }
494  *opIt = function(*opIt, operands.back());
495  operands.pop_back();
496  ++opIt;
497  }
498  }
499  if (nonliterals.empty()) {
500  return literals.front();
501  } else if (literals.empty()) {
502  return nonliterals.front();
503  } else {
504  return function(literals.front(), nonliterals.front());
505  }
506  } else {
507  return apply(expressions, function);
508  }
509  }
510  }
