34  OrderExtender<ValueType>::OrderExtender(std::shared_ptr<storm::models::sparse::Model<ValueType>> model) {
35  this->model = model;
36  this->matrix = model->getTransitionMatrix();
37  this->assumptionSeen = false;
38  uint_fast64_t numberOfStates = this->model->getNumberOfStates();
39 
40  // Build stateMap
41  for (uint_fast64_t i = 0; i < numberOfStates; ++i) {
42  stateMap[i] = new storm::storage::BitVector(numberOfStates, false);
43  auto row = matrix.getRow(i);
44  for (auto rowItr = row.begin(); rowItr != row.end(); ++rowItr) {
45  // ignore self-loops when there are more transitions
46  if (i != rowItr->getColumn() || row.getNumberOfEntries() == 1) {
47  stateMap[i]->set(rowItr->getColumn(), true);
48  }
49  }
50  }
51 
52  // Check if MC contains cycles
53  storm::storage::StronglyConnectedComponentDecompositionOptions const options;
54  this->sccs = storm::storage::StronglyConnectedComponentDecomposition<ValueType>(matrix, options);
55  acyclic = true;
56  for (size_t i = 0; acyclic && i < sccs.size(); ++i) {
57  acyclic &= sccs.getBlock(i).size() <= 1;
58  }
59  }

62  std::tuple<Order*, uint_fast64_t, uint_fast64_t> OrderExtender<ValueType>::toOrder(std::vector<std::shared_ptr<storm::logic::Formula const>> formulas) {
63  STORM_LOG_THROW((++formulas.begin()) == formulas.end(), storm::exceptions::NotSupportedException, "Only one formula allowed for monotonicity analysis");
64  STORM_LOG_THROW((*(formulas[0])).isProbabilityOperatorFormula()
65  && ((*(formulas[0])).asProbabilityOperatorFormula().getSubformula().isUntilFormula()
66  || (*(formulas[0])).asProbabilityOperatorFormula().getSubformula().isEventuallyFormula()), storm::exceptions::NotSupportedException, "Expecting until or eventually formula");
67 
68  uint_fast64_t numberOfStates = this->model->getNumberOfStates();
69 
70  storm::modelchecker::SparsePropositionalModelChecker<storm::models::sparse::Model<ValueType>> propositionalChecker(*model);
71  storm::storage::BitVector phiStates;
72  storm::storage::BitVector psiStates;
73  if ((*(formulas[0])).asProbabilityOperatorFormula().getSubformula().isUntilFormula()) {
74  phiStates = propositionalChecker.check((*(formulas[0])).asProbabilityOperatorFormula().getSubformula().asUntilFormula().getLeftSubformula())->asExplicitQualitativeCheckResult().getTruthValuesVector();
75  psiStates = propositionalChecker.check((*(formulas[0])).asProbabilityOperatorFormula().getSubformula().asUntilFormula().getRightSubformula())->asExplicitQualitativeCheckResult().getTruthValuesVector();
76  } else {
77  phiStates = storm::storage::BitVector(numberOfStates, true);
78  psiStates = propositionalChecker.check((*(formulas[0])).asProbabilityOperatorFormula().getSubformula().asEventuallyFormula().getSubformula())->asExplicitQualitativeCheckResult().getTruthValuesVector();
79  }
80 
81 
82  // Get the maybeStates
83  std::pair<storm::storage::BitVector, storm::storage::BitVector> statesWithProbability01 = storm::utility::graph::performProb01(this->model->getBackwardTransitions(), phiStates, psiStates);
84  storm::storage::BitVector topStates = statesWithProbability01.second;
85  storm::storage::BitVector bottomStates = statesWithProbability01.first;
86 
87  STORM_LOG_THROW(topStates.begin() != topStates.end(), storm::exceptions::NotImplementedException, "Formula yields to no 1 states");
88  STORM_LOG_THROW(bottomStates.begin() != bottomStates.end(), storm::exceptions::NotImplementedException, "Formula yields to no zero states");
89 
90  // Transform to Order
91  auto matrix = this->model->getTransitionMatrix();
92 
93  auto initialMiddleStates = storm::storage::BitVector(numberOfStates);
94  // Add possible cycle breaking states
95  if (!acyclic) {
96  for (size_t i = 0; i < sccs.size(); ++i) {
97  auto scc = sccs.getBlock(i);
98  if (scc.size() > 1) {
99  auto states = scc.getStates();
100  // check if the state has already one successor in bottom of top, in that case pick it
101  for (auto const& state : states) {
102  auto successors = stateMap[state];
103  if (successors->getNumberOfSetBits() == 2) {
104  auto succ1 = successors->getNextSetIndex(0);
105  auto succ2 = successors->getNextSetIndex(succ1 + 1);
106  auto intersection = bottomStates | topStates;
107  if (intersection[succ1] || intersection[succ2]) {
108  initialMiddleStates.set(state);
109  break;
110  }
111  }
112  }
113  }
114  }
115  }
116  std::vector<uint_fast64_t> statesSorted = storm::utility::graph::getTopologicalSort(matrix);
117  Order *order = new Order(&topStates, &bottomStates, &initialMiddleStates, numberOfStates, &statesSorted);
118 
119  return this->extendOrder(order);
120  }

124  std::tuple<Order*, uint_fast64_t, uint_fast64_t> OrderExtender<ValueType>::toOrder(std::vector<std::shared_ptr<storm::logic::Formula const>> formulas, std::vector<double> minValues, std::vector<double> maxValues) {
125  uint_fast64_t numberOfStates = this->model->getNumberOfStates();
126  uint_fast64_t bottom = numberOfStates;
127  uint_fast64_t top = numberOfStates;
128  std::vector<uint_fast64_t> statesSorted = storm::utility::graph::getTopologicalSort(matrix);
129  Order *order = nullptr;
130 
131  for (auto state : statesSorted) {
132  if ((minValues[numberOfStates - 1 - state] == 1 || maxValues[numberOfStates - 1 - state] == 0)
133  && minValues[numberOfStates - 1 - state] == maxValues[numberOfStates - 1 - state]) {
134  if (maxValues[numberOfStates - 1 - state] == 0) {
135  assert (bottom == numberOfStates);
136  bottom = state;
137  }
138  if (minValues[numberOfStates - 1 - state] == 1) {
139  assert (top == numberOfStates);
140  top = state;
141  }
142  if (bottom != numberOfStates && top != numberOfStates) {
143  order = new Order(top, bottom, numberOfStates, &statesSorted);
144  }
145 
146  } else {
147  assert (order != nullptr);
148  auto successors = stateMap[state];
149  if (successors->getNumberOfSetBits() == 1) {
150  auto succ = successors->getNextSetIndex(0);
151  if (succ != state) {
152  if (!order->contains(succ)) {
153  order->add(succ);
154  }
155  order->addToNode(state, order->getNode(succ));
156  }
157  } else if (successors->getNumberOfSetBits() > 1) {
158  uint_fast64_t min = numberOfStates;
159  uint_fast64_t max = numberOfStates;
160  bool allSorted = true;
161 
162  for (auto succ = successors->getNextSetIndex(0);
163  succ < numberOfStates; succ = successors->getNextSetIndex(succ + 1)) {
164  if (min == numberOfStates) {
165  assert (max == numberOfStates);
166  min = succ;
167  max = succ;
168  } else {
169  if (minValues[numberOfStates - 1 - succ] > maxValues[numberOfStates - 1 - max]) {
170  max = succ;
171  } else if (maxValues[numberOfStates - 1 - succ] < minValues[numberOfStates - 1 - min]) {
172  min = succ;
173  } else {
174  allSorted = false;
175  break;
176  }
177  }
178  }
179 
180  if (allSorted && min != max) {
181  if (order->contains(min) && order->contains(max)) {
182  assert (order->compare(min,max) == Order::UNKNOWN || order->compare(min,max) == Order::BELOW);
183  if (order->compare(min, max) == Order::UNKNOWN) {
184  order->addRelation(max, min);
185  }
186  }
187  if (!order->contains(min)) {
188  if (order->contains(max)) {
189  order->addBetween(min, order->getNode(max), order->getBottom());
190  } else {
191  order->add(min);
192  }
193  }
194  if (!order->contains(max)) {
195  // Because of construction min is in the order
196  order->addBetween(max, order->getTop(), order->getNode(min));
197  }
198  assert (order->compare(max, min) == Order::ABOVE);
199  if (order->contains(state)) {
200  if (order->compare(max, state) == Order::UNKNOWN) {
201  order->addRelation(max, state);
202  }
203  if (order->compare(state, min) == Order::UNKNOWN) {
204  order->addRelation(state, min);
205  }
206  } else {
207  order->addBetween(state, max, min);
208  }
209  assert (order->compare(max, state) == Order::ABOVE);
210  assert (order->compare(state, min) == Order::ABOVE);
211  }
212  }
213  }
214  }
215 
216  assert (order != nullptr);
217 
218  // Handle sccs
219  auto addedStates = order->getAddedStates();
220  for (auto scc : sccs) {
221  if (scc.size() > 1) {
222  auto states = scc.getStates();
223  auto candidate = -1;
224  for (auto const& state : states) {
225  if (addedStates->get(state)) {
226  candidate = -1;
227  break;
228  // if there is a state of the scc already present in the order, there is no need to add one.
229  }
230  auto successors = stateMap[state];
231  if (candidate == -1 && successors->getNumberOfSetBits() == 2) {
232  auto succ1 = successors->getNextSetIndex(0);
233  auto succ2 = successors->getNextSetIndex(succ1 + 1);
234  if (addedStates->get(succ1) || addedStates->get(succ2)) {
235  candidate = state;
236  }
237  }
238  }
239  if (candidate != -1) {
240  order->add(candidate);
241  order->statesToHandle->set(candidate);
242  }
243  }
244  }
245  return this->extendOrder(order);
246  }

250  void OrderExtender<ValueType>::handleAssumption(Order* order, std::shared_ptr<storm::expressions::BinaryRelationExpression> assumption) {
251  assert (assumption != nullptr);
252  assumptionSeen = true;
253 
254  storm::expressions::BinaryRelationExpression expr = *assumption;
255  assert (expr.getRelationType() == storm::expressions::BinaryRelationExpression::RelationType::Greater
256  || expr.getRelationType() == storm::expressions::BinaryRelationExpression::RelationType::Equal);
257 
258  if (expr.getRelationType() == storm::expressions::BinaryRelationExpression::RelationType::Equal) {
259  assert (expr.getFirstOperand()->isVariable() && expr.getSecondOperand()->isVariable());
260  storm::expressions::Variable var1 = expr.getFirstOperand()->asVariableExpression().getVariable();
261  storm::expressions::Variable var2 = expr.getSecondOperand()->asVariableExpression().getVariable();
262  auto val1 = std::stoul(var1.getName(), nullptr, 0);
263  auto val2 = std::stoul(var2.getName(), nullptr, 0);
264  auto comp = order->compare(val1, val2);
265 
266  assert (comp == Order::UNKNOWN);
267  Order::Node *n1 = order->getNode(val1);
268  Order::Node *n2 = order->getNode(val2);
269 
270  if (n1 != nullptr && n2 != nullptr) {
271  order->mergeNodes(n1,n2);
272  } else if (n1 != nullptr) {
273  order->addToNode(val2, n1);
274  } else if (n2 != nullptr) {
275  order->addToNode(val1, n2);
276  } else {
277  order->add(val1);
278  order->addToNode(val2, order->getNode(val1));
279  }
280  } else {
281  assert (expr.getFirstOperand()->isVariable() && expr.getSecondOperand()->isVariable());
282  storm::expressions::Variable largest = expr.getFirstOperand()->asVariableExpression().getVariable();
283  storm::expressions::Variable smallest = expr.getSecondOperand()->asVariableExpression().getVariable();
284  auto val1 = std::stoul(largest.getName(), nullptr, 0);
285  auto val2 = std::stoul(smallest.getName(), nullptr, 0);
286  auto compareRes = order->compare(val1, val2);
287 
288  assert(compareRes == Order::UNKNOWN);
289  Order::Node *n1 = order->getNode(val1);
290  Order::Node *n2 = order->getNode(val2);
291 
292  if (n1 != nullptr && n2 != nullptr) {
293  order->addRelationNodes(n1, n2);
294  } else if (n1 != nullptr) {
295  order->addBetween(val2, n1, order->getBottom());
296  } else if (n2 != nullptr) {
297  order->addBetween(val1, order->getTop(), n2);
298  } else {
299  order->add(val1);
300  order->addBetween(val2, order->getNode(val1), order->getBottom());
301  }
302  }
303  }

306  std::tuple<Order*, uint_fast64_t, uint_fast64_t> OrderExtender<ValueType>::extendAllSuccAdded(Order* order, uint_fast64_t const & stateNumber, storm::storage::BitVector* successors) {
307  auto numberOfStates = successors->size();
308  assert (order->getAddedStates()->size() == numberOfStates);
309 
310  if (successors->getNumberOfSetBits() == 1) {
311  // As there is only one successor the current state and its successor must be at the same nodes.
312  order->addToNode(stateNumber, order->getNode(successors->getNextSetIndex(0)));
313  } else if (successors->getNumberOfSetBits() == 2) {
314  // Otherwise, check how the two states compare, and add if the comparison is possible.
315  uint_fast64_t successor1 = successors->getNextSetIndex(0);
316  uint_fast64_t successor2 = successors->getNextSetIndex(successor1 + 1);
317 
318  int compareResult = order->compare(successor1, successor2);
319  if (compareResult == Order::ABOVE) {
320  // successor 1 is closer to top than successor 2
321  order->addBetween(stateNumber, order->getNode(successor1),
322  order->getNode(successor2));
323  } else if (compareResult == Order::BELOW) {
324  // successor 2 is closer to top than successor 1
325  order->addBetween(stateNumber, order->getNode(successor2),
326  order->getNode(successor1));
327  } else if (compareResult == Order::SAME) {
328  // the successors are at the same level
329  order->addToNode(stateNumber, order->getNode(successor1));
330  } else {
331  assert(order->compare(successor1, successor2) == Order::UNKNOWN);
332  return std::make_tuple(order, successor1, successor2);
333  }
334  } else if (successors->getNumberOfSetBits() > 2) {
335  for (auto const& i : *successors) {
336  for (auto j = successors->getNextSetIndex(i+1); j < numberOfStates; j = successors->getNextSetIndex(j+1)) {
337  if (order->compare(i,j) == Order::UNKNOWN) {
338  return std::make_tuple(order, i, j);
339  }
340  }
341  }
342 
343  auto highest = successors->getNextSetIndex(0);
344  auto lowest = highest;
345  for (auto i = successors->getNextSetIndex(highest+1); i < numberOfStates; i = successors->getNextSetIndex(i+1)) {
346  if (order->compare(i, highest) == Order::ABOVE) {
347  highest = i;
348  }
349  if (order->compare(lowest, i) == Order::ABOVE) {
350  lowest = i;
351  }
352  }
353  if (lowest == highest) {
354  order->addToNode(stateNumber, order->getNode(highest));
355  } else {
356  order->addBetween(stateNumber, order->getNode(highest), order->getNode(lowest));
357  }
358  }
359  return std::make_tuple(order, numberOfStates, numberOfStates);
360  }

365  std::tuple<Order*, uint_fast64_t, uint_fast64_t> OrderExtender<ValueType>::extendOrder(Order* order, std::shared_ptr<storm::expressions::BinaryRelationExpression> assumption) {
366  auto numberOfStates = this->model->getNumberOfStates();
367 
368  if (assumption != nullptr) {
369  handleAssumption(order, assumption);
370  }
371 
372  auto oldNumberSet = numberOfStates;
373  while (oldNumberSet != order->getAddedStates()->getNumberOfSetBits()) {
374  oldNumberSet = order->getAddedStates()->getNumberOfSetBits();
375 
376  // Forward reasoning for cycles;
377  if (!acyclic) {
378  auto statesToHandle = order->statesToHandle;
379  auto stateNumber = statesToHandle->getNextSetIndex(0);
380 
381  while (stateNumber != numberOfStates) {
382  storm::storage::BitVector *successors = stateMap[stateNumber];
383  // Checking for states which are already added to the order, and only have one successor left which haven't been added yet
384  auto succ1 = successors->getNextSetIndex(0);
385  auto succ2 = successors->getNextSetIndex(succ1 + 1);
386 
387  assert (order->contains(stateNumber));
388  if (successors->getNumberOfSetBits() == 1) {
389  if (!order->contains(succ1)) {
390  order->addToNode(succ1, order->getNode(stateNumber));
391  statesToHandle->set(succ1, true);
392  if (order->containsStatesSorted(succ1)) {
393  order->removeStatesSorted(succ1);
394  }
395  }
396  statesToHandle->set(stateNumber, false);
397  stateNumber = statesToHandle->getNextSetIndex(0);
398  } else if (successors->getNumberOfSetBits() == 2
399  && ((order->contains(succ1) && !order->contains(succ2))
400  || (!order->contains(succ1) && order->contains(succ2)))) {
401 
402  if (!order->contains(succ1)) {
403  std::swap(succ1, succ2);
404  }
405 
406  auto compare = order->compare(stateNumber, succ1);
407  if (compare == Order::ABOVE) {
408  if (order->containsStatesSorted(succ2)) {
409  order->removeStatesSorted(succ2);
410  }
411  order->addBetween(succ2, order->getTop(), order->getNode(stateNumber));
412  statesToHandle->set(succ2);
413  statesToHandle->set(stateNumber, false);
414  stateNumber = statesToHandle->getNextSetIndex(0);
415  } else if (compare == Order::BELOW) {
416  if (order->containsStatesSorted(succ2)) {
417  order->removeStatesSorted(succ2);
418  }
419  order->addBetween(succ2, order->getNode(stateNumber), order->getBottom());
420  statesToHandle->set(succ2);
421  statesToHandle->set(stateNumber, false);
422  stateNumber = statesToHandle->getNextSetIndex(0);
423  } else {
424  // We don't know positions, so we set the current state number to false
425  statesToHandle->set(stateNumber, false);
426  stateNumber = statesToHandle->getNextSetIndex(0);
427  }
428 
429  } else if (!((order->contains(succ1) && !order->contains(succ2))
430  || (!order->contains(succ1) && order->contains(succ2)))) {
431  stateNumber = statesToHandle->getNextSetIndex(stateNumber + 1);
432  } else {
433  statesToHandle->set(stateNumber, false);
434  stateNumber = statesToHandle->getNextSetIndex(0);
435  }
436  }
437 
438  }
439 
440  // Normal backwardreasoning
441  auto stateNumber = order->getNextSortedState();
442  while (stateNumber != numberOfStates && order->contains(stateNumber)) {
443  order->removeFirstStatesSorted();
444  stateNumber = order->getNextSortedState();
445 
446  if (stateNumber != numberOfStates && order->contains(stateNumber)) {
447  auto resAllAdded = allSuccAdded(order, stateNumber);
448  if (!std::get<0>(resAllAdded)) {
449  return std::make_tuple(order, std::get<1>(resAllAdded), std::get<2>(resAllAdded));
450  }
451  }
452  }
453 
454  if (stateNumber != numberOfStates && !order->contains(stateNumber)) {
455  auto successors = stateMap[stateNumber];
456 
457  auto result = extendAllSuccAdded(order, stateNumber, successors);
458  if (std::get<1>(result) != numberOfStates) {
459  // So we don't know the relation between all successor states
460  return result;
461  } else {
462  assert (order->getNode(stateNumber) != nullptr);
463  if (!acyclic) {
464  order->statesToHandle->set(stateNumber);
465  }
466  order->removeFirstStatesSorted();
467  }
468  }
469  assert (stateNumber == numberOfStates || order->getNode(stateNumber) != nullptr);
470  assert (stateNumber == numberOfStates || order->contains(stateNumber));
471 
472  }
473  assert (order->getAddedStates()->getNumberOfSetBits() == numberOfStates);
474  order->setDoneBuilding(true);
475  return std::make_tuple(order, numberOfStates, numberOfStates);
476  }
