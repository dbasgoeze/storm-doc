104  boost::any RewardAccumulationEliminationVisitor::visit(EventuallyFormula const& f, boost::any const& data) const {
105  std::shared_ptr<Formula> subformula = boost::any_cast<std::shared_ptr<Formula>>(f.getSubformula().accept(*this, data));
106  if (f.hasRewardAccumulation()) {
107  if (f.isTimePathFormula()) {
108  if (model.isDiscreteTimeModel() && ((!f.getRewardAccumulation().isExitSet() && !f.getRewardAccumulation().isStepsSet()) || (f.getRewardAccumulation().isStepsSet() && f.getRewardAccumulation().isExitSet()))) {
109  return std::static_pointer_cast<Formula>(std::make_shared<EventuallyFormula>(subformula, f.getContext(), f.getRewardAccumulation()));
110  } else if (!model.isDiscreteTimeModel() && (!f.getRewardAccumulation().isTimeSet() || f.getRewardAccumulation().isExitSet() || f.getRewardAccumulation().isStepsSet())) {
111  return std::static_pointer_cast<Formula>(std::make_shared<EventuallyFormula>(subformula, f.getContext(), f.getRewardAccumulation()));
112  }
113  } else if (f.isRewardPathFormula()) {
114  STORM_LOG_THROW(!data.empty(), storm::exceptions::UnexpectedException, "Formula " << f << " does not seem to be a subformula of a reward operator.");

116  if (!canEliminate(f.getRewardAccumulation(), rewName)) {
117  return std::static_pointer_cast<Formula>(std::make_shared<EventuallyFormula>(subformula, f.getContext(), f.getRewardAccumulation()));
118  }
119  }

139  bool RewardAccumulationEliminationVisitor::canEliminate(storm::logic::RewardAccumulation const& accumulation, boost::optional<std::string> rewardModelName) const {
140  STORM_LOG_THROW(rewardModelName.is_initialized(), storm::exceptions::InvalidPropertyException, "Unable to find transient variable for unique reward model.");
141  storm::jani::RewardModelInformation info(model, rewardModelName.get());
142 
143  if ((info.hasActionRewards() || info.hasTransitionRewards()) && !accumulation.isStepsSet()) {
144  return false;
145  }
146  if (info.hasStateRewards()) {
147  if (model.isDiscreteTimeModel()) {
148  if (!accumulation.isExitSet()) {
149  return false;
150  }
151  // accumulating over time in discrete time models has no effect, i.e., the value of accumulation.isTimeSet() does not matter here.
152  } else {
153  if (accumulation.isExitSet() || !accumulation.isTimeSet()) {
154  return false;
155  }
156  }
157  }
158  return true;
159  }
