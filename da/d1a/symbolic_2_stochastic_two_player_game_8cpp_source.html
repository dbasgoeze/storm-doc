54  void StochasticTwoPlayerGame<Type, ValueType>::createIllegalMasks() {
55  // Compute legal player 1 mask.
56  this->illegalPlayer1Mask = this->getTransitionMatrix().notZero().existsAbstract(this->getColumnVariables()).existsAbstract(this->getPlayer2Variables());
57 
58  // Correct the mask for player 2. This is necessary, because it is not yet restricted to the legal choices of player 1.
59  illegalPlayer2Mask = this->getIllegalMask() && this->illegalPlayer1Mask;
60 
61  // Then set the illegal mask for player 1 correctly.
62  this->illegalPlayer1Mask = !illegalPlayer1Mask && this->getReachableStates();
63  }
