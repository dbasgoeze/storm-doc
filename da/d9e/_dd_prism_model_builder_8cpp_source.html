449  if (itNewActions != actionIndexToDdMap.end()) {
450  actionIndexToDdMap[renamingIt->second] = DdPrismModelBuilder<Type, ValueType>::combineUnsynchronizedActions(generationInfo, action.second, itNewActions->second);
451 
452  } else {
453  // In this case, we can simply copy the action over.
454  actionIndexToDdMap[renamingIt->second] = action.second;
455  }
456  } else {
457  // If the action is not to be renamed, we need to copy it over. However, if some other action
458  // was renamed to the very same action name before, we need to combine the transitions.
459  auto itNewActions = actionIndexToDdMap.find(action.first);
460  if (itNewActions != actionIndexToDdMap.end()) {
461  actionIndexToDdMap[action.first] = DdPrismModelBuilder<Type, ValueType>::combineUnsynchronizedActions(generationInfo, action.second, itNewActions->second);
462  } else {
463  // In this case, we can simply copy the action over.
464  actionIndexToDdMap[action.first] = action.second;
465  }
466  }
467  }
468 
469  return typename DdPrismModelBuilder<Type, ValueType>::ModuleDecisionDiagram(sub.independentAction, actionIndexToDdMap, sub.identity, sub.numberOfUsedNondeterminismVariables);
470  }

476  void composeInParallel(typename DdPrismModelBuilder<Type, ValueType>::ModuleDecisionDiagram& left, typename DdPrismModelBuilder<Type, ValueType>::ModuleDecisionDiagram& right, std::set<uint_fast64_t> const& synchronizationActionIndices) const {
477  STORM_LOG_TRACE("Composing two modules.");
478 
479  // Combine the tau action.
480  uint_fast64_t numberOfUsedNondeterminismVariables = right.independentAction.numberOfUsedNondeterminismVariables;
481  left.independentAction = DdPrismModelBuilder<Type, ValueType>::combineUnsynchronizedActions(generationInfo, left.independentAction, right.independentAction, left.identity, right.identity);
482  numberOfUsedNondeterminismVariables = std::max(numberOfUsedNondeterminismVariables, left.independentAction.numberOfUsedNondeterminismVariables);
483 
484  // Create an empty action for the case where one of the modules does not have a certain action.
485  typename DdPrismModelBuilder<Type, ValueType>::ActionDecisionDiagram emptyAction(*generationInfo.manager);
486 
487  // Treat all non-tau actions of the left module.
488  for (auto& action : left.synchronizingActionToDecisionDiagramMap) {
489  // If we need to synchronize over this action index, we try to do so now.
490  if (synchronizationActionIndices.find(action.first) != synchronizationActionIndices.end()) {
491  // If we are to synchronize over an action that does not exist in the second module, the result
492  // is that the synchronization is the empty action.
493  if (!right.hasSynchronizingAction(action.first)) {
494  action.second = emptyAction;
495  } else {
496  // Otherwise, the actions of the modules are synchronized.
497  action.second = DdPrismModelBuilder<Type, ValueType>::combineSynchronizingActions(action.second, right.synchronizingActionToDecisionDiagramMap[action.first]);
498  }
499  } else {
500  // If we don't synchronize over this action, we need to construct the interleaving.
501 
502  // If both modules contain the action, we need to mutually multiply the other identity.
503  if (right.hasSynchronizingAction(action.first)) {
504  action.second = DdPrismModelBuilder<Type, ValueType>::combineUnsynchronizedActions(generationInfo, action.second, right.synchronizingActionToDecisionDiagramMap[action.first], left.identity, right.identity);
505  } else {
506  // If only the first module has this action, we need to use a dummy action decision diagram
507  // for the second module.
508  action.second = DdPrismModelBuilder<Type, ValueType>::combineUnsynchronizedActions(generationInfo, action.second, emptyAction, left.identity, right.identity);
509  }
510  }
511  numberOfUsedNondeterminismVariables = std::max(numberOfUsedNondeterminismVariables, action.second.numberOfUsedNondeterminismVariables);
512  }
513 
514  // Treat all non-tau actions of the right module.
515  for (auto const& actionIndex : right.getSynchronizingActionIndices()) {
516  // Here, we only need to treat actions that the first module does not have, because we have handled
517  // this case earlier.
518  if (!left.hasSynchronizingAction(actionIndex)) {
519  if (synchronizationActionIndices.find(actionIndex) != synchronizationActionIndices.end()) {
520  // If we are to synchronize over this action that does not exist in the first module, the
521  // result is that the synchronization is the empty action.
522  left.synchronizingActionToDecisionDiagramMap[actionIndex] = emptyAction;
523  } else {
524  // If only the second module has this action, we need to use a dummy action decision diagram
525  // for the first module.
526  left.synchronizingActionToDecisionDiagramMap[actionIndex] = DdPrismModelBuilder<Type, ValueType>::combineUnsynchronizedActions(generationInfo, emptyAction, right.synchronizingActionToDecisionDiagramMap[actionIndex], left.identity, right.identity);
527  }
528  }
529  numberOfUsedNondeterminismVariables = std::max(numberOfUsedNondeterminismVariables, left.synchronizingActionToDecisionDiagramMap[actionIndex].numberOfUsedNondeterminismVariables);
530  }
531 
532  // Combine identity matrices.
533  left.identity = left.identity * right.identity;
534 
535  // Keep track of the number of nondeterminism variables used.
536  left.numberOfUsedNondeterminismVariables = std::max(left.numberOfUsedNondeterminismVariables, numberOfUsedNondeterminismVariables);
537  }

540  };

605  template <storm::dd::DdType Type, typename ValueType>
606  typename DdPrismModelBuilder<Type, ValueType>::UpdateDecisionDiagram DdPrismModelBuilder<Type, ValueType>::createUpdateDecisionDiagram(GenerationInformation& generationInfo, storm::prism::Module const& module, storm::dd::Add<Type, ValueType> const& guard, storm::prism::Update const& update) {
607  storm::dd::Add<Type, ValueType> updateDd = generationInfo.manager->template getAddOne<ValueType>();
608 
609  STORM_LOG_TRACE("Translating update " << update);
610 
611  // Iterate over all assignments (boolean and integer) and build the DD for it.
612  std::vector<storm::prism::Assignment> assignments = update.getAssignments();
613  std::set<storm::expressions::Variable> assignedVariables;
614  for (auto const& assignment : assignments) {
615  // Record the variable as being written.
616  STORM_LOG_TRACE("Assigning to variable " << generationInfo.variableToRowMetaVariableMap->at(assignment.getVariable()).getName());
617  assignedVariables.insert(assignment.getVariable());
618 
619  // Translate the written variable.
620  auto const& primedMetaVariable = generationInfo.variableToColumnMetaVariableMap->at(assignment.getVariable());
621  storm::dd::Add<Type, ValueType> writtenVariable = generationInfo.manager->template getIdentity<ValueType>(primedMetaVariable);
622 
623  // Translate the expression that is being assigned.
624  storm::dd::Add<Type, ValueType> updateExpression = generationInfo.rowExpressionAdapter->translateExpression(assignment.getExpression());
625 
626  // Combine the update expression with the guard.
627  storm::dd::Add<Type, ValueType> result = updateExpression * guard;
628 
629  // Combine the variable and the assigned expression.
630  storm::dd::Add<Type, ValueType> tmp = result;
631  result = result.equals(writtenVariable).template toAdd<ValueType>();
632  result *= guard;
633 
634  // Restrict the transitions to the range of the written variable.
635  result = result * generationInfo.manager->getRange(primedMetaVariable).template toAdd<ValueType>();
636 
637  updateDd *= result;
638  }
639 
640  // Compute the set of assigned global variables.
641  std::set<storm::expressions::Variable> assignedGlobalVariables;
642  std::set_intersection(assignedVariables.begin(), assignedVariables.end(), generationInfo.allGlobalVariables.begin(), generationInfo.allGlobalVariables.end(), std::inserter(assignedGlobalVariables, assignedGlobalVariables.begin()));
643 
644  // All unassigned boolean variables need to keep their value.
645  for (storm::prism::BooleanVariable const& booleanVariable : module.getBooleanVariables()) {
646  if (assignedVariables.find(booleanVariable.getExpressionVariable()) == assignedVariables.end()) {
647  STORM_LOG_TRACE("Multiplying identity of variable " << booleanVariable.getName());
648  updateDd *= generationInfo.variableToIdentityMap.at(booleanVariable.getExpressionVariable());
649  }
650  }
651 
652  // All unassigned integer variables need to keep their value.
653  for (storm::prism::IntegerVariable const& integerVariable : module.getIntegerVariables()) {
654  if (assignedVariables.find(integerVariable.getExpressionVariable()) == assignedVariables.end()) {
655  STORM_LOG_TRACE("Multiplying identity of variable " << integerVariable.getName());
656  updateDd *= generationInfo.variableToIdentityMap.at(integerVariable.getExpressionVariable());
657  }
658  }
659 
660  return UpdateDecisionDiagram(updateDd, assignedGlobalVariables);
661  }

663  template <storm::dd::DdType Type, typename ValueType>
664  typename DdPrismModelBuilder<Type, ValueType>::ActionDecisionDiagram DdPrismModelBuilder<Type, ValueType>::createCommandDecisionDiagram(GenerationInformation& generationInfo, storm::prism::Module const& module, storm::prism::Command const& command) {
665  STORM_LOG_TRACE("Translating guard " << command.getGuardExpression());
666  storm::dd::Bdd<Type> guard = generationInfo.rowExpressionAdapter->translateBooleanExpression(command.getGuardExpression()) && generationInfo.moduleToRangeMap[module.getName()].notZero();
667  STORM_LOG_WARN_COND(!guard.isZero(), "The guard '" << command.getGuardExpression() << "' is unsatisfiable.");
668 
669  if (!guard.isZero()) {
670  // Create the DDs representing the individual updates.
671  std::vector<UpdateDecisionDiagram> updateResults;
672  for (storm::prism::Update const& update : command.getUpdates()) {
673  updateResults.push_back(createUpdateDecisionDiagram(generationInfo, module, guard.template toAdd<ValueType>(), update));
674 
675  STORM_LOG_WARN_COND(!updateResults.back().updateDd.isZero(), "Update '" << update << "' does not have any effect.");
676  }
677 
678  // Start by gathering all variables that were written in at least one update.
679  std::set<storm::expressions::Variable> globalVariablesInSomeUpdate;
680 
681  // If the command is labeled, we have to analyze which portion of the global variables was written by
682  // any of the updates and make all update results equal w.r.t. this set. If the command is not labeled,
683  // we can already multiply the identities of all global variables.
684  if (command.isLabeled()) {
685  std::for_each(updateResults.begin(), updateResults.end(), [&globalVariablesInSomeUpdate] (UpdateDecisionDiagram const& update) { globalVariablesInSomeUpdate.insert(update.assignedGlobalVariables.begin(), update.assignedGlobalVariables.end()); } );
686  } else {
687  globalVariablesInSomeUpdate = generationInfo.allGlobalVariables;
688  }
689 
690  // Then, multiply the missing identities.
691  for (auto& updateResult : updateResults) {
692  std::set<storm::expressions::Variable> missingIdentities;
693  std::set_difference(globalVariablesInSomeUpdate.begin(), globalVariablesInSomeUpdate.end(), updateResult.assignedGlobalVariables.begin(), updateResult.assignedGlobalVariables.end(), std::inserter(missingIdentities, missingIdentities.begin()));
694 
695  for (auto const& variable : missingIdentities) {
696  STORM_LOG_TRACE("Multiplying identity for variable " << variable.getName() << "[" << variable.getIndex() << "] to update.");
697  updateResult.updateDd *= generationInfo.variableToIdentityMap.at(variable);
698  }
699  }
700 
701  // Now combine the update DDs to the command DD.
702  storm::dd::Add<Type, ValueType> commandDd = generationInfo.manager->template getAddZero<ValueType>();
703  auto updateResultsIt = updateResults.begin();
704  for (auto updateIt = command.getUpdates().begin(), updateIte = command.getUpdates().end(); updateIt != updateIte; ++updateIt, ++updateResultsIt) {
705  storm::dd::Add<Type, ValueType> probabilityDd = generationInfo.rowExpressionAdapter->translateExpression(updateIt->getLikelihoodExpression());
706  commandDd += updateResultsIt->updateDd * probabilityDd;
707  }
708 
709  return ActionDecisionDiagram(guard, guard.template toAdd<ValueType>() * commandDd, globalVariablesInSomeUpdate);
710  } else {
711  return ActionDecisionDiagram(*generationInfo.manager);
712  }
713  }

722

724  continue;

745  }
746  }
747 
748  return result;

838  template <storm::dd::DdType Type, typename ValueType>
839  typename DdPrismModelBuilder<Type, ValueType>::ActionDecisionDiagram DdPrismModelBuilder<Type, ValueType>::combineCommandsToActionMDP(GenerationInformation& generationInfo, std::vector<ActionDecisionDiagram>& commandDds, uint_fast64_t nondeterminismVariableOffset) {
840  storm::dd::Bdd<Type> allGuards = generationInfo.manager->getBddZero();
841  storm::dd::Add<Type, ValueType> allCommands = generationInfo.manager->template getAddZero<ValueType>();
842 
843  // Make all command DDs assign to the same global variables.
844  std::set<storm::expressions::Variable> assignedGlobalVariables = equalizeAssignedGlobalVariables(generationInfo, commandDds);
845 
846  // Sum all guards, so we can read off the maximal number of nondeterministic choices in any given state.
847  storm::dd::Add<Type, uint_fast64_t> sumOfGuards = generationInfo.manager->template getAddZero<uint_fast64_t>();
848  for (auto const& commandDd : commandDds) {
849  sumOfGuards += commandDd.guardDd.template toAdd<uint_fast64_t>();
850  allGuards |= commandDd.guardDd;
851  }
852  uint_fast64_t maxChoices = sumOfGuards.getMax();
853 
854  STORM_LOG_TRACE("Found " << maxChoices << " local choices.");
855 
856  // Depending on the maximal number of nondeterminstic choices, we need to use some variables to encode the nondeterminism.
857  if (maxChoices == 0) {
858  return ActionDecisionDiagram(*generationInfo.manager);
859  } else if (maxChoices == 1) {
860  // Sum up all commands.
861  for (auto const& commandDd : commandDds) {
862  allCommands += commandDd.transitionsDd;
863  }
864  return ActionDecisionDiagram(allGuards, allCommands, assignedGlobalVariables);
865  } else {
866  // Calculate number of required variables to encode the nondeterminism.
867  uint_fast64_t numberOfBinaryVariables = static_cast<uint_fast64_t>(std::ceil(storm::utility::math::log2(maxChoices)));
868 
869  storm::dd::Bdd<Type> equalsNumberOfChoicesDd;
870  std::vector<storm::dd::Add<Type, ValueType>> choiceDds(maxChoices, generationInfo.manager->template getAddZero<ValueType>());
871  std::vector<storm::dd::Bdd<Type>> remainingDds(maxChoices, generationInfo.manager->getBddZero());
872 
873  for (uint_fast64_t currentChoices = 1; currentChoices <= maxChoices; ++currentChoices) {
874  // Determine the set of states with exactly currentChoices choices.
875  equalsNumberOfChoicesDd = sumOfGuards.equals(generationInfo.manager->getConstant(currentChoices));
876 
877  // If there is no such state, continue with the next possible number of choices.
878  if (equalsNumberOfChoicesDd.isZero()) {
879  continue;
880  }
881 
882  // Reset the previously used intermediate storage.
883  for (uint_fast64_t j = 0; j < currentChoices; ++j) {
884  choiceDds[j] = generationInfo.manager->template getAddZero<ValueType>();
885  remainingDds[j] = equalsNumberOfChoicesDd;
886  }
887 
888  for (std::size_t j = 0; j < commandDds.size(); ++j) {
889  // Check if command guard overlaps with equalsNumberOfChoicesDd. That is, there are states with exactly currentChoices
890  // choices such that one outgoing choice is given by the j-th command.
891  storm::dd::Bdd<Type> guardChoicesIntersection = commandDds[j].guardDd && equalsNumberOfChoicesDd;
892 
893  // If there is no such state, continue with the next command.
894  if (guardChoicesIntersection.isZero()) {
895  continue;
896  }
897 
898  // Split the nondeterministic choices.
899  for (uint_fast64_t k = 0; k < currentChoices; ++k) {
900  // Calculate the overlapping part of command guard and the remaining DD.
901  storm::dd::Bdd<Type> remainingGuardChoicesIntersection = guardChoicesIntersection && remainingDds[k];
902 
903  // Check if we can add some overlapping parts to the current index.
904  if (!remainingGuardChoicesIntersection.isZero()) {
905  // Remove overlapping parts from the remaining DD.
906  remainingDds[k] = remainingDds[k] && !remainingGuardChoicesIntersection;
907 
908  // Combine the overlapping part of the guard with command updates and add it to the resulting DD.
909  choiceDds[k] += remainingGuardChoicesIntersection.template toAdd<ValueType>() * commandDds[j].transitionsDd;
910  }
911 
912  // Remove overlapping parts from the command guard DD
913  guardChoicesIntersection = guardChoicesIntersection && !remainingGuardChoicesIntersection;
914 
915  // If the guard DD has become equivalent to false, we can stop here.
916  if (guardChoicesIntersection.isZero()) {
917  break;
918  }
919  }
920  }
921 
922  // Add the meta variables that encode the nondeterminisim to the different choices.
923  for (uint_fast64_t j = 0; j < currentChoices; ++j) {
924  allCommands += encodeChoice(generationInfo, nondeterminismVariableOffset, numberOfBinaryVariables, j) * choiceDds[j];
925  }
926 
927  // Delete currentChoices out of overlapping DD
928  sumOfGuards = sumOfGuards * (!equalsNumberOfChoicesDd).template toAdd<uint_fast64_t>();
929  }
930 
931  return ActionDecisionDiagram(allGuards, allCommands, assignedGlobalVariables, nondeterminismVariableOffset + numberOfBinaryVariables);
932  }

957

959  std::set<storm::expressions::Variable> assignedGlobalVariables = equalizeAssignedGlobalVariables(generationInfo, action1, action2);

966  } else if (action2.transitionsDd.isZero()) {
967  return ActionDecisionDiagram(action1.guardDd, action1.transitionsDd, assignedGlobalVariables, action1.numberOfUsedNondeterminismVariables);
968  }

971  uint_fast64_t numberOfUsedNondeterminismVariables = std::max(action1.numberOfUsedNondeterminismVariables, action2.numberOfUsedNondeterminismVariables);

983  nondeterminismEncoding *= generationInfo.manager->getEncoding(generationInfo.nondeterminismMetaVariables[i], 0).template toAdd<ValueType>();
984  }

986  }

989  storm::dd::Add<Type, ValueType> combinedTransitions = generationInfo.manager->getEncoding(generationInfo.nondeterminismMetaVariables[numberOfUsedNondeterminismVariables], 1).ite(action2.transitionsDd, action1.transitionsDd);
990 
991  return ActionDecisionDiagram(action1.guardDd || action2.guardDd, combinedTransitions, assignedGlobalVariables, numberOfUsedNondeterminismVariables + 1);

1179  template <storm::dd::DdType Type, typename ValueType>
1180  storm::models::symbolic::StandardRewardModel<Type, ValueType> DdPrismModelBuilder<Type, ValueType>::createRewardModelDecisionDiagrams(GenerationInformation& generationInfo, storm::prism::RewardModel const& rewardModel, ModuleDecisionDiagram const& globalModule, storm::dd::Add<Type, ValueType> const& reachableStatesAdd, storm::dd::Add<Type, ValueType> const& transitionMatrix, boost::optional<storm::dd::Add<Type, ValueType>>& stateActionDd) {
1181 
1182  // Start by creating the state reward vector.
1183  boost::optional<storm::dd::Add<Type, ValueType>> stateRewards;
1184  if (rewardModel.hasStateRewards()) {
1185  stateRewards = generationInfo.manager->template getAddZero<ValueType>();
1186 
1187  for (auto const& stateReward : rewardModel.getStateRewards()) {
1188  storm::dd::Add<Type, ValueType> states = generationInfo.rowExpressionAdapter->translateExpression(stateReward.getStatePredicateExpression());
1189  storm::dd::Add<Type, ValueType> rewards = generationInfo.rowExpressionAdapter->translateExpression(stateReward.getRewardValueExpression());
1190 
1191  // Restrict the rewards to those states that satisfy the condition.
1192  rewards = reachableStatesAdd * states * rewards;
1193 
1194  // Add the rewards to the global state reward vector.
1195  stateRewards.get() += rewards;
1196  }
1197  // Perform some sanity checks.
1198  checkRewards(stateRewards.get(), "state rewards");
1199  }
1200 
1201  // Next, build the state-action reward vector.
1202  boost::optional<storm::dd::Add<Type, ValueType>> stateActionRewards;
1203  if (rewardModel.hasStateActionRewards()) {
1204  stateActionRewards = generationInfo.manager->template getAddZero<ValueType>();
1205 
1206  for (auto const& stateActionReward : rewardModel.getStateActionRewards()) {
1207  storm::dd::Add<Type, ValueType> states = generationInfo.rowExpressionAdapter->translateExpression(stateActionReward.getStatePredicateExpression());
1208  storm::dd::Add<Type, ValueType> rewards = generationInfo.rowExpressionAdapter->translateExpression(stateActionReward.getRewardValueExpression());
1209  storm::dd::Add<Type, ValueType> synchronization = generationInfo.manager->template getAddOne<ValueType>();
1210 
1211  if (generationInfo.program.getModelType() == storm::prism::Program::ModelType::MDP) {
1212  synchronization = getSynchronizationDecisionDiagram(generationInfo, stateActionReward.getActionIndex());
1213  }
1214  ActionDecisionDiagram const& actionDd = stateActionReward.isLabeled() ? globalModule.synchronizingActionToDecisionDiagramMap.at(stateActionReward.getActionIndex()) : globalModule.independentAction;
1215  states *= actionDd.guardDd.template toAdd<ValueType>() * reachableStatesAdd;
1216  storm::dd::Add<Type, ValueType> stateActionRewardDd = synchronization * states * rewards;
1217 
1218  // If we are building the state-action rewards for an MDP, we need to make sure that the reward is
1219  // only given on legal nondeterminism encodings, which is why we multiply with the state-action DD.
1220  if (generationInfo.program.getModelType() == storm::prism::Program::ModelType::MDP) {
1221  if (!stateActionDd) {
1222  stateActionDd = transitionMatrix.notZero().existsAbstract(generationInfo.columnMetaVariables).template toAdd<ValueType>();
1223  }
1224  stateActionRewardDd *= stateActionDd.get();
1225  } else if (generationInfo.program.getModelType() == storm::prism::Program::ModelType::DTMC || generationInfo.program.getModelType() == storm::prism::Program::ModelType::CTMC) {
1226  // For DTMCs and CTMC, we need to multiply the entries with the multiplicity/exit rate of the corresponding action.
1227  stateActionRewardDd *= actionDd.transitionsDd.sumAbstract(generationInfo.columnMetaVariables);
1228  }
1229 
1230  // Add the rewards to the global transition reward matrix.
1231  stateActionRewards.get() += stateActionRewardDd;
1232  }
1233 
1234  // Scale state-action rewards for DTMCs and CTMCs.
1235  if (generationInfo.program.getModelType() == storm::prism::Program::ModelType::DTMC || generationInfo.program.getModelType() == storm::prism::Program::ModelType::CTMC) {
1236  if (!stateActionDd) {
1237  stateActionDd = transitionMatrix.sumAbstract(generationInfo.columnMetaVariables);
1238  }
1239 
1240  stateActionRewards.get() /= stateActionDd.get();
1241  }
1242 
1243  // Perform some sanity checks.
1244  checkRewards(stateActionRewards.get(), "action rewards");
1245  }
1246 
1247  // Then build the transition reward matrix.
1248  boost::optional<storm::dd::Add<Type, ValueType>> transitionRewards;
1249  if (rewardModel.hasTransitionRewards()) {
1250  transitionRewards = generationInfo.manager->template getAddZero<ValueType>();
1251 
1252  for (auto const& transitionReward : rewardModel.getTransitionRewards()) {
1253  storm::dd::Add<Type, ValueType> sourceStates = generationInfo.rowExpressionAdapter->translateExpression(transitionReward.getSourceStatePredicateExpression());
1254  storm::dd::Add<Type, ValueType> targetStates = generationInfo.rowExpressionAdapter->translateExpression(transitionReward.getTargetStatePredicateExpression());
1255  storm::dd::Add<Type, ValueType> rewards = generationInfo.rowExpressionAdapter->translateExpression(transitionReward.getRewardValueExpression());
1256 
1257  storm::dd::Add<Type, ValueType> synchronization = generationInfo.manager->template getAddOne<ValueType>();
1258 
1259  storm::dd::Add<Type, ValueType> transitions;
1260  if (transitionReward.isLabeled()) {
1261  if (generationInfo.program.getModelType() == storm::prism::Program::ModelType::MDP) {
1262  synchronization = getSynchronizationDecisionDiagram(generationInfo, transitionReward.getActionIndex());
1263  }
1264  transitions = globalModule.synchronizingActionToDecisionDiagramMap.at(transitionReward.getActionIndex()).transitionsDd;
1265  } else {
1266  if (generationInfo.program.getModelType() == storm::prism::Program::ModelType::MDP) {
1267  synchronization = getSynchronizationDecisionDiagram(generationInfo);
1268  }
1269  transitions = globalModule.independentAction.transitionsDd;
1270  }
1271 
1272  storm::dd::Add<Type, ValueType> transitionRewardDd = synchronization * sourceStates * targetStates * rewards;
1273  if (generationInfo.program.getModelType() == storm::prism::Program::ModelType::DTMC) {
1274  // For DTMCs we need to keep the weighting for the scaling that follows.
1275  transitionRewardDd = transitions * transitionRewardDd;
1276  } else {

1282  transitionRewards.get() += transitionRewardDd;

1286  checkRewards(transitionRewards.get(), "transition rewards");

1289  if (generationInfo.program.getModelType() == storm::prism::Program::ModelType::DTMC) {
1290  transitionRewards.get() /= stateActionDd.get();
1291  }
1292  }
1293 
1294  return storm::models::symbolic::StandardRewardModel<Type, ValueType>(stateRewards, stateActionRewards, transitionRewards);

1310  }

1318  // In particular, this creates the meta variables used to encode the model.
1319  GenerationInformation generationInfo(program);
1320 
1321  SystemResult system = createSystemDecisionDiagram(generationInfo);
1322  storm::dd::Add<Type, ValueType> transitionMatrix = system.allTransitionsDd;
1323 
1324  ModuleDecisionDiagram const& globalModule = system.globalModule;

1327  storm::dd::Bdd<Type> terminalStatesBdd = generationInfo.manager->getBddZero();

1329  storm::expressions::Expression terminalExpression = options.terminalStates.asExpression([&program](std::string const& labelName) {
1330  if (program.hasLabel(labelName)) {
1331  return program.getLabelExpression(labelName);
1332  } else {
1333  STORM_LOG_THROW(labelName == "init" || labelName == "deadlock", storm::exceptions::InvalidArgumentException, "Terminal states refer to illegal label '" << labelName << "'.");
1334  // If the label name is "init" we can abort 'exploration' directly at the initial state. If it is deadlock, we do not have to abort.
1335  return program.getManager().boolean(labelName == "init");
1336  }
1337  });
1338  terminalExpression = terminalExpression.substitute(program.getConstantsSubstitution());
1339  terminalStatesBdd = generationInfo.rowExpressionAdapter->translateExpression(terminalExpression).toBdd();
1340  transitionMatrix *= (!terminalStatesBdd).template toAdd<ValueType>();
1341  }
1342 
1343  // Cut the transitions and rewards to the reachable fragment of the state space.
1344  storm::dd::Bdd<Type> initialStates = createInitialStatesDecisionDiagram(generationInfo);
1345 
1346  storm::dd::Bdd<Type> transitionMatrixBdd = transitionMatrix.notZero();
1347  if (program.getModelType() == storm::prism::Program::ModelType::MDP) {
1348  transitionMatrixBdd = transitionMatrixBdd.existsAbstract(generationInfo.allNondeterminismVariables);
1349  }

1352  storm::dd::Add<Type, ValueType> reachableStatesAdd = reachableStates.template toAdd<ValueType>();

1354  if (system.stateActionDd) {
1355  system.stateActionDd.get() *= reachableStatesAdd;
1356  }
1357 
1358  // Detect deadlocks and 1) fix them if requested 2) throw an error otherwise.
1359  storm::dd::Bdd<Type> statesWithTransition = transitionMatrixBdd.existsAbstract(generationInfo.columnMetaVariables);
1360  storm::dd::Bdd<Type> deadlockStates = reachableStates && !statesWithTransition;
1361

1378  for (auto const& var : generationInfo.allGlobalVariables) {
1379  identity *= generationInfo.variableToIdentityMap.at(var);
1380  }
1381

1383  transitionMatrix += deadlockStatesAdd * identity;

1386  // want to attach a lot of self-loops to the deadlock states.

1389  action *= generationInfo.manager->template getIdentity<ValueType>(metaVariable);
1390  }
1391  // Make sure that global variables do not change along the introduced self-loops.
1392  for (auto const& var : generationInfo.allGlobalVariables) {
1393  action *= generationInfo.variableToIdentityMap.at(var);
1394  }
1395  transitionMatrix += deadlockStatesAdd * globalModule.identity * action;
1396  }
1397  } else {
1398  STORM_LOG_THROW(false, storm::exceptions::InvalidArgumentException, "The model contains " << deadlockStates.getNonZeroCount() << " deadlock states. Please unset the option to not fix deadlocks, if you want to fix them automatically.");
1399  }
1400  }
1401 
1402  // Reduce the deadlock states by the states that we did simply not explore.
1403  deadlockStates = deadlockStates && !terminalStatesBdd;
1404 
1405  // Now build the reward models.
1406  std::vector<std::reference_wrapper<storm::prism::RewardModel const>> selectedRewardModels;
1407

1409  for (auto const& rewardModelName : options.rewardModelsToBuild) {
1410  STORM_LOG_THROW(rewardModelName.empty() || program.hasRewardModel(rewardModelName), storm::exceptions::InvalidArgumentException, "Model does not possess a reward model with the name '" << rewardModelName << "'.");
1411  }
1412

1414  if (options.buildAllRewardModels || options.rewardModelsToBuild.find(rewardModel.getName()) != options.rewardModelsToBuild.end()) {
1415  selectedRewardModels.push_back(rewardModel);
1416  }
1417  }

1419  // the only existing reward model (given that no explicit name was given for the referenced reward model).
1420  if (selectedRewardModels.empty() && program.getNumberOfRewardModels() == 1 && options.rewardModelsToBuild.size() == 1 && *options.rewardModelsToBuild.begin() == "") {
1421  selectedRewardModels.push_back(program.getRewardModel(0));
1422  }

1425 
1426  // Build the labels that can be accessed as a shortcut.
1427  std::map<std::string, storm::expressions::Expression> labelToExpressionMapping;
1428  for (auto const& label : program.getLabels()) {
1429  labelToExpressionMapping.emplace(label.getName(), label.getStatePredicateExpression());
1430  }
1431

1433  if (program.getModelType() == storm::prism::Program::ModelType::DTMC) {
1434  result = std::shared_ptr<storm::models::symbolic::Model<Type, ValueType>>(new storm::models::symbolic::Dtmc<Type, ValueType>(generationInfo.manager, reachableStates, initialStates, deadlockStates, transitionMatrix, generationInfo.rowMetaVariables, generationInfo.rowExpressionAdapter, generationInfo.columnMetaVariables, generationInfo.rowColumnMetaVariablePairs, labelToExpressionMapping, rewardModels));
1435  } else if (program.getModelType() == storm::prism::Program::ModelType::CTMC) {
1436  result = std::shared_ptr<storm::models::symbolic::Model<Type, ValueType>>(new storm::models::symbolic::Ctmc<Type, ValueType>(generationInfo.manager, reachableStates, initialStates, deadlockStates, transitionMatrix, system.stateActionDd, generationInfo.rowMetaVariables, generationInfo.rowExpressionAdapter, generationInfo.columnMetaVariables, generationInfo.rowColumnMetaVariablePairs, labelToExpressionMapping, rewardModels));
1437  } else if (program.getModelType() == storm::prism::Program::ModelType::MDP) {
1438  result = std::shared_ptr<storm::models::symbolic::Model<Type, ValueType>>(new storm::models::symbolic::Mdp<Type, ValueType>(generationInfo.manager, reachableStates, initialStates, deadlockStates, transitionMatrix, generationInfo.rowMetaVariables, generationInfo.rowExpressionAdapter, generationInfo.columnMetaVariables, generationInfo.rowColumnMetaVariablePairs, generationInfo.allNondeterminismVariables, labelToExpressionMapping, rewardModels));
1439  } else {
1440  STORM_LOG_THROW(false, storm::exceptions::InvalidArgumentException, "Invalid model type.");
1441  }
1442

1444  result->addParameters(generationInfo.parameters);
1445  }
1446
