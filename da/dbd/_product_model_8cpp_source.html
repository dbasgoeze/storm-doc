96  storm::storage::MemoryStructure ProductModel<ValueType>::computeMemoryStructure(storm::models::sparse::Model<ValueType> const& model, std::vector<storm::modelchecker::multiobjective::Objective<ValueType>> const& objectives) {
97 
98  storm::modelchecker::SparsePropositionalModelChecker<storm::models::sparse::Model<ValueType>> mc(model);
99 
100  // Create a memory structure that remembers whether (sub)objectives are satisfied
101  storm::storage::MemoryStructure memory = storm::storage::MemoryStructureBuilder<ValueType>::buildTrivialMemoryStructure(model);
102  for (uint64_t objIndex = 0; objIndex < objectives.size(); ++objIndex) {
103  if (!objectives[objIndex].formula->isProbabilityOperatorFormula()) {
104  continue;
105  }
106 
107  std::vector<uint64_t> dimensionIndexMap;
108  for (auto const& globalDimensionIndex : objectiveDimensions[objIndex]) {
109  dimensionIndexMap.push_back(globalDimensionIndex);
110  }
111 
112  // collect the memory states for this objective
113  std::vector<storm::storage::BitVector> objMemStates;
114  storm::storage::BitVector m(dimensionIndexMap.size(), false);
115  for (; !m.full(); m.increment()) {
116  objMemStates.push_back(~m);
117  }
118  objMemStates.push_back(~m);
119  assert(objMemStates.size() == 1ull << dimensionIndexMap.size());
120 
121  // build objective memory
122  auto objMemoryBuilder = storm::storage::MemoryStructureBuilder<ValueType>(objMemStates.size(), model);
123 
124  // Get the set of states that for all subobjectives satisfy either the left or the right subformula
125  storm::storage::BitVector constraintStates(model.getNumberOfStates(), true);
126  for (auto const& dim : objectiveDimensions[objIndex]) {
127  auto const& dimension = dimensions[dim];
128  STORM_LOG_ASSERT(dimension.formula->isBoundedUntilFormula(), "Unexpected Formula type");
129  constraintStates &=
130  (mc.check(dimension.formula->asBoundedUntilFormula().getLeftSubformula())->asExplicitQualitativeCheckResult().getTruthValuesVector() |
131  mc.check(dimension.formula->asBoundedUntilFormula().getRightSubformula())->asExplicitQualitativeCheckResult().getTruthValuesVector());
132  }
133 
134  // Build the transitions between the memory states
135  for (uint64_t memState = 0; memState < objMemStates.size(); ++memState) {
136  auto const& memStateBV = objMemStates[memState];
137  for (uint64_t memStatePrime = 0; memStatePrime < objMemStates.size(); ++memStatePrime) {
138  auto const& memStatePrimeBV = objMemStates[memStatePrime];
139  if (memStatePrimeBV.isSubsetOf(memStateBV)) {
140 
141  std::shared_ptr<storm::logic::Formula const> transitionFormula = storm::logic::Formula::getTrueFormula();
142  for (auto const& subObjIndex : memStateBV) {
143  std::shared_ptr<storm::logic::Formula const> subObjFormula = dimensions[dimensionIndexMap[subObjIndex]].formula->asBoundedUntilFormula().getRightSubformula().asSharedPointer();
144  if (memStatePrimeBV.get(subObjIndex)) {
145  subObjFormula = std::make_shared<storm::logic::UnaryBooleanStateFormula>(storm::logic::UnaryBooleanStateFormula::OperatorType::Not, subObjFormula);
146  }
147  transitionFormula = std::make_shared<storm::logic::BinaryBooleanStateFormula>(storm::logic::BinaryBooleanStateFormula::OperatorType::And, transitionFormula, subObjFormula);
148  }
149 
150  storm::storage::BitVector transitionStates = mc.check(*transitionFormula)->asExplicitQualitativeCheckResult().getTruthValuesVector();
151  if (memStatePrimeBV.empty()) {
152  transitionStates |= ~constraintStates;
153  } else {
154  transitionStates &= constraintStates;
155  }
156  objMemoryBuilder.setTransition(memState, memStatePrime, transitionStates);
157 
158  // Set the initial states
159  if (memStateBV.full()) {
160  storm::storage::BitVector initialTransitionStates = model.getInitialStates() & transitionStates;
161  // At this point we can check whether there is an initial state that already satisfies all subObjectives.
162  // Such a situation can not be reduced (easily) to an expected reward computation and thus requires special treatment
163  if (memStatePrimeBV.empty() && !initialTransitionStates.empty()) {
164  prob1InitialStates[objIndex] = initialTransitionStates;
165  }
166 
167  for (auto const& initState : initialTransitionStates) {
168  objMemoryBuilder.setInitialMemoryState(initState, memStatePrime);
169  }
170  }
171  }
172  }
173  }
174 
175  // Build the memory labels
176  for (uint64_t memState = 0; memState < objMemStates.size(); ++memState) {
177  auto const& memStateBV = objMemStates[memState];
178  for (auto const& subObjIndex : memStateBV) {
179  objMemoryBuilder.setLabel(memState, dimensions[dimensionIndexMap[subObjIndex]].memoryLabel.get());
180  }
181  }
182  auto objMemory = objMemoryBuilder.build();
183  memory = memory.product(objMemory);
184  }
185  return memory;

212 
213  std::vector<uint64_t> inverseMemoryStateMap(memoryStateManager.getUpperMemoryStateBound(), std::numeric_limits<uint64_t>::max());
214  for (uint64_t memStateIndex = 0; memStateIndex < memoryStateMap.size(); ++memStateIndex) {
215  inverseMemoryStateMap[memoryStateMap[memStateIndex]] = memStateIndex;
216  }
217 
218  auto const& memory = productBuilder.getMemory();
219  auto const& model = productBuilder.getOriginalModel();
220  auto const& modelTransitions = model.getTransitionMatrix();
221 
222  std::vector<storm::storage::BitVector> reachableProductStates(memoryStateManager.getUpperMemoryStateBound());
223  for (auto const& memState : memoryStateMap) {
224  reachableProductStates[memState] = storm::storage::BitVector(model.getNumberOfStates(), false);
225  }
226 
227  // Initialize the reachable states with the initial states
228  // If the bound is not known (e.g., when computing quantiles) we don't know the initial epoch class. Hence, we consider all possibilities.
229  std::vector<EpochClass> initEpochClasses;
230  initEpochClasses.push_back(epochManager.getEpochClass(epochManager.getZeroEpoch()));
231  for (uint64_t dim = 0; dim < dimensions.size(); ++dim) {
232  Dimension<ValueType> const& dimension = dimensions[dim];
233  if (dimension.boundType == DimensionBoundType::Unbounded) {
234  // For unbounded dimensions we are only interested in the bottom class.
235  for (auto& ec : initEpochClasses) {
236  epochManager.setDimensionOfEpochClass(ec, dim, true);
237  }
238  } else if (!dimension.maxValue) {
239  // If no max value is known, we have to consider all possibilities
240  std::vector<EpochClass> newEcs = initEpochClasses;
241  for (auto& ec : newEcs) {
242  epochManager.setDimensionOfEpochClass(ec, dim, true);
243  }
244  initEpochClasses.insert(initEpochClasses.end(), newEcs.begin(), newEcs.end());
245  }
246  }
247  for (auto const& initEpochClass : initEpochClasses) {
248  auto memStateIt = memory.getInitialMemoryStates().begin();
249  for (auto const& initState : model.getInitialStates()) {
250  uint64_t transformedMemoryState = transformMemoryState(memoryStateMap[*memStateIt], initEpochClass, memoryStateManager.getInitialMemoryState());
251  reachableProductStates[transformedMemoryState].set(initState, true);
252  ++memStateIt;
253  }
254  assert(memStateIt == memory.getInitialMemoryStates().end());
255  }
256 
257  // Find the reachable epoch classes
258  std::set<Epoch> possibleSteps(originalModelSteps.begin(), originalModelSteps.end());
259  std::set<EpochClass, std::function<bool(EpochClass const&, EpochClass const&)>> reachableEpochClasses(std::bind(&EpochManager::epochClassOrder, &epochManager, std::placeholders::_1, std::placeholders::_2));
260  collectReachableEpochClasses(reachableEpochClasses, possibleSteps);
261 
262 
263  // Iterate over all epoch classes starting from the initial one (i.e., no bottom dimension).
264  for (auto epochClassIt = reachableEpochClasses.rbegin(); epochClassIt != reachableEpochClasses.rend(); ++epochClassIt) {
265  auto const& epochClass = *epochClassIt;
266 
267  // Find the remaining set of reachable states via DFS.
268  std::vector<std::pair<uint64_t, MemoryState>> dfsStack;
269  for (MemoryState const& memState : memoryStateMap) {
270  for (auto const& modelState : reachableProductStates[memState]) {
271  dfsStack.emplace_back(modelState, memState);
272  }
273  }
274 
275  while (!dfsStack.empty()) {
276  uint64_t currentModelState = dfsStack.back().first;
277  MemoryState currentMemoryState = dfsStack.back().second;
278  uint64_t currentMemoryStateIndex = inverseMemoryStateMap[currentMemoryState];
279  dfsStack.pop_back();
280 
281  for (uint64_t choice = modelTransitions.getRowGroupIndices()[currentModelState]; choice != modelTransitions.getRowGroupIndices()[currentModelState + 1]; ++choice) {
282 
283  for (auto transitionIt = modelTransitions.getRow(choice).begin(); transitionIt < modelTransitions.getRow(choice).end(); ++transitionIt) {
284 
285  MemoryState successorMemoryState = memoryStateMap[memory.getSuccessorMemoryState(currentMemoryStateIndex, transitionIt - modelTransitions.begin())];
286  successorMemoryState = transformMemoryState(successorMemoryState, epochClass, currentMemoryState);
287  if (!reachableProductStates[successorMemoryState].get(transitionIt->getColumn())) {
288  reachableProductStates[successorMemoryState].set(transitionIt->getColumn(), true);
289  dfsStack.emplace_back(transitionIt->getColumn(), successorMemoryState);
290  }
291  }
292  }
293  }
294  }
295 
296  for (uint64_t memStateIndex = 0; memStateIndex < memoryStateManager.getMemoryStateCount(); ++memStateIndex) {
297  for (auto const& modelState : reachableProductStates[memoryStateMap[memStateIndex]]) {
298  productBuilder.addReachableState(modelState, memStateIndex);
299  }
300  }

353  std::vector<std::vector<ValueType>> ProductModel<ValueType>::computeObjectiveRewards(EpochClass const& epochClass, std::vector<storm::modelchecker::multiobjective::Objective<ValueType>> const& objectives) const {
354  std::vector<std::vector<ValueType>> objectiveRewards;
355  objectiveRewards.reserve(objectives.size());
356 
357  for (uint64_t objIndex = 0; objIndex < objectives.size(); ++objIndex) {
358  auto const& formula = *objectives[objIndex].formula;
359  if (formula.isProbabilityOperatorFormula()) {
360  storm::modelchecker::SparsePropositionalModelChecker<storm::models::sparse::Model<ValueType>> mc(getProduct());
361  std::vector<uint64_t> dimensionIndexMap;
362  for (auto const& globalDimensionIndex : objectiveDimensions[objIndex]) {
363  dimensionIndexMap.push_back(globalDimensionIndex);
364  }
365 
366  std::shared_ptr<storm::logic::Formula const> sinkStatesFormula;
367  for (auto const& dim : objectiveDimensions[objIndex]) {
368  auto memLabelFormula = std::make_shared<storm::logic::AtomicLabelFormula>(dimensions[dim].memoryLabel.get());
369  if (sinkStatesFormula) {
370  sinkStatesFormula = std::make_shared<storm::logic::BinaryBooleanStateFormula>(storm::logic::BinaryBooleanStateFormula::OperatorType::Or, sinkStatesFormula, memLabelFormula);
371  } else {
372  sinkStatesFormula = memLabelFormula;
373  }
374  }
375  sinkStatesFormula = std::make_shared<storm::logic::UnaryBooleanStateFormula>(storm::logic::UnaryBooleanStateFormula::OperatorType::Not, sinkStatesFormula);
376 
377  std::vector<ValueType> objRew(getProduct().getTransitionMatrix().getRowCount(), storm::utility::zero<ValueType>());
378  storm::storage::BitVector relevantObjectives(objectiveDimensions[objIndex].getNumberOfSetBits());
379 
380  while (!relevantObjectives.full()) {
381  relevantObjectives.increment();
382 
383  // find out whether objective reward should be earned within this epoch class
384  bool collectRewardInEpoch = true;
385  for (auto const& subObjIndex : relevantObjectives) {
386  if (dimensions[dimensionIndexMap[subObjIndex]].boundType == DimensionBoundType::UpperBound && epochManager.isBottomDimensionEpochClass(epochClass, dimensionIndexMap[subObjIndex])) {
387  collectRewardInEpoch = false;
388  break;
389  }
390  }
391 
392  if (collectRewardInEpoch) {
393  std::shared_ptr<storm::logic::Formula const> relevantStatesFormula;
394  std::shared_ptr<storm::logic::Formula const> goalStatesFormula = storm::logic::CloneVisitor().clone(*sinkStatesFormula);
395  for (uint64_t subObjIndex = 0; subObjIndex < dimensionIndexMap.size(); ++subObjIndex) {
396  std::shared_ptr<storm::logic::Formula> memLabelFormula = std::make_shared<storm::logic::AtomicLabelFormula>(dimensions[dimensionIndexMap[subObjIndex]].memoryLabel.get());
397  if (relevantObjectives.get(subObjIndex)) {
398  auto rightSubFormula = dimensions[dimensionIndexMap[subObjIndex]].formula->asBoundedUntilFormula().getRightSubformula().asSharedPointer();
399  goalStatesFormula = std::make_shared<storm::logic::BinaryBooleanStateFormula>(storm::logic::BinaryBooleanStateFormula::OperatorType::And, goalStatesFormula, rightSubFormula);
400  } else {
401  memLabelFormula = std::make_shared<storm::logic::UnaryBooleanStateFormula>(storm::logic::UnaryBooleanStateFormula::OperatorType::Not, memLabelFormula);
402  }
403  if (relevantStatesFormula) {
404  relevantStatesFormula = std::make_shared<storm::logic::BinaryBooleanStateFormula>(storm::logic::BinaryBooleanStateFormula::OperatorType::And, relevantStatesFormula, memLabelFormula);
405  } else {
406  relevantStatesFormula = memLabelFormula;
407  }
408  }
409 
410  storm::storage::BitVector relevantStates = mc.check(*relevantStatesFormula)->asExplicitQualitativeCheckResult().getTruthValuesVector();
411  storm::storage::BitVector relevantChoices = getProduct().getTransitionMatrix().getRowFilter(relevantStates);
412  storm::storage::BitVector goalStates = mc.check(*goalStatesFormula)->asExplicitQualitativeCheckResult().getTruthValuesVector();
413  for (auto const& choice : relevantChoices) {
414  objRew[choice] += getProduct().getTransitionMatrix().getConstrainedRowSum(choice, goalStates);
415  }
416  }
417  }
418 
419  objectiveRewards.push_back(std::move(objRew));
420 
421  } else if (formula.isRewardOperatorFormula()) {
422  auto const& rewModel = getProduct().getRewardModel(formula.asRewardOperatorFormula().getRewardModelName());
423  STORM_LOG_THROW(!rewModel.hasTransitionRewards(), storm::exceptions::NotSupportedException, "Reward model has transition rewards which is not expected.");
424  bool rewardCollectedInEpoch = true;
425  if (formula.getSubformula().isCumulativeRewardFormula()) {
426  for (auto const& dim : objectiveDimensions[objIndex]) {
427  if (epochManager.isBottomDimensionEpochClass(epochClass, dim)) {
428  rewardCollectedInEpoch = false;
429  break;
430  }
431  }
432  } else {
433  STORM_LOG_THROW(formula.getSubformula().isTotalRewardFormula(), storm::exceptions::UnexpectedException, "Unexpected type of formula " << formula);
434  }
435  if (rewardCollectedInEpoch) {
436  objectiveRewards.push_back(rewModel.getTotalRewardVector(getProduct().getTransitionMatrix()));
437  } else {
438  objectiveRewards.emplace_back(getProduct().getTransitionMatrix().getRowCount(), storm::utility::zero<ValueType>());
439  }
440  } else {
441  STORM_LOG_THROW(false, storm::exceptions::UnexpectedException, "Unexpected type of formula " << formula);
442  }
443  }
444 
445  return objectiveRewards;
446  }

476  // For dimensions for which no bound (aka. maxValue) is known, we will later overapproximate the set of reachable classes.
477  Epoch startEpoch = epochManager.getZeroEpoch();
478  for (uint64_t dim = 0; dim < epochManager.getDimensionCount(); ++dim) {
479  if (dimensions[dim].maxValue) {
480  epochManager.setDimensionOfEpoch(startEpoch, dim, dimensions[dim].maxValue.get());
481  } else {
482  epochManager.setBottomDimension(startEpoch, dim);
483  }
484  }
485 
486  std::set<Epoch> seenEpochs({startEpoch});
487  std::vector<Epoch> dfsStack({startEpoch});
488 
489  reachableEpochClasses.insert(epochManager.getEpochClass(startEpoch));
490 
491  // Perform a DFS to find all the reachable epochs
492  while (!dfsStack.empty()) {
493  Epoch currentEpoch = dfsStack.back();
494  dfsStack.pop_back();
495  for (auto const& step : possibleSteps) {
496  Epoch successorEpoch = epochManager.getSuccessorEpoch(currentEpoch, step);
497  if (seenEpochs.insert(successorEpoch).second) {
498  reachableEpochClasses.insert(epochManager.getEpochClass(successorEpoch));
499  dfsStack.push_back(std::move(successorEpoch));
500  }
501  }
502  }
503 
504  // Also treat dimensions without a priori bound. Unbounded dimensions need no further treatment as for these only the 'bottom' class is relevant.
505  for (uint64_t dim = 0; dim < epochManager.getDimensionCount(); ++dim) {
506  if (dimensions[dim].boundType != DimensionBoundType::Unbounded && !dimensions[dim].maxValue) {
507  std::vector<EpochClass> newClasses;
508  for (auto const& c : reachableEpochClasses) {
509  auto newClass = c;
510  epochManager.setDimensionOfEpochClass(newClass, dim, false);
511  newClasses.push_back(newClass);
512  }
513  for (auto const& c: newClasses) {
514  reachableEpochClasses.insert(c);
515  }
516  }
517  }
518  }

520  template<typename ValueType>
521  void ProductModel<ValueType>::computeReachableStates(EpochClass const& epochClass, std::vector<EpochClass> const& predecessors) {
522 
523  storm::storage::BitVector bottomDimensions(epochManager.getDimensionCount(), false);
524  bool considerInitialStates = true;
525  for (uint64_t dim = 0; dim < epochManager.getDimensionCount(); ++dim) {
526  if (epochManager.isBottomDimensionEpochClass(epochClass, dim)) {
527  bottomDimensions.set(dim, true);
528  if (dimensions[dim].boundType != DimensionBoundType::Unbounded && dimensions[dim].maxValue) {
529  considerInitialStates = false;
530  }
531  }
532  }
533  storm::storage::BitVector nonBottomDimensions = ~bottomDimensions;
534 
535  storm::storage::BitVector ecInStates(getProduct().getNumberOfStates(), false);
536  if (considerInitialStates) {
537  for (auto const& initState : getProduct().getInitialStates()) {
538  uint64_t transformedInitState = transformProductState(initState, epochClass, memoryStateManager.getInitialMemoryState());
539  ecInStates.set(transformedInitState, true);
540  }
541  }
542  for (auto const& predecessor : predecessors) {
543  storm::storage::BitVector positiveStepDimensions(epochManager.getDimensionCount(), false);
544  for (uint64_t dim = 0; dim < epochManager.getDimensionCount(); ++dim) {
545  if (!epochManager.isBottomDimensionEpochClass(predecessor, dim) && bottomDimensions.get(dim)) {
546  positiveStepDimensions.set(dim, true);
547  }
548  }
549  STORM_LOG_ASSERT(reachableStates.find(predecessor) != reachableStates.end(), "Could not find reachable states of predecessor epoch class.");
550  storm::storage::BitVector predecessorStates = reachableStates.find(predecessor)->second;
551  for (auto const& predecessorState : predecessorStates) {
552  uint64_t predecessorMemoryState = getMemoryState(predecessorState);
553  for (uint64_t choice = getProduct().getTransitionMatrix().getRowGroupIndices()[predecessorState]; choice < getProduct().getTransitionMatrix().getRowGroupIndices()[predecessorState + 1]; ++choice) {
554  bool choiceLeadsToThisClass = false;
555  Epoch const& choiceStep = getSteps()[choice];
556  for (auto const& dim : positiveStepDimensions) {
557  if (epochManager.getDimensionOfEpoch(choiceStep, dim) > 0) {
558  choiceLeadsToThisClass = true;
559  }
560  }
561 
562  if (choiceLeadsToThisClass) {
563  for (auto const& transition : getProduct().getTransitionMatrix().getRow(choice)) {
564  uint64_t successorState = transformProductState(transition.getColumn(), epochClass, predecessorMemoryState);
565 
566  ecInStates.set(successorState, true);
567  }
568  }
569  }
570  }
571  }
572 
573  // Find all states reachable from an InState via DFS.
574  storm::storage::BitVector ecReachableStates = ecInStates;
575  std::vector<uint64_t> dfsStack(ecReachableStates.begin(), ecReachableStates.end());
576 
577  while (!dfsStack.empty()) {
578  uint64_t currentState = dfsStack.back();
579  uint64_t currentMemoryState = getMemoryState(currentState);
580  dfsStack.pop_back();
581 
582  for (uint64_t choice = getProduct().getTransitionMatrix().getRowGroupIndices()[currentState]; choice != getProduct().getTransitionMatrix().getRowGroupIndices()[currentState + 1]; ++choice) {
583 
584  bool choiceLeadsOutsideOfEpoch = false;
585  Epoch const& choiceStep = getSteps()[choice];
586  for (auto const& dim : nonBottomDimensions) {
587  if (epochManager.getDimensionOfEpoch(choiceStep, dim) > 0) {
588  choiceLeadsOutsideOfEpoch = true;
589  break;
590  }
591  }
592 
593  for (auto const& transition : getProduct().getTransitionMatrix().getRow(choice)) {
594  uint64_t successorState = transformProductState(transition.getColumn(), epochClass, currentMemoryState);
595  if (choiceLeadsOutsideOfEpoch) {
596  ecInStates.set(successorState, true);
597  }
598  if (!ecReachableStates.get(successorState)) {
599  ecReachableStates.set(successorState, true);
600  dfsStack.push_back(successorState);
601  }
602  }
603  }
604  }
605 
606  reachableStates[epochClass] = std::move(ecReachableStates);
607 
608  inStates[epochClass] = std::move(ecInStates);
609  }

615  for (auto const& objDimensions : objectiveDimensions) {

627  }
628  }
