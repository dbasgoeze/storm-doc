71  uint_fast64_t computeStatePenalty(storm::storage::sparse::state_type const& state, storm::storage::FlexibleSparseMatrix<ValueType> const& transitionMatrix, storm::storage::FlexibleSparseMatrix<ValueType> const& backwardTransitions, std::vector<ValueType> const& oneStepProbabilities) {
72  uint_fast64_t penalty = 0;
73  bool hasParametricSelfLoop = false;
74 
75  for (auto const& predecessor : backwardTransitions.getRow(state)) {
76  for (auto const& successor : transitionMatrix.getRow(state)) {
77  penalty += estimateComplexity(predecessor.getValue()) * estimateComplexity(successor.getValue());
78  }
79  if (predecessor.getColumn() == state) {
80  hasParametricSelfLoop = !storm::utility::isConstant(predecessor.getValue());
81  }
82  penalty += estimateComplexity(oneStepProbabilities[predecessor.getColumn()]) * estimateComplexity(predecessor.getValue()) * estimateComplexity(oneStepProbabilities[state]);
83  }
84 
85  // If it is a self-loop that is parametric, we increase the penalty a lot.
86  if (hasParametricSelfLoop) {
87  penalty *= 10;
88  }
89 
90  return penalty;
91  }

112  return std::make_unique<StaticStatePriorityQueue>(sortedStates);

118  } else if (eliminationOrderIsPenaltyBased(order)) {

120  typename DynamicStatePriorityQueue<ValueType>::PenaltyFunctionType penaltyFunction = order == storm::settings::modules::EliminationSettings::EliminationOrder::RegularExpression ? computeStatePenaltyRegularExpression<ValueType> : computeStatePenalty<ValueType>;
121  for (uint_fast64_t index = 0; index < sortedStates.size(); ++index) {
122  statePenalties[index] = std::make_pair(sortedStates[index], penaltyFunction(sortedStates[index], transitionMatrix, backwardTransitions, oneStepProbabilities));
123  }
124

126

129  for (uint_fast64_t index = 0; index < sortedStates.size(); ++index) {
