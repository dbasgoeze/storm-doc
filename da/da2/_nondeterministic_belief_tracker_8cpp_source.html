181  void SparseBeliefState<ValueType>::updateHelper(std::vector<std::map<uint64_t, ValueType>> const& partialBeliefs, std::vector<ValueType> const& sums, typename std::map<uint64_t, ValueType>::const_iterator nextStateIt, uint32_t newObservation, std::unordered_set<SparseBeliefState<ValueType>>& previousBeliefs) const {
182  if(nextStateIt == belief.end()) {
183  for (uint64_t i = 0; i < partialBeliefs.size(); ++i) {
184  auto const& partialBelief = partialBeliefs[i];
185  auto const& sum = sums[i];
186  if (storm::utility::isZero(sum)) {
187  continue;
188  }
189  std::size_t newHash = 0;
190  ValueType risk = storm::utility::zero<ValueType>();
191  std::map<uint64_t, ValueType> finalBelief;
192  for (auto &entry : partialBelief) {
193  assert(!storm::utility::isZero(sum));
194  finalBelief[entry.first] = entry.second / sum;
195  //boost::hash_combine(newHash, std::hash<ValueType>()(entry.second));
196  boost::hash_combine(newHash, entry.first);
197  risk += entry.second / sum * manager->getRisk(entry.first);
198  }
199  previousBeliefs.insert(SparseBeliefState<ValueType>(manager, finalBelief, newHash, risk, id));
200  }
201  } else {
202  uint64_t state = nextStateIt->first;
203  auto newNextStateIt = nextStateIt;
204  newNextStateIt++;
205  std::vector<std::map<uint64_t, ValueType>> newPartialBeliefs;
206  std::vector<ValueType> newSums;
207  for (uint64_t i = 0; i < partialBeliefs.size(); ++i) {
208 
209  for (auto row = manager->getPomdp().getNondeterministicChoiceIndices()[state];
210  row < manager->getPomdp().getNondeterministicChoiceIndices()[state + 1]; ++row) {
211  std::map<uint64_t, ValueType> newPartialBelief = partialBeliefs[i];
212  ValueType newSum = sums[i];
213  for (auto const &transition : manager->getPomdp().getTransitionMatrix().getRow(row)) {
214  if (newObservation != manager->getPomdp().getObservation(transition.getColumn())) {
215  continue;
216  }
217 
218  if (newPartialBelief.count(transition.getColumn()) == 0) {
219  newPartialBelief[transition.getColumn()] = transition.getValue() * nextStateIt->second;
220  } else {
221  newPartialBelief[transition.getColumn()] += transition.getValue() * nextStateIt->second;
222  }
223  newSum += transition.getValue() * nextStateIt->second;
224 
225  }
226  newPartialBeliefs.push_back(newPartialBelief);
227  newSums.push_back(newSum);
228  }
