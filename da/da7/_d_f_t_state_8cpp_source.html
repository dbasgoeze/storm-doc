251  template<typename ValueType>
252  bool DFTState<ValueType>::updateFailableInRestrictions(size_t id) {
253  if (!hasFailed(id)) {
254  // Non-failure does not change anything in a restriction
255  return false;
256  }
257 
258  bool addedFailableEvent = false;
259  for (auto restriction : mDft.getElement(id)->restrictions()) {
260  STORM_LOG_ASSERT(restriction->containsChild(id), "Ids do not match.");
261  if (restriction->isSeqEnforcer()) {
262  for (auto it = restriction->children().cbegin(); it != restriction->children().cend(); ++it) {
263  if ((*it)->id() != id) {
264  if (!hasFailed((*it)->id())) {
265  // Failure should be prevented later on
266  STORM_LOG_TRACE("Child " << (*it)->name() << " should have failed.");
267  break;
268  }
269  } else {
270  // Current event has failed
271  STORM_LOG_ASSERT(hasFailed((*it)->id()), "Child " << (*it)->name() << " should have failed.");
272  ++it;
273  while (it != restriction->children().cend() && !isOperational((*it)->id())) {
274  ++it;
275  }
276  if (it != restriction->children().cend() && (*it)->isBasicElement() && isOperational((*it)->id())) {
277  // Enable next event
278  failableElements.addBE((*it)->id());
279  STORM_LOG_TRACE("Added possible BE failure: " << *(*it));
280  addedFailableEvent = true;
281  }
282  break;
283  }
284  }
285  } else if (restriction->isMutex()) {
286  // Current element has failed and disables all other children
287  for (auto const& child : restriction->children()) {
288  if (child->isBasicElement() && child->id() != id && getElementState(child->id()) == DFTElementState::Operational) {
289  // Disable child
290  failableElements.removeBE(child->id());
291  STORM_LOG_TRACE("Disabled child: " << *child);
292  addedFailableEvent = true;
293  }
294  }
295  } else {

297  }
298  }
