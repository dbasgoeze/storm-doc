37  std::vector<Point> parsePoints(std::vector<std::string> const& input) {
38  std::vector<Point> result;
39  for (auto const& i : input) {
40  Point currPoint;
41  std::size_t pos1 = 0;
42  std::size_t pos2 = i.find(",");
43  while (pos2 != std::string::npos) {
44  currPoint.push_back(storm::utility::convertNumber<storm::RationalNumber, std::string>(i.substr(pos1, pos2 - pos1)));
45  pos1 = pos2 + 1;
46  pos2 = i.find(",", pos1);
47  }
48  currPoint.push_back(storm::utility::convertNumber<storm::RationalNumber, std::string>(i.substr(pos1)));
49  result.push_back(currPoint);
50  }
51  return result;
52  }

54  std::set<Point> setMinus(std::vector<Point> const& lhs, std::vector<Point> const& rhs) {
55  std::set<Point> result(lhs.begin(), lhs.end());
56  for (auto const& r : rhs) {
57  for (auto lIt = result.begin(); lIt != result.end(); ++lIt) {
58  if (*lIt == r) {
59  result.erase(lIt);
60  break;
61  }
62  }
63  }
64  return result;
65  }
