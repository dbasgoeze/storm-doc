97  CompositionInformationVisitor::CompositionInformationVisitor(Model const& model, Composition const& composition) : model(model), composition(composition), nextFreeActionIndex(0) {
98  // Determine the next free index we can give out to a new action.
99  for (auto const& action : model.getActions()) {
100  uint64_t actionIndex = model.getActionIndex(action.getName());
101 
102  nameToIndexMap[action.getName()] = model.getActionIndex(action.getName());
103  indexToNameMap[actionIndex] = action.getName();
104 
105  nextFreeActionIndex = std::max(nextFreeActionIndex, model.getActionIndex(action.getName()));
106  }

116  boost::any CompositionInformationVisitor::visit(AutomatonComposition const& composition, boost::any const& data) {
117  Model const& model = boost::any_cast<Model const&>(data);
118  Automaton const& automaton = model.getAutomaton(composition.getAutomatonName());
119 
120  CompositionInformation result;
121  result.increaseAutomatonMultiplicity(composition.getAutomatonName());
122  for (auto const& actionIndex : automaton.getActionIndices()) {
123  if (actionIndex != storm::jani::Model::SILENT_ACTION_INDEX) {
124  result.addNonSilentActionIndex(actionIndex);
125  }
126  }
127  for (auto const& action : composition.getInputEnabledActions()) {
128  auto it = nameToIndexMap.find(action);
129  STORM_LOG_THROW(it != nameToIndexMap.end(), storm::exceptions::WrongFormatException, "Illegal action name '" << action << "' in when input-enabling automaton '" << composition.getAutomatonName() << "'.");
130  uint64_t actionIndex = it->second;
131  STORM_LOG_THROW(result.hasNonSilentActionIndex(actionIndex), storm::exceptions::WrongFormatException, "Cannot make automaton '" << composition.getAutomatonName() << "' input enabled for action " << action << ", because it has no such action.");
132  result.addInputEnabledActionIndex(actionIndex);
133  }
134  return result;

137  boost::any CompositionInformationVisitor::visit(ParallelComposition const& composition, boost::any const& data) {
138  CompositionInformation result;
139 
140  std::vector<CompositionInformation> subinformation;
141 
142  std::set<uint64_t> nonsilentSubActionIndices;
143  for (auto const& subcomposition : composition.getSubcompositions()) {
144  subinformation.push_back(boost::any_cast<CompositionInformation>(subcomposition->accept(*this, data)));
145  nonsilentSubActionIndices.insert(subinformation.back().getNonSilentActionIndices().begin(), subinformation.back().getNonSilentActionIndices().end());
146  }
147 
148  bool containsNonStandardParallelComposition = false;
149  bool containsSubParallelComposition = false;
150 
151  for (auto const& subinfo : subinformation) {
152  containsNonStandardParallelComposition |= subinfo.containsNonStandardParallelComposition();
153  containsSubParallelComposition |= subinfo.containsParallelComposition();
154  result.addMultiplicityMap(subinfo.getAutomatonToMultiplicityMap());
155  }
156 
157  // Keep track of the synchronization vectors that are effective, meaning that the subcompositions all have
158  // the non-silent actions that are referred to.
159  std::set<uint64_t> effectiveSynchVectors;
160  for (uint64_t synchVectorIndex = 0; synchVectorIndex < composition.getNumberOfSynchronizationVectors(); ++synchVectorIndex) {
161  effectiveSynchVectors.insert(synchVectorIndex);
162  }
163 
164  // Now compute non-silent actions.
165  std::set<uint64_t> nonSilentActionIndices;
166  for (uint_fast64_t infoIndex = 0; infoIndex < subinformation.size(); ++infoIndex) {
167  auto const& subinfo = subinformation[infoIndex];
168 
169  std::set<uint64_t> enabledSynchVectors;
170  std::set<uint64_t> actionsInSynch;
171  for (uint64_t synchVectorIndex = 0; synchVectorIndex < composition.getNumberOfSynchronizationVectors(); ++synchVectorIndex) {
172  auto const& synchVector = composition.getSynchronizationVector(synchVectorIndex);
173  if (synchVector.getInput(infoIndex) != SynchronizationVector::NO_ACTION_INPUT) {
174  for (auto const& nonSilentActionIndex : subinfo.getNonSilentActionIndices()) {
175  std::string const& nonSilentAction = indexToNameMap.at(nonSilentActionIndex);
176  if (synchVector.getInput(infoIndex) == nonSilentAction) {
177  enabledSynchVectors.insert(synchVectorIndex);
178  actionsInSynch.insert(nonSilentActionIndex);
179  }
180  }
181  } else {
182  enabledSynchVectors.insert(synchVectorIndex);
183  }
184  }
185 
186  std::set_difference(subinfo.getNonSilentActionIndices().begin(), subinfo.getNonSilentActionIndices().end(), actionsInSynch.begin(), actionsInSynch.end(), std::inserter(nonSilentActionIndices, nonSilentActionIndices.begin()));
187 
188  std::set<uint64_t> newEffectiveSynchVectors;
189  std::set_intersection(effectiveSynchVectors.begin(), effectiveSynchVectors.end(), enabledSynchVectors.begin(), enabledSynchVectors.end(), std::inserter(newEffectiveSynchVectors, newEffectiveSynchVectors.begin()));
190  effectiveSynchVectors = std::move(newEffectiveSynchVectors);
191  }
192 
193  // Now add all outputs of effective synchronization vectors.
194  for (auto const& synchVectorIndex : effectiveSynchVectors) {
195  auto const& synchVector = composition.getSynchronizationVector(synchVectorIndex);
196  if (synchVector.getOutput() != storm::jani::Model::SILENT_ACTION_NAME) {
197  nonSilentActionIndices.insert(addOrGetActionIndex(synchVector.getOutput()));
198  }
199  }
200 
201  // Finally check whether it's a non-standard parallel composition. We do that by first constructing a set of
202  // all effective synchronization vectors and then checking whether this set is fully contained within the
203  // set of expected synchronization vectors.
204 
205  std::set<storm::jani::SynchronizationVector, storm::jani::SynchronizationVectorLexicographicalLess> synchVectorSet;
206  for (auto synchVectorIndex : effectiveSynchVectors) {
207  synchVectorSet.insert(composition.getSynchronizationVector(synchVectorIndex));
208  }
209 
210  // Construct the set of expected synchronization vectors.
211  std::set<storm::jani::SynchronizationVector, storm::jani::SynchronizationVectorLexicographicalLess> expectedSynchVectorSetUnderApprox;
212  std::set<storm::jani::SynchronizationVector, storm::jani::SynchronizationVectorLexicographicalLess> expectedSynchVectorSetOverApprox;
213  for (auto actionIndex : nonsilentSubActionIndices) {
214  std::string const& actionName = indexToNameMap.at(actionIndex);
215  std::vector<std::string> input;
216  uint64_t numberOfParticipatingAutomata = 0;
217  for (auto const& subcomposition : subinformation) {
218  if (subcomposition.getNonSilentActionIndices().find(actionIndex) != subcomposition.getNonSilentActionIndices().end()) {
219  input.push_back(actionName);
220  ++numberOfParticipatingAutomata;
221  } else {
222  input.push_back(SynchronizationVector::NO_ACTION_INPUT);
223  }
224  }
225 
226  storm::jani::SynchronizationVector newSynchVector(input, actionName);
227  expectedSynchVectorSetOverApprox.insert(newSynchVector);
228  if (numberOfParticipatingAutomata > 1) {
229  expectedSynchVectorSetUnderApprox.insert(newSynchVector);
230  }
231  }
232 
233  containsNonStandardParallelComposition |= !std::includes(expectedSynchVectorSetOverApprox.begin(), expectedSynchVectorSetOverApprox.end(), synchVectorSet.begin(), synchVectorSet.end(), SynchronizationVectorLexicographicalLess());
234 
235  containsNonStandardParallelComposition |= !std::includes(synchVectorSet.begin(), synchVectorSet.end(), expectedSynchVectorSetUnderApprox.begin(), expectedSynchVectorSetUnderApprox.end(), SynchronizationVectorLexicographicalLess());
236 
237  result.setContainsNonStandardParallelComposition(containsNonStandardParallelComposition);
238  result.setContainsParallelComposition(true);
239  result.setContainsNestedParallelComposition(containsSubParallelComposition);
240 
241  result.addNonSilentActionIndices(nonSilentActionIndices);
242  return result;
243  }
