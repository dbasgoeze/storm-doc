24  template<typename SparseModelType>
25  bool SparseParametricModelSimplifier<SparseModelType>::simplify(storm::logic::Formula const& formula) {
26  // Make sure that there is no old result from a previous call
27  simplifiedModel = nullptr;
28  simplifiedFormula = nullptr;
29  if (formula.isProbabilityOperatorFormula()) {
30  storm::logic::ProbabilityOperatorFormula const& probOpForm = formula.asProbabilityOperatorFormula();
31  if (probOpForm.getSubformula().isUntilFormula()) {
32  return simplifyForUntilProbabilities(probOpForm);
33  } else if (probOpForm.getSubformula().isReachabilityProbabilityFormula()) {
34  return simplifyForReachabilityProbabilities(probOpForm);
35  } else if (probOpForm.getSubformula().isBoundedUntilFormula()) {
36  return simplifyForBoundedUntilProbabilities(probOpForm);
37  }
38  } else if (formula.isRewardOperatorFormula()) {
39  storm::logic::RewardOperatorFormula rewOpForm = formula.asRewardOperatorFormula();
40  STORM_LOG_THROW((rewOpForm.hasRewardModelName() && originalModel.hasRewardModel(rewOpForm.getRewardModelName())) || (!rewOpForm.hasRewardModelName() && originalModel.hasUniqueRewardModel()), storm::exceptions::InvalidPropertyException, "The reward model specified by formula " << formula << " is not available in the given model.");
41  if (rewOpForm.getSubformula().isReachabilityRewardFormula()) {
42  return simplifyForReachabilityRewards(rewOpForm);
43  } else if (rewOpForm.getSubformula().isCumulativeRewardFormula()) {
44  return simplifyForCumulativeRewards(rewOpForm);
45  }
46  }
47  // reaching this point means that the provided formula is not supported. Thus, no simplification is possible.
48  STORM_LOG_DEBUG("Simplification not possible because the formula is not supported. Formula: " << formula);
49  return false;
50  }

100  std::shared_ptr<SparseModelType> SparseParametricModelSimplifier<SparseModelType>::eliminateConstantDeterministicStates(SparseModelType const& model, storm::storage::BitVector const& consideredStates, boost::optional<std::string> const& rewardModelName) {
101  storm::storage::SparseMatrix<typename SparseModelType::ValueType> const& sparseMatrix = model.getTransitionMatrix();
102 
103  // get the action-based reward values
104  std::vector<typename SparseModelType::ValueType> actionRewards;
105  if(rewardModelName) {
106  actionRewards = model.getRewardModel(*rewardModelName).getTotalRewardVector(sparseMatrix);
107  } else {
108  actionRewards = std::vector<typename SparseModelType::ValueType>(model.getTransitionMatrix().getRowCount(), storm::utility::zero<typename SparseModelType::ValueType>());
109  }
110 
111  // Find the states that are to be eliminated
112  storm::storage::BitVector selectedStates = consideredStates;
113  for (auto state : consideredStates) {
114  if (sparseMatrix.getRowGroupSize(state) == 1 && (!rewardModelName.is_initialized() || storm::utility::isConstant(actionRewards[sparseMatrix.getRowGroupIndices()[state]]))) {
115  for (auto const& entry : sparseMatrix.getRowGroup(state)) {
116  if(!storm::utility::isConstant(entry.getValue())) {
117  selectedStates.set(state, false);
118  break;
119  }
120  }
121  } else {
122  selectedStates.set(state, false);
123  }
124  }
125 
126  // invoke elimination and obtain resulting transition matrix
127  storm::storage::FlexibleSparseMatrix<typename SparseModelType::ValueType> flexibleMatrix(sparseMatrix);
128  storm::storage::FlexibleSparseMatrix<typename SparseModelType::ValueType> flexibleBackwardTransitions(sparseMatrix.transpose(), true);
129  storm::solver::stateelimination::NondeterministicModelStateEliminator<typename SparseModelType::ValueType> stateEliminator(flexibleMatrix, flexibleBackwardTransitions, actionRewards);
130  for(auto state : selectedStates) {
131  stateEliminator.eliminateState(state, true);
132  }
133  selectedStates.complement();
134  auto keptRows = sparseMatrix.getRowFilter(selectedStates);
135  storm::storage::SparseMatrix<typename SparseModelType::ValueType> newTransitionMatrix = flexibleMatrix.createSparseMatrix(keptRows, selectedStates);
136 
137  // obtain the reward model for the resulting system
138  std::unordered_map<std::string, typename SparseModelType::RewardModelType> rewardModels;
139  if(rewardModelName) {
140  storm::utility::vector::filterVectorInPlace(actionRewards, keptRows);
141  rewardModels.insert(std::make_pair(*rewardModelName, typename SparseModelType::RewardModelType(boost::none, std::move(actionRewards))));
142  }
143 
144  return std::make_shared<SparseModelType>(std::move(newTransitionMatrix), model.getStateLabeling().getSubLabeling(selectedStates), std::move(rewardModels));
145  }
