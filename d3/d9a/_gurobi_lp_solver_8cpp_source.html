175  int error = 0;

185  template<typename ValueType>
186  void GurobiLpSolver<ValueType>::addConstraint(std::string const& name, storm::expressions::Expression const& constraint) {
187  STORM_LOG_TRACE("Adding constraint " << (name == "" ? std::to_string(nextConstraintIndex) : name) << " to GurobiLpSolver:" << std::endl << "\t" << constraint);
188  STORM_LOG_THROW(constraint.isRelationalExpression(), storm::exceptions::InvalidArgumentException, "Illegal constraint is not a relational expression.");
189  STORM_LOG_THROW(constraint.getOperator() != storm::expressions::OperatorType::NotEqual, storm::exceptions::InvalidArgumentException, "Illegal constraint uses inequality operator.");
190 
191  storm::expressions::LinearCoefficientVisitor::VariableCoefficients leftCoefficients = storm::expressions::LinearCoefficientVisitor().getLinearCoefficients(constraint.getOperand(0));
192  storm::expressions::LinearCoefficientVisitor::VariableCoefficients rightCoefficients = storm::expressions::LinearCoefficientVisitor().getLinearCoefficients(constraint.getOperand(1));
193  leftCoefficients.separateVariablesFromConstantPart(rightCoefficients);
194 
195  // Now we need to transform the coefficients to the vector representation.
196  std::vector<int> variables;
197  std::vector<double> coefficients;
198  for (auto const& variableCoefficientPair : leftCoefficients) {
199  auto variableIndexPair = this->variableToIndexMap.find(variableCoefficientPair.first);
200  variables.push_back(variableIndexPair->second);
201  coefficients.push_back(leftCoefficients.getCoefficient(variableIndexPair->first));
202  }
203 
204  // Determine the type of the constraint and add it properly.
205  int error = 0;
206  switch (constraint.getOperator()) {
207  case storm::expressions::OperatorType::Less:
208  error = GRBaddconstr(model, variables.size(), variables.data(), coefficients.data(), GRB_LESS_EQUAL, rightCoefficients.getConstantPart() - storm::settings::getModule<storm::settings::modules::GurobiSettings>().getIntegerTolerance(), name == "" ? nullptr : name.c_str());
209  break;
210  case storm::expressions::OperatorType::LessOrEqual:
211  error = GRBaddconstr(model, variables.size(), variables.data(), coefficients.data(), GRB_LESS_EQUAL, rightCoefficients.getConstantPart(), name == "" ? nullptr : name.c_str());
212  break;
213  case storm::expressions::OperatorType::Greater:
214  error = GRBaddconstr(model, variables.size(), variables.data(), coefficients.data(), GRB_GREATER_EQUAL, rightCoefficients.getConstantPart() + storm::settings::getModule<storm::settings::modules::GurobiSettings>().getIntegerTolerance(), name == "" ? nullptr : name.c_str());
215  break;
216  case storm::expressions::OperatorType::GreaterOrEqual:
217  error = GRBaddconstr(model, variables.size(), variables.data(), coefficients.data(), GRB_GREATER_EQUAL, rightCoefficients.getConstantPart(), name == "" ? nullptr : name.c_str());
218  break;
219  case storm::expressions::OperatorType::Equal:
220  error = GRBaddconstr(model, variables.size(), variables.data(), coefficients.data(), GRB_EQUAL, rightCoefficients.getConstantPart(), name == "" ? nullptr : name.c_str());
221  break;
222  default:
223  STORM_LOG_ASSERT(false, "Illegal operator in LP solver constraint.");
224  }
225  ++nextConstraintIndex;
226  STORM_LOG_THROW(error == 0, storm::exceptions::InvalidStateException, "Could not assert constraint (" << GRBgeterrormsg(env) << ", error code " << error << ").");
227  }

416  void GurobiLpSolver<ValueType>::pop() {
417  if (incrementalData.empty()) {
418  STORM_LOG_ERROR("Tried to pop from a solver without pushing before.");
419  } else {
420  IncrementalLevel const& lvl = incrementalData.back();
421 
422  std::vector<int> indicesToBeRemoved = storm::utility::vector::buildVectorForRange(lvl.firstConstraintIndex, nextConstraintIndex);
423  GRBdelconstrs(model, indicesToBeRemoved.size(), indicesToBeRemoved.data());
424  nextConstraintIndex = lvl.firstConstraintIndex;
425  indicesToBeRemoved.clear();
426 
427  if (!lvl.variables.empty()) {
428  int firstIndex = -1;
429  bool first = true;
430  for (auto const& var : lvl.variables) {
431  if (first) {
432  auto it = variableToIndexMap.find(var);
433  firstIndex = it->second;
434  variableToIndexMap.erase(it);
435  first = false;
436  } else {
437  variableToIndexMap.erase(var);
438  }
439  }
440  std::vector<int> indicesToBeRemoved = storm::utility::vector::buildVectorForRange(firstIndex, nextVariableIndex);
441  GRBdelvars(model, indicesToBeRemoved.size(), indicesToBeRemoved.data());
442  nextVariableIndex = firstIndex;
443  }
444  incrementalData.pop_back();
445  update();
446  }
