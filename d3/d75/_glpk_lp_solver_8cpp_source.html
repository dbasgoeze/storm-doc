168  template<typename ValueType>
169  void GlpkLpSolver<ValueType>::addConstraint(std::string const& name, storm::expressions::Expression const& constraint) {
170  // Add the row that will represent this constraint.
171  int constraintIndex = glp_add_rows(this->lp, 1);
172  glp_set_row_name(this->lp, constraintIndex, name.c_str());
173 
174  STORM_LOG_THROW(constraint.getManager() == this->getManager(), storm::exceptions::InvalidArgumentException, "Constraint was not built over the proper variables.");
175  STORM_LOG_THROW(constraint.isRelationalExpression(), storm::exceptions::InvalidArgumentException, "Illegal constraint is not a relational expression.");
176  STORM_LOG_THROW(constraint.getOperator() != storm::expressions::OperatorType::NotEqual, storm::exceptions::InvalidArgumentException, "Illegal constraint uses inequality operator.");
177 
178  storm::expressions::LinearCoefficientVisitor::VariableCoefficients leftCoefficients = storm::expressions::LinearCoefficientVisitor().getLinearCoefficients(constraint.getOperand(0));
179  storm::expressions::LinearCoefficientVisitor::VariableCoefficients rightCoefficients = storm::expressions::LinearCoefficientVisitor().getLinearCoefficients(constraint.getOperand(1));
180  leftCoefficients.separateVariablesFromConstantPart(rightCoefficients);
181 
182 
183  // Determine the type of the constraint and add it properly.
184  switch (constraint.getOperator()) {
185  case storm::expressions::OperatorType::Less:
186  glp_set_row_bnds(this->lp, constraintIndex, GLP_UP, 0, rightCoefficients.getConstantPart() - storm::settings::getModule<storm::settings::modules::GlpkSettings>().getIntegerTolerance());
187  break;
188  case storm::expressions::OperatorType::LessOrEqual:
189  glp_set_row_bnds(this->lp, constraintIndex, GLP_UP, 0, rightCoefficients.getConstantPart());
190  break;
191  case storm::expressions::OperatorType::Greater:
192  glp_set_row_bnds(this->lp, constraintIndex, GLP_LO, rightCoefficients.getConstantPart() + storm::settings::getModule<storm::settings::modules::GlpkSettings>().getIntegerTolerance(), 0);
193  break;
194  case storm::expressions::OperatorType::GreaterOrEqual:
195  glp_set_row_bnds(this->lp, constraintIndex, GLP_LO, rightCoefficients.getConstantPart(), 0);
196  break;
197  case storm::expressions::OperatorType::Equal:
198  glp_set_row_bnds(this->lp, constraintIndex, GLP_FX, rightCoefficients.getConstantPart(), rightCoefficients.getConstantPart());
199  break;
200  default:
201  STORM_LOG_ASSERT(false, "Illegal operator in LP solver constraint.");
202  }
203 
204  // Now we need to transform the coefficients to the vector representation.
205  int len = std::distance(leftCoefficients.begin(), leftCoefficients.end());
206  // glpk uses 1-based indexing (wtf!?)...
207  std::vector<int> variableIndices(1, -1);
208  std::vector<double> coefficients(1, 0.0);
209  variableIndices.reserve(len + 1);
210  coefficients.reserve(len + 1);
211  for (auto const& variableCoefficientPair : leftCoefficients) {
212  auto variableIndexPair = this->variableToIndexMap.find(variableCoefficientPair.first);
213  variableIndices.push_back(variableIndexPair->second);
214  coefficients.push_back(variableCoefficientPair.second);
215  }
216 
217  glp_set_mat_row(this->lp, constraintIndex, len, variableIndices.data(), coefficients.data());
218 
219  this->currentModelHasBeenOptimized = false;
220  }

222  // Method used within the MIP solver to terminate early
223  void callback(glp_tree* t, void* info) {
224  auto& mipgap = *static_cast<std::pair<double, bool>*>(info);
225  double actualRelativeGap = glp_ios_mip_gap(t);
226  double factor = storm::utility::one<double>();
227  if (!mipgap.second) {
228  // Compute absolute gap
229  factor = storm::utility::abs(glp_mip_obj_val(glp_ios_get_prob(t))) + DBL_EPSILON;
230  assert(factor >= 0.0);
231  }
232  if (actualRelativeGap * factor <= mipgap.first) {
233  // Terminate early
234  mipgap.first = actualRelativeGap;
235  mipgap.second = true; // The gap is relative.
236  glp_ios_terminate(t);
237  }
238  }

240  template<typename ValueType>
241  void GlpkLpSolver<ValueType>::optimize() const {
242  // First, reset the flags.
243  this->isInfeasibleFlag = false;
244  this->isUnboundedFlag = false;
245 
246  // Start by setting the model sense.
247  glp_set_obj_dir(this->lp, this->getOptimizationDirection() == OptimizationDirection::Minimize ? GLP_MIN : GLP_MAX);
248 
249  int error = 0;
250  if (this->modelContainsIntegerVariables) {
251  glp_iocp* parameters = new glp_iocp();
252  glp_init_iocp(parameters);
253  parameters->tol_int = storm::settings::getModule<storm::settings::modules::GlpkSettings>().getIntegerTolerance();
254  this->isInfeasibleFlag = false;
255  if (storm::settings::getModule<storm::settings::modules::GlpkSettings>().isMILPPresolverEnabled()) {
256  parameters->presolve = GLP_ON;
257  } else {
258  // Without presolving, we solve the relaxed model first. This is required because
259  // glp_intopt requires that either presolving is enabled or an optimal initial basis is provided.
260  error = glp_simplex(this->lp, nullptr);
261  STORM_LOG_THROW(error == 0, storm::exceptions::InvalidStateException, "Unable to optimize relaxed glpk model (" << error << ").");
262  // If the relaxed model is already not feasible, we don't have to solve the actual model.
263  if (glp_get_status(this->lp) == GLP_INFEAS || glp_get_status(this->lp) == GLP_NOFEAS) {
264  this->isInfeasibleFlag = true;
265  }
266  // If the relaxed model is unbounded, there could still be no feasible integer solution.
267  // However, since we can not provide an optimal initial basis, we will need to enable presolving
268  if (glp_get_status(this->lp) == GLP_UNBND) {
269  parameters->presolve = GLP_ON;
270  } else {
271  parameters->presolve = GLP_OFF;
272  }
273  }
274  if (!this->isInfeasibleFlag) {
275  // Check whether we allow sub-optimal solutions via a non-zero MIP gap.
276  // parameters->mip_gap = this->maxMILPGap; (only works for relative values. Also, we need to obtain the actual gap anyway.
277  std::pair<double, bool> mipgap(this->maxMILPGap, this->maxMILPGapRelative);
278  if (!storm::utility::isZero(this->maxMILPGap)) {
279  parameters->cb_func = &callback;
280  parameters->cb_info = &mipgap;
281  }
282 
283  // Invoke mip solving
284  error = glp_intopt(this->lp, parameters);
285  int status = glp_mip_status(this->lp);
286  delete parameters;
287 
288  // mipgap.first has been set to the achieved mipgap (either within the callback function or because it has been set to this->maxMILPGap)
289  this->actualRelativeMILPGap = mipgap.first;
290 
291  // In case the error is caused by an infeasible problem, we do not want to view this as an error and
292  // reset the error code.
293  if (error == GLP_ENOPFS || status == GLP_NOFEAS) {
294  this->isInfeasibleFlag = true;
295  error = 0;
296  } else if (error == GLP_ENODFS) {
297  this->isUnboundedFlag = true;
298  error = 0;
299  } else if (error == GLP_ESTOP) {
300  // Early termination due to achieved MIP Gap. That's fine.
301  error = 0;
302  } else if (error == GLP_EBOUND) {
303  throw storm::exceptions::InvalidStateException() << "The bounds of some variables are illegal. Note that glpk only accepts integer bounds for integer variables.";
304  }
305  }
306  } else {
307  error = glp_simplex(this->lp, nullptr);
308  }
309 
310  STORM_LOG_THROW(error == 0, storm::exceptions::InvalidStateException, "Unable to optimize glpk model (" << error << ").");
311  this->currentModelHasBeenOptimized = true;
312  }

453  template<typename ValueType>
454  void GlpkLpSolver<ValueType>::pop() {
455  if (incrementalData.empty()) {
456  STORM_LOG_ERROR("Tried to pop from a solver without pushing before.");
457  } else {
458  IncrementalLevel const& lvl = incrementalData.back();
459  // Since glpk uses 1-based indexing, we need to prepend an additional index
460  std::vector<int> indicesToBeRemoved = storm::utility::vector::buildVectorForRange(lvl.firstConstraintIndex - 1, glp_get_num_rows(this->lp) + 1);
461  if (indicesToBeRemoved.size() > 1) {
462  glp_del_rows(this->lp, indicesToBeRemoved.size() - 1, indicesToBeRemoved.data());
463  }
464  indicesToBeRemoved.clear();
465 
466  if (!lvl.variables.empty()) {
467  int firstIndex = -1;
468  bool first = true;
469  for (auto const& var : lvl.variables) {
470  if (first) {
471  auto it = variableToIndexMap.find(var);
472  firstIndex = it->second;
473  variableToIndexMap.erase(it);
474  first = false;
475  } else {
476  variableToIndexMap.erase(var);
477  }
478  }
479  // Since glpk uses 1-based indexing, we need to prepend an additional index
480  std::vector<int> indicesToBeRemoved = storm::utility::vector::buildVectorForRange(firstIndex - 1, glp_get_num_cols(this->lp) + 1);
481  glp_del_cols(this->lp, indicesToBeRemoved.size() - 1, indicesToBeRemoved.data());
482  }
483  incrementalData.pop_back();
484  update();
485  // Check whether we need to adapt the current basis (i.e. the number of basic variables does not equal the number of constraints)
486  int n = glp_get_num_rows(lp);
487  int m = glp_get_num_cols(lp);
488  int nb(0), mb(0);
489  for (int i = 1; i <= n; ++i) {
490  if (glp_get_row_stat(lp, i) == GLP_BS) {
491  ++nb;
492  }
493  }
494  for (int j = 1; j <= m; ++j) {
495  if (glp_get_col_stat(lp, j) == GLP_BS) {
496  ++mb;
497  }
498  }
499  if (n != (nb + mb)) {
500  glp_std_basis(this->lp);
501  }
502  }
503  }
