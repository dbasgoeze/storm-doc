299  std::unique_ptr<CheckResult> ExplicitQuantitativeCheckResult<ValueType>::compareAgainstBound(storm::logic::ComparisonType comparisonType, ValueType const& bound) const {
300  if (this->isResultForAllStates()) {
301  vector_type const& valuesAsVector = boost::get<vector_type>(values);
302  storm::storage::BitVector result(valuesAsVector.size());
303  switch (comparisonType) {
304  case logic::ComparisonType::Less:
305  for (uint_fast64_t index = 0; index < valuesAsVector.size(); ++index) {
306  if (valuesAsVector[index] < bound) {
307  result.set(index);
308  }
309  }
310  break;
311  case logic::ComparisonType::LessEqual:
312  for (uint_fast64_t index = 0; index < valuesAsVector.size(); ++index) {
313  if (valuesAsVector[index] <= bound) {
314  result.set(index);
315  }
316  }
317  break;
318  case logic::ComparisonType::Greater:
319  for (uint_fast64_t index = 0; index < valuesAsVector.size(); ++index) {
320  if (valuesAsVector[index] > bound) {
321  result.set(index);
322  }
323  }
324  break;
325  case logic::ComparisonType::GreaterEqual:
326  for (uint_fast64_t index = 0; index < valuesAsVector.size(); ++index) {
327  if (valuesAsVector[index] >= bound) {
328  result.set(index);
329  }
330  }
331  break;
332  }
333  return std::unique_ptr<CheckResult>(new ExplicitQualitativeCheckResult(std::move(result)));
334  } else {
335  map_type const& valuesAsMap = boost::get<map_type>(values);
336  std::map<storm::storage::sparse::state_type, bool> result;
337  switch (comparisonType) {
338  case logic::ComparisonType::Less:
339  for (auto const& element : valuesAsMap) {
340  result[element.first] = element.second < bound;
341  }
342  break;
343  case logic::ComparisonType::LessEqual:
344  for (auto const& element : valuesAsMap) {
345  result[element.first] = element.second <= bound;
346  }
347  break;
348  case logic::ComparisonType::Greater:
349  for (auto const& element : valuesAsMap) {
350  result[element.first] = element.second > bound;
351  }
352  break;
353  case logic::ComparisonType::GreaterEqual:
354  for (auto const& element : valuesAsMap) {
355  result[element.first] = element.second >= bound;
356  }
357  break;
358  }
359  return std::unique_ptr<CheckResult>(new ExplicitQualitativeCheckResult(std::move(result)));
