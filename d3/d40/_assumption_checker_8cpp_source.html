131  AssumptionStatus AssumptionChecker<ValueType>::validateAssumption(std::shared_ptr<expressions::BinaryRelationExpression> assumption, Order* order) {
132  // First check if based on sample points the assumption can be discharged
133  auto result = checkOnSamples(assumption);
134  assert (result != AssumptionStatus::VALID);
135 
136  if (result == AssumptionStatus::UNKNOWN) {
137  // If result from sample checking was unknown, the assumption might hold, so we continue,
138  // otherwise we return INVALID
139  std::set<expressions::Variable> vars = std::set<expressions::Variable>({});
140  assumption->gatherVariables(vars);
141 
142  STORM_LOG_THROW(assumption->getRelationType() ==
143  expressions::BinaryRelationExpression::RelationType::Greater ||
144  assumption->getRelationType() ==
145  expressions::BinaryRelationExpression::RelationType::Equal,
146  exceptions::NotSupportedException,
147  "Only Greater Or Equal assumptions supported");
148 
149  // Row with successors of the first state
150  auto row1 = matrix.getRow(
151  std::stoi(assumption->getFirstOperand()->asVariableExpression().getVariableName()));
152  // Row with successors of the second state
153  auto row2 = matrix.getRow(
154  std::stoi(assumption->getSecondOperand()->asVariableExpression().getVariableName()));
155 
156  if (row1.getNumberOfEntries() == 2 && row2.getNumberOfEntries() == 2) {
157  // If the states have the same successors for which we know the position in the order
158  // We can check with a function if the assumption holds
159 
160  auto state1succ1 = row1.begin();
161  auto state1succ2 = (++row1.begin());
162  auto state2succ1 = row2.begin();
163  auto state2succ2 = (++row2.begin());
164 
165  if (state1succ1->getColumn() == state2succ2->getColumn()
166  && state2succ1->getColumn() == state1succ2->getColumn()) {
167  std::swap(state1succ1, state1succ2);
168  }
169 
170  if (state1succ1->getColumn() == state2succ1->getColumn() && state1succ2->getColumn() == state2succ2->getColumn()) {
171  if (assumption->getRelationType() == expressions::BinaryRelationExpression::RelationType::Greater
172  && order->compare(state1succ1->getColumn(), state1succ2->getColumn()) != Order::NodeComparison::UNKNOWN) {
173  // The assumption should be the greater assumption
174  // If the result is unknown, we cannot compare, also SMTSolver will not help
175  result = validateAssumptionSMTSolver(assumption, order);
176 
177 // result = validateAssumptionFunction(order, state1succ1, state1succ2, state2succ1,
178 // state2succ2);
179  } else if (assumption->getRelationType() == expressions::BinaryRelationExpression::RelationType::Equal) {
180  // The assumption is equal, the successors are the same,
181  // so if the probability of reaching the successors is the same, we have a valid assumption
182  if (state1succ1->getValue() == state2succ1->getValue()) {
183  result = AssumptionStatus::VALID;
184  }
185  } else {
186  result = AssumptionStatus::UNKNOWN;
187  }
188  } else {
189  result = validateAssumptionSMTSolver(assumption, order);
190  }
191  } else {
192  result = validateAssumptionSMTSolver(assumption, order);
193  }
194  }
195  return result;

198  template <typename ValueType>
199  AssumptionStatus AssumptionChecker<ValueType>::validateAssumptionFunction(Order* order,
200  typename storage::SparseMatrix<ValueType>::iterator state1succ1,
201  typename storage::SparseMatrix<ValueType>::iterator state1succ2,
202  typename storage::SparseMatrix<ValueType>::iterator state2succ1,
203  typename storage::SparseMatrix<ValueType>::iterator state2succ2) {
204  assert((state1succ1->getColumn() == state2succ1->getColumn()
205  && state1succ2->getColumn() == state2succ2->getColumn())
206  || (state1succ1->getColumn() == state2succ2->getColumn()
207  && state1succ2->getColumn() == state2succ1->getColumn()));
208 
209  AssumptionStatus result;
210 
211  // Calculate the difference in probability for the "highest" successor state
212  ValueType prob;
213  auto comp = order->compare(state1succ1->getColumn(), state1succ2->getColumn());
214  assert (comp == Order::NodeComparison::ABOVE || comp == Order::NodeComparison::BELOW);
215  if (comp == Order::NodeComparison::ABOVE) {
216  prob = state1succ1->getValue() - state2succ1->getValue();
217  } else if (comp == Order::NodeComparison::BELOW) {
218  prob = state1succ2->getValue() - state2succ2->getValue();
219  }
220 
221  auto vars = prob.gatherVariables();
222 
223  // If the result in monotone increasing (decreasing), then choose 0 (1) for the substitutions
224  // This will give the smallest result
225  std::map<typename utility::parametric::VariableType<ValueType>::type, typename utility::parametric::CoefficientType<ValueType>::type> substitutions;
226  for (auto var : vars) {
227  auto monotonicity = MonotonicityChecker<ValueType>::checkDerivative(prob.derivative(var), region);
228  if (monotonicity.first) {
229  // monotone increasing
230  substitutions[var] = 0;
231  } else if (monotonicity.second) {
232  // monotone increasing
233  substitutions[var] = 1;
234  } else {
235  result = AssumptionStatus::UNKNOWN;
236  }
237  }
238 
239  if (prob.evaluate(substitutions) >= 0) {
240  result = AssumptionStatus::VALID;
241  }
242  return result;
243  }

247  AssumptionStatus AssumptionChecker<ValueType>::validateAssumptionSMTSolver(std::shared_ptr<expressions::BinaryRelationExpression> assumption, Order* order) {
248  std::shared_ptr<utility::solver::SmtSolverFactory> smtSolverFactory = std::make_shared<utility::solver::MathsatSmtSolverFactory>();
249  std::shared_ptr<expressions::ExpressionManager> manager(new expressions::ExpressionManager());
250 
251  AssumptionStatus result;
252  auto var1 = assumption->getFirstOperand()->asVariableExpression().getVariableName();
253  auto var2 = assumption->getSecondOperand()->asVariableExpression().getVariableName();
254  auto row1 = matrix.getRow(std::stoi(var1));
255  auto row2 = matrix.getRow(std::stoi(var2));
256 
257  bool orderKnown = true;
258  // Check if the order between the different successors is known
259  // Also start creating expression for order of states
260  expressions::Expression exprOrderSucc = manager->boolean(true);
261  std::set<expressions::Variable> stateVariables;
262  for (auto itr1 = row1.begin(); orderKnown && itr1 != row1.end(); ++itr1) {
263  auto varname1 = "s" + std::to_string(itr1->getColumn());
264  if (!manager->hasVariable(varname1)) {
265  stateVariables.insert(manager->declareRationalVariable(varname1));
266  }
267  for (auto itr2 = row2.begin(); orderKnown && itr2 != row2.end(); ++itr2) {
268  if (itr1->getColumn() != itr2->getColumn()) {
269  auto varname2 = "s" + std::to_string(itr2->getColumn());
270  if (!manager->hasVariable(varname2)) {
271  stateVariables.insert(manager->declareRationalVariable(varname2));
272  }
273  auto comp = order->compare(itr1->getColumn(), itr2->getColumn());
274  if (comp == Order::NodeComparison::ABOVE) {
275  exprOrderSucc = exprOrderSucc && !(manager->getVariable(varname1) <=
276  manager->getVariable(varname2));
277  } else if (comp == Order::NodeComparison::BELOW) {
278  exprOrderSucc = exprOrderSucc && !(manager->getVariable(varname1) >=
279  manager->getVariable(varname2));
280  } else if (comp == Order::NodeComparison::SAME) {
281  exprOrderSucc = exprOrderSucc &&
282  (manager->getVariable(varname1) = manager->getVariable(varname2));
283  } else {
284  orderKnown = false;
285  }
286  }
287  }
288  }
289 
290  if (orderKnown) {
291  solver::Z3SmtSolver s(*manager);
292  auto valueTypeToExpression = expressions::RationalFunctionToExpression<ValueType>(manager);
293  expressions::Expression expr1 = manager->rational(0);
294  for (auto itr1 = row1.begin(); itr1 != row1.end(); ++itr1) {
295  expr1 = expr1 + (valueTypeToExpression.toExpression(itr1->getValue()) * manager->getVariable("s" + std::to_string(itr1->getColumn())));
296  }
297 
298  expressions::Expression expr2 = manager->rational(0);
299  for (auto itr2 = row2.begin(); itr2 != row2.end(); ++itr2) {
300  expr2 = expr2 + (valueTypeToExpression.toExpression(itr2->getValue()) * manager->getVariable("s" + std::to_string(itr2->getColumn())));
301  }
302 
303  // Create expression for the assumption based on the relation to successors
304  // It is the negation of actual assumption
305  expressions::Expression exprToCheck ;
306  if (assumption->getRelationType() ==
307  expressions::BinaryRelationExpression::RelationType::Greater) {
308  exprToCheck = expr1 <= expr2;
309  } else {
310  assert (assumption->getRelationType() ==
311  expressions::BinaryRelationExpression::RelationType::Equal);
312  exprToCheck = expr1 != expr2 ;
313  }
314 
315  auto variables = manager->getVariables();
316  // Bounds for the state probabilities and parameters
317  expressions::Expression exprBounds = manager->boolean(true);
318  for (auto var : variables) {
319  if (find(stateVariables.begin(), stateVariables.end(), var) != stateVariables.end()) {
320  // the var is a state
321  exprBounds = exprBounds && manager->rational(0) <= var && var <= manager->rational(1);
322  } else {
323  // the var is a parameter
324  auto lb = storm::utility::convertNumber<storm::RationalNumber>(region.getLowerBoundary(var.getName()));
325  auto ub = storm::utility::convertNumber<storm::RationalNumber>(region.getUpperBoundary(var.getName()));
326  exprBounds = exprBounds && manager->rational(lb) < var && var < manager->rational(ub);
327  }
328  }
329 
330  s.add(exprOrderSucc);
331  s.add(exprBounds);
332  // assert that sorting of successors in the order and the bounds on the expression are at least satisfiable
333  assert (s.check() == solver::SmtSolver::CheckResult::Sat);
334  s.add(exprToCheck);
335  auto smtRes = s.check();
336  if (smtRes == solver::SmtSolver::CheckResult::Unsat) {
337  // If there is no thing satisfying the negation we are safe.
338  result = AssumptionStatus::VALID;
339  } else if (smtRes == solver::SmtSolver::CheckResult::Sat) {
340  assert (smtRes == solver::SmtSolver::CheckResult::Sat);
341  result = AssumptionStatus::INVALID;
342  } else {
343  result = AssumptionStatus::UNKNOWN;
344  }
345  } else {
346  result = AssumptionStatus::UNKNOWN;
347  }
348 
349  return result;
350  }
