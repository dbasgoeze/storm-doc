60  storm::expressions::Expression Z3ExpressionAdapter::translateExpression(z3::expr const& expr) {
61  if (expr.is_app()) {
62  switch (expr.decl().decl_kind()) {
63  case Z3_OP_TRUE:
64  return manager.boolean(true);
65  case Z3_OP_FALSE:
66  return manager.boolean(false);
67  case Z3_OP_EQ:
68  return this->translateExpression(expr.arg(0)) == this->translateExpression(expr.arg(1));
69  case Z3_OP_DISTINCT: {
70  unsigned args = expr.num_args();
71  STORM_LOG_THROW(args != 0, storm::exceptions::ExpressionEvaluationException, "Failed to convert Z3 expression. DISTINCT (mutual != ) operator with 0-arity is assumed to be an error.");
72  if (args == 1) {
73  return manager.boolean(true);
74  } else {
75  storm::expressions::Expression retVal = this->translateExpression(expr.arg(0)) != this->translateExpression(expr.arg(1));
76  for (unsigned arg2 = 2; arg2 < args; ++arg2) {
77  retVal = retVal && (this->translateExpression(expr.arg(0)) != this->translateExpression(expr.arg(arg2)));
78  }
79  for (unsigned arg1 = 1; arg1 < args; ++arg1) {
80  for (unsigned arg2 = arg1 + 1; arg2 < args; ++arg2) {
81  retVal = retVal && (this->translateExpression(expr.arg(arg1)) != this->translateExpression(expr.arg(arg2)));
82  }
83  }
84  return retVal;
85  }
86  }
87  case Z3_OP_ITE:
88  return storm::expressions::ite(this->translateExpression(expr.arg(0)), this->translateExpression(expr.arg(1)), this->translateExpression(expr.arg(2)));
89  case Z3_OP_AND: {
90  unsigned args = expr.num_args();
91  STORM_LOG_THROW(args != 0, storm::exceptions::ExpressionEvaluationException, "Failed to convert Z3 expression. 0-ary AND is assumed to be an error.");
92  if (args == 1) {
93  return this->translateExpression(expr.arg(0));
94  } else {
95  storm::expressions::Expression retVal = this->translateExpression(expr.arg(0));
96  for (unsigned i = 1; i < args; i++) {
97  retVal = retVal && this->translateExpression(expr.arg(i));
98  }
99  return retVal;
100  }
101  }
102  case Z3_OP_OR: {
103  unsigned args = expr.num_args();
104  STORM_LOG_THROW(args != 0, storm::exceptions::ExpressionEvaluationException, "Failed to convert Z3 expression. 0-ary OR is assumed to be an error.");
105  if (args == 1) {
106  return this->translateExpression(expr.arg(0));
107  } else {
108  storm::expressions::Expression retVal = this->translateExpression(expr.arg(0));
109  for (unsigned i = 1; i < args; i++) {
110  retVal = retVal || this->translateExpression(expr.arg(i));
111  }
112  return retVal;
113  }
114  }
115  case Z3_OP_IFF:
116  return storm::expressions::iff(this->translateExpression(expr.arg(0)), this->translateExpression(expr.arg(1)));
117  case Z3_OP_XOR:
118  return this->translateExpression(expr.arg(0)) ^ this->translateExpression(expr.arg(1));
119  case Z3_OP_NOT:
120  return !this->translateExpression(expr.arg(0));
121  case Z3_OP_IMPLIES:
122  return storm::expressions::implies(this->translateExpression(expr.arg(0)), this->translateExpression(expr.arg(1)));
123  case Z3_OP_LE:
124  return this->translateExpression(expr.arg(0)) <= this->translateExpression(expr.arg(1));
125  case Z3_OP_GE:
126  return this->translateExpression(expr.arg(0)) >= this->translateExpression(expr.arg(1));
127  case Z3_OP_LT:
128  return this->translateExpression(expr.arg(0)) < this->translateExpression(expr.arg(1));
129  case Z3_OP_GT:
130  return this->translateExpression(expr.arg(0)) > this->translateExpression(expr.arg(1));
131  case Z3_OP_ADD:
132  return this->translateExpression(expr.arg(0)) + this->translateExpression(expr.arg(1));
133  case Z3_OP_SUB:
134  return this->translateExpression(expr.arg(0)) - this->translateExpression(expr.arg(1));
135  case Z3_OP_UMINUS:
136  return -this->translateExpression(expr.arg(0));
137  case Z3_OP_MUL:
138  return this->translateExpression(expr.arg(0)) * this->translateExpression(expr.arg(1));
139  case Z3_OP_DIV:
140  return this->translateExpression(expr.arg(0)) / this->translateExpression(expr.arg(1));
141  case Z3_OP_IDIV:
142  return this->translateExpression(expr.arg(0)) / this->translateExpression(expr.arg(1));
143  case Z3_OP_ANUM:
144  // Arithmetic numeral.
145  if (expr.is_int() && expr.is_const()) {
146  Z3_SIGNED_INTEGER value;
147  if (Z3_get_numeral_int64(expr.ctx(), expr, &value)) {
148  return manager.integer(value);
149  } else {
150  STORM_LOG_THROW(false, storm::exceptions::ExpressionEvaluationException, "Failed to convert Z3 expression. Expression is constant integer and value does not fit into 64-bit integer.");
151  }
152  } else {
153  STORM_LOG_ASSERT(expr.is_real() && expr.is_const(), "Cannot handle numerical expression");
154  Z3_SIGNED_INTEGER num;
155  Z3_SIGNED_INTEGER den;
156  if (Z3_get_numeral_rational_int64(expr.ctx(), expr, &num, &den)) {
157  return manager.rational(storm::utility::convertNumber<storm::RationalNumber>(static_cast<int_fast64_t>(num)) / storm::utility::convertNumber<storm::RationalNumber>(static_cast<int_fast64_t>(den)));
158  } else {
159  return manager.rational(storm::utility::convertNumber<storm::RationalNumber>(std::string(Z3_get_numeral_string(expr.ctx(), expr))));
160  }
161  }
162  case Z3_OP_UNINTERPRETED:
163  // Currently, we only support uninterpreted constant functions.
164  STORM_LOG_THROW(expr.is_const(), storm::exceptions::ExpressionEvaluationException, "Failed to convert Z3 expression. Encountered non-constant uninterpreted function.");
165  return manager.getVariable(expr.decl().name().str()).getExpression();
166  default:
167  STORM_LOG_THROW(false, storm::exceptions::ExpressionEvaluationException, "Failed to convert Z3 expression. Encountered unhandled Z3_decl_kind " << expr.decl().decl_kind() <<".");
168  break;
169  }
170  } else {
171  STORM_LOG_THROW(false, storm::exceptions::ExpressionEvaluationException, "Failed to convert Z3 expression. Encountered unknown expression type.");
172  }

175  boost::any Z3ExpressionAdapter::visit(storm::expressions::BinaryBooleanFunctionExpression const& expression, boost::any const& data) {
176  auto cacheIt = expressionCache.find(&expression);
177  if (cacheIt != expressionCache.end()) {
178  return cacheIt->second;
179  }
180 
181  z3::expr leftResult = boost::any_cast<z3::expr>(expression.getFirstOperand()->accept(*this, data));
182  z3::expr rightResult = boost::any_cast<z3::expr>(expression.getSecondOperand()->accept(*this, data));
183 
184  z3::expr result(context);
185  switch(expression.getOperatorType()) {
186  case storm::expressions::BinaryBooleanFunctionExpression::OperatorType::And:
187  result = leftResult && rightResult;
188  break;
189  case storm::expressions::BinaryBooleanFunctionExpression::OperatorType::Or:
190  result = leftResult || rightResult;
191  break;
192  case storm::expressions::BinaryBooleanFunctionExpression::OperatorType::Xor:
193  result = z3::expr(context, Z3_mk_xor(context, leftResult, rightResult));
194  break;
195  case storm::expressions::BinaryBooleanFunctionExpression::OperatorType::Implies:
196  result = z3::expr(context, Z3_mk_implies(context, leftResult, rightResult));
197  break;
198  case storm::expressions::BinaryBooleanFunctionExpression::OperatorType::Iff:
199  result = z3::expr(context, Z3_mk_iff(context, leftResult, rightResult));
200  break;
201  default:
202  STORM_LOG_THROW(false, storm::exceptions::ExpressionEvaluationException, "Cannot evaluate expression: unknown boolean binary operator '" << static_cast<int>(expression.getOperatorType()) << "' in expression " << expression << ".");
203  }
204 
205  expressionCache.emplace(&expression, result);
206  return result;
