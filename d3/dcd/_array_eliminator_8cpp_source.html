513  virtual void traverse(OrderedAssignments& orderedAssignments, boost::any const& data) override {
514  auto const& replacements = boost::any_cast<ResultType*>(data)->replacements;
515 
516  // Replace array occurrences in LValues and assigned expressions.
517  std::vector<Assignment> newAssignments;
518  if (!orderedAssignments.empty()) {
519 
520  int64_t level = orderedAssignments.getLowestLevel();
521  std::unordered_map<storm::expressions::Variable, std::vector<Assignment const*>> collectedArrayAccessAssignments;
522  for (Assignment const& assignment : orderedAssignments) {
523  if (assignment.getLevel() != level) {
524  STORM_LOG_ASSERT(assignment.getLevel() > level, "Ordered Assignment does not have the expected order.");
525  for (auto const& arrayAssignments : collectedArrayAccessAssignments) {
526  insertLValueArrayAccessReplacements(arrayAssignments.second, replacements.at(arrayAssignments.first), level, newAssignments);
527  }
528  collectedArrayAccessAssignments.clear();
529  level = assignment.getLevel();
530  }
531  if (assignment.getLValue().isArrayAccess()) {
532  if (!keepNonTrivialArrayAccess || !assignment.getLValue().getArrayIndex().containsVariables()) {
533  auto insertionRes = collectedArrayAccessAssignments.emplace(assignment.getLValue().getArray().getExpressionVariable(), std::vector<Assignment const*>({&assignment}));
534  if (!insertionRes.second) {
535  insertionRes.first->second.push_back(&assignment);
536  }
537  } else {
538  // Keeping array access LValue
539  LValue newLValue(LValue(assignment.getLValue().getArray()), arrayExprEliminator->eliminate(assignment.getLValue().getArrayIndex()));
540  newAssignments.emplace_back(newLValue, arrayExprEliminator->eliminate(assignment.getAssignedExpression()), assignment.getLevel());
541  }
542  } else if (assignment.getLValue().isVariable() && assignment.getVariable().isArrayVariable()) {
543  STORM_LOG_ASSERT(assignment.getAssignedExpression().getType().isArrayType(), "Assigning a non-array expression to an array variable...");
544  std::vector<storm::jani::Variable const*> const& arrayVariableReplacements = replacements.at(assignment.getExpressionVariable());
545  // Get the maximum size of the array expression on the rhs
546  uint64_t rhsSize = MaxArraySizeExpressionVisitor().getMaxSize(assignment.getAssignedExpression(), arraySizes);
547  STORM_LOG_ASSERT(arrayVariableReplacements.size() >= rhsSize, "Array size too small.");
548  for (uint64_t index = 0; index < arrayVariableReplacements.size(); ++index) {
549  auto const& replacement = *arrayVariableReplacements[index];
550  storm::expressions::Expression newRhs;
551  if (index < rhsSize) {
552  newRhs = std::make_shared<storm::expressions::ArrayAccessExpression>(expressionManager, assignment.getAssignedExpression().getType().getElementType(), assignment.getAssignedExpression().getBaseExpressionPointer(), expressionManager.integer(index).getBaseExpressionPointer())->toExpression();
553  } else {
554  newRhs = getOutOfBoundsValue(replacement);
555  }
556  newRhs = arrayExprEliminator->eliminate(newRhs);
557  newAssignments.emplace_back(LValue(replacement), newRhs, level);
558  }
559  } else {
560  newAssignments.emplace_back(assignment.getLValue(), arrayExprEliminator->eliminate(assignment.getAssignedExpression()), assignment.getLevel());
561  }
562  }
563  for (auto const& arrayAssignments : collectedArrayAccessAssignments) {
564  insertLValueArrayAccessReplacements(arrayAssignments.second, replacements.at(arrayAssignments.first), level, newAssignments);
565  }
566  collectedArrayAccessAssignments.clear();
567  orderedAssignments.clear();
568  for (auto const& assignment : newAssignments) {
569  orderedAssignments.add(assignment);
570  }
571  }
