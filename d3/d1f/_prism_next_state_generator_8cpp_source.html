150  template<typename ValueType, typename StateType>
151  std::vector<StateType> PrismNextStateGenerator<ValueType, StateType>::getInitialStates(StateToIdCallback const& stateToIdCallback) {
152  std::vector<StateType> initialStateIndices;
153 
154  // If all states are initial, we can simplify the enumeration substantially.
155  if (program.hasInitialConstruct() && program.getInitialConstruct().getInitialStatesExpression().isTrue()) {
156  CompressedState initialState(this->variableInformation.getTotalBitOffset(true));
157 
158  std::vector<int_fast64_t> currentIntegerValues;
159  currentIntegerValues.reserve(this->variableInformation.integerVariables.size());
160  for (auto const& variable : this->variableInformation.integerVariables) {
161  STORM_LOG_THROW(variable.lowerBound <= variable.upperBound, storm::exceptions::InvalidArgumentException, "Expecting variable with non-empty set of possible values.");
162  currentIntegerValues.emplace_back(0);
163  initialState.setFromInt(variable.bitOffset, variable.bitWidth, 0);
164  }
165 
166  initialStateIndices.emplace_back(stateToIdCallback(initialState));
167 
168  bool done = false;
169  while (!done) {
170  bool changedBooleanVariable = false;
171  for (auto const& booleanVariable : this->variableInformation.booleanVariables) {
172  if (initialState.get(booleanVariable.bitOffset)) {
173  initialState.set(booleanVariable.bitOffset);
174  changedBooleanVariable = true;
175  break;
176  } else {
177  initialState.set(booleanVariable.bitOffset, false);
178  }
179  }
180 
181  bool changedIntegerVariable = false;
182  if (changedBooleanVariable) {
183  initialStateIndices.emplace_back(stateToIdCallback(initialState));
184  } else {
185  for (uint64_t integerVariableIndex = 0; integerVariableIndex < this->variableInformation.integerVariables.size(); ++integerVariableIndex) {
186  auto const& integerVariable = this->variableInformation.integerVariables[integerVariableIndex];
187  if (currentIntegerValues[integerVariableIndex] < integerVariable.upperBound - integerVariable.lowerBound) {
188  ++currentIntegerValues[integerVariableIndex];
189  changedIntegerVariable = true;
190  } else {
191  currentIntegerValues[integerVariableIndex] = integerVariable.lowerBound;
192  }
193  initialState.setFromInt(integerVariable.bitOffset, integerVariable.bitWidth, currentIntegerValues[integerVariableIndex]);
194 
195  if (changedIntegerVariable) {
196  break;
197  }
198  }
199  }
200 
201  if (changedIntegerVariable) {
202  initialStateIndices.emplace_back(stateToIdCallback(initialState));
203  }
204 
205  done = !changedBooleanVariable && !changedIntegerVariable;
206  }
207 
208  STORM_LOG_DEBUG("Enumerated " << initialStateIndices.size() << " initial states using brute force enumeration.");
209  } else {
210  // Prepare an SMT solver to enumerate all initial states.
211  storm::utility::solver::SmtSolverFactory factory;
212  std::unique_ptr<storm::solver::SmtSolver> solver = factory.create(program.getManager());
213 
214  std::vector<storm::expressions::Expression> rangeExpressions = program.getAllRangeExpressions();
215  for (auto const& expression : rangeExpressions) {
216  solver->add(expression);
217  }
218  solver->add(program.getInitialStatesExpression());
219 
220  // Proceed ss long as the solver can still enumerate initial states.
221  while (solver->check() == storm::solver::SmtSolver::CheckResult::Sat) {
222  // Create fresh state.
223  CompressedState initialState(this->variableInformation.getTotalBitOffset(true));
224 
225  // Read variable assignment from the solution of the solver. Also, create an expression we can use to
226  // prevent the variable assignment from being enumerated again.
227  storm::expressions::Expression blockingExpression;
228  std::shared_ptr<storm::solver::SmtSolver::ModelReference> model = solver->getModel();
229  for (auto const& booleanVariable : this->variableInformation.booleanVariables) {
230  bool variableValue = model->getBooleanValue(booleanVariable.variable);
231  storm::expressions::Expression localBlockingExpression = variableValue ? !booleanVariable.variable : booleanVariable.variable;
232  blockingExpression = blockingExpression.isInitialized() ? blockingExpression || localBlockingExpression : localBlockingExpression;
233  initialState.set(booleanVariable.bitOffset, variableValue);
234  }
235  for (auto const& integerVariable : this->variableInformation.integerVariables) {
236  int_fast64_t variableValue = model->getIntegerValue(integerVariable.variable);
237  storm::expressions::Expression localBlockingExpression = integerVariable.variable != model->getManager().integer(variableValue);
238  blockingExpression = blockingExpression.isInitialized() ? blockingExpression || localBlockingExpression : localBlockingExpression;
239  initialState.setFromInt(integerVariable.bitOffset, integerVariable.bitWidth, static_cast<uint_fast64_t>(variableValue - integerVariable.lowerBound));
240  }
241 
242  // Register initial state and return it.
243  StateType id = stateToIdCallback(initialState);
244  initialStateIndices.push_back(id);
245 
246  // Block the current initial state to search for the next one.
247  if (!blockingExpression.isInitialized()) {
248  break;
249  }
250  solver->add(blockingExpression);
251  }
252 
253  STORM_LOG_DEBUG("Enumerated " << initialStateIndices.size() << " initial states using SMT solving.");
254  }
255 
256  return initialStateIndices;

259  template<typename ValueType, typename StateType>
260  StateBehavior<ValueType, StateType> PrismNextStateGenerator<ValueType, StateType>::expand(StateToIdCallback const& stateToIdCallback) {
261  // Prepare the result, in case we return early.
262  StateBehavior<ValueType, StateType> result;
263 
264  // First, construct the state rewards, as we may return early if there are no choices later and we already
265  // need the state rewards then.
266  for (auto const& rewardModel : rewardModels) {
267  ValueType stateRewardValue = storm::utility::zero<ValueType>();
268  if (rewardModel.get().hasStateRewards()) {
269  for (auto const& stateReward : rewardModel.get().getStateRewards()) {
270  if (this->evaluator->asBool(stateReward.getStatePredicateExpression())) {
271  stateRewardValue += ValueType(this->evaluator->asRational(stateReward.getRewardValueExpression()));
272  }
273  }
274  }
275  result.addStateReward(stateRewardValue);
276  }
277 
278  // If a terminal expression was set and we must not expand this state, return now.
279  if (!this->terminalStates.empty()) {
280  for (auto const& expressionBool : this->terminalStates) {
281  if (this->evaluator->asBool(expressionBool.first) == expressionBool.second) {
282  return result;
283  }
284  }
285  }
286 
287  // Get all choices for the state.
288  result.setExpanded();
289 
290  std::vector<Choice<ValueType>> allChoices;
291  if (this->getOptions().isApplyMaximalProgressAssumptionSet()) {
292  // First explore only edges without a rate
293  allChoices = getUnlabeledChoices(*this->state, stateToIdCallback, CommandFilter::Probabilistic);
294  addLabeledChoices(allChoices, *this->state, stateToIdCallback, CommandFilter::Probabilistic);
295  if (allChoices.empty()) {
296  // Expand the Markovian edges if there are no probabilistic ones.
297  allChoices = getUnlabeledChoices(*this->state, stateToIdCallback, CommandFilter::Markovian);
298  addLabeledChoices(allChoices, *this->state, stateToIdCallback, CommandFilter::Markovian);
299  }
300  } else {
301  allChoices = getUnlabeledChoices(*this->state, stateToIdCallback);
302  addLabeledChoices(allChoices, *this->state, stateToIdCallback);
303  }
304 
305  std::size_t totalNumberOfChoices = allChoices.size();
306 
307  // If there is not a single choice, we return immediately, because the state has no behavior (other than
308  // the state reward).
309  if (totalNumberOfChoices == 0) {
310  return result;
311  }
312 
313  // If the model is a deterministic model, we need to fuse the choices into one.
314  if (this->isDeterministicModel() && totalNumberOfChoices > 1) {
315  Choice<ValueType> globalChoice;
316 
317  if (this->options.isAddOverlappingGuardLabelSet()) {
318  this->overlappingGuardStates->push_back(stateToIdCallback(*this->state));
319  }
320 
321  // For CTMCs, we need to keep track of the total exit rate to scale the action rewards later. For DTMCs
322  // this is equal to the number of choices, which is why we initialize it like this here.
323  ValueType totalExitRate = this->isDiscreteTimeModel() ? static_cast<ValueType>(totalNumberOfChoices) : storm::utility::zero<ValueType>();
324 
325  // Iterate over all choices and combine the probabilities/rates into one choice.
326  for (auto const& choice : allChoices) {
327  for (auto const& stateProbabilityPair : choice) {
328  if (this->isDiscreteTimeModel()) {
329  globalChoice.addProbability(stateProbabilityPair.first, stateProbabilityPair.second / totalNumberOfChoices);
330  } else {
331  globalChoice.addProbability(stateProbabilityPair.first, stateProbabilityPair.second);
332  }
333  }
334 
335  if (hasStateActionRewards && !this->isDiscreteTimeModel()) {
336  totalExitRate += choice.getTotalMass();
337  }
338 
339  if (this->options.isBuildChoiceLabelsSet() && choice.hasLabels()) {
340  globalChoice.addLabels(choice.getLabels());
341  }
342 
343  if (this->options.isBuildChoiceOriginsSet() && choice.hasOriginData()) {
344  globalChoice.addOriginData(choice.getOriginData());
345  }
346  }
347 
348  // Now construct the state-action reward for all selected reward models.
349  for (auto const& rewardModel : rewardModels) {
350  ValueType stateActionRewardValue = storm::utility::zero<ValueType>();
351  if (rewardModel.get().hasStateActionRewards()) {
352  for (auto const& stateActionReward : rewardModel.get().getStateActionRewards()) {
353  for (auto const& choice : allChoices) {
354  if (stateActionReward.getActionIndex() == choice.getActionIndex() && this->evaluator->asBool(stateActionReward.getStatePredicateExpression())) {
355  stateActionRewardValue += ValueType(this->evaluator->asRational(stateActionReward.getRewardValueExpression())) * choice.getTotalMass();
356  }
357  }
358 
359  }
360  }
361  if (hasStateActionRewards) {
362  globalChoice.addReward(stateActionRewardValue / totalExitRate);
363  }
364  }
365 
366  // Move the newly fused choice in place.
367  allChoices.clear();
368  allChoices.push_back(std::move(globalChoice));
369  }
370 
371  // Move all remaining choices in place.
372  for (auto& choice : allChoices) {
373  result.addChoice(std::move(choice));
374  }
375 
376  this->postprocess(result);
377 
378  return result;
379  }

381  template<typename ValueType, typename StateType>
382  CompressedState PrismNextStateGenerator<ValueType, StateType>::applyUpdate(CompressedState const& state, storm::prism::Update const& update) {
383  CompressedState newState(state);
384 
385  // NOTE: the following process assumes that the assignments of the update are ordered in such a way that the
386  // assignments to boolean variables precede the assignments to all integer variables and that within the
387  // types, the assignments to variables are ordered (in ascending order) by the expression variables.
388  // This is guaranteed for PRISM models, by sorting the assignments as soon as an update is created.
389 
390  auto assignmentIt = update.getAssignments().begin();
391  auto assignmentIte = update.getAssignments().end();
392 
393  // Iterate over all boolean assignments and carry them out.
394  auto boolIt = this->variableInformation.booleanVariables.begin();
395  for (; assignmentIt != assignmentIte && assignmentIt->getExpression().hasBooleanType(); ++assignmentIt) {
396  while (assignmentIt->getVariable() != boolIt->variable) {
397  ++boolIt;
398  }
399  newState.set(boolIt->bitOffset, this->evaluator->asBool(assignmentIt->getExpression()));
400  }
401 
402  // Iterate over all integer assignments and carry them out.
403  auto integerIt = this->variableInformation.integerVariables.begin();
404  for (; assignmentIt != assignmentIte && assignmentIt->getExpression().hasIntegerType(); ++assignmentIt) {
405  while (assignmentIt->getVariable() != integerIt->variable) {
406  ++integerIt;
407  }
408  int_fast64_t assignedValue = this->evaluator->asInt(assignmentIt->getExpression());
409  if (this->options.isAddOutOfBoundsStateSet()) {
410  if (assignedValue < integerIt->lowerBound || assignedValue > integerIt->upperBound) {
411  return this->outOfBoundsState;
412  }
413  } else if (integerIt->forceOutOfBoundsCheck || this->options.isExplorationChecksSet()) {
414  STORM_LOG_THROW(assignedValue >= integerIt->lowerBound, storm::exceptions::WrongFormatException, "The update " << update << " leads to an out-of-bounds value (" << assignedValue << ") for the variable '" << assignmentIt->getVariableName() << "'.");
415  STORM_LOG_THROW(assignedValue <= integerIt->upperBound, storm::exceptions::WrongFormatException, "The update " << update << " leads to an out-of-bounds value (" << assignedValue << ") for the variable '" << assignmentIt->getVariableName() << "'.");
416  }
417  newState.setFromInt(integerIt->bitOffset, integerIt->bitWidth, assignedValue - integerIt->lowerBound);
418  STORM_LOG_ASSERT(static_cast<int_fast64_t>(newState.getAsInt(integerIt->bitOffset, integerIt->bitWidth)) + integerIt->lowerBound == assignedValue, "Writing to the bit vector bucket failed (read " << newState.getAsInt(integerIt->bitOffset, integerIt->bitWidth) << " but wrote " << assignedValue << ").");
419  }
420 
421  // Check that we processed all assignments.
422  STORM_LOG_ASSERT(assignmentIt == assignmentIte, "Not all assignments were consumed.");
423 
424  return newState;
425  }

436  template<typename ValueType, typename StateType>
437  boost::optional<std::vector<std::vector<std::reference_wrapper<storm::prism::Command const>>>> PrismNextStateGenerator<ValueType, StateType>::getActiveCommandsByActionIndex(uint_fast64_t const& actionIndex, CommandFilter const& commandFilter) {
438 
439  // First check whether there is at least one enabled command at each module
440  // This avoids evaluating unnecessarily many guards.
441  // If we find one module without an enabled command, we return boost::none.
442  // At the same time, we store pointers to the relevant modules, the relevant command sets and the first enabled command within each set.
443 
444  // Iterate over all modules.
445  std::vector<ActiveCommandData> activeCommands;
446  for (uint_fast64_t i = 0; i < program.getNumberOfModules(); ++i) {
447  storm::prism::Module const& module = program.getModule(i);
448 
449  // If the module has no command labeled with the given action, we can skip this module.
450  if (!module.hasActionIndex(actionIndex)) {
451  continue;
452  }
453 
454  std::set<uint_fast64_t> const& commandIndices = module.getCommandIndicesByActionIndex(actionIndex);
455 
456  // If the module contains the action, but there is no command in the module that is labeled with
457  // this action, we don't have any feasible command combinations.
458  if (commandIndices.empty()) {
459  return boost::none;
460  }
461 
462  // Look up commands by their indices and check if the guard evaluates to true in the given state.
463  bool hasOneEnabledCommand = false;
464  for (auto commandIndexIt = commandIndices.begin(), commandIndexIte = commandIndices.end(); commandIndexIt != commandIndexIte; ++commandIndexIt) {
465  storm::prism::Command const& command = module.getCommand(*commandIndexIt);
466  if (commandFilter != CommandFilter::All) {
467  STORM_LOG_ASSERT(commandFilter == CommandFilter::Markovian || commandFilter == CommandFilter::Probabilistic, "Unexpected command filter.");
468  if ((commandFilter == CommandFilter::Markovian) != command.isMarkovian()) {
469  continue;
470  }
471  }
472  if (this->evaluator->asBool(command.getGuardExpression())) {
473  // Found the first enabled command for this module.
474  hasOneEnabledCommand = true;
475  activeCommands.emplace_back(&module, &commandIndices, commandIndexIt);
476  break;
477  }
478  }
479 
480  if (!hasOneEnabledCommand) {
481  return boost::none;
482  }
483  }
484 
485  // If we reach this point, there has to be at least one active command for each relevant module.
486  std::vector<std::vector<std::reference_wrapper<storm::prism::Command const>>> result;
487 
488  // Iterate over all command sets.
489  for (auto const& activeCommand : activeCommands) {
490  std::vector<std::reference_wrapper<storm::prism::Command const>> commands;
491 
492  auto commandIndexIt = activeCommand.currentCommandIndexIt;
493  // The command at the current position is already known to be enabled
494  commands.push_back(activeCommand.modulePtr->getCommand(*commandIndexIt));
495 
496  // Look up commands by their indices and add them if the guard evaluates to true in the given state.
497  auto commandIndexIte = activeCommand.commandIndicesPtr->end();
498  for (++commandIndexIt; commandIndexIt != commandIndexIte; ++commandIndexIt) {
499  storm::prism::Command const& command = activeCommand.modulePtr->getCommand(*commandIndexIt);
500  if (commandFilter != CommandFilter::All) {
501  STORM_LOG_ASSERT(commandFilter == CommandFilter::Markovian || commandFilter == CommandFilter::Probabilistic, "Unexpected command filter.");
502  if ((commandFilter == CommandFilter::Markovian) != command.isMarkovian()) {
503  continue;
504  }
505  }
506  if (this->evaluator->asBool(command.getGuardExpression())) {
507  commands.push_back(command);
508  }
509  }
510 
511  result.push_back(std::move(commands));
512  }
513 
514  STORM_LOG_ASSERT(!result.empty(), "Expected non-empty list.");
515  return result;
516  }

518  template<typename ValueType, typename StateType>
519  std::vector<Choice<ValueType>> PrismNextStateGenerator<ValueType, StateType>::getUnlabeledChoices(CompressedState const& state, StateToIdCallback stateToIdCallback, CommandFilter const& commandFilter) {
520  std::vector<Choice<ValueType>> result;
521 
522  // Iterate over all modules.
523  for (uint_fast64_t i = 0; i < program.getNumberOfModules(); ++i) {
524  storm::prism::Module const& module = program.getModule(i);
525 
526  // Iterate over all commands.
527  for (uint_fast64_t j = 0; j < module.getNumberOfCommands(); ++j) {
528  storm::prism::Command const& command = module.getCommand(j);
529 
530  // Only consider unlabeled commands.
531  if (command.isLabeled()) continue;
532 
533  if (commandFilter != CommandFilter::All) {
534  STORM_LOG_ASSERT(commandFilter == CommandFilter::Markovian || commandFilter == CommandFilter::Probabilistic, "Unexpected command filter.");
535  if ((commandFilter == CommandFilter::Markovian) != command.isMarkovian()) {
536  continue;
537  }
538  }
539 
540  // Skip the command, if it is not enabled.
541  if (!this->evaluator->asBool(command.getGuardExpression())) {
542  continue;
543  }
544 
545  result.push_back(Choice<ValueType>(command.getActionIndex(), command.isMarkovian()));
546  Choice<ValueType>& choice = result.back();
547 
548  // Remember the choice origin only if we were asked to.
549  if (this->options.isBuildChoiceOriginsSet()) {
550  CommandSet commandIndex { command.getGlobalIndex() };
551  choice.addOriginData(boost::any(std::move(commandIndex)));
552  }
553 
554  // Iterate over all updates of the current command.
555  ValueType probabilitySum = storm::utility::zero<ValueType>();
556  for (uint_fast64_t k = 0; k < command.getNumberOfUpdates(); ++k) {
557  storm::prism::Update const& update = command.getUpdate(k);
558 
559  ValueType probability = this->evaluator->asRational(update.getLikelihoodExpression());
560  if (probability != storm::utility::zero<ValueType>()) {
561  // Obtain target state index and add it to the list of known states. If it has not yet been
562  // seen, we also add it to the set of states that have yet to be explored.
563  StateType stateIndex = stateToIdCallback(applyUpdate(state, update));
564 
565  // Update the choice by adding the probability/target state to it.
566  choice.addProbability(stateIndex, probability);
567  if (this->options.isExplorationChecksSet()) {
568  probabilitySum += probability;
569  }
570  }
571  }
572 
573  // Create the state-action reward for the newly created choice.
574  for (auto const& rewardModel : rewardModels) {
575  ValueType stateActionRewardValue = storm::utility::zero<ValueType>();
576  if (rewardModel.get().hasStateActionRewards()) {
577  for (auto const& stateActionReward : rewardModel.get().getStateActionRewards()) {
578  if (stateActionReward.getActionIndex() == choice.getActionIndex() && this->evaluator->asBool(stateActionReward.getStatePredicateExpression())) {
579  stateActionRewardValue += ValueType(this->evaluator->asRational(stateActionReward.getRewardValueExpression()));
580  }
581  }
582  }
583  choice.addReward(stateActionRewardValue);
584  }
585 
586  if (this->options.isExplorationChecksSet()) {
587  // Check that the resulting distribution is in fact a distribution.
588  STORM_LOG_THROW(!program.isDiscreteTimeModel() || this->comparator.isOne(probabilitySum), storm::exceptions::WrongFormatException, "Probabilities do not sum to one for command '" << command << "' (actually sum to " << probabilitySum << ").");
589  }
590  }
591  }
592 
593  return result;
594  }

596  template<typename ValueType, typename StateType>
597  void PrismNextStateGenerator<ValueType, StateType>::generateSynchronizedDistribution(storm::storage::BitVector const& state, ValueType const& probability, uint64_t position, std::vector<std::vector<std::reference_wrapper<storm::prism::Command const>>::const_iterator> const& iteratorList, storm::builder::jit::Distribution<StateType, ValueType>& distribution, StateToIdCallback stateToIdCallback) {
598 
599  if (storm::utility::isZero<ValueType>(probability)) {
600  return;
601  }
602 
603  if (position >= iteratorList.size()) {
604  StateType id = stateToIdCallback(state);
605  distribution.add(id, probability);
606  } else {
607  storm::prism::Command const& command = *iteratorList[position];
608  for (uint_fast64_t j = 0; j < command.getNumberOfUpdates(); ++j) {
609  storm::prism::Update const& update = command.getUpdate(j);
610  generateSynchronizedDistribution(applyUpdate(state, update), probability * this->evaluator->asRational(update.getLikelihoodExpression()), position + 1, iteratorList, distribution, stateToIdCallback);
611  }
612  }

615  template<typename ValueType, typename StateType>
616  void PrismNextStateGenerator<ValueType, StateType>::addLabeledChoices(std::vector<Choice<ValueType>>& choices, CompressedState const& state, StateToIdCallback stateToIdCallback, CommandFilter const& commandFilter) {
617 
618  for (uint_fast64_t actionIndex : program.getSynchronizingActionIndices()) {
619  boost::optional<std::vector<std::vector<std::reference_wrapper<storm::prism::Command const>>>> optionalActiveCommandLists = getActiveCommandsByActionIndex(actionIndex, commandFilter);
620 
621  // Only process this action label, if there is at least one feasible solution.
622  if (optionalActiveCommandLists) {
623  std::vector<std::vector<std::reference_wrapper<storm::prism::Command const>>> const& activeCommandList = optionalActiveCommandLists.get();
624  std::vector<std::vector<std::reference_wrapper<storm::prism::Command const>>::const_iterator> iteratorList(activeCommandList.size());
625 
626  // Initialize the list of iterators.
627  for (size_t i = 0; i < activeCommandList.size(); ++i) {
628  iteratorList[i] = activeCommandList[i].cbegin();
629  }
630 
631  storm::builder::jit::Distribution<StateType, ValueType> distribution;
632 
633  // As long as there is one feasible combination of commands, keep on expanding it.
634  bool done = false;
635  while (!done) {
636  distribution.clear();
637  generateSynchronizedDistribution(state, storm::utility::one<ValueType>(), 0, iteratorList, distribution, stateToIdCallback);
638  distribution.compress();
639 
640  // At this point, we applied all commands of the current command combination and newTargetStates
641  // contains all target states and their respective probabilities. That means we are now ready to
642  // add the choice to the list of transitions.
643  choices.push_back(Choice<ValueType>(actionIndex));
644 
645  // Now create the actual distribution.
646  Choice<ValueType>& choice = choices.back();
647 
648  // Remember the choice label and origins only if we were asked to.
649  if (this->options.isBuildChoiceLabelsSet()) {
650  choice.addLabel(program.getActionName(actionIndex));
651  }
652  if (this->options.isBuildChoiceOriginsSet()) {
653  CommandSet commandIndices;
654  for (uint_fast64_t i = 0; i < iteratorList.size(); ++i) {
655  commandIndices.insert(iteratorList[i]->get().getGlobalIndex());
656  }
657  choice.addOriginData(boost::any(std::move(commandIndices)));
658  }
659 
660  // Add the probabilities/rates to the newly created choice.
661  ValueType probabilitySum = storm::utility::zero<ValueType>();
662  choice.reserve(std::distance(distribution.begin(), distribution.end()));
663  for (auto const& stateProbability : distribution) {
664  choice.addProbability(stateProbability.getState(), stateProbability.getValue());
665  if (this->options.isExplorationChecksSet()) {
666  probabilitySum += stateProbability.getValue();
667  }
668  }
669 
670  if (this->options.isExplorationChecksSet()) {
671  // Check that the resulting distribution is in fact a distribution.
672  STORM_LOG_THROW(!program.isDiscreteTimeModel() || !this->comparator.isConstant(probabilitySum) || this->comparator.isOne(probabilitySum), storm::exceptions::WrongFormatException, "Sum of update probabilities do not some to one for some command (actually sum to " << probabilitySum << ").");
673  }
674 
675  // Create the state-action reward for the newly created choice.
676  for (auto const& rewardModel : rewardModels) {
677  ValueType stateActionRewardValue = storm::utility::zero<ValueType>();
678  if (rewardModel.get().hasStateActionRewards()) {
679  for (auto const& stateActionReward : rewardModel.get().getStateActionRewards()) {
680  if (stateActionReward.getActionIndex() == choice.getActionIndex() && this->evaluator->asBool(stateActionReward.getStatePredicateExpression())) {
681  stateActionRewardValue += ValueType(this->evaluator->asRational(stateActionReward.getRewardValueExpression()));
682  }
683  }
684  }
685  choice.addReward(stateActionRewardValue);
686  }
687 
688  // Now, check whether there is one more command combination to consider.
689  bool movedIterator = false;
690  for (int_fast64_t j = iteratorList.size() - 1; !movedIterator && j >= 0; --j) {
691  ++iteratorList[j];
692  if (iteratorList[j] != activeCommandList[j].end()) {
693  movedIterator = true;
694  } else {
695  // Reset the iterator to the beginning of the list.
696  iteratorList[j] = activeCommandList[j].begin();
697  }
698  }
699 
700  done = !movedIterator;
701  }
702  }
703  }
704  }

752  std::shared_ptr<storm::storage::sparse::ChoiceOrigins> PrismNextStateGenerator<ValueType, StateType>::generateChoiceOrigins(std::vector<boost::any>& dataForChoiceOrigins) const {
753  if (!this->getOptions().isBuildChoiceOriginsSet()) {
754  return nullptr;
755  }
756 
757  std::vector<uint_fast64_t> identifiers;
758  identifiers.reserve(dataForChoiceOrigins.size());
759 
760  std::map<CommandSet, uint_fast64_t> commandSetToIdentifierMap;
761  // The empty commandset (i.e., the choices without origin) always has to get identifier getIdentifierForChoicesWithNoOrigin() -- which is assumed to be 0
762  STORM_LOG_ASSERT(storm::storage::sparse::ChoiceOrigins::getIdentifierForChoicesWithNoOrigin() == 0, "The no origin identifier is assumed to be zero");
763  commandSetToIdentifierMap.insert(std::make_pair(CommandSet(), 0));
764  uint_fast64_t currentIdentifier = 1;
765  for (boost::any& originData : dataForChoiceOrigins) {
766  STORM_LOG_ASSERT(originData.empty() || boost::any_cast<CommandSet>(&originData) != nullptr, "Origin data has unexpected type: " << originData.type().name() << ".");
767 
768  CommandSet currentCommandSet = originData.empty() ? CommandSet() : boost::any_cast<CommandSet>(std::move(originData));
769  auto insertionRes = commandSetToIdentifierMap.insert(std::make_pair(std::move(currentCommandSet), currentIdentifier));
770  identifiers.push_back(insertionRes.first->second);
771  if (insertionRes.second) {
772  ++currentIdentifier;
773  }
774  }
775 
776  std::vector<CommandSet> identifierToCommandSetMapping(currentIdentifier);
777  for (auto const& setIdPair : commandSetToIdentifierMap) {
778  identifierToCommandSetMapping[setIdPair.second] = setIdPair.first;
779  }
780 
781  return std::make_shared<storm::storage::sparse::PrismChoiceOrigins>(std::make_shared<storm::prism::Program>(program), std::move(identifiers), std::move(identifierToCommandSetMapping));
782  }
