68  void performSccDecompositionGCM(storm::storage::SparseMatrix<ValueType> const& transitionMatrix, uint_fast64_t startState, storm::storage::BitVector& nonTrivialStates, storm::storage::BitVector const* subsystem, storm::storage::BitVector const* choices, uint_fast64_t& currentIndex, storm::storage::BitVector& hasPreorderNumber, std::vector<uint_fast64_t>& preorderNumbers, std::vector<uint_fast64_t>& recursionStateStack, std::vector<uint_fast64_t>& s, std::vector<uint_fast64_t>& p, storm::storage::BitVector& stateHasScc, std::vector<uint_fast64_t>& stateToSccMapping, uint_fast64_t& sccCount, bool /*forceTopologicalSort*/, std::vector<uint_fast64_t>* sccDepths) {
69  // The forceTopologicalSort flag can be ignored as this method always generates a topological sort.
70 
71  // Prepare the stack used for turning the recursive procedure into an iterative one.
72  STORM_LOG_ASSERT(recursionStateStack.empty(), "Expected an empty recursion stack.");
73  recursionStateStack.push_back(startState);
74 
75  while (!recursionStateStack.empty()) {
76  // Peek at the topmost state in the stack, but leave it on there for now.
77  uint_fast64_t currentState = recursionStateStack.back();
78 
79  // If the state has not yet been seen, we need to assign it a preorder number and iterate over its successors.
80  if (!hasPreorderNumber.get(currentState)) {
81  preorderNumbers[currentState] = currentIndex++;
82  hasPreorderNumber.set(currentState, true);
83 
84  s.push_back(currentState);
85  p.push_back(currentState);
86 
87  for (uint64_t row = transitionMatrix.getRowGroupIndices()[currentState], rowEnd = transitionMatrix.getRowGroupIndices()[currentState + 1]; row != rowEnd; ++row) {
88  if (choices && !choices->get(row)) {
89  continue;
90  }
91 
92  for (auto const& successor : transitionMatrix.getRow(row)) {
93  if ((!subsystem || subsystem->get(successor.getColumn())) && successor.getValue() != storm::utility::zero<ValueType>()) {
94  if (currentState == successor.getColumn()) {
95  nonTrivialStates.set(currentState, true);
96  }
97 
98  if (!hasPreorderNumber.get(successor.getColumn())) {
99  // In this case, we must recursively visit the successor. We therefore push the state
100  // onto the recursion stack.
101  recursionStateStack.push_back(successor.getColumn());
102  } else {
103  if (!stateHasScc.get(successor.getColumn())) {
104  while (preorderNumbers[p.back()] > preorderNumbers[successor.getColumn()]) {
105  p.pop_back();
106  }
107  }
108  }
109  }
110  }
111  }
112  } else {
113  // In this case, we have searched all successors of the current state and can exit the "recursion"
114  // on the current state.
115  if (currentState == p.back()) {
116  p.pop_back();
117  if (sccDepths) {
118  uint_fast64_t sccDepth = 0;
119  // Find the largest depth over successor SCCs.
120  auto stateIt = s.end();
121  do {
122  --stateIt;
123  for (uint64_t row = transitionMatrix.getRowGroupIndices()[*stateIt], rowEnd = transitionMatrix.getRowGroupIndices()[*stateIt + 1]; row != rowEnd; ++row) {
124  if (choices && !choices->get(row)) {
125  continue;
126  }
127  for (auto const& successor : transitionMatrix.getRow(row)) {
128  if ((!subsystem || subsystem->get(successor.getColumn())) && successor.getValue() != storm::utility::zero<ValueType>() && stateHasScc.get(successor.getColumn())) {
129  sccDepth = std::max(sccDepth, (*sccDepths)[stateToSccMapping[successor.getColumn()]] + 1);
130  }
131  }
132  }
133  } while (*stateIt != currentState);
134  sccDepths->push_back(sccDepth);
135  }
136  bool nonSingletonScc = s.back() != currentState;
137  uint_fast64_t poppedState = 0;
138  do {
139  poppedState = s.back();
140  s.pop_back();
141  stateToSccMapping[poppedState] = sccCount;
142  stateHasScc.set(poppedState);
143  if (nonSingletonScc) {
144  nonTrivialStates.set(poppedState, true);
145  }
146  } while (poppedState != currentState);
147  ++sccCount;
148  }
149 
150  recursionStateStack.pop_back();
151  }
152  }
153  }

156  void StronglyConnectedComponentDecomposition<ValueType>::performSccDecomposition(storm::storage::SparseMatrix<ValueType> const& transitionMatrix, StronglyConnectedComponentDecompositionOptions const& options) {
157 
158  STORM_LOG_ASSERT(!options.choicesPtr || options.subsystemPtr, "Expecting subsystem if choices are given.");
159 
160  uint_fast64_t numberOfStates = transitionMatrix.getRowGroupCount();
161  uint_fast64_t sccCount = 0;
162 
163  // We need to keep of trivial states (singleton SCCs without selfloop).
164  storm::storage::BitVector nonTrivialStates(numberOfStates, false);
165 
166  // Obtain a mapping from states to the SCC it belongs to
167  std::vector<uint_fast64_t> stateToSccMapping(numberOfStates);
168  {
169 
170  // Set up the environment of the algorithm.
171  // Start with the two stacks it maintains.
172  // This is to reduce memory (re-)allocations
173  std::vector<uint_fast64_t> s;
174  s.reserve(numberOfStates);
175  std::vector<uint_fast64_t> p;
176  p.reserve(numberOfStates);
177  std::vector<uint_fast64_t> recursionStateStack;
178  recursionStateStack.reserve(numberOfStates);
179 
180  // We also need to store the preorder numbers of states and which states have been assigned to which SCC.
181  std::vector<uint_fast64_t> preorderNumbers(numberOfStates);
182  storm::storage::BitVector hasPreorderNumber(numberOfStates);
183  storm::storage::BitVector stateHasScc(numberOfStates);
184 
185  // Store scc depths if requested
186  std::vector<uint_fast64_t>* sccDepthsPtr = nullptr;
187  sccDepths = boost::none;
188  if (options.isComputeSccDepthsSet || options.areOnlyBottomSccsConsidered) {
189  sccDepths = std::vector<uint_fast64_t>();
190  sccDepthsPtr = &sccDepths.get();
191  }
192 
193 
194  // Start the search for SCCs from every state in the block.
195  uint_fast64_t currentIndex = 0;
196  if (options.subsystemPtr) {
197  for (auto state : *options.subsystemPtr) {
198  if (!hasPreorderNumber.get(state)) {
199  performSccDecompositionGCM(transitionMatrix, state, nonTrivialStates, options.subsystemPtr, options.choicesPtr, currentIndex, hasPreorderNumber, preorderNumbers, recursionStateStack, s, p, stateHasScc, stateToSccMapping, sccCount, options.isTopologicalSortForced, sccDepthsPtr);
200  }
201  }
202  } else {
203  for (uint64_t state = 0; state < transitionMatrix.getRowGroupCount(); ++state) {
204  if (!hasPreorderNumber.get(state)) {
205  performSccDecompositionGCM(transitionMatrix, state, nonTrivialStates, options.subsystemPtr, options.choicesPtr, currentIndex, hasPreorderNumber, preorderNumbers, recursionStateStack, s, p, stateHasScc, stateToSccMapping, sccCount, options.isTopologicalSortForced, sccDepthsPtr);
206  }
207  }
208  }
209  }
210  // After we obtained the state-to-SCC mapping, we build the actual blocks.
211  this->blocks.resize(sccCount);
212  for (uint64_t state = 0; state < transitionMatrix.getRowGroupCount(); ++state) {
213  // Check if this state (and is SCC) should be considered in this decomposition.
214  if ((!options.subsystemPtr || options.subsystemPtr->get(state))) {
215  if (!options.areNaiveSccsDropped || nonTrivialStates.get(state)) {
216  uint_fast64_t sccIndex = stateToSccMapping[state];
217  if (!options.areOnlyBottomSccsConsidered || sccDepths.get()[sccIndex] == 0) {
218  this->blocks[sccIndex].insert(state);
219  if (!nonTrivialStates.get(state)) {
220  this->blocks[sccIndex].setIsTrivial(true);
221  STORM_LOG_ASSERT(this->blocks[sccIndex].size() == 1, "Unexpected number of states in a trivial SCC.");
222  }
223  }
224  }
225  }
226  }
227 
228  // If requested, we need to delete all naive SCCs.
229  if (options.areOnlyBottomSccsConsidered || options.areNaiveSccsDropped) {
230  storm::storage::BitVector blocksToDrop(sccCount);
231  for (uint_fast64_t sccIndex = 0; sccIndex < sccCount; ++sccIndex) {
232  if (this->blocks[sccIndex].empty()) {
233  blocksToDrop.set(sccIndex, true);
234  }
235  }
236  // Create the new set of blocks by moving all the blocks we need to keep into it.
237  storm::utility::vector::filterVectorInPlace(this->blocks, ~blocksToDrop);
238  if (this->sccDepths) {
239  storm::utility::vector::filterVectorInPlace(this->sccDepths.get(), ~blocksToDrop);
240  }
241  }
