302  [] (storm::storage::MatrixEntry<uint_fast64_t, ExportValueType> const& a, storm::storage::MatrixEntry<uint_fast64_t, ExportValueType> const& b) {
303  return a.getColumn() < b.getColumn();
304  });

311  }
312

522  } else {

524  bool nextVariableIsNondeterminismVariable = !Cudd_IsConstant(nondeterminismVariables) && Cudd_NodeReadIndex(nondeterminismVariables) == Cudd_NodeReadIndex(variables);
525

528  DdNodePtr e;

531  if (Cudd_NodeReadIndex(transitionMatrixNode) == Cudd_NodeReadIndex(variables)) {
532  t = Cudd_T(transitionMatrixNode);
533  e = Cudd_E(transitionMatrixNode);
534  } else {
535  t = e = transitionMatrixNode;
536  }
537

539  extractTransitionMatrixRec(e, sourceOdd.getElseSuccessor(), sourceOffset, targetPartitionNode, representativesNode, Cudd_T(variables), Cudd_T(nondeterminismVariables), stateOdd, stateOffset);
540  extractTransitionMatrixRec(t, sourceOdd.getThenSuccessor(), sourceOffset + sourceOdd.getElseOffset(), targetPartitionNode, representativesNode, Cudd_T(variables), Cudd_T(nondeterminismVariables), stateOdd, stateOffset);
541  } else {
542  DdNodePtr t;
543  DdNodePtr tt;
544  DdNodePtr te;
545  DdNodePtr e;
546  DdNodePtr et;
547  DdNodePtr ee;
548  if (Cudd_NodeReadIndex(transitionMatrixNode) == Cudd_NodeReadIndex(variables)) {
549  // Source node was not skipped in transition matrix.
550  t = Cudd_T(transitionMatrixNode);
551  e = Cudd_E(transitionMatrixNode);
552  } else {

560  } else {

562  tt = te = t;
563  }
564  if (t != e) {
565  if (Cudd_NodeReadIndex(e) == Cudd_NodeReadIndex(variables) + 1) {
566  // Target node was not skipped in transition matrix.
567  et = Cudd_T(e);
568  ee = Cudd_E(e);
569  } else {
570  // Target node was skipped in transition matrix.
571  et = ee = e;
572  }
573  } else {
574  et = tt;
575  ee = te;
576  }
577 
578  DdNodePtr targetT;
579  DdNodePtr targetE;
580  if (Cudd_NodeReadIndex(targetPartitionNode) == Cudd_NodeReadIndex(variables)) {
581  // Node was not skipped in target partition.
582  targetT = Cudd_T(targetPartitionNode);
583  targetE = Cudd_E(targetPartitionNode);
584 
585  if (Cudd_IsComplement(targetPartitionNode)) {
586  targetT = Cudd_Not(targetT);
587  targetE = Cudd_Not(targetE);
588  }
589  } else {

591  targetT = targetE = targetPartitionNode;
592  }
593

595  DdNodePtr representativesE;
596  if (Cudd_NodeReadIndex(representativesNode) == Cudd_NodeReadIndex(variables)) {
597  // Node was not skipped in representatives.

599  representativesE = Cudd_E(representativesNode);
600  } else {

602  representativesT = representativesE = representativesNode;
603  }
604

607  representativesE = Cudd_Not(representativesE);
608  }
609 
610  extractTransitionMatrixRec(ee, sourceOdd.getElseSuccessor(), sourceOffset, targetE, representativesE, Cudd_T(variables), nondeterminismVariables, stateOdd ? &stateOdd->getElseSuccessor() : stateOdd, stateOffset);
611  extractTransitionMatrixRec(et, sourceOdd.getElseSuccessor(), sourceOffset, targetT, representativesE, Cudd_T(variables), nondeterminismVariables, stateOdd ? &stateOdd->getElseSuccessor() : stateOdd, stateOffset);
612  extractTransitionMatrixRec(te, sourceOdd.getThenSuccessor(), sourceOffset + sourceOdd.getElseOffset(), targetE, representativesT, Cudd_T(variables), nondeterminismVariables, stateOdd ? &stateOdd->getThenSuccessor() : stateOdd, stateOffset + (stateOdd ? stateOdd->getElseOffset() : 0));
