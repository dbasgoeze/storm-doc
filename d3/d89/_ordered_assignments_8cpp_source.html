28  bool OrderedAssignments::add(Assignment const& assignment, bool addToExisting) {
29  // If the element is contained in this set of assignment, nothing needs to be added.
30  if (!addToExisting && this->contains(assignment)) {
31  return false;
32  }
33 
34  // Otherwise, we find the spot to insert it.
35  auto it = lowerBound(assignment, allAssignments);
36 
37  // Check if an assignment to this variable is already present
38  if (it != allAssignments.end() && assignment.getLValue() == (*it)->getLValue()) {
39  STORM_LOG_THROW(addToExisting && assignment.getLValue().isVariable() && assignment.getExpressionVariable().hasNumericalType(), storm::exceptions::InvalidArgumentException, "Cannot add assignment ('" << assignment.getAssignedExpression() << "') as an assignment ('" << (*it)->getAssignedExpression() << "') to LValue '" << (*it)->getLValue() << "' already exists.");
40  (*it)->setAssignedExpression((*it)->getAssignedExpression() + assignment.getAssignedExpression());
41  } else {
42  // Finally, insert the new element in the correct vectors.
43  auto elementToInsert = std::make_shared<Assignment>(assignment);
44  allAssignments.emplace(it, elementToInsert);
45  if (assignment.isTransient()) {
46  auto transientIt = lowerBound(assignment, transientAssignments);
47  transientAssignments.emplace(transientIt, elementToInsert);
48  } else {
49  auto nonTransientIt = lowerBound(assignment, nonTransientAssignments);
50  nonTransientAssignments.emplace(nonTransientIt, elementToInsert);
51  }
52  }
53 
54  return true;
