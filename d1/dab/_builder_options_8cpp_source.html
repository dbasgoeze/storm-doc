70  void BuilderOptions::preserveFormula(storm::logic::Formula const& formula, storm::storage::SymbolicModelDescription const& modelDescription) {
71  // If we already had terminal states, we need to erase them.
72  if (hasTerminalStates()) {
73  clearTerminalStates();
74  }
75 
76  // Determine the reward models we need to build.
77  std::set<std::string> referencedRewardModels = formula.getReferencedRewardModels();
78  for (auto const& rewardModelName : referencedRewardModels) {
79  rewardModelNames.emplace(rewardModelName);
80  }
81 
82  // Extract all the labels used in the formula.
83  std::vector<std::shared_ptr<storm::logic::AtomicLabelFormula const>> atomicLabelFormulas = formula.getAtomicLabelFormulas();
84  for (auto const& formula : atomicLabelFormulas) {
85  addLabel(formula->getLabel());
86  }
87 
88  // Extract all the expressions used in the formula.
89  std::vector<std::shared_ptr<storm::logic::AtomicExpressionFormula const>> atomicExpressionFormulas = formula.getAtomicExpressionFormulas();
90  for (auto const& formula : atomicExpressionFormulas) {
91  addLabel(formula->getExpression());
92  }
93 
94  scaleAndLiftTransitionRewards = scaleAndLiftTransitionRewards && storm::logic::LiftableTransitionRewardsVisitor(modelDescription).areTransitionRewardsLiftable(formula);
95  }
