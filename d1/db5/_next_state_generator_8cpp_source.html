219  template<typename ValueType, typename StateType>
220  void NextStateGenerator<ValueType, StateType>::postprocess(StateBehavior<ValueType, StateType>& result) {
221  // If the model we build is a Markov Automaton, we postprocess the choices to sum all Markovian choices
222  // and make the Markovian choice the very first one (if there is any).
223  bool foundPreviousMarkovianChoice = false;
224  if (this->getModelType() == ModelType::MA) {
225  uint64_t numberOfChoicesToDelete = 0;
226 
227  for (uint_fast64_t index = 0; index + numberOfChoicesToDelete < result.getNumberOfChoices();) {
228  Choice<ValueType>& choice = result.getChoices()[index];
229 
230  if (choice.isMarkovian()) {
231  if (foundPreviousMarkovianChoice) {
232  // If there was a previous Markovian choice, we need to sum them. Note that we can assume
233  // that the previous Markovian choice is the very first one in the choices vector.
234  result.getChoices().front().add(choice);
235 
236  // Swap the choice to the end to indicate it can be removed (if it's not already there).
237  if (index != result.getNumberOfChoices() - 1 - numberOfChoicesToDelete) {
238  choice = std::move(result.getChoices()[result.getNumberOfChoices() - 1 - numberOfChoicesToDelete]);
239  }
240  ++numberOfChoicesToDelete;
241  } else {
242  // If there is no previous Markovian choice, just move the Markovian choice to the front.
243  if (index != 0) {
244  std::swap(result.getChoices().front(), choice);
245  }
246  foundPreviousMarkovianChoice = true;
247  ++index;
248  }
249  } else {
250  ++index;
251  }
252  }
253 
254  // Finally remove the choices that were added to other Markovian choices.
255  if (numberOfChoicesToDelete > 0) {
256  result.getChoices().resize(result.getChoices().size() - numberOfChoicesToDelete);
257  }
258  }
259  }
