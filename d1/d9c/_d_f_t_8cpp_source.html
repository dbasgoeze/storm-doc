94  template<typename ValueType>
95  void DFT<ValueType>::setDynamicBehaviorInfo() {
96  std::vector<bool> dynamicBehaviorVector(mElements.size(), false);
97 
98  std::queue <DFTElementPointer> elementQueue;
99 
100  // deal with all dynamic elements
101  for (auto const &element : mElements) {
102  switch (element->type()) {
103  case storage::DFTElementType::PAND:
104  case storage::DFTElementType::POR:
105  case storage::DFTElementType::MUTEX: {
106  auto gate = std::static_pointer_cast<storm::storage::DFTChildren<ValueType>>(element);
107  dynamicBehaviorVector[gate->id()] = true;
108  for (auto const &child : gate->children()) {
109  // only enqueue static children
110  if (!dynamicBehaviorVector.at(child->id())) {
111  elementQueue.push(child);
112  }
113  }
114  break;
115  }
116  // TODO different cases
117  case storage::DFTElementType::SPARE: {
118  auto spare = std::static_pointer_cast<storm::storage::DFTSpare<ValueType>>(element);
119 
120  // Iterate over all children (representatives of spare modules)
121  for (auto const &child : spare->children()) {
122  // Case 1: Shared Module
123  // If child only has one parent, it is this SPARE -> nothing to check
124  if (child->nrParents() > 1) {
125  // TODO make more efficient by directly setting ALL spares which share a module to be dynamic
126  for (auto const &parent : child->parents()) {
127  if (parent->isSpareGate() and parent->id() != spare->id()) {
128  dynamicBehaviorVector[spare->id()] = true;
129  break; // inner loop
130  }
131  }
132  }
133  // Case 2: Triggering outside events
134  // If the SPARE was already detected to have dynamic behavior, do not proceed
135  if (!dynamicBehaviorVector[spare->id()]) {
136  for (auto const &memberID : module(child->id())) {
137  // Iterate over all members of the module child represents
138  auto member = getElement(memberID);
139  for (auto const dep : member->outgoingDependencies()) {
140  // If the member has outgoing dependencies, check if those trigger something outside the module
141  for (auto const depEvent : dep->dependentEvents()) {
142  // If a dependent event is not found in the module, SPARE is dynamic
143  if (std::find(module(child->id()).begin(), module(child->id()).end(),
144  depEvent->id()) == module(child->id()).end()) {
145  dynamicBehaviorVector[spare->id()] = true;
146  break; //depEvent-loop
147  }
148  }
149  if (dynamicBehaviorVector[spare->id()]) { break; } //dependency-loop
150  }
151  if (dynamicBehaviorVector[spare->id()]) { break; } //module-loop
152  }
153 
154  }
155  if (dynamicBehaviorVector[spare->id()]) { break; } //child-loop
156  }
157  // if during the computation, dynamic behavior was detected, add children to queue
158  if (dynamicBehaviorVector[spare->id()]) {
159  for (auto const &child : spare->children()) {
160  // only enqueue static children
161  if (!dynamicBehaviorVector.at(child->id())) {
162  elementQueue.push(child);
163  }
164  }
165  }
166  break;
167  }
168  case storage::DFTElementType::SEQ: {
169  auto seq = std::static_pointer_cast<storm::storage::DFTSeq<ValueType>>(element);
170  // A SEQ only has dynamic behavior if not all children are BEs
171  if (!seq->allChildrenBEs()) {
172  dynamicBehaviorVector[seq->id()] = true;
173  for (auto const &child : seq->children()) {
174  // only enqueue static children
175  if (!dynamicBehaviorVector.at(child->id())) {
176  elementQueue.push(child);
177  }
178  }
179  }
180  break;
181  }
182  default: {
183  break;
184  }
185 
186  }
187  }
188  // propagate dynamic behavior
189  while (!elementQueue.empty()) {
190  DFTElementPointer currentElement = elementQueue.front();
191  elementQueue.pop();
192  switch (currentElement->type()) {
193  // Static Gates
194  case storage::DFTElementType::AND:
195  case storage::DFTElementType::OR:
196  case storage::DFTElementType::VOT: {
197  // check all parents and if one has dynamic behavior, propagate it
198  dynamicBehaviorVector[currentElement->id()] = true;
199  auto gate = std::static_pointer_cast<storm::storage::DFTGate<ValueType>>(currentElement);
200  for (auto const &child : gate->children()) {
201  // only enqueue static children
202  if (!dynamicBehaviorVector.at(child->id())) {
203  elementQueue.push(child);
204  }
205  }
206  break;
207  }
208  //BEs
209  case storage::DFTElementType::BE: {
210  auto be = std::static_pointer_cast<storm::storage::DFTBE<ValueType>>(currentElement);
211  dynamicBehaviorVector[be->id()] = true;
212  // add all ingoing dependencies to queue
213  for (auto const &dep : be->ingoingDependencies()) {
214  if (!dynamicBehaviorVector.at(dep->id())) {
215  elementQueue.push(dep);
216  }
217  }
218  break;
219  }
220  case storage::DFTElementType::PDEP: {
221  auto dep = std::static_pointer_cast<storm::storage::DFTDependency<ValueType>>(currentElement);
222  dynamicBehaviorVector[dep->id()] = true;
223  // add all ingoing dependencies to queue
224  auto trigger = dep->triggerEvent();
225  if (!dynamicBehaviorVector.at(trigger->id())) {
226  elementQueue.push(trigger);
227  }
228  break;
229  }
230  default:
231  break;
232  }
233  }
234  mDynamicBehavior = dynamicBehaviorVector;
235  }

237  template<typename ValueType>
238  DFTStateGenerationInfo DFT<ValueType>::buildStateGenerationInfo(storm::storage::DFTIndependentSymmetries const& symmetries) const {
239  DFTStateGenerationInfo generationInfo(nrElements(), mNrOfSpares, mNrRepresentatives, mMaxSpareChildCount);
240 
241  // Generate Pre and Post info for restrictions, and mutexes
242  for (auto const& elem : mElements) {
243  if (!elem->isDependency() && !elem->isRestriction()) {
244  generationInfo.setRestrictionPreElements(elem->id(), elem->seqRestrictionPres());
245  generationInfo.setRestrictionPostElements(elem->id(), elem->seqRestrictionPosts());
246  generationInfo.setMutexElements(elem->id(), elem->mutexRestrictionElements());
247  }
248  }
249 
250  size_t stateIndex = 0;
251  std::queue<size_t> visitQueue;
252  storm::storage::BitVector visited(nrElements(), false);
253 
254  if (symmetries.groups.empty()) {
255  // Perform DFS for whole tree
256  visitQueue.push(mTopLevelIndex);
257  stateIndex = performStateGenerationInfoDFS(generationInfo, visitQueue, visited, stateIndex);
258  } else {
259  // Generate information according to symmetries
260  for (size_t symmetryIndex : symmetries.sortedSymmetries) {
261  STORM_LOG_ASSERT(!visited[symmetryIndex], "Element already considered for symmetry.");
262  auto const& symmetryGroup = symmetries.groups.at(symmetryIndex);
263  STORM_LOG_ASSERT(!symmetryGroup.empty(), "No symmetry available.");
264 
265  // Insert all elements of first subtree of each symmetry
266  size_t groupIndex = stateIndex;
267  for (std::vector<size_t> const& symmetryElement : symmetryGroup) {
268  if (visited[symmetryElement[0]]) {
269  groupIndex = std::min(groupIndex, generationInfo.getStateIndex(symmetryElement[0]));
270  } else {
271  stateIndex = generateStateInfo(generationInfo, symmetryElement[0], visited, stateIndex);
272  }
273  }
274  size_t offset = stateIndex - groupIndex;
275 
276  // Mirror symmetries
277  size_t noSymmetricElements = symmetryGroup.front().size();
278  STORM_LOG_ASSERT(noSymmetricElements > 1, "No symmetry available.");
279 
280  for (std::vector<size_t> symmetricElements : symmetryGroup) {
281  STORM_LOG_ASSERT(symmetricElements.size() == noSymmetricElements, "No. of symmetric elements do not coincide.");
282  if (visited[symmetricElements[1]]) {
283  // Elements already mirrored
284  for (size_t index : symmetricElements) {
285  STORM_LOG_ASSERT(visited[index], "Element not mirrored.");
286  }
287  continue;
288  }
289 
290  // Initialize for original element
291  size_t originalElement = symmetricElements[0];
292  size_t index = generationInfo.getStateIndex(originalElement);
293  size_t activationIndex = isRepresentative(originalElement) ? generationInfo.getSpareActivationIndex(originalElement) : 0;
294  size_t usageIndex = mElements[originalElement]->isSpareGate() ? generationInfo.getSpareUsageIndex(originalElement) : 0;
295 
296  // Mirror symmetry for each element
297  for (size_t i = 1; i < symmetricElements.size(); ++i) {
298  size_t symmetricElement = symmetricElements[i];
299  visited.set(symmetricElement);
300 
301  generationInfo.addStateIndex(symmetricElement, index + offset * i);
302  stateIndex += 2;
303 
304  STORM_LOG_ASSERT((activationIndex > 0) == isRepresentative(symmetricElement), "Bits for representative incorrect.");
305  if (activationIndex > 0) {
306  generationInfo.addSpareActivationIndex(symmetricElement, activationIndex + offset * i);
307  ++stateIndex;
308  }
309 
310  STORM_LOG_ASSERT((usageIndex > 0) == mElements[symmetricElement]->isSpareGate(), "Bits for usage incorrect.");
311  if (usageIndex > 0) {
312  generationInfo.addSpareUsageIndex(symmetricElement, usageIndex + offset * i);
313  stateIndex += generationInfo.usageInfoBits();
314  }
315  }
316  }
317 
318  // Store starting indices of symmetry groups
319  std::vector<size_t> symmetryIndices;
320  for (size_t i = 0; i < noSymmetricElements; ++i) {
321  symmetryIndices.push_back(groupIndex + i * offset);
322  }
323  generationInfo.addSymmetry(offset, symmetryIndices);
324  }
325  }
326 
327  // TODO symmetries in dependencies?
328  // Consider dependencies
329  for (size_t idDependency : getDependencies()) {
330  std::shared_ptr<DFTDependency<ValueType> const> dependency = getDependency(idDependency);
331  visitQueue.push(dependency->id());
332  visitQueue.push(dependency->triggerEvent()->id());
333  STORM_LOG_THROW(dependency->dependentEvents().size() == 1, storm::exceptions::NotSupportedException,
334  "Direct state generation does not support n-ary dependencies. Consider rewriting them by setting the binary dependency flag.");
335  visitQueue.push(dependency->dependentEvents()[0]->id());
336  }
337  stateIndex = performStateGenerationInfoDFS(generationInfo, visitQueue, visited, stateIndex);
338 
339  // Visit all remaining states
340  for (size_t i = 0; i < visited.size(); ++i) {
341  if (!visited[i]) {
342  visitQueue.push(i);
343  stateIndex = performStateGenerationInfoDFS(generationInfo, visitQueue, visited, stateIndex);
344  }
345  }
346 
347  generationInfo.generateSymmetries(symmetries);
348 
349  STORM_LOG_TRACE(generationInfo);
350  STORM_LOG_ASSERT(stateIndex == mStateVectorSize, "Id incorrect.");
351  STORM_LOG_ASSERT(visited.full(), "Not all elements considered.");
352  generationInfo.checkSymmetries();
353 
354  return generationInfo;
355  }

456  template<typename ValueType>
457  DFT<ValueType> DFT<ValueType>::optimize() const {
458  std::vector<size_t> modIdea = findModularisationRewrite();
459  STORM_LOG_DEBUG("Modularisation idea: " << storm::utility::vector::toString(modIdea));
460 
461  if (modIdea.empty()) {
462  // No rewrite needed
463  return *this;
464  }
465 
466  std::vector<std::vector<size_t>> rewriteIds;
467  rewriteIds.push_back(modIdea);
468 
469  storm::builder::DFTBuilder<ValueType> builder;
470 
471  // Accumulate elements which must be rewritten
472  std::set<size_t> rewriteSet;
473  for (std::vector<size_t> rewrites : rewriteIds) {
474  rewriteSet.insert(rewrites.front());
475  }
476  // Copy all other elements which do not change
477  for (auto elem : mElements) {
478  if (rewriteSet.count(elem->id()) == 0) {
479  builder.copyElement(elem);
480  }
481  }
482 
483  // Add rewritten elements
484  for (std::vector<size_t> rewrites : rewriteIds) {
485  STORM_LOG_ASSERT(rewrites.size() > 1, "No rewritten elements.");
486  STORM_LOG_ASSERT(mElements[rewrites[1]]->hasParents(), "Rewritten elements has no parents.");
487  STORM_LOG_ASSERT(mElements[rewrites[1]]->parents().front()->isGate(), "Rewritten element has no parent gate.");
488  DFTGatePointer originalParent = std::static_pointer_cast<DFTGate<ValueType>>(mElements[rewrites[1]]->parents().front());
489  std::string newParentName = builder.getUniqueName(originalParent->name());
490 
491  // Accumulate children names
492  std::vector<std::string> childrenNames;
493  for (size_t i = 1; i < rewrites.size(); ++i) {
494  STORM_LOG_ASSERT(mElements[rewrites[i]]->parents().front()->id() == originalParent->id(),
495  "Children have not the same father for rewrite " << mElements[rewrites[i]]->name());
496  childrenNames.push_back(mElements[rewrites[i]]->name());
497  }
498 
499  // Add element inbetween parent and children
500  switch (originalParent->type()) {
501  case DFTElementType::AND:
502  builder.addAndElement(newParentName, childrenNames);
503  break;
504  case DFTElementType::OR:
505  builder.addOrElement(newParentName, childrenNames);
506  break;
507  case DFTElementType::BE:
508  case DFTElementType::VOT:
509  case DFTElementType::PAND:
510  case DFTElementType::SPARE:
511  case DFTElementType::POR:
512  case DFTElementType::PDEP:
513  case DFTElementType::SEQ:
514  case DFTElementType::MUTEX:
515  // Other elements are not supported
516  default:
517  STORM_LOG_ASSERT(false, "Dft type can not be rewritten.");
518  break;
519  }
520 
521  // Add parent with the new child newParent and all its remaining children
522  childrenNames.clear();
523  childrenNames.push_back(newParentName);
524  for (auto const& child : originalParent->children()) {
525  if (std::find(rewrites.begin() + 1, rewrites.end(), child->id()) == rewrites.end()) {
526  // Child was not rewritten and must be kept
527  childrenNames.push_back(child->name());
528  }
529  }
530  builder.copyGate(originalParent, childrenNames);
531  }
532 
533  builder.setTopLevel(mElements[mTopLevelIndex]->name());
534  // TODO use reference?
535  DFT<ValueType> newDft = builder.build();
536  STORM_LOG_TRACE(newDft.getElementsString());
537  return newDft.optimize();
538  }

796  template<typename ValueType>
797  std::map<size_t, size_t> DFT<ValueType>::findBijection(size_t index1, size_t index2, DFTColouring<ValueType> const& colouring, bool sparesAsLeaves) const {
798  STORM_LOG_TRACE("Considering ids " << index1 << ", " << index2 << " for isomorphism.");
799  bool sharedSpareMode = false;
800  std::map<size_t, size_t> bijection;
801 
802  if (getElement(index1)->isRelevant() || getElement(index2)->isRelevant()) {
803  // Relevant events need to be uniquely identified and cannot be symmetric.
804  return {};
805  }
806 
807  if (isBasicElement(index1)) {
808  if (!isBasicElement(index2)) {
809  return {};
810  }
811  if (colouring.hasSameColour(index1, index2)) {
812  bijection[index1] = index2;
813  return bijection;
814  } else {
815  return {};
816  }
817  }
818 
819  STORM_LOG_ASSERT(isGate(index1), "Element is no gate.");
820  STORM_LOG_ASSERT(isGate(index2), "Element is no gate.");
821  std::vector<size_t> isubdft1 = getGate(index1)->independentSubDft(false);
822  std::vector<size_t> isubdft2 = getGate(index2)->independentSubDft(false);
823  if (isubdft1.empty() || isubdft2.empty() || isubdft1.size() != isubdft2.size()) {
824  if (isubdft1.empty() && isubdft2.empty() && sparesAsLeaves) {
825  // Check again for shared spares
826  sharedSpareMode = true;
827  isubdft1 = getGate(index1)->independentSubDft(false, true);
828  isubdft2 = getGate(index2)->independentSubDft(false, true);
829  if (isubdft1.empty() || isubdft2.empty() || isubdft1.size() != isubdft2.size()) {
830  return {};
831  }
832  } else {
833  return {};
834  }
835  }
836  STORM_LOG_TRACE("Checking subdfts from " << index1 << ", " << index2 << " for isomorphism.");
837  auto LHS = colouring.colourSubdft(isubdft1);
838  auto RHS = colouring.colourSubdft(isubdft2);
839  auto IsoCheck = DFTIsomorphismCheck<ValueType>(LHS, RHS, *this);
840 
841  while (IsoCheck.findNextIsomorphism()) {
842  bijection = IsoCheck.getIsomorphism();
843  if (sharedSpareMode) {
844  bool bijectionSpareCompatible = true;
845  for (size_t elementId : isubdft1) {
846  if (getElement(elementId)->isSpareGate()) {
847  std::shared_ptr<DFTSpare<ValueType>> spareLeft = std::static_pointer_cast<DFTSpare<ValueType>>(mElements[elementId]);
848  std::shared_ptr<DFTSpare<ValueType>> spareRight = std::static_pointer_cast<DFTSpare<ValueType>>(mElements[bijection.at(elementId)]);
849 
850  if (spareLeft->nrChildren() != spareRight->nrChildren()) {
851  bijectionSpareCompatible = false;
852  break;
853  }
854  // Check bijection for spare children
855  for (size_t i = 0; i < spareLeft->nrChildren(); ++i) {
856  size_t childLeftId = spareLeft->children().at(i)->id();
857  size_t childRightId = spareRight->children().at(i)->id();
858 
859  STORM_LOG_ASSERT(bijection.count(childLeftId) == 0, "Child already part of bijection.");
860  if (childLeftId == childRightId) {
861  // Ignore shared child
862  continue;
863  }
864 
865  // TODO generalize for more than one parent
866  if (spareLeft->children().at(i)->nrParents() != 1 || spareRight->children().at(i)->nrParents() != 1) {
867  bijectionSpareCompatible = false;
868  break;
869  }
870 
871  std::map<size_t, size_t> tmpBijection = findBijection(childLeftId, childRightId, colouring, false);
872  if (!tmpBijection.empty()) {
873  bijection.insert(tmpBijection.begin(), tmpBijection.end());
874  } else {
875  bijectionSpareCompatible = false;
876  break;
877  }
878  }
879  if (!bijectionSpareCompatible) {
880  break;
881  }
882  }
883  }
884  if (bijectionSpareCompatible) {
885  return bijection;
886  }
887  } else {
888  return bijection;
889  }
890  } // end while
891  return {};
892  }

915  template<typename ValueType>
916  void DFT<ValueType>::findSymmetriesHelper(std::vector<size_t> const& candidates, DFTColouring<ValueType> const& colouring,
917  std::map<size_t, std::vector<std::vector<size_t>>>& result) const {
918  if (candidates.size() <= 0) {
919  return;
920  }
921 
922  std::set<size_t> foundEqClassFor;
923  for (auto it1 = candidates.cbegin(); it1 != candidates.cend(); ++it1) {
924  std::vector<std::vector<size_t>> symClass;
925  if (foundEqClassFor.count(*it1) > 0) {
926  // This item is already in a class.
927  continue;
928  }
929  if (!getElement(*it1)->hasOnlyStaticParents()) {
930  continue;
931  }
932 
933  std::tuple<std::vector<size_t>, std::vector<size_t>, std::vector<size_t>> influencedElem1Ids = getSortedParentAndDependencyIds(*it1);
934  auto it2 = it1;
935  for (++it2; it2 != candidates.cend(); ++it2) {
936  if (!getElement(*it2)->hasOnlyStaticParents()) {
937  continue;
938  }
939 
940  if (influencedElem1Ids == getSortedParentAndDependencyIds(*it2)) {
941  std::map<size_t, size_t> bijection = findBijection(*it1, *it2, colouring, true);
942  if (!bijection.empty()) {
943  STORM_LOG_TRACE("Subdfts are symmetric");
944  foundEqClassFor.insert(*it2);
945  if (symClass.empty()) {
946  for (auto const& i : bijection) {
947  symClass.push_back(std::vector<size_t>({i.first}));
948  }
949  }
950  auto symClassIt = symClass.begin();
951  for (auto const& i : bijection) {
952  symClassIt->emplace_back(i.second);
953  ++symClassIt;
954 
955  }
956  }
957  }
958  }
959 
960  if (!symClass.empty()) {
961  result.emplace(*it1, symClass);
962  }
963  }
