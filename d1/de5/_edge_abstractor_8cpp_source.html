107  auto start = std::chrono::high_resolution_clock::now();
108 
109  // compute a decomposition of the command
110  // * start with all relevant blocks: blocks of assignment variables and variables in the rhs of assignments
111  // * go through all assignments of all updates and merge relevant blocks that are related via an assignment
112  // * repeat this until nothing changes anymore
113  // * the resulting blocks are the decomposition
114 
115  // Start by constructing the relevant blocks.
116  std::set<uint64_t> allRelevantBlocks;
117  std::map<storm::expressions::Variable, uint64_t> variableToBlockIndex;
118  for (auto const& destination : edge.get().getDestinations()) {
119  for (auto const& assignment : destination.getOrderedAssignments().getAllAssignments()) {
120  allRelevantBlocks.insert(localExpressionInformation.getBlockIndexOfVariable(assignment.getExpressionVariable()));
121 
122  auto rhsVariableBlocks = localExpressionInformation.getBlockIndicesOfVariables(assignment.getAssignedExpression().getVariables());
123  allRelevantBlocks.insert(rhsVariableBlocks.begin(), rhsVariableBlocks.end());
124  }
125  }
126  STORM_LOG_TRACE("Found " << allRelevantBlocks.size() << " relevant block(s).");
127 
128  // Create a block partition.
129  std::vector<std::set<uint64_t>> relevantBlockPartition;
130  std::map<storm::expressions::Variable, uint64_t> variableToLocalBlockIndex;
131  uint64_t index = 0;
132  for (auto const& blockIndex : allRelevantBlocks) {
133  relevantBlockPartition.emplace_back(std::set<uint64_t>({blockIndex}));
134  for (auto const& variable : localExpressionInformation.getVariableBlockWithIndex(blockIndex)) {
135  variableToLocalBlockIndex[variable] = index;
136  }
137  ++index;
138  }
139 
140  // Merge all blocks that are related via the right-hand side of assignments.
141  for (auto const& destination : edge.get().getDestinations()) {
142  for (auto const& assignment : destination.getOrderedAssignments().getAllAssignments()) {
143  std::set<storm::expressions::Variable> rhsVariables = assignment.getAssignedExpression().getVariables();
144 
145  if (!rhsVariables.empty()) {
146  uint64_t blockToKeep = variableToLocalBlockIndex.at(*rhsVariables.begin());
147  for (auto const& variable : rhsVariables) {
148  uint64_t block = variableToLocalBlockIndex.at(variable);
149  if (block != blockToKeep) {
150  for (auto const& blockIndex : relevantBlockPartition[block]) {
151  for (auto const& variable : localExpressionInformation.getVariableBlockWithIndex(blockIndex)) {
152  variableToLocalBlockIndex[variable] = blockToKeep;
153  }
154  }
155  relevantBlockPartition[blockToKeep].insert(relevantBlockPartition[block].begin(), relevantBlockPartition[block].end());
156  relevantBlockPartition[block].clear();
157  }
158  }
159  }
160  }
161  }
162 
163  // Proceed by relating the blocks via assignment-variables and the expressions of their assigned expressions.
164  bool changed = false;
165  do {
166  changed = false;
167  for (auto const& destination : edge.get().getDestinations()) {
168  for (auto const& assignment : destination.getOrderedAssignments().getAllAssignments()) {
169  std::set<storm::expressions::Variable> rhsVariables = assignment.getAssignedExpression().getVariables();
170

172  storm::expressions::Variable const& representativeVariable = *rhsVariables.begin();
173  uint64_t representativeBlock = variableToLocalBlockIndex.at(representativeVariable);
174  uint64_t assignmentVariableBlock = variableToLocalBlockIndex.at(assignment.getExpressionVariable());
175 
176  // If the blocks are different, we merge them now
177  if (assignmentVariableBlock != representativeBlock) {
178  changed = true;
179 
180  for (auto const& blockIndex : relevantBlockPartition[assignmentVariableBlock]) {
181  for (auto const& variable : localExpressionInformation.getVariableBlockWithIndex(blockIndex)) {
182  variableToLocalBlockIndex[variable] = representativeBlock;
183  }
184  }
185  relevantBlockPartition[representativeBlock].insert(relevantBlockPartition[assignmentVariableBlock].begin(), relevantBlockPartition[assignmentVariableBlock].end());
186  relevantBlockPartition[assignmentVariableBlock].clear();
187  }
188  }
189  }
190  }
191  } while (changed);
192 
193  // Now remove all blocks that are empty and obtain the partition.
194  std::vector<std::set<uint64_t>> cleanedRelevantBlockPartition;
195  for (auto& outerBlock : relevantBlockPartition) {
196  if (!outerBlock.empty()) {
197  cleanedRelevantBlockPartition.emplace_back();
198 
199  for (auto const& innerBlock : outerBlock) {
200  if (!localExpressionInformation.getExpressionBlock(innerBlock).empty()) {
201  cleanedRelevantBlockPartition.back().insert(innerBlock);
202  }
203  }
204

206  cleanedRelevantBlockPartition.pop_back();
207  }
208  }
209  }

233  // are not contained within a single block of our decomposition.
234  bool enumerateAbstractGuard = true;

236  for (auto const& block : relevantBlockPartition) {
237  bool allContained = true;
238  for (auto const& guardBlock : guardBlocks) {
239  if (block.find(guardBlock) == block.end()) {
240  allContained = false;
241  break;
242  }
243  }
244  if (allContained) {

254  std::set<uint64_t> relatedGuardPredicates = localExpressionInformation.getRelatedExpressions(variablesContainedInGuard);
255  std::vector<storm::expressions::Variable> guardDecisionVariables;
256  std::vector<std::pair<storm::expressions::Variable, uint_fast64_t>> guardVariablesAndPredicates;
257  for (auto const& element : relevantPredicatesAndVariables.first) {
258  if (relatedGuardPredicates.find(element.second) != relatedGuardPredicates.end()) {
259  guardDecisionVariables.push_back(element.first);

261  }

266  ++numberOfSolutions;
267  return true;
268  });

270 
271  // Now that we have the abstract guard, we can add it as an assertion to the solver before enumerating
272  // the other solutions.
273 
274  // Create a new backtracking point before adding the guard.
275  smtSolver->push();
276 
277  // Create the guard constraint.
278  std::pair<std::vector<storm::expressions::Expression>, std::unordered_map<uint_fast64_t, storm::expressions::Variable>> result = abstractGuard.toExpression(this->getAbstractionInformation().getExpressionManager());
279 
280  // Then add it to the solver.
281  for (auto const& expression : result.first) {
282  smtSolver->add(expression);
283  }
284 
285  // Finally associate the level variables with the predicates.
286  for (auto const& indexVariablePair : result.second) {
287  smtSolver->add(storm::expressions::iff(indexVariablePair.second, this->getAbstractionInformation().getPredicateForDdVariableIndex(indexVariablePair.first)));
288  }
289  }

291  // Then enumerate the solutions for each of the blocks of the decomposition
292  uint64_t usedNondeterminismVariables = 0;
293  uint64_t blockCounter = 0;
294  std::vector<storm::dd::Bdd<DdType>> blockBdds;
295  for (auto const& block : relevantBlockPartition) {
296  std::set<uint64_t> relevantPredicates;
297  for (auto const& innerBlock : block) {
298  relevantPredicates.insert(localExpressionInformation.getExpressionBlock(innerBlock).begin(), localExpressionInformation.getExpressionBlock(innerBlock).end());
299  }
300 
301  if (relevantPredicates.empty()) {
302  STORM_LOG_TRACE("Block does not contain relevant predicates, skipping it.");
303  continue;
304  }
305 
306  std::vector<storm::expressions::Variable> transitionDecisionVariables;
307  std::vector<std::pair<storm::expressions::Variable, uint_fast64_t>> sourceVariablesAndPredicates;
308  for (auto const& element : relevantPredicatesAndVariables.first) {
309  if (relevantPredicates.find(element.second) != relevantPredicates.end()) {
310  transitionDecisionVariables.push_back(element.first);
311  sourceVariablesAndPredicates.push_back(element);
312  }
313  }
314 
315  std::vector<std::vector<std::pair<storm::expressions::Variable, uint_fast64_t>>> destinationVariablesAndPredicates;
316  for (uint64_t destinationIndex = 0; destinationIndex < edge.get().getNumberOfDestinations(); ++destinationIndex) {
317  destinationVariablesAndPredicates.emplace_back();
318  for (auto const& assignment : edge.get().getDestination(destinationIndex).getOrderedAssignments().getAllAssignments()) {
319  uint64_t assignmentVariableBlockIndex = localExpressionInformation.getBlockIndexOfVariable(assignment.getVariable().getExpressionVariable());
320 
321  if (block.find(assignmentVariableBlockIndex) != block.end()) {
322  std::set<uint64_t> const& assignmentVariableBlock = localExpressionInformation.getExpressionBlock(assignmentVariableBlockIndex);
323  for (auto const& element : relevantPredicatesAndVariables.second[destinationIndex]) {
324  if (assignmentVariableBlock.find(element.second) != assignmentVariableBlock.end()) {
325  destinationVariablesAndPredicates.back().push_back(element);
326  transitionDecisionVariables.push_back(element.first);
327  }
328  }
329  }
330  }

334  numberOfSolutions = 0;

337  ++numberOfSolutions;
338  return true;
339  });

341  numberOfTotalSolutions += numberOfSolutions;
342 
343  // Now we search for the maximal number of choices of player 2 to determine how many DD variables we
344  // need to encode the nondeterminism.
345  uint_fast64_t maximalNumberOfChoices = 0;
346  for (auto const& sourceDistributionsPair : sourceToDistributionsMap) {
347  maximalNumberOfChoices = std::max(maximalNumberOfChoices, static_cast<uint_fast64_t>(sourceDistributionsPair.second.size()));
348  }
349 
350  // We now compute how many variables we need to encode the choices. We add one to the maximal number of
351  // choices to account for a possible transition to a bottom state.
352  uint_fast64_t numberOfVariablesNeeded = (maximalNumberOfChoices > 1) ? (static_cast<uint_fast64_t>(std::ceil(std::log2(maximalNumberOfChoices + (blockCounter == 0 ? 1 : 0))))) : (blockCounter == 0 ? 1 : 0);
353 
354  // Finally, build overall result.
355  storm::dd::Bdd<DdType> resultBdd = this->getAbstractionInformation().getDdManager().getBddZero();
356 
357  for (auto const& sourceDistributionsPair : sourceToDistributionsMap) {
358  STORM_LOG_ASSERT(!sourceDistributionsPair.first.isZero(), "The source BDD must not be empty.");
359  STORM_LOG_ASSERT(!sourceDistributionsPair.second.empty(), "The distributions must not be empty.");

362  uint_fast64_t distributionIndex = blockCounter == 0 ? 1 : 0;

366  ++distributionIndex;
367  STORM_LOG_ASSERT(!allDistributions.isZero(), "The BDD must not be empty.");
368  }

370  STORM_LOG_ASSERT(!resultBdd.isZero(), "The BDD must not be empty.");
371  }
372  usedNondeterminismVariables += numberOfVariablesNeeded;
373 
374  blockBdds.push_back(resultBdd);
375  ++blockCounter;
376  }
377

380  }
381 
382  // multiply the results
383  storm::dd::Bdd<DdType> resultBdd = getAbstractionInformation().getDdManager().getBddOne();

385  for (auto const& blockBdd : blockBdds) {

387  ++blockIndex;
388  }

415 
416  auto end = std::chrono::high_resolution_clock::now();
417 
418  STORM_LOG_TRACE("Enumerated " << numberOfTotalSolutions << " solutions in " << std::chrono::duration_cast<std::chrono::milliseconds>(end - start).count() << "ms.");

431  sourceToDistributionsMap[getSourceStateBdd(model, relevantPredicatesAndVariables.first)].push_back(getDistributionBdd(model, relevantPredicatesAndVariables.second));
432  ++numberOfSolutions;

434  });
435 
436  // Now we search for the maximal number of choices of player 2 to determine how many DD variables we
437  // need to encode the nondeterminism.

483  template <storm::dd::DdType DdType, typename ValueType>
484  std::pair<std::set<uint_fast64_t>, std::set<uint_fast64_t>> EdgeAbstractor<DdType, ValueType>::computeRelevantPredicates(storm::jani::OrderedAssignments const& assignments) const {
485  std::pair<std::set<uint_fast64_t>, std::set<uint_fast64_t>> result;
486 
487  std::set<storm::expressions::Variable> assignedVariables;
488  for (auto const& assignment : assignments.getAllAssignments()) {
489  // Also, variables appearing on the right-hand side of an assignment are relevant for source state.
490  auto const& rightHandSidePredicates = localExpressionInformation.getExpressionsUsingVariables(assignment.getAssignedExpression().getVariables());
491  result.first.insert(rightHandSidePredicates.begin(), rightHandSidePredicates.end());
492 
493  // Variables that are being assigned are relevant for the successor state.
494  storm::expressions::Variable const& assignedVariable = assignment.getExpressionVariable();
495  auto const& leftHandSidePredicates = localExpressionInformation.getExpressionsUsingVariable(assignedVariable);
496  result.second.insert(leftHandSidePredicates.begin(), leftHandSidePredicates.end());
497 
498  // Predicates that are indirectly related to the assigned variables are relevant for the source state (if requested).
499  if (this->addPredicatesForValidBlocks) {
