109  template <typename ValueType, bool Nondeterministic>
110  std::vector<ValueType> SparseInfiniteHorizonHelper<ValueType, Nondeterministic>::computeLongRunAverageValues(Environment const& env, ValueGetter const& stateRewardsGetter, ValueGetter const& actionRewardsGetter) {
111  // We will compute the long run average value for each MEC individually and then set-up an Equation system to compute the value also at non-mec states.
112  // For a description of this approach see, e.g., Guck et al.: Modelling and Analysis of Markov Reward Automata (ATVA'14), https://doi.org/10.1007/978-3-319-11936-6_13
113 
114  // Prepare an environment for the underlying solvers.
115  auto underlyingSolverEnvironment = env;
116  if (env.solver().isForceSoundness()) {
117  // For sound computations, the error in the MECS plus the error in the remaining system should not exceed the user defined precsion.
118  storm::RationalNumber newPrecision = env.solver().lra().getPrecision() / storm::utility::convertNumber<storm::RationalNumber>(2);
119  underlyingSolverEnvironment.solver().minMax().setPrecision(newPrecision);
120  underlyingSolverEnvironment.solver().minMax().setRelativeTerminationCriterion(env.solver().lra().getRelativeTerminationCriterion());
121  underlyingSolverEnvironment.solver().setLinearEquationSolverPrecision(newPrecision, env.solver().lra().getRelativeTerminationCriterion());
122  underlyingSolverEnvironment.solver().lra().setPrecision(newPrecision);
123  }
124 
125  // If requested, allocate memory for the choices made
126  if (Nondeterministic && this->isProduceSchedulerSet()) {
127  if (!_producedOptimalChoices.is_initialized()) {
128  _producedOptimalChoices.emplace();
129  }
130  _producedOptimalChoices->resize(_transitionMatrix.getRowGroupCount());
131  }
132  STORM_LOG_ASSERT(Nondeterministic || !this->isProduceSchedulerSet(), "Scheduler production enabled for deterministic model.");
133 
134  // Decompose the model to their bottom components (MECS or BSCCS)
135  createDecomposition();
136 
137  // Compute the long-run average for all components in isolation.
138  // Set up some logging
139  std::string const componentString = (Nondeterministic ? std::string("Maximal end") : std::string("Bottom strongly connected")) + (_longRunComponentDecomposition->size() == 1 ? std::string(" component") : std::string(" components"));
140  storm::utility::ProgressMeasurement progress(componentString);
141  progress.setMaxCount( _longRunComponentDecomposition->size());
142  progress.startNewMeasurement(0);
143  STORM_LOG_INFO("Computing long run average values for " << _longRunComponentDecomposition->size() << " " << componentString << " individually...");
144  std::vector<ValueType> componentLraValues;
145  componentLraValues.reserve(_longRunComponentDecomposition->size());
146  for (auto const& c : *_longRunComponentDecomposition) {
147  componentLraValues.push_back(computeLraForComponent(underlyingSolverEnvironment, stateRewardsGetter, actionRewardsGetter, c));
148  progress.updateProgress(componentLraValues.size());
149  }
150 
151  // Solve the resulting SSP where end components are collapsed into single auxiliary states
152  STORM_LOG_INFO("Solving stochastic shortest path problem.");
153  return buildAndSolveSsp(underlyingSolverEnvironment, componentLraValues);
154  }
