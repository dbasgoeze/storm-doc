285  // This is because it could be that the commands are taken to enable other synchronizations. Therefore, we need
286  // to add an additional clause that says that the right-hand side of the implication is also true if all commands
287  // of the current choice have enabled synchronization options.
288  if (labelSet.size() > 1) {

309  }

312  }

315  } else {

551  for (uint_fast64_t commandIndex = 0; commandIndex < module.getNumberOfCommands(); ++commandIndex) {
552  storm::prism::Command const& command = module.getCommand(commandIndex);
553 
554  // If the current command is one of the commands we need to consider, store a reference to it in the container.
555  if (precedingLabelSet.find(command.getGlobalIndex()) != precedingLabelSet.end()) {
556  preceedingGuardConjunction = preceedingGuardConjunction && command.getGuardExpression();
557 
558  currentPreceedingVariableUpdates.emplace_back();
559 
560  for (uint64_t updateIndex = 0; updateIndex < command.getNumberOfUpdates(); ++updateIndex) {
561  storm::prism::Update const& update = command.getUpdate(updateIndex);
562  boost::container::flat_map<storm::expressions::Variable, storm::expressions::Expression> variableUpdates;
563  for (auto const& assignment : update.getAssignments()) {
564  variableUpdates.emplace(assignment.getVariable(), assignment.getExpression());
565  }
566  currentPreceedingVariableUpdates.back().emplace_back(std::move(variableUpdates));
567  }
568  }
569  }
570  }
571  } else {
572  storm::jani::Model const& janiModel = symbolicModel.asJaniModel();
573  for (uint_fast64_t automatonIndex = 0; automatonIndex < janiModel.getNumberOfAutomata(); ++automatonIndex) {
574  storm::jani::Automaton const& automaton = janiModel.getAutomaton(automatonIndex);
575 
576  for (uint_fast64_t edgeIndex = 0; edgeIndex < automaton.getNumberOfEdges(); ++edgeIndex) {
577  // If the current command is one of the commands we need to consider, store a reference to it in the container.
578  if (precedingLabelSet.find(janiModel.encodeAutomatonAndEdgeIndices(automatonIndex, edgeIndex)) != precedingLabelSet.end()) {
579  storm::jani::Edge const& edge = automaton.getEdge(edgeIndex);
580 
581  preceedingGuardConjunction = preceedingGuardConjunction && edge.getGuard();
582 
583  currentPreceedingVariableUpdates.emplace_back();
584 
585  for (uint64_t destinationIndex = 0; destinationIndex < edge.getNumberOfDestinations(); ++destinationIndex) {
586  storm::jani::EdgeDestination const& destination = edge.getDestination(destinationIndex);
587  boost::container::flat_map<storm::expressions::Variable, storm::expressions::Expression> variableUpdates;
588  for (auto const& assignment : destination.getOrderedAssignments().getNonTransientAssignments()) {
589  variableUpdates.emplace(assignment.getVariable().getExpressionVariable(), assignment.getAssignedExpression());
590  }
591  currentPreceedingVariableUpdates.back().emplace_back(std::move(variableUpdates));
592  }
593  }
594  }
595  }
596  }
597 
598  //std::cout << "pgc: " << preceedingGuardConjunction << std::endl;
599 
600  // Assert all the guards of the preceding command set.
601  localSolver->add(preceedingGuardConjunction);
602 
603  std::vector<std::vector<boost::container::flat_map<storm::expressions::Variable, storm::expressions::Expression>>::const_iterator> iteratorVector;
604  for (auto const& variableUpdates : currentPreceedingVariableUpdates) {
605  iteratorVector.push_back(variableUpdates.begin());
606  }
607 
608  // Iterate over all possible combinations of updates of the preceding command set.
609  std::vector<storm::expressions::Expression> formulae;
610  while (true) {
611  std::map<storm::expressions::Variable, storm::expressions::Expression> currentVariableUpdateCombinationMap;
612  for (auto const& updateIterator : iteratorVector) {
613  for (auto const& variableUpdatePair : *updateIterator) {
614  currentVariableUpdateCombinationMap.emplace(variableUpdatePair.first, variableUpdatePair.second);
615  }
616  }
617 
618  STORM_LOG_DEBUG("About to assert a weakest precondition.");
619  storm::expressions::Expression wp = guardConjunction.substitute(currentVariableUpdateCombinationMap);
620  formulae.push_back(wp);
621  STORM_LOG_DEBUG("Asserted weakest precondition.");
622 
623  // Now try to move iterators to the next position if possible. If we could properly
624  // move it, we can directly move on to the next combination of updates. If we have
625  // to reset it to the start, we do so unless there is no more iterator to reset.
626  uint_fast64_t k = iteratorVector.size();
627  for (; k > 0; --k) {
628  ++iteratorVector[k - 1];
629  if (iteratorVector[k - 1] == currentPreceedingVariableUpdates[k - 1].end()) {
630  iteratorVector[k - 1] = currentPreceedingVariableUpdates[k - 1].begin();
631  } else {
632  break;
633  }
634  }
635 
636  // If we had to reset all iterator to the start, we are done.
637  if (k == 0) {
638  break;
639  }
640  }
641 
642  // Now assert the disjunction of all weakest preconditions of all considered update combinations.
643  assertDisjunction(*localSolver, formulae, symbolicModel.isPrismProgram() ? symbolicModel.asPrismProgram().getManager() : symbolicModel.asJaniModel().getManager());
644 
645  STORM_LOG_DEBUG("Asserted disjunction of all weakest preconditions.");
646  storm::solver::SmtSolver::CheckResult result = localSolver->check();
647 
648  if (result == storm::solver::SmtSolver::CheckResult::Sat) {
649  backwardImplications[labelSetAndPrecedingLabelSetsPair.first].insert(precedingLabelSet);
650  backwardImplicationAdded = true;
651  } else if (result == storm::solver::SmtSolver::CheckResult::Unknown) {
652  STORM_LOG_ERROR("The SMT solver does not come to a conclusive answer. Does your model contain integer division?");
653  }
654 
655  localSolver->pop();
656  }
657 
658  // Popping the disjunction of negated guards from the solver stack.
659  localSolver->pop();
660  STORM_LOG_ERROR_COND(backwardImplicationAdded, "Error in adding cuts for counterexample generation (backward implication misses a label set).");
661  } else {
662  STORM_LOG_DEBUG("Selection is enabled in initial state.");
663  }
664 
665  }
666  } else if (symbolicModel.isJaniModel()) {
667  STORM_LOG_WARN("Model uses assignment levels, did not assert backward implications.");
668  }
669 
670  STORM_LOG_DEBUG("Successfully gathered data for cuts.");
671 
672  // Compute the sets of labels such that the transitions labeled with this set possess at least one known label.
673  storm::storage::FlatSet<storm::storage::FlatSet<uint_fast64_t>> hasKnownSuccessor;
674  for (auto const& labelSetFollowingSetsPair : followingLabels) {
675  for (auto const& set : labelSetFollowingSetsPair.second) {
676  if (std::includes(relevancyInformation.knownLabels.begin(), relevancyInformation.knownLabels.end(), set.begin(), set.end())) {
677  hasKnownSuccessor.insert(set);
678  break;
679  }
680  }
681  }
682 
683  // Compute the sets of labels such that the transitions labeled with this set possess at least one known predecessor.
684  storm::storage::FlatSet<storm::storage::FlatSet<uint_fast64_t>> hasKnownPredecessor;
685  for (auto const& labelSetImplicationsPair : backwardImplications) {
686  for (auto const& set : labelSetImplicationsPair.second) {
687  if (std::includes(relevancyInformation.knownLabels.begin(), relevancyInformation.knownLabels.end(), set.begin(), set.end())) {
688  hasKnownPredecessor.insert(labelSetImplicationsPair.first);
689  break;
690  }
691  }
692  }
693 
694  STORM_LOG_DEBUG("Asserting initial combination is taken.");
695  {
696  std::vector<storm::expressions::Expression> formulae;
697 
698  // Start by asserting that we take at least one initial label. We may do so only if there is no initial
699  // combination that is already known. Otherwise this condition would be too strong.
700  bool initialCombinationKnown = false;
701  for (auto const& combination : initialCombinations) {
702  storm::storage::FlatSet<uint_fast64_t> tmpSet;
703  std::set_difference(combination.begin(), combination.end(), relevancyInformation.knownLabels.begin(), relevancyInformation.knownLabels.end(), std::inserter(tmpSet, tmpSet.end()));
704  if (tmpSet.size() == 0) {
705  initialCombinationKnown = true;
706  break;
707  } else {
708  storm::expressions::Expression conj = variableInformation.manager->boolean(true);
709  for (auto label : tmpSet) {
710  conj = conj && variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(label));
711  }
712  formulae.push_back(conj);
713  }
714  }
715  if (!initialCombinationKnown) {
716  assertDisjunction(solver, formulae, *variableInformation.manager);
717  }
718  }
719 
720  STORM_LOG_DEBUG("Asserting target combination is taken.");
721  {
722  std::vector<storm::expressions::Expression> formulae;
723 
724  // Likewise, if no target combination is known, we may assert that there is at least one.
725  bool targetCombinationKnown = false;
726  for (auto const& combination : targetCombinations) {
727  storm::storage::FlatSet<uint_fast64_t> tmpSet;
728  std::set_difference(combination.begin(), combination.end(), relevancyInformation.knownLabels.begin(), relevancyInformation.knownLabels.end(), std::inserter(tmpSet, tmpSet.end()));
729  if (tmpSet.size() == 0) {
730  targetCombinationKnown = true;
731  break;
732  } else {
733  storm::expressions::Expression conj = variableInformation.manager->boolean(true);
734  for (auto label : tmpSet) {
735  conj = conj && variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(label));
736  }
737  formulae.push_back(conj);
738  }
739  }
740  if (!targetCombinationKnown) {
741  assertDisjunction(solver, formulae, *variableInformation.manager);
742  }
743  }
744

746

748  boost::container::flat_map<storm::storage::FlatSet<uint_fast64_t>, storm::expressions::Expression> labelSetToFormula;
749  for (auto const &labelSet : relevancyInformation.relevantLabelSets) {
750  storm::expressions::Expression labelSetFormula = variableInformation.manager->boolean(false);
751 
752  // Compute the set of unknown labels on the left-hand side of the implication.
753  storm::storage::FlatSet<uint_fast64_t> unknownLhsLabels;
754  std::set_difference(labelSet.begin(), labelSet.end(), relevancyInformation.knownLabels.begin(), relevancyInformation.knownLabels.end(), std::inserter(unknownLhsLabels, unknownLhsLabels.end()));
755  for (auto label : unknownLhsLabels) {
756  labelSetFormula = labelSetFormula || !variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(label));
757  }
758 
759  // Only build a constraint if the combination does not lead to a target state and
760  // no successor set is already known.
761  storm::expressions::Expression successorExpression;
762  if (targetCombinations.find(labelSet) == targetCombinations.end() && hasKnownSuccessor.find(labelSet) == hasKnownSuccessor.end()) {
763  successorExpression = variableInformation.manager->boolean(false);
764 
765  auto const &followingLabelSets = followingLabels.at(labelSet);
766 
767  for (auto const &followingSet : followingLabelSets) {
768  storm::storage::FlatSet<uint_fast64_t> tmpSet;
769 
770  // Check which labels of the current following set are not known. This set must be non-empty, because
771  // otherwise a successor combination would already be known and control cannot reach this point.
772  std::set_difference(followingSet.begin(), followingSet.end(), relevancyInformation.knownLabels.begin(), relevancyInformation.knownLabels.end(), std::inserter(tmpSet, tmpSet.end()));
773 
774  // Construct an expression that enables all unknown labels of the current following set.
775  storm::expressions::Expression conj = variableInformation.manager->boolean(true);
776  for (auto label : tmpSet) {
777  conj = conj && variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(label));
778  }
779  successorExpression = successorExpression || conj;
780  }
781  } else {
782  successorExpression = variableInformation.manager->boolean(true);
783  }
784  // Constructed following cuts at this point.
785 
786  // Only build a constraint if the combination is no initial combination and no
787  // predecessor set is already known.
788  storm::expressions::Expression predecessorExpression;
789  if (initialCombinations.find(labelSet) == initialCombinations.end() && hasKnownPredecessor.find(labelSet) == hasKnownPredecessor.end()) {
790  predecessorExpression = variableInformation.manager->boolean(false);
791 
792 // std::cout << "labelSet" << std::endl;
793 // for (auto const& e : labelSet) {
794 // std::cout << e << ", ";
795 // }
796 // std::cout << std::endl;
797 
798  auto const &preceedingLabelSets = backwardImplications.at(labelSet);
799 
800  for (auto const &preceedingSet : preceedingLabelSets) {
801  storm::storage::FlatSet<uint_fast64_t> tmpSet;
802 
803  // Check which labels of the current following set are not known. This set must be non-empty, because
804  // otherwise a predecessor combination would already be known and control cannot reach this point.
805  std::set_difference(preceedingSet.begin(), preceedingSet.end(), relevancyInformation.knownLabels.begin(), relevancyInformation.knownLabels.end(), std::inserter(tmpSet, tmpSet.end()));
806 
807  // Construct an expression that enables all unknown labels of the current following set.
808  storm::expressions::Expression conj = variableInformation.manager->boolean(true);
809  for (auto label : tmpSet) {
810  conj = conj && variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(label));
811  }
812  predecessorExpression = predecessorExpression || conj;
813  }
814  } else {
815  predecessorExpression = variableInformation.manager->boolean(true);
816  }
817 
818  labelSetFormula = labelSetFormula || (successorExpression && predecessorExpression);
819 
820  labelSetToFormula[labelSet] = labelSetFormula;
821  }
822 
823  for (auto const &labelSetFormula : labelSetToFormula) {
824  solver.add(labelSetFormula.second || getOtherSynchronizationEnabledFormula(labelSetFormula.first, synchronizingLabels, labelSetToFormula, variableInformation, relevancyInformation));
825  }
826  }
827 
828  STORM_LOG_DEBUG("Asserting synchronization cuts.");
829  // Finally, assert that if we take one of the synchronizing labels, we also take one of the combinations
830  // the label appears in.
831  for (auto const& labelSynchronizingSetsPair : synchronizingLabels) {
832  std::vector<storm::expressions::Expression> formulae;
833 
834  if (relevancyInformation.knownLabels.find(labelSynchronizingSetsPair.first) == relevancyInformation.knownLabels.end()) {
835  formulae.push_back(!variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(labelSynchronizingSetsPair.first)));
836  }
837 
838  // We need to be careful, because there may be one synchronisation set out of which all labels are
839  // known, which means we must not assert anything.
840  bool allImplicantsKnownForOneSet = false;
841  for (auto const& synchronizingSet : labelSynchronizingSetsPair.second) {
842  storm::expressions::Expression currentCombination = variableInformation.manager->boolean(true);
843  bool allImplicantsKnownForCurrentSet = true;
844  for (auto label : synchronizingSet) {
845  if (relevancyInformation.knownLabels.find(label) == relevancyInformation.knownLabels.end() && label != labelSynchronizingSetsPair.first) {
846  currentCombination = currentCombination && variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(label));
847  }
848  }
849  formulae.push_back(currentCombination);
850 
851  // If all implicants of the current set are known, we do not need to further build the constraint.
852  if (allImplicantsKnownForCurrentSet) {
853  allImplicantsKnownForOneSet = true;
854  break;
855  }
856  }
857 
858  if (!allImplicantsKnownForOneSet) {
859  assertDisjunction(solver, formulae, *variableInformation.manager);
860  }
861  }
862 
863  auto endTime = std::chrono::high_resolution_clock::now();
864  return std::chrono::duration_cast<std::chrono::milliseconds>(endTime - assertCutsClock);
865  }

870  static void assertReachabilityCuts(storm::models::sparse::Model<T> const& model, std::vector<storm::storage::FlatSet<uint_fast64_t>> const& labelSets, storm::storage::BitVector const& psiStates, VariableInformation const& variableInformation, RelevancyInformation const& relevancyInformation, storm::solver::SmtSolver& solver) {
871 
872  if (!variableInformation.hasReachabilityVariables) {
873  throw storm::exceptions::InvalidStateException() << "Impossible to assert reachability cuts without the necessary variables.";
874  }
875 
876  // Get some data from the model for convenient access.
877  storm::storage::SparseMatrix<T> const& transitionMatrix = model.getTransitionMatrix();
878  storm::storage::SparseMatrix<T> backwardTransitions = model.getBackwardTransitions();
879 
880  // First, we add the formulas that encode
881  // (1) if an incoming transition is chosen, an outgoing one is chosen as well (for non-initial states)
882  // (2) an outgoing transition out of the initial states is taken.
883  storm::expressions::Expression initialStateExpression = variableInformation.manager->boolean(false);
884  for (auto relevantState : relevancyInformation.relevantStates) {
885  if (!model.getInitialStates().get(relevantState)) {
886  // Assert the constraints (1).
887  storm::storage::FlatSet<uint_fast64_t> relevantPredecessors;
888  for (auto const& predecessorEntry : backwardTransitions.getRow(relevantState)) {
889  if (relevantState != predecessorEntry.getColumn() && relevancyInformation.relevantStates.get(predecessorEntry.getColumn())) {
890  relevantPredecessors.insert(predecessorEntry.getColumn());
891  }
892  }
893 
894  storm::storage::FlatSet<uint_fast64_t> relevantSuccessors;
895  for (auto const& relevantChoice : relevancyInformation.relevantChoicesForRelevantStates.at(relevantState)) {
896  for (auto const& successorEntry : transitionMatrix.getRow(relevantChoice)) {
897  if (relevantState != successorEntry.getColumn() && (relevancyInformation.relevantStates.get(successorEntry.getColumn()) || psiStates.get(successorEntry.getColumn()))) {
898  relevantSuccessors.insert(successorEntry.getColumn());
899  }
900  }
901  }
902 
903  storm::expressions::Expression expression = variableInformation.manager->boolean(true);
904  for (auto predecessor : relevantPredecessors) {
905  expression = expression && !variableInformation.statePairVariables.at(variableInformation.statePairToIndexMap.at(std::make_pair(predecessor, relevantState)));
906  }
907  for (auto successor : relevantSuccessors) {
908  expression = expression || variableInformation.statePairVariables.at(variableInformation.statePairToIndexMap.at(std::make_pair(relevantState, successor)));
909  }
910 
911  solver.add(expression);
912  } else {
913  // Assert the constraints (2).
914  storm::storage::FlatSet<uint_fast64_t> relevantSuccessors;
915  for (auto const& relevantChoice : relevancyInformation.relevantChoicesForRelevantStates.at(relevantState)) {
916  for (auto const& successorEntry : transitionMatrix.getRow(relevantChoice)) {
917  if (relevantState != successorEntry.getColumn() && (relevancyInformation.relevantStates.get(successorEntry.getColumn()) || psiStates.get(successorEntry.getColumn()))) {
918  relevantSuccessors.insert(successorEntry.getColumn());
919  }
920  }
921  }
922 
923  for (auto successor : relevantSuccessors) {
924  initialStateExpression = initialStateExpression || variableInformation.statePairVariables.at(variableInformation.statePairToIndexMap.at(std::make_pair(relevantState, successor)));
925  }
926  }
927  }
928  solver.add(initialStateExpression);
929 
930  // Finally, add constraints that
931  // (1) if a transition is selected, a valid labeling is selected as well.
932  // (2) enforce that if a transition from s to s' is selected, the ordering variables become strictly larger.
933  for (auto const& statePairIndexPair : variableInformation.statePairToIndexMap) {
934  uint_fast64_t sourceState = statePairIndexPair.first.first;
935  uint_fast64_t targetState = statePairIndexPair.first.second;
936 
937  // Assert constraint for (1).
938  storm::storage::FlatSet<uint_fast64_t> choicesForStatePair;
939  for (auto const& relevantChoice : relevancyInformation.relevantChoicesForRelevantStates.at(sourceState)) {
940  for (auto const& successorEntry : transitionMatrix.getRow(relevantChoice)) {
941  if (successorEntry.getColumn() == targetState) {
942  choicesForStatePair.insert(relevantChoice);
943  }
944  }
945  }
946  storm::expressions::Expression labelExpression = !variableInformation.statePairVariables.at(statePairIndexPair.second);
947  for (auto choice : choicesForStatePair) {
948  storm::expressions::Expression choiceExpression = variableInformation.manager->boolean(true);
949  for (auto element : labelSets.at(choice)) {
950  if (relevancyInformation.knownLabels.find(element) == relevancyInformation.knownLabels.end()) {
951  choiceExpression = choiceExpression && variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(element));
952  }
953  }
954  labelExpression = labelExpression || choiceExpression;
955  }
956  solver.add(labelExpression);
957 
958  // Assert constraint for (2).
959  storm::expressions::Expression orderExpression = !variableInformation.statePairVariables.at(statePairIndexPair.second) || variableInformation.stateOrderVariables.at(variableInformation.relevantStatesToOrderVariableIndexMap.at(sourceState)).getExpression() < variableInformation.stateOrderVariables.at(variableInformation.relevantStatesToOrderVariableIndexMap.at(targetState)).getExpression();
960  solver.add(orderExpression);
961  }
962  }

1852  lastSize = commandSet.size();
1853  } else {
1854  STORM_LOG_DEBUG("Lower bound on label set size is " << currentBound << " after " << milliseconds << "ms (checked " << iterations << " models, " << zeroProbabilityCount << " could not reach the target set).");

1856  }
1857  }

1861  totalTime = std::chrono::high_resolution_clock::now() - totalClock;
1862

1865  stats.modelCheckingTime = std::chrono::duration_cast<std::chrono::milliseconds>(totalModelCheckingTime);
1866  stats.solverTime = std::chrono::duration_cast<std::chrono::milliseconds>(totalSolverTime);
1867  stats.iterations = iterations;

1900  static void extendLabelSetLowerBound(storm::models::sparse::Model<T> const& model, storm::storage::FlatSet<uint_fast64_t>& commandSet, storm::storage::BitVector const& phiStates, storm::storage::BitVector const& psiStates, bool silent = false) {
1901  auto startTime = std::chrono::high_resolution_clock::now();
1902 
1903  // Create sub-model that only contains the choices allowed by the given command set.
1904  std::shared_ptr<storm::models::sparse::Model<T>> subModel = restrictModelToLabelSet(model, commandSet).first;
1905 
1906  // Then determine all prob0E(psi) states that are reachable in the sub-model.
1907  storm::storage::BitVector reachableProb0EStates = storm::utility::graph::getReachableStates(subModel->getTransitionMatrix(), subModel->getInitialStates(), phiStates, psiStates);
1908 
1909  // Create a queue of reachable prob0E(psi) states so we can check which commands need to be added
1910  // to give them a strategy that avoids psi states.
1911  std::queue<uint_fast64_t> prob0EWorklist;
1912  for (auto const& e : reachableProb0EStates) {
1913  prob0EWorklist.push(e);
1914  }
1915 
1916  // As long as there are reachable prob0E(psi) states, we add commands so they can stay within
1917  // prob0E(states).
1918  while (!prob0EWorklist.empty()) {
1919  uint_fast64_t state = prob0EWorklist.front();
1920  prob0EWorklist.pop();
1921 
1922  // Now iterate over the original choices of the prob0E(psi) state and add at least one.
1923  bool hasLabeledChoice = false;
1924  uint64_t smallestCommandSetSize = 0;
1925  uint64_t smallestCommandChoice = model.getTransitionMatrix().getRowGroupIndices()[state];
1926 
1927  // Determine the choice with the least amount of commands (bad heuristic).
1928  for (uint64_t choice = smallestCommandChoice; choice < model.getTransitionMatrix().getRowGroupIndices()[state + 1]; ++choice) {
1929  bool onlyProb0ESuccessors = true;
1930  for (auto const& successorEntry : model.getTransitionMatrix().getRow(choice)) {
1931  if (!psiStates.get(successorEntry.getColumn())) {
1932  onlyProb0ESuccessors = false;
1933  break;
1934  }
1935  }
1936 
1937  if (onlyProb0ESuccessors) {
1938  uint64_t labelSetSize = 0;
1939  if (model.getChoiceOrigins()->isPrismChoiceOrigins()) {
1940  labelSetSize = model.getChoiceOrigins()->asPrismChoiceOrigins().getCommandSet(choice).size();
1941  } else {
1942  assert(model.getChoiceOrigins()->isJaniChoiceOrigins());
1943  labelSetSize = model.getChoiceOrigins()->asJaniChoiceOrigins().getEdgeIndexSet(choice).size();
1944  }
1945  hasLabeledChoice |= (labelSetSize != 0);
1946 
1947  if (smallestCommandChoice == 0 || labelSetSize < smallestCommandSetSize) {
1948  smallestCommandSetSize = labelSetSize;
1949  smallestCommandChoice = choice;
1950  }
1951  }
1952  }
1953 
1954  if (hasLabeledChoice) {
1955  // Take all labels of the selected choice.
1956  if (model.getChoiceOrigins()->isPrismChoiceOrigins()) {
1957  auto const& labelSet = model.getChoiceOrigins()->asPrismChoiceOrigins().getCommandSet(smallestCommandChoice);
1958  commandSet.insert(labelSet.begin(), labelSet.end());
1959  } else {
1960  assert(model.getChoiceOrigins()->isJaniChoiceOrigins());
1961  auto const& labelSet = model.getChoiceOrigins()->asJaniChoiceOrigins().getEdgeIndexSet(smallestCommandChoice);
1962  commandSet.insert(labelSet.begin(), labelSet.end());
1963  }
1964 
1965  // Check for which successor states choices need to be added
1966  for (auto const& successorEntry : model.getTransitionMatrix().getRow(smallestCommandChoice)) {
1967  if (!storm::utility::isZero(successorEntry.getValue())) {
1968  if (!reachableProb0EStates.get(successorEntry.getColumn())) {
1969  reachableProb0EStates.set(successorEntry.getColumn());
1970  prob0EWorklist.push(successorEntry.getColumn());
1971  }
1972  }
1973  }
1974  }
1975  }
1976 
1977  auto endTime = std::chrono::high_resolution_clock::now();
1978  if (!silent) {
1979  std::cout << std::endl << "Extended command for lower bounded property to size " << commandSet.size() << " in " << std::chrono::duration_cast<std::chrono::milliseconds>(endTime - startTime).count() << "ms." << std::endl;
1980  }
