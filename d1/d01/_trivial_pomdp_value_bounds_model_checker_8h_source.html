62  std::vector<ValueType> computeValuesForGuessedScheduler(std::vector<ValueType> const& stateValues, std::vector<ValueType>* actionBasedRewards, storm::logic::Formula const& formula, storm::pomdp::analysis::FormulaInformation const& info, std::shared_ptr<storm::models::sparse::Mdp<ValueType>> underlyingMdp, ValueType const& scoreThreshold, bool relativeScore) {
63  // Create some positional scheduler for the POMDP
64  storm::storage::Scheduler<ValueType> pomdpScheduler(pomdp.getNumberOfStates());
65  // For each state, we heuristically find a good distribution over output actions.
66  auto choiceValues = getChoiceValues(stateValues, actionBasedRewards);
67  auto const& choiceIndices = pomdp.getTransitionMatrix().getRowGroupIndices();
68  std::vector<storm::storage::Distribution<ValueType, uint_fast64_t>> choiceDistributions(pomdp.getNrObservations());
69  for (uint64_t state = 0; state < pomdp.getNumberOfStates(); ++state) {
70  auto& choiceDistribution = choiceDistributions[pomdp.getObservation(state)];
71  ValueType const& stateValue = stateValues[state];
72  assert(stateValue >= storm::utility::zero<ValueType>());
73  for (auto choice = choiceIndices[state]; choice < choiceIndices[state + 1]; ++choice) {
74  ValueType const& choiceValue = choiceValues[choice];
75  assert(choiceValue >= storm::utility::zero<ValueType>());
76  // Rate this choice by considering the relative difference between the choice value and the (optimal) state value
77  // A high score shall mean that the choice is "good"
78  if (storm::utility::isInfinity(stateValue)) {
79  // For infinity states, we simply distribute uniformly.
80  // FIXME: This case could be handled a bit more sensible
81  choiceDistribution.addProbability(choice - choiceIndices[state], scoreThreshold);
82  } else {
83  ValueType choiceScore = info.minimize() ? (choiceValue - stateValue) : (stateValue - choiceValue);
84  if (relativeScore) {
85  ValueType avg = (stateValue + choiceValue) / storm::utility::convertNumber<ValueType, uint64_t>(2);
86  if (!storm::utility::isZero(avg)) {
87  choiceScore /= avg;
88  }
89  }
90  choiceScore = storm::utility::one<ValueType>() - choiceScore;
91  if (choiceScore >= scoreThreshold) {
92  choiceDistribution.addProbability(choice - choiceIndices[state], choiceScore);
93  }
94  }
95 
96  }
97  STORM_LOG_ASSERT(choiceDistribution.size() > 0, "Empty choice distribution.");
98  }
99  // Normalize all distributions
100  for (auto& choiceDistribution : choiceDistributions) {
101  choiceDistribution.normalize();
102  }
103  // Set the scheduler for all states
104  for (uint64_t state = 0; state < pomdp.getNumberOfStates(); ++state) {
105  pomdpScheduler.setChoice(choiceDistributions[pomdp.getObservation(state)], state);
106  }
107  STORM_LOG_ASSERT(!pomdpScheduler.isPartialScheduler(), "Expected a fully defined scheduler.");
108  auto scheduledModel = underlyingMdp->applyScheduler(pomdpScheduler, false);
109 
110  auto resultPtr = storm::api::verifyWithSparseEngine<ValueType>(scheduledModel, storm::api::createTask<ValueType>(formula.asSharedPointer(), false));
111  STORM_LOG_THROW(resultPtr, storm::exceptions::UnexpectedException, "No check result obtained.");
112  STORM_LOG_THROW(resultPtr->isExplicitQuantitativeCheckResult(), storm::exceptions::UnexpectedException, "Unexpected Check result Type");
113  std::vector<ValueType> pomdpSchedulerResult = std::move(resultPtr->template asExplicitQuantitativeCheckResult<ValueType>().getValueVector());
114  return pomdpSchedulerResult;
115  }

118  STORM_LOG_THROW(info.isNonNestedReachabilityProbability() || info.isNonNestedExpectedRewardFormula(), storm::exceptions::NotSupportedException, "The property type is not supported for this analysis.");
119 
120  // Compute the values on the fully observable MDP
121  // We need an actual MDP so that we can apply schedulers below.
122  // Also, the api call in the next line will require a copy anyway.
123  auto underlyingMdp = std::make_shared<storm::models::sparse::Mdp<ValueType>>(pomdp.getTransitionMatrix(), pomdp.getStateLabeling(), pomdp.getRewardModels());
124  auto resultPtr = storm::api::verifyWithSparseEngine<ValueType>(underlyingMdp, storm::api::createTask<ValueType>(formula.asSharedPointer(), false));
125  STORM_LOG_THROW(resultPtr, storm::exceptions::UnexpectedException, "No check result obtained.");
126  STORM_LOG_THROW(resultPtr->isExplicitQuantitativeCheckResult(), storm::exceptions::UnexpectedException, "Unexpected Check result Type");
127  std::vector<ValueType> fullyObservableResult = std::move(resultPtr->template asExplicitQuantitativeCheckResult<ValueType>().getValueVector());
128 
129  std::vector<ValueType> actionBasedRewards;
130  std::vector<ValueType>* actionBasedRewardsPtr = nullptr;
131  if (info.isNonNestedExpectedRewardFormula()) {
132  actionBasedRewards = pomdp.getRewardModel(info.getRewardModelName()).getTotalRewardVector(pomdp.getTransitionMatrix());
133  actionBasedRewardsPtr = &actionBasedRewards;
134  }
135  std::vector<std::vector<ValueType>> guessedSchedulerValues;
136 
137  std::vector<std::pair<double, bool>> guessParameters({{0.875,false},{0.875,true},{0.75,false},{0.75,true}});
138  for (auto const& pars : guessParameters) {
139  guessedSchedulerValues.push_back(computeValuesForGuessedScheduler(fullyObservableResult, actionBasedRewardsPtr, formula, info, underlyingMdp, storm::utility::convertNumber<ValueType>(pars.first), pars.second));
140  }
141 
142  // compute the 'best' guess and do a few iterations on it
143  uint64_t bestGuess = 0;
144  ValueType bestGuessSum = std::accumulate(guessedSchedulerValues.front().begin(), guessedSchedulerValues.front().end(), storm::utility::zero<ValueType>());
145  for (uint64_t guess = 1; guess < guessedSchedulerValues.size(); ++guess) {
146  ValueType guessSum = std::accumulate(guessedSchedulerValues[guess].begin(), guessedSchedulerValues[guess].end(), storm::utility::zero<ValueType>());
147  if ((info.minimize() && guessSum < bestGuessSum) || (info.maximize() && guessSum > bestGuessSum)) {
148  bestGuess = guess;
149  bestGuessSum = guessSum;
150  }
151  }
152  guessedSchedulerValues.push_back(computeValuesForGuessedScheduler(guessedSchedulerValues[bestGuess], actionBasedRewardsPtr, formula, info, underlyingMdp, storm::utility::convertNumber<ValueType>(guessParameters[bestGuess].first), guessParameters[bestGuess].second));
153  guessedSchedulerValues.push_back(computeValuesForGuessedScheduler(guessedSchedulerValues.back(), actionBasedRewardsPtr, formula, info, underlyingMdp, storm::utility::convertNumber<ValueType>(guessParameters[bestGuess].first), guessParameters[bestGuess].second));
154  guessedSchedulerValues.push_back(computeValuesForGuessedScheduler(guessedSchedulerValues.back(), actionBasedRewardsPtr, formula, info, underlyingMdp, storm::utility::convertNumber<ValueType>(guessParameters[bestGuess].first), guessParameters[bestGuess].second));
155 
156  // Check if one of the guesses is worse than one of the others (and potentially delete it)
157  // Avoid deleting entries during the loop to ensure that indices remain valid
158  storm::storage::BitVector keptGuesses(guessedSchedulerValues.size(), true);
159  for (uint64_t i = 0; i < guessedSchedulerValues.size() - 1; ++i) {
160  if (!keptGuesses.get(i)) {
161  continue;
162  }
163  for (uint64_t j = i + 1; j < guessedSchedulerValues.size(); ++j) {
164  if (!keptGuesses.get(j)) {
165  continue;
166  }
167  if (storm::utility::vector::compareElementWise(guessedSchedulerValues[i], guessedSchedulerValues[j], std::less_equal<ValueType>())) {
168  if (info.minimize()) {
169  // In this case we are guessing upper bounds (and smaller upper bounds are better)
170  keptGuesses.set(j, false);
171  } else {
172  // In this case we are guessing lower bounds (and larger lower bounds are better)
173  keptGuesses.set(i, false);
174  break;
175  }
176  } else if (storm::utility::vector::compareElementWise(guessedSchedulerValues[j], guessedSchedulerValues[i], std::less_equal<ValueType>())) {
177  if (info.minimize()) {
178  keptGuesses.set(i, false);
179  break;
180  } else {
181  keptGuesses.set(j, false);
182  }
183  }
184  }
185  }
186  STORM_LOG_INFO("Keeping scheduler guesses " << keptGuesses);
187  storm::utility::vector::filterVectorInPlace(guessedSchedulerValues, keptGuesses);
188 
189  // Finally prepare the result
190  ValueBounds result;
191  if (info.minimize()) {
192  result.lower.push_back(std::move(fullyObservableResult));
193  result.upper = std::move(guessedSchedulerValues);
194  } else {
195  result.lower = std::move(guessedSchedulerValues);
196  result.upper.push_back(std::move(fullyObservableResult));
197  }
198  STORM_LOG_WARN_COND_DEBUG(storm::utility::vector::compareElementWise(result.lower.front(), result.upper.front(), std::less_equal<ValueType>()), "Lower bound is larger than upper bound");
199  return result;
200  }
