331  modelComponents.markovianStates = markovianStatesNew;
332

343  // Set id for skipped states as we remap it later

345  } else {

496  bool isAddLabelsClaiming = storm::settings::getModule<storm::settings::modules::FaultTreeSettings>().isAddLabelsClaiming();
497 
498  // Build state labeling
499  modelComponents.stateLabeling = storm::models::sparse::StateLabeling(modelComponents.transitionMatrix.getRowGroupCount());
500  // Initial state
501  modelComponents.stateLabeling.addLabel("init");
502  STORM_LOG_ASSERT(matrixBuilder.getRemapping(initialStateIndex) == initialStateIndex, "Initial state should not be remapped.");
503  modelComponents.stateLabeling.addLabelToState("init", initialStateIndex);
504  // System failure
505  modelComponents.stateLabeling.addLabel("failed");
506 
507  // Label all states corresponding to their status (failed, don't care BE)
508  // Collect labels for all necessary elements
509  for (size_t id = 0; id < dft.nrElements(); ++id) {
510  std::shared_ptr<storage::DFTElement<ValueType> const> element = dft.getElement(id);
511  if (element->isRelevant()) {
512  modelComponents.stateLabeling.addLabel(element->name() + "_failed");
513  modelComponents.stateLabeling.addLabel(element->name() + "_dc");
514  }
515  }
516  std::vector<std::shared_ptr<storm::storage::DFTGate<ValueType> const>> spares; // Only filled if needed
517  if (isAddLabelsClaiming) {
518  // Collect labels for claiming
519  for (size_t spareId : dft.getSpareIndices()) {
520  auto const& spare = dft.getGate(spareId);
521  spares.push_back(spare);
522  for (auto const& child : spare->children()) {

753  template<typename ValueType, typename StateType>
754  ValueType ExplicitDFTModelBuilder<ValueType, StateType>::getUpperBound(DFTStatePointer const& state) const {
755  if (state->hasFailed(dft.getTopLevelIndex())) {
756  return storm::utility::zero<ValueType>();
757  }
758 
759  // Get the upper bound by considering the failure of all BEs
760  ValueType upperBound = storm::utility::one<ValueType>();
761  ValueType rateSum = storm::utility::zero<ValueType>();
762 
763  // Compute for each independent subtree
764  for (std::vector<size_t> const& subtree : subtreeBEs) {
765  // Get all possible rates
766  std::vector<ValueType> rates;
767  storm::storage::BitVector coldBEs(subtree.size(), false);
768  for (size_t i = 0; i < subtree.size(); ++i) {
769  size_t id = subtree[i];
770  // Consider only still operational BEs
771  if (state->isOperational(id)) {
772  auto be = dft.getBasicElement(id);
773  switch (be->beType()) {
774  case storm::storage::BEType::CONSTANT:
775  // Ignore BE which cannot fail
776  continue;
777  case storm::storage::BEType::EXPONENTIAL:
778  {
779  // Get BE rate
780  ValueType rate = state->getBERate(id);
781  if (storm::utility::isZero<ValueType>(rate)) {
782  // Get active failure rate for cold BE
783  auto beExp = std::static_pointer_cast<storm::storage::BEExponential<ValueType> const>(be);
784  rate = beExp->activeFailureRate();
785  STORM_LOG_ASSERT(!storm::utility::isZero<ValueType>(rate), "Failure rate should not be zero.");
786  // Mark BE as cold
787  coldBEs.set(i, true);
788  }
789  rates.push_back(rate);
790  rateSum += rate;
791  break;
792  }
793  default:
794  STORM_LOG_THROW(false, storm::exceptions::InvalidArgumentException, "BE of type '" << be->type() << "' is not known.");
795  break;
796  }
797  }
798  }
799  STORM_LOG_ASSERT(rates.size() > 0, "No rates failable");
800 
801  // Sort rates
802  std::sort(rates.begin(), rates.end());
803  std::vector<size_t> rateCount(rates.size(), 0);
804  size_t lastIndex = 0;
805  for (size_t i = 0; i < rates.size(); ++i) {
806  if (rates[lastIndex] != rates[i]) {
807  lastIndex = i;
808  }
809  ++rateCount[lastIndex];
810  }
811 
812  for (size_t i = 0; i < rates.size(); ++i) {
813  // Cold BEs cannot fail in the first step
814  if (!coldBEs.get(i) && rateCount[i] > 0) {
815  std::iter_swap(rates.begin() + i, rates.end() - 1);
816  // Compute AND MTTF of subtree without current rate and scale with current rate
817  upperBound += rates.back() * rateCount[i] * computeMTTFAnd(rates, rates.size() - 1);
818  // Swap back
819  // TODO try to avoid swapping back
820  std::iter_swap(rates.begin() + i, rates.end() - 1);
821  }
822  }
823  }
824 
825  STORM_LOG_TRACE("Upper bound is " << (rateSum / upperBound) << " for state " << state->getId());
826  STORM_LOG_ASSERT(!storm::utility::isZero(upperBound), "Upper bound is 0");
827  STORM_LOG_ASSERT(!storm::utility::isZero(rateSum), "State is absorbing");
828  return rateSum / upperBound;
829  }
