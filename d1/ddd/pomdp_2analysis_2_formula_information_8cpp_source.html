73  template <typename PomdpType>
74  FormulaInformation::StateSet getStateSet(PomdpType const& pomdp, storm::storage::BitVector&& inputStates) {
75  FormulaInformation::StateSet result;
76  result.states = std::move(inputStates);
77  for (auto const& state : result.states) {
78  result.observations.insert(pomdp.getObservation(state));
79  }
80  // check if this set is observation-closed, i.e., whether there is a state outside of this set with one of the observations collected above
81  result.observationClosed = true;
82  for (uint64_t state = result.states.getNextUnsetIndex(0); state < result.states.size(); state = result.states.getNextUnsetIndex(state + 1)) {
83  if (result.observations.count(pomdp.getObservation(state)) > 0) {
84  result.observationClosed = false;
85  break;
86  }
87  }
88  return result;
89  }

114  template <typename PomdpType>
115  FormulaInformation getFormulaInformation(PomdpType const& pomdp, storm::logic::ProbabilityOperatorFormula const& formula) {
116  STORM_LOG_THROW(formula.hasOptimalityType(), storm::exceptions::InvalidPropertyException, "The property does not specify an optimization direction (min/max)");
117  STORM_LOG_WARN_COND(!formula.hasBound(), "The probability threshold for the given property will be ignored.");
118  auto const& subformula = formula.getSubformula();
119  std::shared_ptr<storm::logic::Formula const> targetStatesFormula, constraintsStatesFormula;
120  if (subformula.isEventuallyFormula()) {
121  targetStatesFormula = subformula.asEventuallyFormula().getSubformula().asSharedPointer();
122  constraintsStatesFormula = storm::logic::Formula::getTrueFormula()->asSharedPointer();
123  } else if (subformula.isUntilFormula()) {
124  storm::logic::UntilFormula const &untilFormula = subformula.asUntilFormula();
125  targetStatesFormula = untilFormula.getRightSubformula().asSharedPointer();
126  constraintsStatesFormula = untilFormula.getLeftSubformula().asSharedPointer();
127  }
128  if (targetStatesFormula && targetStatesFormula->isInFragment(storm::logic::propositional()) && constraintsStatesFormula && constraintsStatesFormula->isInFragment(storm::logic::propositional())) {
129  FormulaInformation result(FormulaInformation::Type::NonNestedReachabilityProbability, formula.getOptimalityType());
130  result.updateTargetStates(pomdp, getStates(*targetStatesFormula, false, pomdp));
131  result.updateSinkStates(pomdp, getStates(*constraintsStatesFormula, true, pomdp));
132  return result;
133  }
134  return FormulaInformation();
135  }
