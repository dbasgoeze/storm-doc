48  bool performPreprocessing(std::shared_ptr<storm::models::sparse::Pomdp<ValueType>>& pomdp, storm::pomdp::analysis::FormulaInformation& formulaInfo, storm::logic::Formula const& formula) {
49  auto const& pomdpSettings = storm::settings::getModule<storm::settings::modules::POMDPSettings>();
50  bool preprocessingPerformed = false;
51  if (pomdpSettings.isSelfloopReductionSet()) {
52  storm::transformer::GlobalPOMDPSelfLoopEliminator<ValueType> selfLoopEliminator(*pomdp);
53  if (selfLoopEliminator.preservesFormula(formula)) {
54  STORM_PRINT_AND_LOG("Eliminating self-loop choices ...");
55  uint64_t oldChoiceCount = pomdp->getNumberOfChoices();
56  pomdp = selfLoopEliminator.transform();
57  STORM_PRINT_AND_LOG(oldChoiceCount - pomdp->getNumberOfChoices() << " choices eliminated through self-loop elimination." << std::endl);
58  preprocessingPerformed = true;
59  } else {
60  STORM_PRINT_AND_LOG("Not eliminating self-loop choices as it does not preserve the formula." << std::endl);
61  }
62  }
63  if (pomdpSettings.isQualitativeReductionSet() && formulaInfo.isNonNestedReachabilityProbability()) {
64  storm::analysis::QualitativeAnalysisOnGraphs<ValueType> qualitativeAnalysis(*pomdp);
65  STORM_PRINT_AND_LOG("Computing states with probability 0 ...");
66  storm::storage::BitVector prob0States = qualitativeAnalysis.analyseProb0(formula.asProbabilityOperatorFormula());
67  std::cout << prob0States << std::endl;
68  STORM_PRINT_AND_LOG(" done. " << prob0States.getNumberOfSetBits() << " states found." << std::endl);
69  STORM_PRINT_AND_LOG("Computing states with probability 1 ...");
70  storm::storage::BitVector prob1States = qualitativeAnalysis.analyseProb1(formula.asProbabilityOperatorFormula());
71  STORM_PRINT_AND_LOG(" done. " << prob1States.getNumberOfSetBits() << " states found." << std::endl);
72  storm::pomdp::transformer::KnownProbabilityTransformer<ValueType> kpt = storm::pomdp::transformer::KnownProbabilityTransformer<ValueType>();
73  pomdp = kpt.transform(*pomdp, prob0States, prob1States);
74  // Update formulaInfo to changes from Preprocessing
75  formulaInfo.updateTargetStates(*pomdp, std::move(prob1States));
76  formulaInfo.updateSinkStates(*pomdp, std::move(prob0States));
77  preprocessingPerformed = true;
78  }
79  return preprocessingPerformed;

270  template<typename ValueType, storm::dd::DdType DdType>
271  bool performTransformation(std::shared_ptr<storm::models::sparse::Pomdp<ValueType>>& pomdp, storm::logic::Formula const& formula) {
272  auto const& pomdpSettings = storm::settings::getModule<storm::settings::modules::POMDPSettings>();
273  auto const& ioSettings = storm::settings::getModule<storm::settings::modules::IOSettings>();
274  auto const& transformSettings = storm::settings::getModule<storm::settings::modules::ToParametricSettings>();
275  bool transformationPerformed = false;
276  bool memoryUnfolded = false;
277  if (pomdpSettings.getMemoryBound() > 1) {
278  STORM_PRINT_AND_LOG("Computing the unfolding for memory bound " << pomdpSettings.getMemoryBound() << " and memory pattern '" << storm::storage::toString(pomdpSettings.getMemoryPattern()) << "' ...");
279  storm::storage::PomdpMemory memory = storm::storage::PomdpMemoryBuilder().build(pomdpSettings.getMemoryPattern(), pomdpSettings.getMemoryBound());
280  std::cout << memory.toString() << std::endl;
281  storm::transformer::PomdpMemoryUnfolder<ValueType> memoryUnfolder(*pomdp, memory);
282  pomdp = memoryUnfolder.transform();
283  STORM_PRINT_AND_LOG(" done." << std::endl);
284  pomdp->printModelInformationToStream(std::cout);
285  transformationPerformed = true;
286  memoryUnfolded = true;
287  }
288 
289  // From now on the pomdp is considered memoryless
290 
291  if (transformSettings.isMecReductionSet()) {
292  STORM_PRINT_AND_LOG("Eliminating mec choices ...");
293  // Note: Elimination of mec choices only preserves memoryless schedulers.
294  uint64_t oldChoiceCount = pomdp->getNumberOfChoices();
295  storm::transformer::GlobalPomdpMecChoiceEliminator<ValueType> mecChoiceEliminator(*pomdp);
296  pomdp = mecChoiceEliminator.transform(formula);
297  STORM_PRINT_AND_LOG(" done." << std::endl);
298  STORM_PRINT_AND_LOG(oldChoiceCount - pomdp->getNumberOfChoices() << " choices eliminated through MEC choice elimination." << std::endl);
299  pomdp->printModelInformationToStream(std::cout);
300  transformationPerformed = true;
301  }
302 
303  if (transformSettings.isTransformBinarySet() || transformSettings.isTransformSimpleSet()) {
304  if (transformSettings.isTransformSimpleSet()) {
305  STORM_PRINT_AND_LOG("Transforming the POMDP to a simple POMDP.");
306  pomdp = storm::transformer::BinaryPomdpTransformer<ValueType>().transform(*pomdp, true);
307  } else {
308  STORM_PRINT_AND_LOG("Transforming the POMDP to a binary POMDP.");
309  pomdp = storm::transformer::BinaryPomdpTransformer<ValueType>().transform(*pomdp, false);
310  }
311  pomdp->printModelInformationToStream(std::cout);
312  STORM_PRINT_AND_LOG(" done." << std::endl);
313  transformationPerformed = true;
314  }
315 
316  if (pomdpSettings.isExportToParametricSet()) {
317  STORM_PRINT_AND_LOG("Transforming memoryless POMDP to pMC...");
318  storm::transformer::ApplyFiniteSchedulerToPomdp<ValueType> toPMCTransformer(*pomdp);
319  std::string transformMode = transformSettings.getFscApplicationTypeString();
320  auto pmc = toPMCTransformer.transform(storm::transformer::parsePomdpFscApplicationMode(transformMode));
321  STORM_PRINT_AND_LOG(" done." << std::endl);
322  pmc->printModelInformationToStream(std::cout);
323  if (transformSettings.allowPostSimplifications()) {
324  STORM_PRINT_AND_LOG("Simplifying pMC...");
325  pmc = storm::api::performBisimulationMinimization<storm::RationalFunction>(pmc->template as<storm::models::sparse::Dtmc<storm::RationalFunction>>(),{formula.asSharedPointer()}, storm::storage::BisimulationType::Strong)->template as<storm::models::sparse::Dtmc<storm::RationalFunction>>();
326  STORM_PRINT_AND_LOG(" done." << std::endl);
327  pmc->printModelInformationToStream(std::cout);
328  }
329  STORM_PRINT_AND_LOG("Exporting pMC...");
330  storm::analysis::ConstraintCollector<storm::RationalFunction> constraints(*pmc);
331  auto const& parameterSet = constraints.getVariables();
332  std::vector<storm::RationalFunctionVariable> parameters(parameterSet.begin(), parameterSet.end());
333  std::vector<std::string> parameterNames;
334  for (auto const& parameter : parameters) {
335  parameterNames.push_back(parameter.name());
336  }
337  storm::api::exportSparseModelAsDrn(pmc, pomdpSettings.getExportToParametricFilename(), parameterNames, !ioSettings.isExplicitExportPlaceholdersDisabled());
338  STORM_PRINT_AND_LOG(" done." << std::endl);
339  transformationPerformed = true;
340  }
341  if (transformationPerformed && !memoryUnfolded) {
342  STORM_PRINT_AND_LOG("Implicitly assumed restriction to memoryless schedulers for at least one transformation." << std::endl);
343  }
344  return transformationPerformed;
345  }

348  void processOptionsWithValueTypeAndDdLib(storm::cli::SymbolicInput const& symbolicInput, storm::cli::ModelProcessingInformation const& mpi) {

350 
351  auto model = storm::cli::buildPreprocessExportModelWithValueTypeAndDdlib<DdType, ValueType>(symbolicInput, mpi);
352  if (!model) {
353  STORM_PRINT_AND_LOG("No input model given." << std::endl);
354  return;
355  }
356  STORM_LOG_THROW(model->getType() == storm::models::ModelType::Pomdp && model->isSparseModel(), storm::exceptions::WrongFormatException, "Expected a POMDP in sparse representation.");
357 
358  std::shared_ptr<storm::models::sparse::Pomdp<ValueType>> pomdp = model->template as<storm::models::sparse::Pomdp<ValueType>>();
359  if (!pomdpSettings.isNoCanonicSet()) {

361  pomdp = makeCanonic.transform();
362  }
363 
364  std::shared_ptr<storm::logic::Formula const> formula;
365  if (!symbolicInput.properties.empty()) {
366  formula = symbolicInput.properties.front().getRawFormula();
367  STORM_PRINT_AND_LOG("Analyzing property '" << *formula << "'" << std::endl);
368  STORM_LOG_WARN_COND(symbolicInput.properties.size() == 1, "There is currently no support for multiple properties. All other properties will be ignored.");
369  }
370 
371  if (pomdpSettings.isAnalyzeUniqueObservationsSet()) {
372  STORM_PRINT_AND_LOG("Analyzing states with unique observation ..." << std::endl);
373  storm::analysis::UniqueObservationStates<ValueType> uniqueAnalysis(*pomdp);
374  std::cout << uniqueAnalysis.analyse() << std::endl;
375  }
376 
377  if (formula) {
378  auto formulaInfo = storm::pomdp::analysis::getFormulaInformation(*pomdp, *formula);
379  STORM_LOG_THROW(!formulaInfo.isUnsupported(), storm::exceptions::InvalidPropertyException, "The formula '" << *formula << "' is not supported by storm-pomdp.");
380 
381  storm::utility::Stopwatch sw(true);
382  // Note that formulaInfo contains state-based information which potentially needs to be updated during preprocessing
383  if (performPreprocessing<ValueType, DdType>(pomdp, formulaInfo, *formula)) {
384  sw.stop();
385  STORM_PRINT_AND_LOG("Time for graph-based POMDP (pre-)processing: " << sw << "." << std::endl);
386  pomdp->printModelInformationToStream(std::cout);
387  }
388 
389  sw.restart();
390  if (performTransformation<ValueType, DdType>(pomdp, *formula)) {
391  sw.stop();
392  STORM_PRINT_AND_LOG("Time for POMDP transformation(s): " << sw << "s." << std::endl);
393  }
394 
395  sw.restart();
396  if (performAnalysis<ValueType, DdType>(pomdp, formulaInfo, *formula)) {
397  sw.stop();
398  STORM_PRINT_AND_LOG("Time for POMDP analysis: " << sw << "s." << std::endl);
399  }
400 
401 
402  } else {
403  STORM_LOG_WARN("Nothing to be done. Did you forget to specify a formula?");
404  }

421  }
