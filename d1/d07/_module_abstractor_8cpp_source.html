62  template <storm::dd::DdType DdType, typename ValueType>
63  GameBddResult<DdType> ModuleAbstractor<DdType, ValueType>::abstract() {
64  // First, we retrieve the abstractions of all commands.
65  std::vector<GameBddResult<DdType>> commandDdsAndUsedOptionVariableCounts;
66  uint_fast64_t maximalNumberOfUsedOptionVariables = 0;
67  for (auto& command : commands) {
68  commandDdsAndUsedOptionVariableCounts.push_back(command.abstract());
69  maximalNumberOfUsedOptionVariables = std::max(maximalNumberOfUsedOptionVariables, commandDdsAndUsedOptionVariableCounts.back().numberOfPlayer2Variables);
70  }
71 
72  // Then, we build the module BDD by adding the single command DDs. We need to make sure that all command
73  // DDs use the same amount DD variable encoding the choices of player 2.
74  storm::dd::Bdd<DdType> result = this->getAbstractionInformation().getDdManager().getBddZero();
75  for (auto const& commandDd : commandDdsAndUsedOptionVariableCounts) {
76  result |= commandDd.bdd && this->getAbstractionInformation().encodePlayer2Choice(1, commandDd.numberOfPlayer2Variables, maximalNumberOfUsedOptionVariables);
77  }
78  return GameBddResult<DdType>(result, maximalNumberOfUsedOptionVariables);
79  }
