121  void SettingsManager::setFromExplodedString(std::vector<std::string> const& commandLineArguments) {
122  // In order to assign the parsed arguments to an option, we need to keep track of the "active" option's name.
123  bool optionActive = false;
124  bool activeOptionIsShortName = false;
125  std::string activeOptionName = "";
126  std::vector<std::string> argumentCache;
127 
128  // Walk through all arguments.
129  for (uint_fast64_t i = 0; i < commandLineArguments.size(); ++i) {
130  std::string const& currentArgument = commandLineArguments[i];
131 
132  // Check if the given argument is a new option or belongs to a previously given option.
133  if (!currentArgument.empty() && currentArgument.at(0) == '-') {
134  if (optionActive) {
135  // At this point we know that a new option is about to come. Hence, we need to assign the current
136  // cache content to the option that was active until now.
137  setOptionsArguments(activeOptionName, activeOptionIsShortName ? this->shortNameToOptions : this->longNameToOptions, argumentCache);
138 
139  // After the assignment, the argument cache needs to be cleared.
140  argumentCache.clear();
141  } else {
142  optionActive = true;
143  }
144 
145  if (currentArgument.at(1) == '-') {
146  // In this case, the argument has to be the long name of an option. Try to get all options that
147  // match the long name.
148  std::string optionName = currentArgument.substr(2);
149  auto optionIterator = this->longNameToOptions.find(optionName);
150  if (optionIterator == this->longNameToOptions.end()) {
151  handleUnknownOption(optionName, false);
152  }
153  activeOptionIsShortName = false;
154  activeOptionName = optionName;
155  } else {
156  // In this case, the argument has to be the short name of an option. Try to get all options that
157  // match the short name.
158  std::string optionName = currentArgument.substr(1);
159  auto optionIterator = this->shortNameToOptions.find(optionName);
160  if (optionIterator == this->shortNameToOptions.end()) {
161  handleUnknownOption(optionName, true);
162  }
163  activeOptionIsShortName = true;
164  activeOptionName = optionName;
165  }
166  } else if (optionActive) {
167  // Add the current argument to the list of arguments for the currently active options.
168  argumentCache.push_back(currentArgument);
169  } else {
170  STORM_LOG_THROW(false, storm::exceptions::OptionParserException, "Found stray argument '" << currentArgument << "' that is not preceeded by a matching option.");
171  }
172  }
173 
174  // If an option is still active at this point, we need to set it.
175  if (optionActive) {
176  setOptionsArguments(activeOptionName, activeOptionIsShortName ? this->shortNameToOptions : this->longNameToOptions, argumentCache);
177  }
178 
179  // Include the options from a possibly specified configuration file, but don't overwrite existing settings.
180  if (storm::settings::hasModule<storm::settings::modules::GeneralSettings>() && storm::settings::getModule<storm::settings::modules::GeneralSettings>().isConfigSet()) {
181  this->setFromConfigurationFile(storm::settings::getModule<storm::settings::modules::GeneralSettings>().getConfigFilename());
182  }
183 
184  // Finally, check whether all modules are okay with the current settings.
185  this->finalizeAllModules();
186  }

294  std::string SettingsManager::getHelpForSelection(std::vector<std::string> const& selectedModuleNames, std::vector<std::string> const& selectedLongOptionNames, std::string modulesHeader, std::string optionsHeader) const {
295  std::stringstream stream;
296 
297  // Remember which options we printed, so we don't display options twice.
298  std::set<std::shared_ptr<Option>> printedOptions;
299 
300  // Try to match the regular expression against the known modules.
301  uint_fast64_t maxLengthModules = 0;
302  for (auto const& moduleName : selectedModuleNames) {
303  maxLengthModules = std::max(maxLengthModules, getPrintLengthOfLongestOption(moduleName, true));
304  // Add all options of this module to the list of printed options so we don't print them twice.
305  auto optionIterator = this->moduleOptions.find(moduleName);
306  STORM_LOG_ASSERT(optionIterator != this->moduleOptions.end(), "Unable to find selected module " << moduleName << ".");
307  printedOptions.insert(optionIterator->second.begin(), optionIterator->second.end());
308  }
309 
310  // Try to match the regular expression against the known options.
311  std::vector<std::shared_ptr<Option>> matchingOptions;
312  uint_fast64_t maxLengthOptions = 0;
313  for (auto const& optionName : selectedLongOptionNames) {
314  auto optionIterator = this->longNameToOptions.find(optionName);
315  STORM_LOG_ASSERT(optionIterator != this->longNameToOptions.end(), "Unable to find selected option " << optionName << ".");
316  for (auto const& option : optionIterator->second) {
317  // Only add the option if we have not already added it to the list of options that is going
318  // to be printed anyway.
319  if (printedOptions.find(option) == printedOptions.end()) {
320  maxLengthOptions = std::max(maxLengthOptions, option->getPrintLength());
321  matchingOptions.push_back(option);
322  printedOptions.insert(option);
323  }
324  }
325  }
326 
327  // Print the matching modules.
328  uint_fast64_t maxLength = std::max(maxLengthModules, maxLengthOptions);
329  if (selectedModuleNames.size() > 0) {
330  if (modulesHeader != "") {
331  stream << modulesHeader << std::endl;
332  }
333  for (auto const& matchingModuleName : selectedModuleNames) {
334  stream << getHelpForModule(matchingModuleName, maxLength, true);
335  }
336  }
337 
338  // Print the matching options.
339  if (matchingOptions.size() > 0) {
340  if (optionsHeader != "") {
341  stream << optionsHeader << std::endl;
342  }
343  for (auto const& option : matchingOptions) {
344  stream << std::setw(maxLength) << std::left << *option << std::endl;
345  }
346  }
347  return stream.str();
348  }

421  void SettingsManager::addOption(std::shared_ptr<Option> const& option) {
422  // First, we register to which module the given option belongs.
423  auto moduleOptionIterator = this->moduleOptions.find(option->getModuleName());
424  STORM_LOG_THROW(moduleOptionIterator != this->moduleOptions.end(), storm::exceptions::IllegalFunctionCallException, "Cannot add option for unknown module '" << option->getModuleName() << "'.");
425  moduleOptionIterator->second.emplace_back(option);
426 
427  // Then, we add the option's name (and possibly short name) to the registered options. If a module prefix is
428  // not required for this option, we have to add both versions to our mappings, the prefixed one and the
429  // non-prefixed one.
430  if (!option->getRequiresModulePrefix()) {
431  bool isCompatible = this->isCompatible(option, option->getLongName(), this->longNameToOptions);
432  STORM_LOG_THROW(isCompatible, storm::exceptions::IllegalFunctionCallException, "Unable to add option '" << option->getLongName() << "', because an option with the same name is incompatible with it.");
433  addOptionToMap(option->getLongName(), option, this->longNameToOptions);
434  }
435  // For the prefixed name, we don't need a compatibility check, because a module is not allowed to register the same option twice.
436  addOptionToMap(option->getModuleName() + ":" + option->getLongName(), option, this->longNameToOptions);
437  longOptionNames.push_back(option->getModuleName() + ":" + option->getLongName());
438 
439  if (option->getHasShortName()) {
440  if (!option->getRequiresModulePrefix()) {
441  bool isCompatible = this->isCompatible(option, option->getShortName(), this->shortNameToOptions);
442  STORM_LOG_THROW(isCompatible, storm::exceptions::IllegalFunctionCallException, "Unable to add option '" << option->getLongName() << "', because an option with the same name is incompatible with it.");
443  addOptionToMap(option->getShortName(), option, this->shortNameToOptions);
444  }
445  addOptionToMap(option->getModuleName() + ":" + option->getShortName(), option, this->shortNameToOptions);
446  }

534  std::map<std::string, std::vector<std::string>> SettingsManager::parseConfigFile(std::string const& filename) const {
535  std::map<std::string, std::vector<std::string>> result;
536 
537  std::ifstream input;
538  storm::utility::openFile(filename, input);
539 
540  bool globalScope = true;
541  std::string activeModule = "";
542  uint_fast64_t lineNumber = 1;
543  for (std::string line; storm::utility::getline(input, line); ++lineNumber) {
544  // If the first character of the line is a "[", we expect the settings of a new module to start and
545  // the line to be of the shape [<module>].
546  if (line.at(0) == '[') {
547  STORM_LOG_THROW(line.at(0) == '[' && line.find("]") == line.length() - 1 && line.find("[", 1) == line.npos, storm::exceptions::OptionParserException, "Illegal module name header in configuration file '" << filename << " in line " << std::to_string(lineNumber) << ". Expected [<module>] where <module> is a placeholder for a known module.");
548 
549  // Extract the module name and check whether it's a legal one.
550  std::string moduleName = line.substr(1, line.length() - 2);
551  STORM_LOG_THROW(moduleName != "" && (moduleName == "global" || (this->modules.find(moduleName) != this->modules.end())), storm::exceptions::OptionParserException, "Module header in configuration file '" << filename << " in line " << std::to_string(lineNumber) << " refers to unknown module '" << moduleName << ".");
552 
553  // If the module name is "global", we unset the currently active module and treat all options to follow as unprefixed.
554  if (moduleName == "global") {
555  globalScope = true;
556  } else {
557  activeModule = moduleName;
558  globalScope = false;
559  }
560  } else {
561  // In this case, we expect the line to be of the shape o or o=a b c, where o is an option and a, b
562  // and c are the values that are supposed to be assigned to the arguments of the option.
563  std::size_t assignmentSignIndex = line.find("=");
564  bool containsAssignment = false;
565  if (assignmentSignIndex != line.npos) {
566  containsAssignment = true;
567  }
568 
569  std::string optionName;
570  if (containsAssignment) {
571  optionName = line.substr(0, assignmentSignIndex);
572  } else {
573  optionName = line;
574  }
575 
576  if (globalScope) {
577  STORM_LOG_THROW(this->longNameToOptions.find(optionName) != this->longNameToOptions.end(), storm::exceptions::OptionParserException, "Option assignment in configuration file '" << filename << " in line " << lineNumber << " refers to unknown option '" << optionName << "'.");
578  } else {
579  STORM_LOG_THROW(this->longNameToOptions.find(activeModule + ":" + optionName) != this->longNameToOptions.end(), storm::exceptions::OptionParserException, "Option assignment in configuration file '" << filename << " in line " << lineNumber << " refers to unknown option '" << activeModule << ":" << optionName << "'.");
580  }
581 
582  std::string fullOptionName = (!globalScope ? activeModule + ":" : "") + optionName;
583  STORM_LOG_WARN_COND(result.find(fullOptionName) == result.end(), "Option '" << fullOptionName << "' is set in line " << lineNumber << " of configuration file " << filename << ", but has been set before.");
584 
585  // If the current line is an assignment, split the right-hand side of the assignment into parts
586  // enclosed by quotation marks.
587  if (containsAssignment) {
588  std::string assignedValues = line.substr(assignmentSignIndex + 1);
589  std::vector<std::string> argumentCache;
590 
591  // As horrible as it may look, this regular expression matches either a quoted string (possibly
592  // containing escaped quotes) or a simple word (without whitespaces and quotes).
593  std::regex argumentRegex("\"(([^\\\\\"]|((\\\\\\\\)*\\\\\")|\\\\[^\"])*)\"|(([^ \\\\\"]|((\\\\\\\\)*\\\\\")|\\\\[^\"])+)");
594  boost::algorithm::trim_left(assignedValues);
595 
596  while (!assignedValues.empty()) {
597  std::smatch match;
598  bool hasMatch = std::regex_search(assignedValues, match, argumentRegex);
599 
600  // If the input could not be matched, we have a parsing error.
601  STORM_LOG_THROW(hasMatch, storm::exceptions::OptionParserException, "Parsing error in configuration file '" << filename << "' in line " << lineNumber << ". Unexpected input '" << assignedValues << "'.");
602 
603  // Extract the matched argument and cut off the quotation marks if necessary.
604  std::string matchedArgument = std::string(match[0].first, match[0].second);
605  if (matchedArgument.at(0) == '"') {
606  matchedArgument = matchedArgument.substr(1, matchedArgument.length() - 2);
607  }
608  argumentCache.push_back(matchedArgument);
609 
610  assignedValues = assignedValues.substr(match.length());
611  boost::algorithm::trim_left(assignedValues);
612  }
613 
614 
615  // After successfully parsing the argument values, we store them in the result map.
616  result.emplace(fullOptionName, argumentCache);
617  } else {
618  // In this case, we can just insert the option to indicate it should be set (without arguments).
619  result.emplace(fullOptionName, std::vector<std::string>());
620  }
621  }
622  }
623 
624  storm::utility::closeFile(input);
625  return result;
626  }
