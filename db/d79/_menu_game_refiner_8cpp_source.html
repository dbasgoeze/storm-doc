112  storm::dd::Bdd<Type> spanningTree = game.getManager().getBddZero();
113 
114  storm::dd::Add<Type, ValueType> transitionMatrix = ((transitionFilter && game.getExtendedTransitionMatrix().maxAbstractRepresentative(game.getProbabilisticBranchingVariables())).template toAdd<ValueType>() * game.getExtendedTransitionMatrix()).sumAbstract(game.getPlayer2Variables());
115 
116  std::set<storm::expressions::Variable> variablesToAbstract(game.getRowVariables());
117  variablesToAbstract.insert(game.getPlayer1Variables().begin(), game.getPlayer1Variables().end());
118  variablesToAbstract.insert(game.getProbabilisticBranchingVariables().begin(), game.getProbabilisticBranchingVariables().end());

120  // Determine the new maximal probabilities to all states.
121  storm::dd::Add<Type, ValueType> tmp = border.template toAdd<ValueType>() * transitionMatrix * maxProbabilities;

123  storm::dd::Add<Type, ValueType> newMaxProbabilities = tmp.maxAbstract(variablesToAbstract).swapVariables(game.getRowColumnMetaVariablePairs());

130  spanningTree &= ((!updateStates).swapVariables(game.getRowColumnMetaVariablePairs()));

133  spanningTree |= updateStates.swapVariables(game.getRowColumnMetaVariablePairs()) && newMaxProbabilityChoices;

136  border = updateStates;

327  }

357  ++refinementPredicateIndexToCount[index];
358  }
359  }
360  }
361

363  uint64_t chosenPredicateIndex = 0;
364  for (uint64_t index = 0; index < possibleRefinementPredicates.size(); ++index) {
365  if (refinementPredicateIndexToCount[index] > refinementPredicateIndexToCount[chosenPredicateIndex]) {
366  chosenPredicateIndex = index;
367  }
368  }
369  newPredicate = possibleRefinementPredicates[chosenPredicateIndex];
370  STORM_LOG_DEBUG("Derived new predicate (based on weakest-precondition): " << newPredicate << ", (equivalent to " << (refinementPredicateIndexToCount[chosenPredicateIndex] - 1) << " other refinement predicates).");
371  } else {
372  newPredicate = possibleRefinementPredicates.front();
373  STORM_LOG_DEBUG("Derived new predicate (based on weakest-precondition): " << newPredicate << ".");
374  }
375 
376  STORM_LOG_ASSERT(newPredicate.isInitialized(), "Could not derive new predicate as there is no deviation.");
377  }
378 
379  return RefinementPredicates(fromGuard ? RefinementPredicates::Source::Guard : RefinementPredicates::Source::WeakestPrecondition, {newPredicate});
380  }

554  template<storm::dd::DdType Type, typename ValueType>
555  std::pair<std::vector<std::vector<storm::expressions::Expression>>, std::map<storm::expressions::Variable, storm::expressions::Expression>> MenuGameRefiner<Type, ValueType>::buildTrace(storm::expressions::ExpressionManager& expressionManager, storm::abstraction::MenuGame<Type, ValueType> const& game, storm::dd::Bdd<Type> const& spanningTree, storm::dd::Bdd<Type> const& pivotState) const {
556  std::vector<std::vector<storm::expressions::Expression>> predicates;
557 
558  // Prepare some variables.
559  AbstractionInformation<Type> const& abstractionInformation = abstractor.get().getAbstractionInformation();
560  std::set<storm::expressions::Variable> variablesToAbstract(game.getColumnVariables());
561  variablesToAbstract.insert(game.getPlayer1Variables().begin(), game.getPlayer1Variables().end());
562  variablesToAbstract.insert(game.getProbabilisticBranchingVariables().begin(), game.getProbabilisticBranchingVariables().end());
563 
564  storm::expressions::Expression initialExpression = abstractor.get().getInitialExpression();
565 
566  std::set<storm::expressions::Variable> oldVariables = initialExpression.getVariables();
567  for (auto const& predicate : abstractionInformation.getPredicates()) {
568  std::set<storm::expressions::Variable> usedVariables = predicate.getVariables();
569  oldVariables.insert(usedVariables.begin(), usedVariables.end());
570  }
571 
572  std::map<storm::expressions::Variable, storm::expressions::Variable> oldToNewVariables;
573  for (auto const& variable : oldVariables) {
574  oldToNewVariables[variable] = expressionManager.getVariable(variable.getName());
575  }
576  std::map<storm::expressions::Variable, storm::expressions::Expression> lastSubstitution;
577  for (auto const& variable : oldToNewVariables) {
578  lastSubstitution[variable.second] = variable.second;
579  }
580  std::map<storm::expressions::Variable, storm::expressions::Expression> stepVariableToCopiedVariableMap;
581 
582  // Start with the target state part of the trace.
583  storm::storage::BitVector decodedTargetState = abstractionInformation.decodeState(pivotState);
584  predicates.emplace_back(abstractionInformation.getPredicates(decodedTargetState));
585  for (auto& predicate : predicates.back()) {
586  predicate = predicate.changeManager(expressionManager);
587  }
588  // Add ranges of variables.
589  for (auto const& pred : abstractionInformation.getConstraints()) {
590  predicates.back().push_back(pred.changeManager(expressionManager));
591  }
592 
593  // Perform a backward search for an initial state.
594  storm::dd::Bdd<Type> currentState = pivotState;
595  while ((currentState && game.getInitialStates()).isZero()) {
596  storm::dd::Bdd<Type> predecessorTransition = currentState.swapVariables(game.getRowColumnMetaVariablePairs()) && spanningTree;
597  std::tuple<storm::storage::BitVector, uint64_t, uint64_t> decodedPredecessor = abstractionInformation.decodeStatePlayer1ChoiceAndUpdate(predecessorTransition);
598 
599  // Create a new copy of each variable to use for this step.
600  std::map<storm::expressions::Variable, storm::expressions::Expression> substitution;
601  for (auto const& variablePair : oldToNewVariables) {
602  storm::expressions::Variable variableCopy = expressionManager.declareVariableCopy(variablePair.second);
603  substitution[variablePair.second] = variableCopy;
604  stepVariableToCopiedVariableMap[variableCopy] = variablePair.second;
605  }
606 
607  // Retrieve the variable updates that the predecessor needs to perform to get to the current state.
608  auto variableUpdates = abstractor.get().getVariableUpdates(std::get<1>(decodedPredecessor), std::get<2>(decodedPredecessor));
609  for (auto const& oldNewVariablePair : oldToNewVariables) {
610  storm::expressions::Variable const& newVariable = oldNewVariablePair.second;
611 
612  // If the variable was set, use its update expression.
613  auto updateIt = variableUpdates.find(oldNewVariablePair.first);
614  if (updateIt != variableUpdates.end()) {
615  auto const& update = *updateIt;
616 
617  if (update.second.hasBooleanType()) {
618  predicates.back().push_back(storm::expressions::iff(lastSubstitution.at(newVariable), update.second.changeManager(expressionManager).substitute(substitution)));
619  } else {
620  predicates.back().push_back(lastSubstitution.at(newVariable) == update.second.changeManager(expressionManager).substitute(substitution));
621  }
622  } else {
623  // Otherwise, make sure that the new variable maintains the old value.
624  if (newVariable.hasBooleanType()) {
625  predicates.back().push_back(storm::expressions::iff(lastSubstitution.at(newVariable), substitution.at(newVariable)));
626  } else {
627  predicates.back().push_back(lastSubstitution.at(newVariable) == substitution.at(newVariable));
628  }
629  }
630  }
631 
632  // Add the guard of the choice.
633  predicates.back().push_back(abstractor.get().getGuard(std::get<1>(decodedPredecessor)).changeManager(expressionManager).substitute(substitution));
634 
635  // Retrieve the predicate valuation in the predecessor.
636  predicates.emplace_back(abstractionInformation.getPredicates(std::get<0>(decodedPredecessor)));
637  for (auto& predicate : predicates.back()) {
638  predicate = predicate.changeManager(expressionManager).substitute(substitution);
639  }
640  // Add ranges of variables.
641  for (auto const& pred : abstractionInformation.getConstraints()) {
642  predicates.back().push_back(pred.changeManager(expressionManager).substitute(substitution));
643  }
644  // Move backwards one step.
645  lastSubstitution = std::move(substitution);
646  currentState = predecessorTransition.existsAbstract(variablesToAbstract);
647  }
648 
649  predicates.back().push_back(initialExpression.changeManager(expressionManager).substitute(lastSubstitution));
650  return std::make_pair(predicates, stepVariableToCopiedVariableMap);
651  }

698  }

723  predicates.back().push_back(initialExpression.changeManager(expressionManager));

726  auto actionIt = reversedLabels.rbegin();

729 
730  // Add new predicate frame.
731  predicates.emplace_back();

734  predicates.back().emplace_back(abstractor.get().getGuard(*actionIt).changeManager(expressionManager).substitute(currentSubstitution));

737  std::set<storm::expressions::Variable> const& assignedVariables = abstractor.get().getAssignedVariables(*actionIt);

741  for (auto const& variable : assignedVariables) {

743  newVariableMaps[oldToNewVariables.at(variable)] = variableCopy;
744  stepVariableToCopiedVariableMap[variableCopy] = variable;
745  }

748  auto variableUpdateVector = abstractor.get().getVariableUpdates(*actionIt);

752  for (auto const& variableUpdates : variableUpdateVector) {
753  storm::expressions::Expression variableUpdateExpression;
754  for (auto const& update : variableUpdates) {

757  } else {
758  variableUpdateExpression = variableUpdateExpression && newVariableMaps.at(update.first) == update.second.changeManager(expressionManager).substitute(currentSubstitution);
759  }

763  }
764  if (!allVariableUpdateExpression.isInitialized()) {
765  allVariableUpdateExpression = expressionManager.boolean(true);
766  }
767  predicates.back().emplace_back(allVariableUpdateExpression);
768 
769  // Incorporate the new variables in the current substitution.
770  for (auto const& variablePair : newVariableMaps) {
771  currentSubstitution[variablePair.first] = variablePair.second;
772  }
773

775  extendedPredicateValuation = odd.getEncoding(stateToOffset ? (*stateToOffset)[*pathIt] : *pathIt, abstractionInformation.getNumberOfPredicates() + predicateValuationOffset);
776 
777  // Encode the predicates whose value might have changed.
778  // FIXME: could be optimized by precomputation.
779  for (uint64_t predicateIndex = 0; predicateIndex < abstractionInformation.getNumberOfPredicates(); ++predicateIndex) {
780  auto const& predicate = abstractionInformation.getPredicateByIndex(predicateIndex);
781  std::set<storm::expressions::Variable> usedVariables = predicate.getVariables();
782

784  for (auto usedIt = usedVariables.begin(), assignedIt = assignedVariables.begin();;) {
785  if (usedIt == usedVariables.end() || assignedIt == assignedVariables.end()) {
786  break;
787  }
788 
789  if (*usedIt == *assignedIt) {
790  containsAssignedVariables = true;
791  break;
792  }
793 
794  if (*usedIt < *assignedIt) {
795  ++usedIt;
796  } else {
797  ++assignedIt;
798  }
799  }
800 
801  if (containsAssignedVariables) {
802  auto transformedPredicate = predicate.changeManager(expressionManager).substitute(currentSubstitution);
803  predicates.back().emplace_back(extendedPredicateValuation.get(predicateIndex + predicateValuationOffset) ? transformedPredicate : !transformedPredicate);
804  }
805  }
806 
807  // Enforce constraints of all assigned variables.
808  for (auto const& constraint : abstractionInformation.getConstraints()) {
809  std::set<storm::expressions::Variable> usedVariables = constraint.getVariables();
810 
811  bool containsAssignedVariables = false;
812  for (auto usedIt = usedVariables.begin(), assignedIt = assignedVariables.begin();;) {
813  if (usedIt == usedVariables.end() || assignedIt == assignedVariables.end()) {
814  break;
815  }
816 
817  if (*usedIt == *assignedIt) {
818  containsAssignedVariables = true;
819  break;
820  }
821 
822  if (*usedIt < *assignedIt) {
823  ++usedIt;
824  } else {
825  ++assignedIt;
826  }
827  }
828 
829  if (containsAssignedVariables) {
830  auto transformedConstraint = constraint.changeManager(expressionManager).substitute(currentSubstitution);
831  predicates.back().emplace_back(transformedConstraint);
832  }
833  }
834 
835  ++actionIt;
836  ++step;
837  }
838 
839  return std::make_pair(predicates, stepVariableToCopiedVariableMap);
840  }

1108  template<typename ValueType>
1109  boost::optional<ExplicitPivotStateResult<ValueType>> pickPivotState(bool generatePredecessors, AbstractionSettings::PivotSelectionHeuristic pivotSelectionHeuristic, storm::storage::SparseMatrix<ValueType> const& transitionMatrix, std::vector<uint64_t> const& player1Grouping, std::vector<uint64_t> const& player1Labeling, storm::storage::BitVector const& initialStates, storm::storage::BitVector const& relevantStates, storm::storage::BitVector const& targetStates, ExplicitGameStrategyPair const& minStrategyPair, ExplicitGameStrategyPair const& maxStrategyPair, std::vector<ValueType> const* lowerValues = nullptr, std::vector<ValueType> const* upperValues = nullptr) {
1110 
1111  STORM_LOG_ASSERT(!lowerValues || upperValues, "Expected none or both value results.");
1112  STORM_LOG_ASSERT(!upperValues || lowerValues, "Expected none or both value results.");
1113 
1114  // Perform Dijkstra search that stays within the relevant states and searches for a (pivot) state in which
1115  // the strategies the lower or upper player 1 action leads to a player 2 state in which the choices differ.
1116  // To guarantee that the pivot state is reachable by either of the strategies, we do a parallel Dijkstra
1117  // search in both the lower and upper strategy.
1118 
1119  bool probabilityDistances = pivotSelectionHeuristic == storm::settings::modules::AbstractionSettings::PivotSelectionHeuristic::MostProbablePath;
1120  uint64_t numberOfStates = initialStates.size();
1121  ValueType inftyDistance = probabilityDistances ? storm::utility::zero<ValueType>() : storm::utility::infinity<ValueType>();
1122  ValueType zeroDistance = probabilityDistances ? storm::utility::one<ValueType>() : storm::utility::zero<ValueType>();
1123 
1124  // Create storages for the lower and upper Dijkstra search.
1125  std::vector<ValueType> lowerDistances(numberOfStates, inftyDistance);
1126  std::vector<std::pair<uint64_t, uint64_t>> lowerPredecessors;
1127  std::vector<ValueType> upperDistances(numberOfStates, inftyDistance);
1128  std::vector<std::pair<uint64_t, uint64_t>> upperPredecessors;
1129 
1130  if (generatePredecessors) {
1131  lowerPredecessors.resize(numberOfStates, std::make_pair(ExplicitPivotStateResult<ValueType>::NO_PREDECESSOR, ExplicitPivotStateResult<ValueType>::NO_PREDECESSOR));
1132  upperPredecessors.resize(numberOfStates, std::make_pair(ExplicitPivotStateResult<ValueType>::NO_PREDECESSOR, ExplicitPivotStateResult<ValueType>::NO_PREDECESSOR));
1133  }
1134 
1135  // Use set as priority queue with unique membership.
1136  std::set<ExplicitDijkstraQueueElement<ValueType>, ExplicitDijkstraQueueElementLess<ValueType>> dijkstraQueue;
1137 
1138  for (auto initialState : initialStates) {
1139  if (!relevantStates.get(initialState)) {
1140  continue;
1141  }
1142 
1143  lowerDistances[initialState] = zeroDistance;
1144  upperDistances[initialState] = zeroDistance;
1145  dijkstraQueue.emplace(zeroDistance, initialState, true);
1146  dijkstraQueue.emplace(zeroDistance, initialState, false);
1147  }
1148 
1149  // For some heuristics, we need to potentially find more than just one pivot.
1150  bool considerDeviation = (pivotSelectionHeuristic == storm::settings::modules::AbstractionSettings::PivotSelectionHeuristic::NearestMaximalDeviation || pivotSelectionHeuristic == storm::settings::modules::AbstractionSettings::PivotSelectionHeuristic::MaxWeightedDeviation) && lowerValues && upperValues;
1151  bool foundPivotState = false;
1152 
1153  ExplicitDijkstraQueueElement<ValueType> pivotState(inftyDistance, 0, true);
1154  ValueType pivotStateDeviation = storm::utility::zero<ValueType>();
1155  auto const& player2Grouping = transitionMatrix.getRowGroupIndices();
1156 
1157  while (!dijkstraQueue.empty()) {
1158  // Take out currently best state.
1159  auto currentDijkstraElement = *dijkstraQueue.begin();
1160  ValueType currentDistance = currentDijkstraElement.distance;
1161  uint64_t currentState = currentDijkstraElement.state;
1162  bool currentLower = currentDijkstraElement.lower;
1163  dijkstraQueue.erase(dijkstraQueue.begin());
1164 
1165  if (foundPivotState && (probabilityDistances ? currentDistance < pivotState.distance : currentDistance > pivotState.distance)) {
1166  if (pivotSelectionHeuristic != storm::settings::modules::AbstractionSettings::PivotSelectionHeuristic::MaxWeightedDeviation) {
1167  // For the nearest maximal deviation and most probable path heuristics, future pivot states are
1168  // not important any more, because their distance will be strictly larger, so we can return the
1169  // current pivot state.
1170 
1171  return ExplicitPivotStateResult<ValueType>(pivotState.state, pivotState.distance, pivotState.lower ? std::move(lowerPredecessors) : std::move(upperPredecessors));
1172  } else if (pivotStateDeviation >= currentDistance) {
1173  // If the heuristic is maximal weighted deviation and the weighted deviation for any future pivot
1174  // state is necessarily at most as high as the current one, we can abort the search.
1175  return ExplicitPivotStateResult<ValueType>(pivotState.state, pivotState.distance, pivotState.lower ? std::move(lowerPredecessors) : std::move(upperPredecessors));
1176  }
1177  }
1178 
1179  // Determine whether the current state is a pivot state.
1180  bool isPivotState = false;
1181  if (minStrategyPair.getPlayer1Strategy().hasDefinedChoice(currentState)) {
1182  uint64_t player2Successor = minStrategyPair.getPlayer1Strategy().getChoice(currentState);
1183  if (minStrategyPair.getPlayer2Strategy().hasDefinedChoice(player2Successor) && maxStrategyPair.getPlayer2Strategy().hasDefinedChoice(player2Successor) && minStrategyPair.getPlayer2Strategy().getChoice(player2Successor) != maxStrategyPair.getPlayer2Strategy().getChoice(player2Successor)) {
1184  isPivotState = true;
1185  }
1186  }
1187  if (!isPivotState && maxStrategyPair.getPlayer1Strategy().hasDefinedChoice(currentState)) {
1188  uint64_t player2Successor = maxStrategyPair.getPlayer1Strategy().getChoice(currentState);
1189  if (minStrategyPair.getPlayer2Strategy().hasDefinedChoice(player2Successor) && maxStrategyPair.getPlayer2Strategy().hasDefinedChoice(player2Successor) && minStrategyPair.getPlayer2Strategy().getChoice(player2Successor) != maxStrategyPair.getPlayer2Strategy().getChoice(player2Successor)) {
1190  isPivotState = true;
1191  }
1192  }
1193 
1194  // If it is indeed a pivot state, we can potentially abort the search here.
1195  if (isPivotState) {
1196  if (considerDeviation && foundPivotState) {
1197  ValueType deviationOfCurrentState = (*upperValues)[currentState] - (*lowerValues)[currentState];
1198 
1199  if (deviationOfCurrentState > pivotStateDeviation) {
1200  pivotState = currentDijkstraElement;
1201  pivotStateDeviation = deviationOfCurrentState;
1202  if (pivotSelectionHeuristic == storm::settings::modules::AbstractionSettings::PivotSelectionHeuristic::MaxWeightedDeviation) {
1203  // Scale the deviation with the distance (probability) for this heuristic.
1204  pivotStateDeviation *= currentDistance;
1205  }
1206  }
1207  } else if (!foundPivotState) {
1208  pivotState = currentDijkstraElement;
1209  foundPivotState = true;
1210  }
1211 
1212  // If there is no need to look at other deviations, stop here.
1213  if (!considerDeviation) {
1214  return ExplicitPivotStateResult<ValueType>(pivotState.state, pivotState.distance, pivotState.lower ? std::move(lowerPredecessors) : std::move(upperPredecessors));
1215  }
1216  }
1217 
1218  // We only need to search further if the state has some value deviation.
1219  if (!lowerValues || !upperValues || (*lowerValues)[currentState] < (*upperValues)[currentState]) {
1220  if (currentLower) {
1221  performDijkstraStep(dijkstraQueue, probabilityDistances, lowerDistances, lowerPredecessors, generatePredecessors, true, currentState, currentDistance, isPivotState, minStrategyPair, maxStrategyPair, player1Labeling, player2Grouping, transitionMatrix, targetStates, relevantStates);
1222  } else {
1223  performDijkstraStep(dijkstraQueue, probabilityDistances, upperDistances, upperPredecessors, generatePredecessors, false, currentState, currentDistance, isPivotState, maxStrategyPair, minStrategyPair, player1Labeling, player2Grouping, transitionMatrix, targetStates, relevantStates);
1224  }
1225  }
1226  }
1227 
1228  if (foundPivotState) {
1229  return ExplicitPivotStateResult<ValueType>(pivotState.state, pivotState.distance, pivotState.lower ? std::move(lowerPredecessors) : std::move(upperPredecessors));
1230  }
1231 
1232  // If we arrived at this point, we have explored all relevant states, but none of them was a pivot state,
1233  // which can happen when trying to refine using the qualitative result only.
1234  STORM_LOG_TRACE("Did not find pivot state in explicit Dijkstra search.");
1235  return boost::none;
1236  }
