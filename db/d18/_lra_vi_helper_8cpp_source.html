27  template <typename ValueType, typename ComponentType, LraViTransitionsType TransitionsType>
28  LraViHelper<ValueType, ComponentType, TransitionsType>::LraViHelper(ComponentType const& component, storm::storage::SparseMatrix<ValueType> const& transitionMatrix, ValueType const& aperiodicFactor, storm::storage::BitVector const* timedStates, std::vector<ValueType> const* exitRates) : _transitionMatrix(transitionMatrix), _timedStates(timedStates), _hasInstantStates(TransitionsType == LraViTransitionsType::DetTsNondetIs || TransitionsType == LraViTransitionsType::DetTsDetIs), _Tsx1IsCurrent(false) {
29  setComponent(component);
30 
31  // Run through the component and collect some data:
32  // We create two submodels, one consisting of the timed states of the component and one consisting of the instant states of the component.
33  // For this, we create a state index map that point from state indices of the input model to indices of the corresponding submodel of that state.
34  std::map<uint64_t, uint64_t> toSubModelStateMapping;
35  // We also obtain state and choices counts of the two submodels
36  uint64_t numTsSubModelStates(0), numTsSubModelChoices(0);
37  uint64_t numIsSubModelStates(0), numIsSubModelChoices(0);
38  // We will need to uniformize the timed MEC states by introducing a selfloop.
39  // For this, we need to find a uniformization rate which will be a little higher (given by aperiodicFactor) than the maximum rate occurring in the component.
40  _uniformizationRate = exitRates == nullptr ? storm::utility::one<ValueType>() : storm::utility::zero<ValueType>();
41  // Now run over the MEC and collect the required data.
42  for (auto const& element : _component) {
43  uint64_t componentState = element.first;
44  if (isTimedState(componentState)) {
45  toSubModelStateMapping.emplace(componentState, numTsSubModelStates);
46  ++numTsSubModelStates;
47  numTsSubModelChoices += element.second.size();
48  STORM_LOG_ASSERT(nondetTs() || element.second.size() == 1, "Timed state has multiple choices but only a single choice was expected.");
49  if (exitRates) {
50  _uniformizationRate = std::max(_uniformizationRate, (*exitRates)[componentState]);
51  }
52  } else {
53  toSubModelStateMapping.emplace(componentState, numIsSubModelStates);
54  ++numIsSubModelStates;
55  numIsSubModelChoices += element.second.size();
56  STORM_LOG_ASSERT(nondetIs() || element.second.size() == 1, "Instant state has multiple choices but only a single choice was expected.");
57  }
58  }
59  assert(numIsSubModelStates + numTsSubModelStates == _component.size());
60  assert(_hasInstantStates || numIsSubModelStates == 0);
61  STORM_LOG_ASSERT(nondetTs() || numTsSubModelStates == numTsSubModelChoices, "Unexpected choice count of deterministic timed submodel.");
62  STORM_LOG_ASSERT(nondetIs() || numIsSubModelStates == numIsSubModelChoices, "Unexpected choice count of deterministic instant submodel.");
63  _hasInstantStates = _hasInstantStates && numIsSubModelStates > 0;
64  STORM_LOG_THROW(numTsSubModelStates > 0, storm::exceptions::InvalidOperationException, "Bottom Component has no timed states. Computation of Long Run Average values not supported. Is this a Markov Automaton with Zeno behavior?");
65 
66  // We make sure that every timed state gets a selfloop to make the model aperiodic
67  _uniformizationRate *= storm::utility::one<ValueType>() + aperiodicFactor;
68 
69  // Now build the timed and the instant submodels.
70  // In addition, we also need the transitions between the two.
71  storm::storage::SparseMatrixBuilder<ValueType> tsTransitionsBuilder(numTsSubModelChoices, numTsSubModelStates, 0, true, nondetTs(), nondetTs() ? numTsSubModelStates : 0);
72  storm::storage::SparseMatrixBuilder<ValueType> tsToIsTransitionsBuilder, isTransitionsBuilder, isToTsTransitionsBuilder;
73  if (_hasInstantStates) {
74  tsToIsTransitionsBuilder = storm::storage::SparseMatrixBuilder<ValueType>(numTsSubModelChoices, numIsSubModelStates, 0, true, nondetTs(), nondetTs() ? numTsSubModelStates : 0);
75  isTransitionsBuilder = storm::storage::SparseMatrixBuilder<ValueType>(numIsSubModelChoices, numIsSubModelStates, 0, true, nondetIs(), nondetIs() ? numIsSubModelStates : 0);
76  isToTsTransitionsBuilder = storm::storage::SparseMatrixBuilder<ValueType>(numIsSubModelChoices, numTsSubModelStates, 0, true, nondetIs(), nondetIs() ? numIsSubModelStates : 0);
77  _IsChoiceValues.reserve(numIsSubModelChoices);
78  }
79  ValueType uniformizationFactor = storm::utility::one<ValueType>() / _uniformizationRate;
80  uint64_t currTsRow = 0;
81  uint64_t currIsRow = 0;
82  for (auto const& element : _component) {
83  uint64_t componentState = element.first;
84  if (isTimedState(componentState)) {
85  // The currently processed state is timed.
86  if (nondetTs()) {
87  tsTransitionsBuilder.newRowGroup(currTsRow);
88  if (_hasInstantStates) {
89  tsToIsTransitionsBuilder.newRowGroup(currTsRow);
90  }
91  }
92  // If there are exit rates, the uniformization factor needs to be updated.
93  if (exitRates) {
94  uniformizationFactor = (*exitRates)[componentState] / _uniformizationRate;
95  }
96  // We need to uniformize which means that a diagonal entry for the selfloop will be inserted.
97  ValueType selfLoopProb = storm::utility::one<ValueType>() - uniformizationFactor;
98  for (auto const& componentChoice : element.second) {
99  tsTransitionsBuilder.addDiagonalEntry(currTsRow, selfLoopProb);
100  for (auto const& entry : this->_transitionMatrix.getRow(componentChoice)) {
101  uint64_t subModelColumn = toSubModelStateMapping[entry.getColumn()];
102  if (isTimedState(entry.getColumn())) {
103  // We have a transition from a timed state to a timed state
104  STORM_LOG_ASSERT(subModelColumn < numTsSubModelStates, "Invalid state for timed submodel");
105  tsTransitionsBuilder.addNextValue(currTsRow, subModelColumn, uniformizationFactor * entry.getValue());
106  } else {
107  // We have a transition from a timed to a instant state
108  STORM_LOG_ASSERT(subModelColumn < numIsSubModelStates, "Invalid state for instant submodel");
109  tsToIsTransitionsBuilder.addNextValue(currTsRow, subModelColumn, uniformizationFactor * entry.getValue());
110  }
111  }
112  ++currTsRow;
113  }
114  } else {
115  // The currently processed state is instant
116  if (nondetIs()) {
117  isTransitionsBuilder.newRowGroup(currIsRow);
118  isToTsTransitionsBuilder.newRowGroup(currIsRow);
119  }
120  for (auto const& componentChoice : element.second) {
121  for (auto const& entry : this->_transitionMatrix.getRow(componentChoice)) {
122  uint64_t subModelColumn = toSubModelStateMapping[entry.getColumn()];
123  if (isTimedState(entry.getColumn())) {
124  // We have a transition from an instant state to a timed state
125  STORM_LOG_ASSERT(subModelColumn < numTsSubModelStates, "Invalid state for timed submodel");
126  isToTsTransitionsBuilder.addNextValue(currIsRow, subModelColumn, entry.getValue());
127  } else {
128  // We have a transition from an instant to an instant state
129  STORM_LOG_ASSERT(subModelColumn < numIsSubModelStates, "Invalid state for instant submodel");
130  isTransitionsBuilder.addNextValue(currIsRow, subModelColumn, entry.getValue());
131  }
132  }
133  ++currIsRow;
134  }
135  }
136  }
137  _TsTransitions = tsTransitionsBuilder.build();
138  if (_hasInstantStates) {
139  _TsToIsTransitions = tsToIsTransitionsBuilder.build();
140  _IsTransitions = isTransitionsBuilder.build();
141  _IsToTsTransitions = isToTsTransitionsBuilder.build();
142  }

159  template <typename ValueType, typename ComponentType, LraViTransitionsType TransitionsType>
160  ValueType LraViHelper<ValueType, ComponentType, TransitionsType>::performValueIteration(Environment const& env, ValueGetter const& stateValueGetter, ValueGetter const& actionValueGetter, std::vector<ValueType> const* exitRates, storm::solver::OptimizationDirection const* dir, std::vector<uint64_t>* choices) {
161  initializeNewValues(stateValueGetter, actionValueGetter, exitRates);
162  ValueType precision = storm::utility::convertNumber<ValueType>(env.solver().lra().getPrecision());
163  bool relative = env.solver().lra().getRelativeTerminationCriterion();
164  boost::optional<uint64_t> maxIter;
165  if (env.solver().lra().isMaximalIterationCountSet()) {
166  maxIter = env.solver().lra().getMaximalIterationCount();
167  }
168 
169  // start the iterations
170  ValueType result = storm::utility::zero<ValueType>();
171  uint64_t iter = 0;
172  while (!maxIter.is_initialized() || iter < maxIter.get()) {
173  ++iter;
174  performIterationStep(env, dir);
175 
176  // Check if we are done
177  auto convergenceCheckResult = checkConvergence(relative, precision);
178  result = convergenceCheckResult.currentValue;
179  if (convergenceCheckResult.isPrecisionAchieved) {
180  break;
181  }
182  if (storm::utility::resources::isTerminate()) {
183  break;
184  }
185  // If there will be a next iteration, we have to prepare it.
186  prepareNextIteration(env);
187 
188  }
189  if (maxIter.is_initialized() && iter == maxIter.get()) {
190  STORM_LOG_WARN("LRA computation did not converge within " << iter << " iterations.");
191  } else if (storm::utility::resources::isTerminate()) {
192  STORM_LOG_WARN("LRA computation aborted after " << iter << " iterations.");
193  } else {
194  STORM_LOG_TRACE("LRA computation converged after " << iter << " iterations.");
195  }
196 
197  if (choices) {
198  // We will be doing one more iteration step and track scheduler choices this time.
199  prepareNextIteration(env);
200  performIterationStep(env, dir, choices);
201  }
202  return result;
203  }

205  template <typename ValueType, typename ComponentType, LraViTransitionsType TransitionsType>
206  void LraViHelper<ValueType, ComponentType, TransitionsType>::initializeNewValues(ValueGetter const& stateValueGetter, ValueGetter const& actionValueGetter, std::vector<ValueType> const* exitRates) {
207  // clear potential old values and reserve enough space for new values
208  _TsChoiceValues.clear();
209  _TsChoiceValues.reserve(_TsTransitions.getRowCount());
210  if (_hasInstantStates) {
211  _IsChoiceValues.clear();
212  _IsChoiceValues.reserve(_IsTransitions.getRowCount());
213  }
214 
215  // Set the new choice-based values
216  ValueType actionRewardScalingFactor = storm::utility::one<ValueType>() / _uniformizationRate;
217  for (auto const& element : _component) {
218  uint64_t componentState = element.first;
219  if (isTimedState(componentState)) {
220  if (exitRates) {
221  actionRewardScalingFactor = (*exitRates)[componentState] / _uniformizationRate;
222  }
223  for (auto const& componentChoice : element.second) {
224  // Compute the values obtained for this choice.
225  _TsChoiceValues.push_back(stateValueGetter(componentState) / _uniformizationRate + actionValueGetter(componentChoice) * actionRewardScalingFactor);
226  }
227  } else {
228  for (auto const& componentChoice : element.second) {
229  // Compute the values obtained for this choice.
230  // State values do not count here since no time passes in instant states.
231  _IsChoiceValues.push_back(actionValueGetter(componentChoice));
232  }
233  }
234  }
235 
236  // Set-up new iteration vectors for timed states
237  _Tsx1.assign(_TsTransitions.getRowGroupCount(), storm::utility::zero<ValueType>());
238  _Tsx2 = _Tsx1;
239 
240  if (_hasInstantStates) {
241  // Set-up vectors for storing intermediate results for instant states.
242  _Isx.resize(_IsTransitions.getRowGroupCount(), storm::utility::zero<ValueType>());
243  _Isb = _IsChoiceValues;
244  }

247  template <typename ValueType, typename ComponentType, LraViTransitionsType TransitionsType>
248  void LraViHelper<ValueType, ComponentType, TransitionsType>::prepareSolversAndMultipliers(const Environment& env, storm::solver::OptimizationDirection const* dir) {
249  _TsMultiplier = storm::solver::MultiplierFactory<ValueType>().create(env, _TsTransitions);
250  if (_hasInstantStates) {
251  if (_IsTransitions.getNonzeroEntryCount() > 0) {
252  // Set-up a solver for transitions within instant states
253  _IsSolverEnv = std::make_unique<storm::Environment>(env);
254  if (env.solver().isForceSoundness()) {
255  // To get correct results, the inner equation systems are solved exactly.
256  // TODO investigate how an error would propagate
257  _IsSolverEnv->solver().setForceExact(true);
258  }
259  bool isAcyclic = !storm::utility::graph::hasCycle(_IsTransitions);
260  if (isAcyclic) {
261  STORM_LOG_INFO("Instant transitions are acyclic.");
262  _IsSolverEnv->solver().minMax().setMethod(storm::solver::MinMaxMethod::Acyclic);
263  _IsSolverEnv->solver().setLinearEquationSolverType(storm::solver::EquationSolverType::Acyclic);
264  }
265  if (nondetIs()) {
266  storm::solver::GeneralMinMaxLinearEquationSolverFactory<ValueType> factory;
267  _NondetIsSolver = factory.create(*_IsSolverEnv, _IsTransitions);
268  _NondetIsSolver->setHasUniqueSolution(true); // Assume non-zeno MA
269  _NondetIsSolver->setHasNoEndComponents(true); // assume non-zeno MA
270  _NondetIsSolver->setCachingEnabled(true);
271  auto req = _NondetIsSolver->getRequirements(*_IsSolverEnv, *dir);
272  req.clearUniqueSolution();
273  if (isAcyclic) {
274  req.clearAcyclic();
275  }
276  // Computing a priori lower/upper bounds is not particularly easy, as there might be selfloops with high probabilities
277  // Which accumulate a lot of reward. Moreover, the right-hand-side of the equation system changes dynamically.
278  STORM_LOG_THROW(!req.hasEnabledCriticalRequirement(), storm::exceptions::UnmetRequirementException, "The solver requirement " << req.getEnabledRequirementsAsString() << " has not been cleared.");
279  _NondetIsSolver->setRequirementsChecked(true);
280  } else {
281  storm::solver::GeneralLinearEquationSolverFactory<ValueType> factory;
282  if (factory.getEquationProblemFormat(*_IsSolverEnv) != storm::solver::LinearEquationSolverProblemFormat::FixedPointSystem) {
283  // We need to convert the transition matrix connecting instant states
284  // TODO: This could have been done already during construction of the matrix.
285  // Insert diagonal entries.
286  storm::storage::SparseMatrix<ValueType> converted(_IsTransitions, true);
287  // Compute A' = 1-A
288  converted.convertToEquationSystem();
289  STORM_LOG_WARN("The selected equation solver requires to create a temporary " << converted.getDimensionsAsString());
290  // Note that the solver has ownership of the converted matrix.
291  _DetIsSolver = factory.create(*_IsSolverEnv, std::move(converted));
292  } else {
293  _DetIsSolver = factory.create(*_IsSolverEnv, _IsTransitions);
294  }
295  _DetIsSolver->setCachingEnabled(true);
296  auto req = _DetIsSolver->getRequirements(*_IsSolverEnv);
297  if (isAcyclic) {
298  req.clearAcyclic();
299  }
300  // A priori lower/upper bounds are hard (see MinMax version of this above)
301  STORM_LOG_THROW(!req.hasEnabledCriticalRequirement(), storm::exceptions::UnmetRequirementException, "The solver requirement " << req.getEnabledRequirementsAsString() << " has not been cleared.");
302  }
303  }
304 
305  // Set up multipliers for transitions connecting timed and instant states
306  _TsToIsMultiplier = storm::solver::MultiplierFactory<ValueType>().create(env, _TsToIsTransitions);
307  _IsToTsMultiplier = storm::solver::MultiplierFactory<ValueType>().create(env, _IsToTsTransitions);
308  }

312  void LraViHelper<ValueType, ComponentType, TransitionsType>::setInputModelChoices(std::vector<uint64_t>& choices, std::vector<uint64_t> const& localMecChoices, bool setChoiceZeroToTimedStates, bool setChoiceZeroToInstantStates) const {
313  // Transform the local choices (within this mec) to choice indices for the input model
314  uint64_t localState = 0;
315  for (auto const& element : _component) {
316  uint64_t elementState = element.first;
317  if ((setChoiceZeroToTimedStates && isTimedState(elementState)) || (setChoiceZeroToInstantStates && !isTimedState(elementState))) {
318  choices[elementState] = 0;
319  } else {
320  uint64_t choice = localMecChoices[localState];
321  STORM_LOG_ASSERT(choice < element.second.size(), "The selected choice does not seem to exist.");
322  auto globalChoiceIndexIt = element.second.begin();
323  for (uint64_t i = 0; i < choice; ++i) {
324  ++globalChoiceIndexIt;
325  }
326  uint64_t globalChoiceIndex = *(globalChoiceIndexIt);
327  choices[elementState] = globalChoiceIndex - _transitionMatrix.getRowGroupIndices()[elementState];
328  ++localState;
329  }
330  }
331  STORM_LOG_ASSERT(localState == localMecChoices.size(), "Did not traverse all component states.");
332  }

334  template <typename ValueType, typename ComponentType, LraViTransitionsType TransitionsType>
335  void LraViHelper<ValueType, ComponentType, TransitionsType>::performIterationStep(Environment const& env, storm::solver::OptimizationDirection const* dir, std::vector<uint64_t>* choices) {
336  STORM_LOG_ASSERT(!((nondetTs() || nondetIs()) && dir == nullptr), "No optimization direction provided for model with nondeterminism");
337  // Initialize value vectors, multiplers, and solver if this has not been done, yet
338  if (!_TsMultiplier) {
339  prepareSolversAndMultipliers(env, dir);
340  }
341 
342  // Compute new x values for the timed states
343  // Flip what is new and what is old
344  _Tsx1IsCurrent = !_Tsx1IsCurrent;
345  // At this point, xOld() points to what has been computed in the most recent call of performIterationStep (initially, this is the 0-vector).
346  // The result of this ongoing computation will be stored in xNew()
347 
348  // Compute the values obtained by a single uniformization step between timed states only
349  if (nondetTs()) {
350  if (choices == nullptr) {
351  _TsMultiplier->multiplyAndReduce(env, *dir, xOld(), &_TsChoiceValues, xNew());
352  } else {
353  // Also keep track of the choices made.
354  std::vector<uint64_t> tsChoices(_TsTransitions.getRowGroupCount());
355  _TsMultiplier->multiplyAndReduce(env, *dir, xOld(), &_TsChoiceValues, xNew(), &tsChoices);
356  // Note that nondeterminism within the timed states means that there can not be instant states (We either have MDPs or MAs)
357  // Hence, in this branch we don't have to care for choices at instant states.
358  STORM_LOG_ASSERT(!_hasInstantStates, "Nondeterministic timed states are only supported if there are no instant states.");
359  setInputModelChoices(*choices, tsChoices);
360  }
361  } else {
362  _TsMultiplier->multiply(env, xOld(), &_TsChoiceValues, xNew());
363  }
364  if (_hasInstantStates) {
365  // Add the values obtained by taking a single uniformization step that leads to an instant state followed by arbitrarily many instant steps.
366  // First compute the total values when taking arbitrarily many instant transitions (in no time)
367  if (_NondetIsSolver) {
368  // We might need to track the optimal choices.
369  if (choices == nullptr) {
370  _NondetIsSolver->solveEquations(*_IsSolverEnv, *dir, _Isx, _Isb);
371  } else {
372  _NondetIsSolver->setTrackScheduler();
373  _NondetIsSolver->solveEquations(*_IsSolverEnv, *dir, _Isx, _Isb);
374  setInputModelChoices(*choices, _NondetIsSolver->getSchedulerChoices(), true);
375  }
376  } else if (_DetIsSolver) {
377  _DetIsSolver->solveEquations(*_IsSolverEnv, _Isx, _Isb);
378  } else {
379  STORM_LOG_ASSERT(_IsTransitions.getNonzeroEntryCount() == 0, "If no solver was initialized, an empty matrix would have been expected.");
380  if (nondetIs()) {
381  if (choices == nullptr) {
382  storm::utility::vector::reduceVectorMinOrMax(*dir, _Isb, _Isx, _IsTransitions.getRowGroupIndices());
383  } else {
384  std::vector<uint64_t> psChoices(_IsTransitions.getRowGroupCount());
385  storm::utility::vector::reduceVectorMinOrMax(*dir, _Isb, _Isx, _IsTransitions.getRowGroupIndices(), &psChoices);
386  setInputModelChoices(*choices, psChoices, true);
387  }
388  } else {
389  // For deterministic instant states, there is nothing to reduce, i.e., we could just set _Isx = _Isb.
390  // For efficiency reasons, we do a swap instead:
391  _Isx.swap(_Isb);
392  // Note that at this point we have changed the contents of _Isb, but they will be overwritten anyway.
393  if (choices) {
394  // Set choice 0 to all states.
395  setInputModelChoices(*choices, {}, true, true);
396  }
397  }
398  }
399  // Now add the (weighted) values of the instant states to the values of the timed states.
400  _TsToIsMultiplier->multiply(env, _Isx, &xNew(), xNew());
401  }

404  template <typename ValueType, typename ComponentType, LraViTransitionsType TransitionsType>
405  typename LraViHelper<ValueType, ComponentType, TransitionsType>::ConvergenceCheckResult LraViHelper<ValueType, ComponentType, TransitionsType>::checkConvergence(bool relative, ValueType precision) const {
406  STORM_LOG_ASSERT(_TsMultiplier, "tried to check for convergence without doing an iteration first.");
407  // All values are scaled according to the uniformizationRate.
408  // We need to 'revert' this scaling when computing the absolute precision.
409  // However, for relative precision, the scaling cancels out.
410  ValueType threshold = relative ? precision : ValueType(precision / _uniformizationRate);
411 
412  ConvergenceCheckResult res = { true, storm::utility::one<ValueType>() };
413  // Now check whether the currently produced results are precise enough
414  STORM_LOG_ASSERT(threshold > storm::utility::zero<ValueType>(), "Did not expect a non-positive threshold.");
415  auto x1It = xOld().begin();
416  auto x1Ite = xOld().end();
417  auto x2It = xNew().begin();
418  ValueType maxDiff = (*x2It - *x1It);
419  ValueType minDiff = maxDiff;
420  // The difference between maxDiff and minDiff is zero at this point. Thus, it doesn't make sense to check the threshold now.
421  for (++x1It, ++x2It; x1It != x1Ite; ++x1It, ++x2It) {
422  ValueType diff = (*x2It - *x1It);
423  // Potentially update maxDiff or minDiff
424  bool skipCheck = false;
425  if (maxDiff < diff) {
426  maxDiff = diff;
427  } else if (minDiff > diff) {
428  minDiff = diff;
429  } else {
430  skipCheck = true;
431  }
432  // Check convergence
433  if (!skipCheck && (maxDiff - minDiff) > (relative ? (threshold * minDiff) : threshold)) {
434  res.isPrecisionAchieved = false;
435  break;
436  }
437  }
438 
439  // Compute the average of the maximal and the minimal difference.
440  ValueType avgDiff = (maxDiff + minDiff) / (storm::utility::convertNumber<ValueType>(2.0));
441 
442  // "Undo" the scaling of the values
443  res.currentValue = avgDiff * _uniformizationRate;
444  return res;
445  }

447  template <typename ValueType, typename ComponentType, LraViTransitionsType TransitionsType>
448  void LraViHelper<ValueType, ComponentType, TransitionsType>::prepareNextIteration(Environment const& env) {
449  // To avoid large (and numerically unstable) x-values, we substract a reference value.
450  ValueType referenceValue = xNew().front();
451  storm::utility::vector::applyPointwise<ValueType, ValueType>(xNew(), xNew(), [&referenceValue] (ValueType const& x_i) -> ValueType { return x_i - referenceValue; });
452  if (_hasInstantStates) {
453  // Update the RHS of the equation system for the instant states by taking the new values of timed states into account.
454  STORM_LOG_ASSERT(!nondetTs(), "Nondeterministic timed states not expected when there are also instant states.");
455  _IsToTsMultiplier->multiply(env, xNew(), &_IsChoiceValues, _Isb);
456  }

459  template <typename ValueType, typename ComponentType, LraViTransitionsType TransitionsType>
