86  template<typename ValueType>
87  storm::storage::BitVector QualitativeAnalysisOnGraphs<ValueType>::analyseProb1Max(storm::storage::BitVector const& okay, storm::storage::BitVector const& good) const {
88  storm::storage::BitVector newGoalStates = storm::utility::graph::performProb1A(pomdp.getTransitionMatrix(), pomdp.getTransitionMatrix().getRowGroupIndices(), pomdp.getBackwardTransitions(), okay, good);
89  STORM_LOG_TRACE("Prob1A states according to MDP: " << newGoalStates);
90  // Now find a set of observations such that there is a memoryless scheduler inducing prob. 1 for each state whose observation is in the set.
91  storm::storage::BitVector potentialGoalStates = storm::utility::graph::performProb1E(pomdp.getTransitionMatrix(), pomdp.getTransitionMatrix().getRowGroupIndices(), pomdp.getBackwardTransitions(), okay, newGoalStates);
92  STORM_LOG_TRACE("Prob1E states according to MDP: " << potentialGoalStates);
93 
94  storm::storage::BitVector notGoalStates = ~potentialGoalStates;
95  storm::storage::BitVector potentialGoalObservations(pomdp.getNrObservations(), true);
96  for (uint64_t observation = 0; observation < pomdp.getNrObservations(); ++observation) {
97  for (uint64_t state : notGoalStates) {
98  potentialGoalObservations.set(pomdp.getObservation(state), false);
99  }
100  }
101  STORM_LOG_TRACE("Prob1E observations according to MDP: " << potentialGoalObservations);
102 
103  std::vector<std::vector<uint64_t>> statesPerObservation(pomdp.getNrObservations(), std::vector<uint64_t>());
104  for (uint64_t state : potentialGoalStates) {
105  statesPerObservation[pomdp.getObservation(state)].push_back(state);
106  }
107  storm::storage::BitVector singleObservationStates(pomdp.getNumberOfStates());
108  for (uint64_t state = 0; state < pomdp.getNumberOfStates(); ++state) {
109  if(statesPerObservation[pomdp.getObservation(state)].size() == 1) {
110  singleObservationStates.set(state);
111  }
112  }
113 
114 
115  storm::storage::BitVector goalStates(pomdp.getNumberOfStates());
116  while (goalStates != newGoalStates) {
117  goalStates = storm::utility::graph::performProb1A(pomdp.getTransitionMatrix(), pomdp.getTransitionMatrix().getRowGroupIndices(), pomdp.getBackwardTransitions(), okay, newGoalStates);
118  goalStates = storm::utility::graph::performProb1E(pomdp.getTransitionMatrix(), pomdp.getTransitionMatrix().getRowGroupIndices(), pomdp.getBackwardTransitions(), okay & singleObservationStates, goalStates);
119  newGoalStates = goalStates;
120  STORM_LOG_TRACE("Prob1A states according to MDP: " << newGoalStates);
121  for (uint64_t observation : potentialGoalObservations) {
122  STORM_LOG_TRACE("Consider observation " << observation);
123  uint64_t actsForObservation = pomdp.getTransitionMatrix().getRowGroupSize(statesPerObservation[observation][0]);
124  // Search whether we find an action that works for this observation.
125  for (uint64_t act = 0; act < actsForObservation; act++) {
126  STORM_LOG_TRACE("Consider action " << act);
127 
128  bool isGoalAction = true; // Assume that this works, then check whether we find a violation.
129  for (uint64_t state : statesPerObservation[observation]) {
130  STORM_LOG_TRACE("Consider state " << state);
131  if (newGoalStates.get(state)) {
132  STORM_LOG_TRACE("Already a goal state " << state);
133 
134  // A state is only a goal state if all actions work,
135  // or if all states with the same observation are goal states (and then, it does not matter which action is a goal action).
136  // Notice that this can mean that we wrongly conclude that some action is okay even if this is not the correct action (but then some other action exists which is okay for all states).
137  continue;
138  }
139  uint64_t row = pomdp.getNondeterministicChoiceIndices()[state] + act;
140  // Check whether all actions lead with a positive probabilty to a goal, and with zero probability to another (non-goal) state.
141  bool hasGoalEntry = false;
142  for (auto const& entry : pomdp.getTransitionMatrix().getRow(row)) {
143  assert(!storm::utility::isZero(entry.getValue()));
144 
145  if(newGoalStates.get(entry.getColumn())) {
146  STORM_LOG_TRACE("Reaches state " << entry.getColumn() << " which is a PROB1e state");
147  hasGoalEntry = true;
148  }
149  else if(pomdp.getObservation(entry.getColumn()) != observation) {
150  STORM_LOG_TRACE("Reaches state " << entry.getColumn() << " which is not a PROB1e state");
151  isGoalAction = false;
152  break;
153  }
154  }
155  if (!isGoalAction || !hasGoalEntry) {
156  // Action is definitiely not a goal action, no need to check further states.
157  isGoalAction = false;
158  break;
159  }
160  }
161  if (isGoalAction) {
162  for (uint64_t state : statesPerObservation[observation]) {
163  newGoalStates.set(state);
164  }
165  // No need to check other actions.
166  break;
167  }
168  }
169  }
170  }
171  // Notice that the goal states are not observable, i.e., the observations may not be sufficient to decide whether we are in a goal state.
172  return goalStates;
173
