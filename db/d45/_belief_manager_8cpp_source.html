225  template<typename PomdpType, typename BeliefValueType, typename StateType>
226  bool BeliefManager<PomdpType, BeliefValueType, StateType>::assertBelief(BeliefType const &belief) const {
227  BeliefValueType sum = storm::utility::zero<ValueType>();
228  boost::optional<uint32_t> observation;
229  for (auto const &entry : belief) {
230  if (entry.first >= pomdp.getNumberOfStates()) {
231  STORM_LOG_ERROR("Belief does refer to non-existing pomdp state " << entry.first << ".");
232  return false;
233  }
234  uint64_t entryObservation = pomdp.getObservation(entry.first);
235  if (observation) {
236  if (observation.get() != entryObservation) {
237  STORM_LOG_ERROR("Beliefsupport contains different observations.");
238  return false;
239  }
240  } else {
241  observation = entryObservation;
242  }
243  // Don't use cc for these checks, because computations with zero are usually fine
244  if (storm::utility::isZero(entry.second)) {
245  // We assume that beliefs only consider their support.
246  STORM_LOG_ERROR("Zero belief probability.");
247  return false;
248  }
249  if (entry.second < storm::utility::zero<BeliefValueType>()) {
250  STORM_LOG_ERROR("Negative belief probability.");
251  return false;
252  }
253  if (cc.isLess(storm::utility::one<BeliefValueType>(), entry.second)) {
254  STORM_LOG_ERROR("Belief probability greater than one.");
255  return false;
256  }
257  sum += entry.second;
258  }
259  if (!cc.isOne(sum)) {
260  STORM_LOG_ERROR("Belief does not sum up to one. (" << sum << " instead).");
261  return false;
262  }
263  return true;
264  }

266  template<typename PomdpType, typename BeliefValueType, typename StateType>
267  bool BeliefManager<PomdpType, BeliefValueType, StateType>::assertTriangulation(BeliefType const &belief, Triangulation const &triangulation) const {
268  if (triangulation.weights.size() != triangulation.gridPoints.size()) {
269  STORM_LOG_ERROR("Number of weights and points in triangulation does not match.");
270  return false;
271  }
272  if (triangulation.size() == 0) {
273  STORM_LOG_ERROR("Empty triangulation.");
274  return false;
275  }
276  BeliefType triangulatedBelief;
277  BeliefValueType weightSum = storm::utility::zero<BeliefValueType>();
278  for (uint64_t i = 0; i < triangulation.weights.size(); ++i) {
279  if (cc.isZero(triangulation.weights[i])) {
280  STORM_LOG_ERROR("Zero weight in triangulation.");
281  return false;
282  }
283  if (cc.isLess(triangulation.weights[i], storm::utility::zero<BeliefValueType>())) {
284  STORM_LOG_ERROR("Negative weight in triangulation.");
285  return false;
286  }
287  if (cc.isLess(storm::utility::one<BeliefValueType>(), triangulation.weights[i])) {
288  STORM_LOG_ERROR("Weight greater than one in triangulation.");
289  }
290  weightSum += triangulation.weights[i];
291  BeliefType const &gridPoint = getBelief(triangulation.gridPoints[i]);
292  for (auto const &pointEntry : gridPoint) {
293  BeliefValueType &triangulatedValue = triangulatedBelief.emplace(pointEntry.first, storm::utility::zero<ValueType>()).first->second;
294  triangulatedValue += triangulation.weights[i] * pointEntry.second;
295  }
296  }
297  if (!cc.isOne(weightSum)) {
298  STORM_LOG_ERROR("Triangulation weights do not sum up to one.");
299  return false;
300  }
301  if (!assertBelief(triangulatedBelief)) {
302  STORM_LOG_ERROR("Triangulated belief is not a belief.");
303  }
304  if (!isEqual(belief, triangulatedBelief)) {
305  STORM_LOG_ERROR("Belief:\n\t" << toString(belief) << "\ndoes not match triangulated belief:\n\t" << toString(triangulatedBelief) << ".");
306  return false;
307  }
308  return true;
309  }

317  template<typename PomdpType, typename BeliefValueType, typename StateType>
318  void
319  BeliefManager<PomdpType, BeliefValueType, StateType>::triangulateBeliefFreudenthal(BeliefType const &belief, BeliefValueType const &resolution, Triangulation &result) {
320  STORM_LOG_ASSERT(resolution != 0, "Invalid resolution: 0");
321  STORM_LOG_ASSERT(storm::utility::isInteger(resolution), "Expected an integer resolution");
322  StateType numEntries = belief.size();
323  // This is the Freudenthal Triangulation as described in Lovejoy (a whole lotta math)
324  // Probabilities will be triangulated to values in 0/N, 1/N, 2/N, ..., N/N
325  // Variable names are mostly based on the paper
326  // However, we speed this up a little by exploiting that belief states usually have sparse support (i.e. numEntries is much smaller than pomdp.getNumberOfStates()).
327  // Initialize diffs and the first row of the 'qs' matrix (aka v)
328  std::set<FreudenthalDiff, std::greater<FreudenthalDiff>> sorted_diffs; // d (and p?) in the paper
329  std::vector<BeliefValueType> qsRow; // Row of the 'qs' matrix from the paper (initially corresponds to v
330  qsRow.reserve(numEntries);
331  std::vector<StateType> toOriginalIndicesMap; // Maps 'local' indices to the original pomdp state indices
332  toOriginalIndicesMap.reserve(numEntries);
333  BeliefValueType x = resolution;
334  for (auto const &entry : belief) {
335  qsRow.push_back(storm::utility::floor(x)); // v
336  sorted_diffs.emplace(toOriginalIndicesMap.size(), x - qsRow.back()); // x-v
337  toOriginalIndicesMap.push_back(entry.first);
338  x -= entry.second * resolution;
339  }
340  // Insert a dummy 0 column in the qs matrix so the loops below are a bit simpler
341  qsRow.push_back(storm::utility::zero<BeliefValueType>());
342 
343  result.weights.reserve(numEntries);
344  result.gridPoints.reserve(numEntries);
345  auto currentSortedDiff = sorted_diffs.begin();
346  auto previousSortedDiff = sorted_diffs.end();
347  --previousSortedDiff;
348  for (StateType i = 0; i < numEntries; ++i) {
349  // Compute the weight for the grid points
350  BeliefValueType weight = previousSortedDiff->diff - currentSortedDiff->diff;
351  if (i == 0) {
352  // The first weight is a bit different
353  weight += storm::utility::one<ValueType>();
354  } else {
355  // 'compute' the next row of the qs matrix
356  qsRow[previousSortedDiff->dimension] += storm::utility::one<BeliefValueType>();
357  }
358  if (!cc.isZero(weight)) {
359  result.weights.push_back(weight);
360  // Compute the grid point
361  BeliefType gridPoint;
362  for (StateType j = 0; j < numEntries; ++j) {
363  BeliefValueType gridPointEntry = qsRow[j] - qsRow[j + 1];
364  if (!cc.isZero(gridPointEntry)) {
365  gridPoint[toOriginalIndicesMap[j]] = gridPointEntry / resolution;
366  }
367  }
368  result.gridPoints.push_back(getOrAddBeliefId(gridPoint));
369  }
370  previousSortedDiff = currentSortedDiff++;
371  }
372  }

374  template<typename PomdpType, typename BeliefValueType, typename StateType>
375  void BeliefManager<PomdpType, BeliefValueType, StateType>::triangulateBeliefDynamic(BeliefType const &belief, BeliefValueType const &resolution, Triangulation &result) {
376  // Find the best resolution for this belief, i.e., N such that the largest distance between one of the belief values to a value in {i/N | 0 ≤ i ≤ N} is minimal
377  STORM_LOG_ASSERT(storm::utility::isInteger(resolution), "Expected an integer resolution");
378  BeliefValueType finalResolution = resolution;
379  uint64_t finalResolutionMisses = belief.size() + 1;
380  // We don't need to check resolutions that are smaller than the maximal resolution divided by 2 (as we already checked multiples of these)
381  for (BeliefValueType currResolution = resolution; currResolution > resolution / 2; --currResolution) {
382  uint64_t currResMisses = 0;
383  bool continueWithNextResolution = false;
384  for (auto const &belEntry : belief) {
385  BeliefValueType product = belEntry.second * currResolution;
386  if (!cc.isZero(product - storm::utility::round(product))) {
387  ++currResMisses;
388  if (currResMisses >= finalResolutionMisses) {
389  // This resolution is not better than a previous resolution
390  continueWithNextResolution = true;
391  break;
392  }
393  }
394  }
395  if (!continueWithNextResolution) {
396  STORM_LOG_ASSERT(currResMisses < finalResolutionMisses, "Distance for this resolution should not be larger than a previously checked one.");
397  finalResolution = currResolution;
398  finalResolutionMisses = currResMisses;
399  if (currResMisses == 0) {
400  break;
401  }
402  }
403  }
404 
405  STORM_LOG_TRACE("Picking resolution " << finalResolution << " for belief " << toString(belief));
406 
407  // do standard freudenthal with the found resolution
408  triangulateBeliefFreudenthal(belief, finalResolution, result);
409  }

411  template<typename PomdpType, typename BeliefValueType, typename StateType>
412  typename BeliefManager<PomdpType, BeliefValueType, StateType>::Triangulation
413  BeliefManager<PomdpType, BeliefValueType, StateType>::triangulateBelief(BeliefType const &belief, BeliefValueType const &resolution) {
414  STORM_LOG_ASSERT(assertBelief(belief), "Input belief for triangulation is not valid.");
415  Triangulation result;
416  // Quickly triangulate Dirac beliefs
417  if (belief.size() == 1u) {
418  result.weights.push_back(storm::utility::one<BeliefValueType>());
419  result.gridPoints.push_back(getOrAddBeliefId(belief));
420  } else {

422  switch (triangulationMode) {
423  case TriangulationMode::Static:
424  triangulateBeliefFreudenthal(belief, ceiledResolution, result);

431  }
432  }

434  return result;

437  template<typename PomdpType, typename BeliefValueType, typename StateType>
438  std::vector<std::pair<typename BeliefManager<PomdpType, BeliefValueType, StateType>::BeliefId, typename BeliefManager<PomdpType, BeliefValueType, StateType>::ValueType>>
439  BeliefManager<PomdpType, BeliefValueType, StateType>::expandInternal(BeliefId const &beliefId, uint64_t actionIndex,
440  boost::optional<std::vector<BeliefValueType>> const &observationTriangulationResolutions) {
441  std::vector<std::pair<BeliefId, ValueType>> destinations;
442 
443  BeliefType belief = getBelief(beliefId);
444 
445  // Find the probability we go to each observation
446  BeliefType successorObs; // This is actually not a belief but has the same type
447  for (auto const &pointEntry : belief) {
448  uint64_t state = pointEntry.first;
449  for (auto const &pomdpTransition : pomdp.getTransitionMatrix().getRow(state, actionIndex)) {
450  if (!storm::utility::isZero(pomdpTransition.getValue())) {
451  auto obs = pomdp.getObservation(pomdpTransition.getColumn());
452  addToDistribution(successorObs, obs, pointEntry.second * pomdpTransition.getValue());
453  }
454  }
455  }
456 
457  // Now for each successor observation we find and potentially triangulate the successor belief
458  for (auto const &successor : successorObs) {
459  BeliefType successorBelief;
460  for (auto const &pointEntry : belief) {
461  uint64_t state = pointEntry.first;
462  for (auto const &pomdpTransition : pomdp.getTransitionMatrix().getRow(state, actionIndex)) {
463  if (pomdp.getObservation(pomdpTransition.getColumn()) == successor.first) {
464  ValueType prob = pointEntry.second * pomdpTransition.getValue() / successor.second;
465  addToDistribution(successorBelief, pomdpTransition.getColumn(), prob);
466  }
467  }
468  }
469  STORM_LOG_ASSERT(assertBelief(successorBelief), "Invalid successor belief.");
470 
471  // Insert the destination. We know that destinations have to be disjoined since they have different observations
472  if (observationTriangulationResolutions) {
473  Triangulation triangulation = triangulateBelief(successorBelief, observationTriangulationResolutions.get()[successor.first]);
474  for (size_t j = 0; j < triangulation.size(); ++j) {
475  // Here we additionally assume that triangulation.gridPoints does not contain the same point multiple times
476  destinations.emplace_back(triangulation.gridPoints[j], triangulation.weights[j] * successor.second);
477  }
478  } else {
479  destinations.emplace_back(getOrAddBeliefId(successorBelief), successor.second);
480  }
481  }
482 
483  return destinations;
484 
485  }
