121  void ExplicitModelBuilder<ValueType, RewardModelType, StateType>::buildMatrices(storm::storage::SparseMatrixBuilder<ValueType>& transitionMatrixBuilder, std::vector<RewardModelBuilder<typename RewardModelType::ValueType>>& rewardModelBuilders, ChoiceInformationBuilder& choiceInformationBuilder, boost::optional<storm::storage::BitVector>& markovianStates, boost::optional<storm::storage::sparse::StateValuationsBuilder>& stateValuationsBuilder) {
122 
123  // Create markovian states bit vector, if required.
124  if (generator->getModelType() == storm::generator::ModelType::MA) {
125  // The bit vector will be resized when the correct size is known.
126  markovianStates = storm::storage::BitVector(1000);
127  }
128 
129  // Create a callback for the next-state generator to enable it to request the index of states.
130  std::function<StateType (CompressedState const&)> stateToIdCallback = std::bind(&ExplicitModelBuilder<ValueType, RewardModelType, StateType>::getOrAddStateIndex, this, std::placeholders::_1);
131 
132  // If the exploration order is something different from breadth-first, we need to keep track of the remapping
133  // from state ids to row groups. For this, we actually store the reversed mapping of row groups to state-ids
134  // and later reverse it.
135  if (options.explorationOrder != ExplorationOrder::Bfs) {
136  stateRemapping = std::vector<uint_fast64_t>();
137  }
138 
139  // Let the generator create all initial states.
140  this->stateStorage.initialStateIndices = generator->getInitialStates(stateToIdCallback);
141  STORM_LOG_THROW(!this->stateStorage.initialStateIndices.empty(), storm::exceptions::WrongFormatException, "The model does not have a single initial state.");
142 
143  // Now explore the current state until there is no more reachable state.
144  uint_fast64_t currentRowGroup = 0;
145  uint_fast64_t currentRow = 0;
146 
147  auto timeOfStart = std::chrono::high_resolution_clock::now();
148  auto timeOfLastMessage = std::chrono::high_resolution_clock::now();
149  uint64_t numberOfExploredStates = 0;
150  uint64_t numberOfExploredStatesSinceLastMessage = 0;
151 
152  // Perform a search through the model.
153  while (!statesToExplore.empty()) {
154  // Get the first state in the queue.
155  CompressedState currentState = statesToExplore.front().first;
156  StateType currentIndex = statesToExplore.front().second;
157  statesToExplore.pop_front();
158 
159  // If the exploration order differs from breadth-first, we remember that this row group was actually
160  // filled with the transitions of a different state.
161  if (options.explorationOrder != ExplorationOrder::Bfs) {
162  stateRemapping.get()[currentIndex] = currentRowGroup;
163  }
164 
165  if (currentIndex % 100000 == 0) {
166  STORM_LOG_TRACE("Exploring state with id " << currentIndex << ".");
167  }
168 
169  generator->load(currentState);
170  if (stateValuationsBuilder) {
171  generator->addStateValuation(currentIndex, stateValuationsBuilder.get());
172  }
173  storm::generator::StateBehavior<ValueType, StateType> behavior = generator->expand(stateToIdCallback);
174 
175  // If there is no behavior, we might have to introduce a self-loop.
176  if (behavior.empty()) {
177  if (!storm::settings::getModule<storm::settings::modules::BuildSettings>().isDontFixDeadlocksSet() || !behavior.wasExpanded()) {
178  // If the behavior was actually expanded and yet there are no transitions, then we have a deadlock state.
179  if (behavior.wasExpanded()) {
180  this->stateStorage.deadlockStateIndices.push_back(currentIndex);
181  }
182 
183  if (markovianStates) {
184  markovianStates.get().grow(currentRowGroup + 1, false);
185  markovianStates.get().set(currentRowGroup);
186  }
187 
188  if (!generator->isDeterministicModel()) {
189  transitionMatrixBuilder.newRowGroup(currentRow);
190  }
191 
192  transitionMatrixBuilder.addNextValue(currentRow, currentIndex, storm::utility::one<ValueType>());
193 
194  for (auto& rewardModelBuilder : rewardModelBuilders) {
195  if (rewardModelBuilder.hasStateRewards()) {
196  rewardModelBuilder.addStateReward(storm::utility::zero<ValueType>());
197  }
198 
199  if (rewardModelBuilder.hasStateActionRewards()) {
200  rewardModelBuilder.addStateActionReward(storm::utility::zero<ValueType>());
201  }
202  }
203 
204  ++currentRow;
205  ++currentRowGroup;
206  } else {
207  STORM_LOG_THROW(false, storm::exceptions::WrongFormatException, "Error while creating sparse matrix from probabilistic program: found deadlock state (" << generator->stateToString(currentState) << "). For fixing these, please provide the appropriate option.");
208  }
209  } else {
210  // Add the state rewards to the corresponding reward models.
211  auto stateRewardIt = behavior.getStateRewards().begin();
212  for (auto& rewardModelBuilder : rewardModelBuilders) {
213  if (rewardModelBuilder.hasStateRewards()) {
214  rewardModelBuilder.addStateReward(*stateRewardIt);
215  }
216  ++stateRewardIt;
217  }
218 
219  // If the model is nondeterministic, we need to open a row group.
220  if (!generator->isDeterministicModel()) {
221  transitionMatrixBuilder.newRowGroup(currentRow);
222  }
223 
224  // Now add all choices.
225  for (auto const& choice : behavior) {
226 
227  // add the generated choice information
228  if (choice.hasLabels()) {
229  for (auto const& label : choice.getLabels()) {
230  choiceInformationBuilder.addLabel(label, currentRow);
231  }
232  }
233  if (choice.hasOriginData()) {
234  choiceInformationBuilder.addOriginData(choice.getOriginData(), currentRow);
235  }
236 
237  // If we keep track of the Markovian choices, store whether the current one is Markovian.
238  if (markovianStates && choice.isMarkovian()) {
239  markovianStates.get().grow(currentRowGroup + 1, false);
240  markovianStates.get().set(currentRowGroup);
241  }
242 
243  // Add the probabilistic behavior to the matrix.
244  for (auto const& stateProbabilityPair : choice) {
245  transitionMatrixBuilder.addNextValue(currentRow, stateProbabilityPair.first, stateProbabilityPair.second);
246  }
247 
248  // Add the rewards to the reward models.
249  auto choiceRewardIt = choice.getRewards().begin();
250  for (auto& rewardModelBuilder : rewardModelBuilders) {
251  if (rewardModelBuilder.hasStateActionRewards()) {
252  rewardModelBuilder.addStateActionReward(*choiceRewardIt);
253  }
254  ++choiceRewardIt;
255  }
256  ++currentRow;
257  }
258  ++currentRowGroup;
259  }
260 
261  ++numberOfExploredStates;
262  if (generator->getOptions().isShowProgressSet()) {
263  ++numberOfExploredStatesSinceLastMessage;
264 
265  auto now = std::chrono::high_resolution_clock::now();
266  auto durationSinceLastMessage = std::chrono::duration_cast<std::chrono::seconds>(now - timeOfLastMessage).count();
267  if (static_cast<uint64_t>(durationSinceLastMessage) >= generator->getOptions().getShowProgressDelay()) {
268  auto statesPerSecond = numberOfExploredStatesSinceLastMessage / durationSinceLastMessage;
269  auto durationSinceStart = std::chrono::duration_cast<std::chrono::seconds>(now - timeOfStart).count();
270  std::cout << "Explored " << numberOfExploredStates << " states in " << durationSinceStart << " seconds (currently " << statesPerSecond << " states per second)." << std::endl;
271  timeOfLastMessage = std::chrono::high_resolution_clock::now();
272  numberOfExploredStatesSinceLastMessage = 0;
273  }
274  }
275 
276  if (storm::utility::resources::isTerminate()) {
277  auto durationSinceStart = std::chrono::duration_cast<std::chrono::seconds>(std::chrono::high_resolution_clock::now() - timeOfStart).count();
278  std::cout << "Explored " << numberOfExploredStates << " states in " << durationSinceStart << " seconds before abort." << std::endl;
279  STORM_LOG_THROW(false, storm::exceptions::AbortException, "Aborted in state space exploration.");
280  break;
281  }
282  }
283 
284  if (markovianStates) {
285  // Since we now know the correct size, cut the bit vector to the correct length.
286  markovianStates->resize(currentRowGroup, false);
287  }
288 
289  // If the exploration order was not breadth-first, we need to fix the entries in the matrix according to
290  // (reversed) mapping of row groups to indices.
291  if (options.explorationOrder != ExplorationOrder::Bfs) {
292  STORM_LOG_ASSERT(stateRemapping, "Unable to fix columns without mapping.");
293  std::vector<uint_fast64_t> const& remapping = stateRemapping.get();
294 
295  // We need to fix the following entities:
296  // (a) the transition matrix
297  // (b) the initial states
298  // (c) the hash map storing the mapping states -> ids
299  // (d) fix remapping for state-generation labels
300 
301  // Fix (a).
302  transitionMatrixBuilder.replaceColumns(remapping, 0);
303 
304  // Fix (b).
305  std::vector<StateType> newInitialStateIndices(this->stateStorage.initialStateIndices.size());
306  std::transform(this->stateStorage.initialStateIndices.begin(), this->stateStorage.initialStateIndices.end(), newInitialStateIndices.begin(), [&remapping] (StateType const& state) { return remapping[state]; } );
307  std::sort(newInitialStateIndices.begin(), newInitialStateIndices.end());
308  this->stateStorage.initialStateIndices = std::move(newInitialStateIndices);
309 
310  // Fix (c).
311  this->stateStorage.stateToId.remap([&remapping] (StateType const& state) { return remapping[state]; } );
312 
313  this->generator->remapStateIds([&remapping] (StateType const& state) { return remapping[state]; });
314  }

317  template <typename ValueType, typename RewardModelType, typename StateType>
318  storm::storage::sparse::ModelComponents<ValueType, RewardModelType> ExplicitModelBuilder<ValueType, RewardModelType, StateType>::buildModelComponents() {
319 
320  // Determine whether we have to combine different choices to one or whether this model can have more than
321  // one choice per state.
322  bool deterministicModel = generator->isDeterministicModel();
323 
324  // Prepare the component builders
325  storm::storage::SparseMatrixBuilder<ValueType> transitionMatrixBuilder(0, 0, 0, false, !deterministicModel, 0);
326  std::vector<RewardModelBuilder<typename RewardModelType::ValueType>> rewardModelBuilders;
327  for (uint64_t i = 0; i < generator->getNumberOfRewardModels(); ++i) {
328  rewardModelBuilders.emplace_back(generator->getRewardModelInformation(i));
329  }
330  ChoiceInformationBuilder choiceInformationBuilder;
331  boost::optional<storm::storage::BitVector> markovianStates;
332 
333  // If we need to build state valuations, initialize them now.
334  boost::optional<storm::storage::sparse::StateValuationsBuilder> stateValuationsBuilder;
335  if (generator->getOptions().isBuildStateValuationsSet()) {
336  stateValuationsBuilder = generator->initializeStateValuationsBuilder();
337  }
338 
339  buildMatrices(transitionMatrixBuilder, rewardModelBuilders, choiceInformationBuilder, markovianStates, stateValuationsBuilder);
340 
341  // Initialize the model components with the obtained information.
342  storm::storage::sparse::ModelComponents<ValueType, RewardModelType> modelComponents(transitionMatrixBuilder.build(0, transitionMatrixBuilder.getCurrentRowGroupCount()), buildStateLabeling(), std::unordered_map<std::string, RewardModelType>(), !generator->isDiscreteTimeModel(), std::move(markovianStates));
343 
344  // Now finalize all reward models.
345  for (auto& rewardModelBuilder : rewardModelBuilders) {
346  modelComponents.rewardModels.emplace(rewardModelBuilder.getName(), rewardModelBuilder.build(modelComponents.transitionMatrix.getRowCount(), modelComponents.transitionMatrix.getColumnCount(), modelComponents.transitionMatrix.getRowGroupCount()));
347  }
348  // Build the choice labeling
349  modelComponents.choiceLabeling = choiceInformationBuilder.buildChoiceLabeling(modelComponents.transitionMatrix.getRowCount());
350 
351  // If requested, build the state valuations and choice origins
352  if (stateValuationsBuilder) {
353  modelComponents.stateValuations = stateValuationsBuilder->build(modelComponents.transitionMatrix.getRowGroupCount());
354  }
355  if (generator->getOptions().isBuildChoiceOriginsSet()) {
356  auto originData = choiceInformationBuilder.buildDataOfChoiceOrigins(modelComponents.transitionMatrix.getRowCount());
357  modelComponents.choiceOrigins = generator->generateChoiceOrigins(originData);
358  }
359  if (generator->isPartiallyObservable()) {
360  std::vector<uint32_t> classes;
361  classes.resize(stateStorage.getNumberOfStates());
362  std::unordered_map<uint32_t, std::vector<std::pair<std::vector<std::string>, uint32_t>>> observationActions;
363  for (auto const& bitVectorIndexPair : stateStorage.stateToId) {
364  uint32_t varObservation = generator->observabilityClass(bitVectorIndexPair.first);
365  classes[bitVectorIndexPair.second] = varObservation;
366  }
367 
368  modelComponents.observabilityClasses = classes;
369  if(generator->getOptions().isBuildObservationValuationsSet()) {
370  modelComponents.observationValuations = generator->makeObservationValuation();
371  }
372  }
373  return modelComponents;
374  }
