79  template <class SparseModelType>
80  void SparseCbAchievabilityQuery<SparseModelType>::initializeConstraintSystem() {
81  uint_fast64_t numStates = this->preprocessedModel->getNumberOfStates();
82  uint_fast64_t numChoices = this->preprocessedModel->getNumberOfChoices();
83  uint_fast64_t numBottomStates = this->reward0EStates.getNumberOfSetBits();
84  STORM_LOG_THROW(numBottomStates > 0, storm::exceptions::UnexpectedException, "No bottom states in the preprocessed model.");
85  storm::expressions::Expression zero = this->expressionManager->rational(storm::utility::zero<ValueType>());
86  storm::expressions::Expression one = this->expressionManager->rational(storm::utility::one<ValueType>());
87 
88  // Declare the variables for the choices and bottom states
89  expectedChoiceVariables.reserve(numChoices);
90  for (uint_fast64_t choice = 0; choice < numChoices; ++choice) {
91  expectedChoiceVariables.push_back(this->expressionManager->declareRationalVariable("y" + std::to_string(choice)));
92  }
93  bottomStateVariables.reserve(numBottomStates);
94  for (uint_fast64_t bottomState = 0; bottomState < numBottomStates; ++bottomState) {
95  bottomStateVariables.push_back(this->expressionManager->declareRationalVariable("z" + std::to_string(bottomState)));
96  }
97 
98  // assert that the values are greater zero and that the bottom state values sum up to one
99  for (auto& var : expectedChoiceVariables) {
100  solver->add(var.getExpression() >= zero);
101  }
102  std::vector<storm::expressions::Expression> bottomStateVarsAsExpression;
103  bottomStateVarsAsExpression.reserve(bottomStateVariables.size());
104  for (auto& var : bottomStateVariables) {
105  solver->add(var.getExpression() >= zero);
106  bottomStateVarsAsExpression.push_back(var.getExpression());
107  }
108  auto bottomStateSum = storm::expressions::sum(bottomStateVarsAsExpression).simplify();
109  solver->add(bottomStateSum == one);
110 
111  // assert that the "incoming" value of each state equals the "outgoing" value
112  storm::storage::SparseMatrix<ValueType> backwardsTransitions = this->preprocessedModel->getTransitionMatrix().transpose();
113  auto bottomStateVariableIt = bottomStateVariables.begin();
114  for (uint_fast64_t state = 0; state < numStates; ++state) {
115  // get the "incomming" value
116  storm::expressions::Expression value = this->preprocessedModel->getInitialStates().get(state) ? one : zero;
117  for (auto const& backwardsEntry : backwardsTransitions.getRow(state)) {
118  value = value + (this->expressionManager->rational(backwardsEntry.getValue()) * expectedChoiceVariables[backwardsEntry.getColumn()].getExpression());
119  }
120 
121  // subtract the "outgoing" value
122  for (uint_fast64_t choice = this->preprocessedModel->getTransitionMatrix().getRowGroupIndices()[state]; choice < this->preprocessedModel->getTransitionMatrix().getRowGroupIndices()[state + 1]; ++choice) {
123  value = value - expectedChoiceVariables[choice];
124  }
125  if (this->reward0EStates.get(state)) {
126  value = value - (*bottomStateVariableIt);
127  ++bottomStateVariableIt;
128  }
129  solver->add(value == zero);
130  }
131  assert(bottomStateVariableIt == bottomStateVariables.end());
132 
133  }

135  template <class SparseModelType>
136  void SparseCbAchievabilityQuery<SparseModelType>::addObjectiveConstraints() {
137  storm::expressions::Expression zero = this->expressionManager->rational(storm::utility::zero<ValueType>());
138  for (Objective<ValueType> const& obj : this->objectives) {
139  STORM_LOG_THROW(obj.formula->isRewardOperatorFormula() && obj.formula->getSubformula().isTotalRewardFormula(), storm::exceptions::InvalidOperationException, "Constraint-based solver only supports total-reward objectives. Got " << *obj.formula << " instead.");
140  STORM_LOG_THROW(obj.formula->hasBound(), storm::exceptions::InvalidOperationException, "Invoked achievability query but no bound was specified for at least one objective.");
141  STORM_LOG_THROW(obj.formula->asRewardOperatorFormula().hasRewardModelName(), storm::exceptions::InvalidOperationException, "Expected reward operator with a reward model name. Got " << *obj.formula << " instead.");
142  std::vector<ValueType> rewards = getActionBasedExpectedRewards(obj.formula->asRewardOperatorFormula().getRewardModelName());
143 
144  // Get the sum of all objective values
145  std::vector<storm::expressions::Expression> objectiveValues;
146  for (uint_fast64_t choice = 0; choice < rewards.size(); ++choice) {
147  if (!storm::utility::isZero(rewards[choice])) {
148  objectiveValues.push_back(this->expressionManager->rational(rewards[choice]) * expectedChoiceVariables[choice].getExpression());
149  }
150  }
151  if (objectiveValues.empty()) {
152  objectiveValues.push_back(this->expressionManager->rational(storm::utility::zero<storm::RationalNumber>()));
153  }
154  auto objValue = storm::expressions::sum(objectiveValues).simplify();
155 
156  // We need to actually evaluate the threshold as rational number. Otherwise a threshold like '<=16/9' might be considered as 1 due to integer division
157  storm::expressions::Expression threshold = this->expressionManager->rational(obj.formula->getThreshold().evaluateAsRational());
158  switch (obj.formula->getBound().comparisonType) {
159  case storm::logic::ComparisonType::Greater:
160  solver->add( objValue > threshold);
161  break;
162  case storm::logic::ComparisonType::GreaterEqual:
163  solver->add( objValue >= threshold);
164  break;
165  case storm::logic::ComparisonType::Less:
166  solver->add( objValue < threshold);
167  break;
168  case storm::logic::ComparisonType::LessEqual:
169  solver->add( objValue <= threshold);
170  break;
171  default:
172  STORM_LOG_THROW(false, storm::exceptions::InvalidOperationException, "One or more objectives have an invalid comparison type");
173  }
174  }
