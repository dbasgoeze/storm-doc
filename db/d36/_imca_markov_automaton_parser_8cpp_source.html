117  storm::storage::sparse::ModelComponents<ValueType> ImcaParserGrammar<ValueType, StateType>::createModelComponents() {
118 
119  // Prepare the statelabeling
120  initialStates.resize(numStates);
121  goalStates.resize(numStates);
122  markovianStates.resize(numStates);
123  storm::models::sparse::StateLabeling stateLabeling(numStates);
124  stateLabeling.addLabel("init", std::move(initialStates));
125  stateLabeling.addLabel("goal", std::move(goalStates));
126 
127  // Fix deadlocks (if required)
128  assert(stateBehaviors.size() == numStates);
129  if (!storm::settings::getModule<storm::settings::modules::BuildSettings>().isDontFixDeadlocksSet()) {
130  StateType state = 0;
131  for (auto& behavior : stateBehaviors) {
132  if (!behavior.wasExpanded()) {
133  storm::generator::Choice<ValueType, StateType> choice(0, true);
134  choice.addProbability(state, storm::utility::one<ValueType>());
135  behavior.setExpanded(true);
136  behavior.addChoice(std::move(choice));
137  markovianStates.set(state);
138  ++numChoices;
139  ++numTransitions;
140  }
141  ++state;
142  }
143  }
144 
145  // Build the transition matrix together with exit rates, reward models, and choice labeling
146  storm::storage::SparseMatrixBuilder<ValueType> matrixBuilder(numChoices, numStates, numTransitions, true, true, numStates);
147  std::vector<ValueType> exitRates;
148  exitRates.reserve(numStates);
149  boost::optional<std::vector<ValueType>> stateRewards, actionRewards;
150  if (hasStateReward) {
151  stateRewards = std::vector<ValueType>(numStates, storm::utility::zero<ValueType>());
152  }
153  if (hasActionReward) {
154  actionRewards = std::vector<ValueType>(numChoices, storm::utility::zero<ValueType>());
155  }
156  boost::optional<storm::models::sparse::ChoiceLabeling> choiceLabeling;
157  if (buildChoiceLabels) {
158  choiceLabeling = storm::models::sparse::ChoiceLabeling(numChoices);
159  }
160  StateType state = 0;
161  StateType row = 0;
162  for (auto const& behavior : stateBehaviors) {
163  matrixBuilder.newRowGroup(row);
164  if (!behavior.getStateRewards().empty()) {
165  assert(behavior.getStateRewards().size() == 1);
166  stateRewards.get()[state] = behavior.getStateRewards().front();
167  }
168  if (markovianStates.get(state)) {
169  //For Markovian states, the Markovian choice has to be the first one in the resulting transition matrix.
170  bool markovianChoiceFound = false;
171  for (auto const& choice : behavior) {
172  if (choice.isMarkovian()) {
173  STORM_LOG_THROW(!markovianChoiceFound, storm::exceptions::WrongFormatException, "Multiple Markovian choices defined for state " << state << ".");
174  markovianChoiceFound = true;
175  if (!choice.getRewards().empty()) {
176  assert(choice.getRewards().size() == 1);
177  actionRewards.get()[row] = choice.getRewards().front();
178  }
179  if (buildChoiceLabels && choice.hasLabels()) {
180  assert(choice.getLabels().size() == 1);
181  std::string const& label = *choice.getLabels().begin();
182  if (!choiceLabeling->containsLabel(label)) {
183  choiceLabeling->addLabel(label);
184  }
185  choiceLabeling->addLabelToChoice(label, row);
186  }
187  exitRates.push_back(choice.getTotalMass());
188  for (auto const& transition : choice) {
189  matrixBuilder.addNextValue(row, transition.first, static_cast<ValueType>(transition.second / exitRates.back()));
190  }
191  ++row;
192  }
193  }
194  } else {
195  exitRates.push_back(storm::utility::zero<ValueType>());
196  }
197  // Now add all probabilistic choices.
198  for (auto const& choice : behavior) {
199  if (!choice.isMarkovian()) {
200  if (!choice.getRewards().empty()) {
201  assert(choice.getRewards().size() == 1);
202  actionRewards.get()[row] = choice.getRewards().front();
203  }
204  if (buildChoiceLabels && choice.hasLabels()) {
205  assert(choice.getLabels().size() == 1);
206  std::string const& label = *choice.getLabels().begin();
207  if (!choiceLabeling->containsLabel(label)) {
208  choiceLabeling->addLabel(label);
209  }
210  choiceLabeling->addLabelToChoice(label, row);
211  }
212  for (auto const& transition : choice) {
213  matrixBuilder.addNextValue(row, transition.first, transition.second);
214  }
215  ++row;
216  }
217  }
218  ++state;
219  }
220 
221  // Finalize the model components
222  std::unordered_map<std::string, storm::models::sparse::StandardRewardModel<ValueType>> rewardModel;
223  if (hasStateReward || hasActionReward) {
224  rewardModel.insert(std::make_pair("", storm::models::sparse::StandardRewardModel<ValueType>(stateRewards, actionRewards)));
225  }
226  storm::storage::sparse::ModelComponents<ValueType> components(matrixBuilder.build(), std::move(stateLabeling), std::move(rewardModel), false, std::move(markovianStates));
227  components.exitRates = std::move(exitRates);
228  components.choiceLabeling = std::move(choiceLabeling);
229 
230  return components;
231  }
