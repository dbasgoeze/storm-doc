175  template <typename ValueType>
176  std::map<uint64_t, storm::expressions::Expression> processEc(storm::storage::MaximalEndComponent const& ec, storm::storage::SparseMatrix<ValueType> const& transitions, std::string const& varNameSuffix, std::vector<storm::expressions::Expression> const& choiceVars, storm::solver::LpSolver<ValueType>& lpModel) {
177  std::map<uint64_t, storm::expressions::Expression> ecStateVars, ecChoiceVars, ecFlowChoiceVars;
178  // Compute an upper bound on the expected number of visits of the states in this ec.
179  // First get a lower bound l on the probability of a path that leaves this MEC. 1-l is an upper bound on Pr_s(X F s).
180  // The desired upper bound is thus 1/(1-(1-l)) = 1/l. See Baier et al., CAV'17
181 
182  // To compute l, we multiply the smallest transition probabilities occurring at each state and MEC-Choice
183  // as well as the smallest 'exit' probability
184  // Observe that the actual transition probabilities do not matter for this encoding.
185  // Hence, we assume that all distributions are uniform to achieve a better (numerically more stable) bound
186  ValueType lpath = storm::utility::one<ValueType>();
187  for (auto const& stateChoices : ec) {
188  uint64_t maxEntryCount = transitions.getColumnCount();
189  // Choices that leave the EC are not considered in the in[s] below. Hence, also do not need to consider them here.
190  for (auto const& choice : stateChoices.second) {
191  maxEntryCount = std::max(maxEntryCount, transitions.getRow(choice).getNumberOfEntries());
192  }
193  lpath *= storm::utility::one<ValueType>() / storm::utility::convertNumber<ValueType>(maxEntryCount);
194  }
195  ValueType expVisitsUpperBound = storm::utility::one<ValueType>() / lpath;
196  STORM_LOG_WARN_COND(expVisitsUpperBound <= storm::utility::convertNumber<ValueType>(1000.0), "Large upper bound for expected visiting times: " << expVisitsUpperBound);
197  // create variables
198  for (auto const& stateChoices : ec) {
199  ecStateVars.emplace(stateChoices.first, lpModel.addBoundedIntegerVariable("e" + std::to_string(stateChoices.first) + varNameSuffix, storm::utility::zero<ValueType>(), storm::utility::one<ValueType>()).getExpression());
200  for (auto const& choice : stateChoices.second) {
201  ecChoiceVars.emplace(choice, lpModel.addBoundedIntegerVariable("ec" + std::to_string(choice) + varNameSuffix, storm::utility::zero<ValueType>(), storm::utility::one<ValueType>()).getExpression());
202  ecFlowChoiceVars.emplace(choice, lpModel.addBoundedContinuousVariable("f" + std::to_string(choice) + varNameSuffix, storm::utility::zero<ValueType>(), expVisitsUpperBound).getExpression());
203  }
204  }
205 
206  // create constraints
207  std::map<uint64_t, std::vector<storm::expressions::Expression>> ins, outs, ecIns;
208  for (auto const& stateChoices : ec) {
209  std::vector<storm::expressions::Expression> ecChoiceVarsAtState;
210  std::vector<storm::expressions::Expression> out;
211  for (auto const& choice : stateChoices.second) {
212  if (choiceVars[choice].isInitialized()) {
213  lpModel.addConstraint("", ecChoiceVars[choice] <= choiceVars[choice]);
214  lpModel.addConstraint("", ecFlowChoiceVars[choice] <= lpModel.getConstant(expVisitsUpperBound) * choiceVars[choice]);
215  }
216  ecChoiceVarsAtState.push_back(ecChoiceVars[choice]);
217  out.push_back(ecFlowChoiceVars[choice]);
218  ValueType fakeProbability = storm::utility::one<ValueType>() / storm::utility::convertNumber<ValueType, uint64_t>(transitions.getRow(choice).getNumberOfEntries());
219  for (auto const& transition : transitions.getRow(choice)) {
220  lpModel.addConstraint("", ecChoiceVars[choice] <= ecStateVars[transition.getColumn()]);
221  ins[transition.getColumn()].push_back(lpModel.getConstant(fakeProbability) * ecFlowChoiceVars[choice]);
222  ecIns[transition.getColumn()].push_back(ecChoiceVars[choice]);
223  }
224  }
225  lpModel.addConstraint("", ecStateVars[stateChoices.first] == storm::expressions::sum(ecChoiceVarsAtState));
226  out.push_back(lpModel.getConstant(expVisitsUpperBound) * ecStateVars[stateChoices.first]);
227  // Iterate over choices that leave the ec
228  for (uint64_t choice = transitions.getRowGroupIndices()[stateChoices.first]; choice < transitions.getRowGroupIndices()[stateChoices.first + 1]; ++choice) {
229  if (stateChoices.second.find(choice) == stateChoices.second.end()) {
230  assert(choiceVars[choice].isInitialized());
231  out.push_back(lpModel.getConstant(expVisitsUpperBound) * choiceVars[choice]);
232  }
233  }
234  outs.emplace(stateChoices.first, out);
235  }
236  uint64_t numStates = std::distance(ec.begin(), ec.end());
237  for (auto const& stateChoices : ec) {
238  auto in = ins.find(stateChoices.first);
239  STORM_LOG_ASSERT(in != ins.end(), "ec state does not seem to have an incoming transition.");
240  // Assume a uniform initial distribution
241  in->second.push_back(lpModel.getConstant(storm::utility::one<ValueType>() / storm::utility::convertNumber<ValueType>(numStates)));
242  auto out = outs.find(stateChoices.first);
243  STORM_LOG_ASSERT(out != outs.end(), "out flow of ec state was not set.");
244  lpModel.addConstraint("", storm::expressions::sum(in->second) <= storm::expressions::sum(out->second));
245  auto ecIn = ecIns.find(stateChoices.first);
246  STORM_LOG_ASSERT(ecIn != ecIns.end(), "ec state does not seem to have an incoming transition.");
247  lpModel.addConstraint("", storm::expressions::sum(ecIn->second) >= ecStateVars[stateChoices.first]);
248  }
249 
250  return ecStateVars;
251  }

253  template <typename ModelType, typename GeometryValueType>
254  bool DeterministicSchedsLpChecker<ModelType, GeometryValueType>::processEndComponents(std::vector<std::vector<storm::expressions::Expression>>& ecVars) {
255  uint64_t ecCounter = 0;
256  auto backwardTransitions = model.getBackwardTransitions();
257 
258  // Get the choices that do not induce a value (i.e. reward) for all objectives.
259  // Only MECS consisting of these choices are relevant
260  storm::storage::BitVector choicesWithValueZero(model.getNumberOfChoices(), true);
261  for (auto const& objHelper : objectiveHelper) {
262  for (auto const& value : objHelper.getChoiceValueOffsets()) {
263  STORM_LOG_ASSERT(!storm::utility::isZero(value.second), "Expected non-zero choice-value offset.");
264  choicesWithValueZero.set(value.first, false);
265  }
266  }
267  storm::storage::MaximalEndComponentDecomposition<ValueType> mecs(model.getTransitionMatrix(), backwardTransitions, storm::storage::BitVector(model.getNumberOfStates(), true), choicesWithValueZero);
268  for (auto const& mec : mecs) {
269  // For each objective we might need to split this mec into several subECs, if the objective yields a non-zero scheduler-independent state value for some states of this ec.
270  // However, note that this split happens objective-wise which is why we can not consider a subsystem in the mec-decomposition above
271  std::map<std::set<uint64_t>, std::vector<uint64_t>> excludedStatesToObjIndex;
272  bool mecContainsSchedulerDependentValue = false;
273  for (uint64_t objIndex = 0; objIndex < objectiveHelper.size(); ++objIndex) {
274  std::set<uint64_t> excludedStates;
275  for (auto const& stateChoices : mec) {
276  auto schedIndValueIt = objectiveHelper[objIndex].getSchedulerIndependentStateValues().find(stateChoices.first);
277  if (schedIndValueIt == objectiveHelper[objIndex].getSchedulerIndependentStateValues().end()) {
278  mecContainsSchedulerDependentValue = true;
279  } else if (!storm::utility::isZero(schedIndValueIt->second)) {
280  excludedStates.insert(stateChoices.first);
281  }
282  }
283  excludedStatesToObjIndex[excludedStates].push_back(objIndex);
284  }
285 
286  // Skip this mec if all state values are independent of the scheduler (e.g. no reward is reachable from here).
287  if (mecContainsSchedulerDependentValue) {
288  for (auto const& exclStates : excludedStatesToObjIndex) {
289  if (exclStates.first.empty()) {
290  auto varsForMec = processEc(mec, model.getTransitionMatrix(), "", choiceVariables, *lpModel);
291  ++ecCounter;
292  for (auto const& stateVar : varsForMec) {
293  for (auto const& objIndex : exclStates.second) {
294  ecVars[objIndex][stateVar.first] = stateVar.second;
295  }
296  }
297  } else {
298  // Compute sub-end components
299  storm::storage::BitVector subEcStates(model.getNumberOfStates(), false), subEcChoices(model.getNumberOfChoices(), false);
300  for (auto const& stateChoices : mec) {
301  if (exclStates.first.count(stateChoices.first) == 0) {
302  subEcStates.set(stateChoices.first, true);
303  for (auto const& choice : stateChoices.second) {
304  subEcChoices.set(choice, true);
305  }
306  }
307  }
308  storm::storage::MaximalEndComponentDecomposition<ValueType> subEcs(model.getTransitionMatrix(), backwardTransitions, subEcStates, subEcChoices);
309  for (auto const& subEc : subEcs) {
310  auto varsForSubEc = processEc(subEc, model.getTransitionMatrix(), "o" + std::to_string(*exclStates.second.begin()), choiceVariables, *lpModel);
311  ++ecCounter;
312  for (auto const& stateVar : varsForSubEc) {
313  for (auto const& objIndex : exclStates.second) {
314  ecVars[objIndex][stateVar.first] = stateVar.second;
315  }
316  }
317  }
318  }
319  }
320  }
321  }
322  bool hasEndComponents = ecCounter > 0 || storm::utility::graph::checkIfECWithChoiceExists(model.getTransitionMatrix(), backwardTransitions, storm::storage::BitVector(model.getNumberOfStates(), true), ~choicesWithValueZero);
323  STORM_LOG_WARN_COND(!hasEndComponents, "Processed " << ecCounter << " End components.");
324  return hasEndComponents;
325  }

327  template <typename ModelType, typename GeometryValueType>
328  void DeterministicSchedsLpChecker<ModelType, GeometryValueType>::initializeLpModel(Environment const& env) {
329  STORM_LOG_INFO("Initializing LP model with " << model.getNumberOfStates() << " states.");
330  if (env.modelchecker().multi().getEncodingType() == storm::MultiObjectiveModelCheckerEnvironment::EncodingType::Auto) {
331  flowEncoding = true;
332  for (auto const& helper : objectiveHelper) {
333  if (!helper.isTotalRewardObjective()) {
334  flowEncoding = false;
335  }
336  }
337  } else {
338  flowEncoding = env.modelchecker().multi().getEncodingType() == storm::MultiObjectiveModelCheckerEnvironment::EncodingType::Flow;
339  }
340  STORM_LOG_INFO("Using " << (flowEncoding ? "flow" : "classical") << " encoding." << std::endl);
341 
342  uint64_t numStates = model.getNumberOfStates();
343  uint64_t initialState = *model.getInitialStates().begin();
344  STORM_LOG_WARN_COND(!storm::settings::getModule<storm::settings::modules::CoreSettings>().isLpSolverSetFromDefaultValue() || storm::settings::getModule<storm::settings::modules::CoreSettings>().getLpSolver() == storm::solver::LpSolverType::Gurobi, "The selected MILP solver might not perform well. Consider installing / using Gurobi.");
345  lpModel = storm::utility::solver::getLpSolver<ValueType>("model");
346 
347  lpModel->setOptimizationDirection(storm::solver::OptimizationDirection::Maximize);
348  initialStateResults.clear();
349 
350  auto zero = lpModel->getConstant(storm::utility::zero<ValueType>());
351  auto one = lpModel->getConstant(storm::utility::one<ValueType>());
352  auto const& groups = model.getTransitionMatrix().getRowGroupIndices();
353  // Create choice variables.
354  choiceVariables.reserve(model.getNumberOfChoices());
355  for (uint64_t state = 0; state < numStates; ++state) {
356  uint64_t numChoices = model.getNumberOfChoices(state);
357  if (numChoices == 1) {
358  choiceVariables.emplace_back();
359  } else {
360  std::vector<storm::expressions::Expression> localChoices;
361  for (uint64_t choice = 0; choice < numChoices; ++choice) {
362  localChoices.push_back(lpModel->addBoundedIntegerVariable("c" + std::to_string(state) + "_" + std::to_string(choice), 0, 1).getExpression());
363  choiceVariables.push_back(localChoices.back());
364  }
365  lpModel->addConstraint("", storm::expressions::sum(localChoices) == one);
366  }
367  }
368  // Create ec Variables for each state/objective
369  std::vector<std::vector<storm::expressions::Expression>> ecVars(objectiveHelper.size(), std::vector<storm::expressions::Expression>(model.getNumberOfStates()));
370  bool hasEndComponents = processEndComponents(ecVars);
371 
372  if (flowEncoding) {
373  storm::storage::BitVector bottomStates(model.getNumberOfStates(), true);
374  for (auto const& helper : objectiveHelper) {
375  STORM_LOG_THROW(helper.isTotalRewardObjective(), storm::exceptions::InvalidOperationException, "The given type of encoding is only supported if the objectives can be reduced to total reward objectives.");
376  storm::storage::BitVector objBottomStates(model.getNumberOfStates(), false);
377  for (auto const& stateVal : helper.getSchedulerIndependentStateValues()) {
378  STORM_LOG_THROW(storm::utility::isZero(stateVal.second), storm::exceptions::InvalidOperationException, "Non-zero constant state-values not allowed for flow encoding.");
379  objBottomStates.set(stateVal.first, true);
380  }
381  bottomStates &= objBottomStates;
382  }
383  storm::storage::BitVector nonBottomStates = ~bottomStates;
384  STORM_LOG_TRACE("Found " << bottomStates.getNumberOfSetBits() << " bottom states.");
385 
386  // Compute upper bounds for each state
387  std::vector<ValueType> visitingTimesUpperBounds = DeterministicSchedsObjectiveHelper<ModelType>::computeUpperBoundOnExpectedVisitingTimes(model.getTransitionMatrix(), bottomStates, nonBottomStates, hasEndComponents);
388  ValueType largestUpperBound = *std::max_element(visitingTimesUpperBounds.begin(), visitingTimesUpperBounds.end());
389  STORM_LOG_WARN_COND(largestUpperBound < storm::utility::convertNumber<ValueType>(1e5), "Found a large upper bound '" << storm::utility::convertNumber<double>(largestUpperBound) << "' in the LP encoding. This might trigger numerical instabilities.");
390  // create choiceValue variables and assert deterministic ones.
391  std::vector<storm::expressions::Expression> choiceValVars(model.getNumberOfChoices());
392  for (auto const& state : nonBottomStates) {
393  for (uint64_t globalChoice = groups[state]; globalChoice < groups[state + 1]; ++globalChoice) {
394  choiceValVars[globalChoice] = lpModel->addBoundedContinuousVariable("y" + std::to_string(globalChoice), storm::utility::zero<ValueType>(), visitingTimesUpperBounds[state]).getExpression();
395  if (model.getNumberOfChoices(state) > 1) {;
396  lpModel->addConstraint("", choiceValVars[globalChoice] <= lpModel->getConstant(visitingTimesUpperBounds[state]) * choiceVariables[globalChoice]);
397  }
398  }
399  }
400  // create EC 'slack' variables for states that lie in an ec
401  std::vector<storm::expressions::Expression> ecValVars(model.getNumberOfStates());
402  if (hasEndComponents) {
403  for (auto const& state : nonBottomStates) {
404  // For the in-out-encoding, all objectives have the same ECs (because there are no non-zero scheduler independend state values).
405  // Hence, we only care for the variables of the first objective.
406  if (ecVars.front()[state].isInitialized()) {
407  ecValVars[state] = lpModel->addBoundedContinuousVariable("z" + std::to_string(state), storm::utility::zero<ValueType>(), visitingTimesUpperBounds[state]).getExpression();
408  lpModel->addConstraint("", ecValVars[state] <= lpModel->getConstant(visitingTimesUpperBounds[state]) * ecVars.front()[state]);
409  }
410  }
411  }
412  // Get 'in' and 'out' expressions
413  std::vector<storm::expressions::Expression> bottomStatesIn;
414  std::vector<std::vector<storm::expressions::Expression>> ins(numStates), outs(numStates);
415  ins[initialState].push_back(one);
416  for (auto const& state : nonBottomStates) {
417  for (uint64_t globalChoice = groups[state]; globalChoice < groups[state + 1]; ++globalChoice) {
418  for (auto const& transition : model.getTransitionMatrix().getRow(globalChoice)) {
419  uint64_t successor = transition.getColumn();
420  storm::expressions::Expression exp = lpModel->getConstant(transition.getValue()) * choiceValVars[globalChoice];
421  if (bottomStates.get(successor)) {
422  bottomStatesIn.push_back(exp);
423  } else {
424  ins[successor].push_back(exp);
425  }
426  }
427  outs[state].push_back(choiceValVars[globalChoice]);
428  }
429  if (ecValVars[state].isInitialized()) {
430  outs[state].push_back(ecValVars[state]);
431  bottomStatesIn.push_back(ecValVars[state]);
432  }
433  }
434 
435  // Assert 'in == out' at each state
436  for (auto const& state : nonBottomStates) {
437  lpModel->addConstraint("", storm::expressions::sum(ins[state]) == storm::expressions::sum(outs[state]));
438 
439  }
440  // Assert the sum for the bottom states
441  lpModel->addConstraint("", storm::expressions::sum(bottomStatesIn) == one);
442 
443  // create initial state results for each objective
444  for (uint64_t objIndex = 0; objIndex < objectiveHelper.size(); ++objIndex) {
445  auto choiceValueOffsets = objectiveHelper[objIndex].getChoiceValueOffsets();
446  std::vector<storm::expressions::Expression> objValue;
447  for (auto const& state : nonBottomStates) {
448  for (uint64_t globalChoice = groups[state]; globalChoice < groups[state + 1]; ++globalChoice) {
449  auto choiceValueIt = choiceValueOffsets.find(globalChoice);
450  if (choiceValueIt != choiceValueOffsets.end()) {
451  assert(!storm::utility::isZero(choiceValueIt->second));
452  objValue.push_back(lpModel->getConstant(choiceValueIt->second) * choiceValVars[globalChoice]);
453  }
454  }
455  }
456  ValueType lowerBound = objectiveHelper[objIndex].getLowerValueBoundAtState(env, initialState);
457  ValueType upperBound = objectiveHelper[objIndex].getUpperValueBoundAtState(env, initialState);
458  storm::expressions::Expression objValueVariable;
459  if (lowerBound == upperBound) {
460  // GLPK does not like point-intervals......
461  objValueVariable = lpModel->getConstant(lowerBound);
462  } else {
463  objValueVariable = lpModel->addBoundedContinuousVariable("x" + std::to_string(objIndex), lowerBound, upperBound).getExpression();
464  }
465  if (objValue.empty()) {
466  lpModel->addConstraint("", objValueVariable == zero);
467  } else {
468  lpModel->addConstraint("", objValueVariable == storm::expressions::sum(objValue));
469  }
470  initialStateResults.push_back(objValueVariable);
471  }
472  } else {
473  // 'classic' backward encoding.
474  for (uint64_t objIndex = 0; objIndex < objectiveHelper.size(); ++objIndex) {
475  STORM_LOG_WARN_COND(objectiveHelper[objIndex].getLargestUpperBound(env) < storm::utility::convertNumber<ValueType>(1e5), "Found a large upper value bound '" << storm::utility::convertNumber<double>(objectiveHelper[objIndex].getLargestUpperBound(env)) << "' in the LP encoding. This might trigger numerical instabilities.");
476  auto const& schedulerIndependentStates = objectiveHelper[objIndex].getSchedulerIndependentStateValues();
477  // Create state variables and store variables of ecs which contain a state with a scheduler independent value
478  std::vector<storm::expressions::Expression> stateVars;
479  stateVars.reserve(numStates);
480  for (uint64_t state = 0; state < numStates; ++state) {
481  auto valIt = schedulerIndependentStates.find(state);
482  if (valIt == schedulerIndependentStates.end()) {
483  ValueType lowerBound = objectiveHelper[objIndex].getLowerValueBoundAtState(env, state);
484  ValueType upperBound = objectiveHelper[objIndex].getUpperValueBoundAtState(env, state);
485  if (lowerBound == upperBound) {
486  // glpk does not like variables with point-interval bounds...
487  if (objectiveHelper[objIndex].minimizing()) {
488  stateVars.push_back(lpModel->getConstant(-lowerBound));
489  } else {
490  stateVars.push_back(lpModel->getConstant(lowerBound));
491  }
492  } else {
493  if (objectiveHelper[objIndex].minimizing()) {
494  stateVars.push_back(lpModel->addBoundedContinuousVariable("x" + std::to_string(objIndex) + "_" + std::to_string(state), -upperBound, -lowerBound).getExpression());
495  } else {
496  stateVars.push_back(lpModel->addBoundedContinuousVariable("x" + std::to_string(objIndex) + "_" + std::to_string(state), lowerBound, upperBound).getExpression());
497  }
498  }
499  } else {
500  ValueType value = valIt->second;
501  if (objectiveHelper[objIndex].minimizing()) {
502  value = -value;
503  }
504  stateVars.push_back(lpModel->getConstant(value));
505  }
506  if (state == initialState) {
507  initialStateResults.push_back(stateVars.back());
508  }
509  }
510 
511  // Create and assert choice values
512  auto const& choiceValueOffsets = objectiveHelper[objIndex].getChoiceValueOffsets();
513  for (uint64_t state = 0; state < numStates; ++state) {
514  if (schedulerIndependentStates.find(state) != schedulerIndependentStates.end()) {
515  continue;
516  }
517  storm::expressions::Expression stateValue;
518  uint64_t numChoices = model.getNumberOfChoices(state);
519  uint64_t choiceOffset = groups[state];
520  storm::expressions::Expression upperValueBoundAtState = lpModel->getConstant(objectiveHelper[objIndex].getUpperValueBoundAtState(env, state));
521  for (uint64_t choice = 0; choice < numChoices; ++choice) {
522  storm::expressions::Expression choiceValue;
523  auto valIt = choiceValueOffsets.find(choiceOffset + choice);
524  if (valIt != choiceValueOffsets.end()) {
525  if (objectiveHelper[objIndex].minimizing()) {
526  choiceValue = lpModel->getConstant(-valIt->second);
527  } else {
528  choiceValue = lpModel->getConstant(valIt->second);
529  }
530  }
531  for (auto const& transition : model.getTransitionMatrix().getRow(state, choice)) {
532  if (!storm::utility::isZero(transition.getValue())) {
533  storm::expressions::Expression transitionValue = lpModel->getConstant(transition.getValue()) * stateVars[transition.getColumn()];
534  if (choiceValue.isInitialized()) {
535  choiceValue = choiceValue + transitionValue;
536  } else {
537  choiceValue = transitionValue;
538  }
539  }
540  }
541  choiceValue = choiceValue.simplify().reduceNesting();
542  if (numChoices == 1) {
543  stateValue = choiceValue;
544  } else {
545  uint64_t globalChoiceIndex = groups[state] + choice;
546  storm::expressions::Expression choiceValVar;
547  if (objectiveHelper[objIndex].minimizing()) {
548  choiceValVar = lpModel->addBoundedContinuousVariable("x" + std::to_string(objIndex) + "_" + std::to_string(state) + "_" + std::to_string(choice), -objectiveHelper[objIndex].getUpperValueBoundAtState(env, state), storm::utility::zero<ValueType>()).getExpression();
549  } else {
550  choiceValVar = lpModel->addBoundedContinuousVariable("x" + std::to_string(objIndex) + "_" + std::to_string(state) + "_" + std::to_string(choice), storm::utility::zero<ValueType>(), objectiveHelper[objIndex].getUpperValueBoundAtState(env, state)).getExpression();
551  }
552  lpModel->addConstraint("", choiceValVar <= choiceValue);
553  if (objectiveHelper[objIndex].minimizing()) {
554  lpModel->addConstraint("", choiceValVar <= -upperValueBoundAtState * (one - choiceVariables[globalChoiceIndex]));
555  } else {
556  lpModel->addConstraint("", choiceValVar <= upperValueBoundAtState * choiceVariables[globalChoiceIndex]);
557  }
558  if (choice == 0) {
559  stateValue = choiceValVar;
560  } else {
561  stateValue = stateValue + choiceValVar;
562  }
563  }
564  }
565  stateValue.simplify().reduceNesting();
566  if (objectiveHelper[objIndex].minimizing()) {
567  lpModel->addConstraint("", stateVars[state] <= stateValue + (lpModel->getConstant(storm::utility::convertNumber<ValueType>(numChoices - 1)) * upperValueBoundAtState));
568  } else {
569  lpModel->addConstraint("", stateVars[state] <= stateValue);
570  }
571  if (numChoices > 1 && hasEndComponents) {
572  auto& ecVar = ecVars[objIndex][state];
573  if (ecVar.isInitialized()) {
574  // if this state is part of an ec, make sure to assign a value of zero.
575  if (objectiveHelper[objIndex].minimizing()) {
576  // This part is optional
577  lpModel->addConstraint("", stateVars[state] >= (ecVar - one) * lpModel->getConstant(objectiveHelper[objIndex].getUpperValueBoundAtState(env, state)));
578  } else {
579  lpModel->addConstraint("", stateVars[state] <= (one - ecVar) * lpModel->getConstant(objectiveHelper[objIndex].getUpperValueBoundAtState(env, state)));
580  }
581  }
582  }
583  }
584  }
585  }
586  lpModel->update();
587  STORM_LOG_INFO("Done initializing LP model.");
588  }

590  template <typename ModelType, typename GeometryValueType>
591  void DeterministicSchedsLpChecker<ModelType, GeometryValueType>::checkRecursive(Environment const& env, storm::storage::geometry::PolytopeTree<GeometryValueType>& polytopeTree, Point const& eps, std::vector<Point>& foundPoints, std::vector<Polytope>& infeasableAreas, uint64_t const& depth) {
592  STORM_LOG_ASSERT(!polytopeTree.isEmpty(), "Tree node is empty");
593  STORM_LOG_ASSERT(!polytopeTree.getPolytope()->isEmpty(), "Tree node is empty.");
594  STORM_LOG_TRACE("Checking at depth " << depth << ": " << polytopeTree.toString());
595 
596  lpModel->push();
597  // Assert the constraints of the current polytope
598  auto nodeConstraints = polytopeTree.getPolytope()->getConstraints(lpModel->getManager(), currentObjectiveVariables);
599  for (auto const& constr : nodeConstraints) {
600  lpModel->addConstraint("", constr);
601  }
602  lpModel->update();
603 
604  if (polytopeTree.getChildren().empty()) {
605  // At leaf nodes we need to perform the actual check.
606 
607  // Numerical instabilities might yield a point that is not actually inside the nodeConstraints.
608  // If the downward closure is disjoint from this tree node, we will not make further progress.
609  // In this case, we sharpen the violating constraints just a little bit.
610  // This way, valid solutions might be excluded, so technically, this can yield false negatives.
611  // However, since we apparently are dealing with numerical algorithms, we can't be sure about correctness anyway.
612  uint64_t num_sharpen = 0;
613  auto halfspaces = polytopeTree.getPolytope()->getHalfspaces();
614  while (true) {
615  STORM_LOG_TRACE("\tSolving MILP...");
616  swCheckAreas.start();
617  lpModel->optimize();
618  swCheckAreas.stop();
619  ++numLpQueries;
620  STORM_LOG_TRACE("\tDone solving MILP...");
621 
622  // STORM_PRINT_AND_LOG("Writing model to file '" << polytopeTree.toId() << ".lp'" << std::endl;);
623  // lpModel->writeModelToFile(polytopeTree.toId() + ".lp");
624 
625  if (lpModel->isInfeasible()) {
626  infeasableAreas.push_back(polytopeTree.getPolytope());
627  polytopeTree.clear();
628  break;
629  } else {
630  STORM_LOG_ASSERT(!lpModel->isUnbounded(), "LP result is unbounded.");
631  swValidate.start();
632  Point newPoint = validateCurrentModel(env);
633  swValidate.stop();
634  // Check whether this new point yields any progress.
635  // There is no progress if (due to numerical inaccuracies) the downwardclosure (including points that are epsilon close to it) contained in this polytope.
636  // We multiply eps by 0.999 so that points that lie on the boundary of polytope and downw. do not count in the intersection.
637  Point newPointPlusEps = newPoint;
638  storm::utility::vector::addScaledVector(newPointPlusEps, eps, storm::utility::convertNumber<GeometryValueType>(0.999));
639  if (polytopeTree.getPolytope()->contains(newPoint) || !polytopeTree.getPolytope()->intersection(storm::storage::geometry::Polytope<GeometryValueType>::createDownwardClosure({newPointPlusEps}))->isEmpty()) {
640  GeometryValueType offset = storm::utility::convertNumber<GeometryValueType>(lpModel->getObjectiveValue());
641  // Get the gap between the found solution and the known bound.
642  offset += storm::utility::convertNumber<GeometryValueType>(lpModel->getMILPGap(false));
643  // we might want to shift the halfspace to guarantee that our point is included.
644  offset = std::max(offset, storm::utility::vector::dotProduct(currentWeightVector, newPoint));
645  auto halfspace = storm::storage::geometry::Halfspace<GeometryValueType>(currentWeightVector, offset).invert();
646  infeasableAreas.push_back(polytopeTree.getPolytope()->intersection(halfspace));
647  if (infeasableAreas.back()->isEmpty()) {
648  infeasableAreas.pop_back();
649  }
650  polytopeTree.setMinus(storm::storage::geometry::Polytope<GeometryValueType>::create({halfspace}));
651  foundPoints.push_back(newPoint);
652  polytopeTree.substractDownwardClosure(newPoint, eps);
653  if (!polytopeTree.isEmpty()) {
654  checkRecursive(env, polytopeTree, eps, foundPoints, infeasableAreas, depth);
655  }
656  break;
657  } else {
658  // If we end up here, we have to sharpen the violated constraints for this polytope
659  for (auto& h : halfspaces) {
660  GeometryValueType distance = h.distance(newPoint);
661  // Check if the found point is outside of this halfspace
662  if (!storm::utility::isZero(distance)) {
663  // The issue has to be for some normal vector with a negative entry. Otherwise, the intersection with the downward closure wouldn't be empty
664  bool normalVectorContainsNegative = false;
665  for (auto const& hi : h.normalVector()) {
666  if (hi < storm::utility::zero<GeometryValueType>()) {
667  normalVectorContainsNegative = true;
668  break;
669  }
670  }
671  if (normalVectorContainsNegative) {
672  h.offset() -= distance / storm::utility::convertNumber<GeometryValueType, uint64_t>(2);
673  if (num_sharpen == 0) {
674  lpModel->push();
675  }
676  lpModel->addConstraint("", h.toExpression(lpModel->getManager(), currentObjectiveVariables));
677  ++num_sharpen;
678  break;
679  }
680  }
681  }
682  STORM_LOG_TRACE("\tSharpened LP");
683  }
684  }
685  }
686  if (num_sharpen > 0) {
687  STORM_LOG_WARN("Numerical instabilities detected: LP Solver found an achievable point outside of the search area. The search area had to be sharpened " << num_sharpen << " times.");
688  // pop sharpened constraints
689  lpModel->pop();
690  }
691 
692  } else {
693  // Traverse all the non-empty children.
694  for (uint64_t childId = 0; childId < polytopeTree.getChildren().size(); ++childId) {
695  if (polytopeTree.getChildren()[childId].isEmpty()) {
696  continue;
697  }
698  uint64_t newPointIndex = foundPoints.size();
699  checkRecursive(env, polytopeTree.getChildren()[childId], eps, foundPoints, infeasableAreas, depth + 1);
700  STORM_LOG_ASSERT(polytopeTree.getChildren()[childId].isEmpty(), "expected empty children.");
701  // Make the new points known to the right siblings
702  for (; newPointIndex < foundPoints.size(); ++newPointIndex) {
703  for (uint64_t siblingId = childId + 1; siblingId < polytopeTree.getChildren().size(); ++siblingId) {
704  polytopeTree.getChildren()[siblingId].substractDownwardClosure(foundPoints[newPointIndex], eps);
705  }
706  }
707  }
708  // All children are empty now, so this node becomes empty.
709  polytopeTree.clear();
710  }
711  STORM_LOG_TRACE("Checking DONE at depth " << depth << " with node " << polytopeTree.toString());
712 
713  lpModel->pop();
714  lpModel->update();
715  }

717  template <typename ModelType, typename GeometryValueType>
718  typename DeterministicSchedsLpChecker<ModelType, GeometryValueType>::Point DeterministicSchedsLpChecker<ModelType, GeometryValueType>::validateCurrentModel(Environment const& env) const {
719  storm::storage::Scheduler<ValueType> scheduler(model.getNumberOfStates());
720  for (uint64_t state = 0; state < model.getNumberOfStates(); ++state) {
721  uint64_t numChoices = model.getNumberOfChoices(state);
722  if (numChoices == 1) {
723  scheduler.setChoice(0, state);
724  } else {
725  uint64_t globalChoiceOffset = model.getTransitionMatrix().getRowGroupIndices()[state];
726  bool choiceFound = false;
727  for (uint64_t localChoice = 0; localChoice < numChoices; ++localChoice) {
728  bool localChoiceEnabled = false;
729  localChoiceEnabled = (lpModel->getIntegerValue(choiceVariables[globalChoiceOffset + localChoice].getBaseExpression().asVariableExpression().getVariable()) == 1);
730  if (localChoiceEnabled) {
731  STORM_LOG_THROW(!choiceFound, storm::exceptions::UnexpectedException, "Multiple choices selected at state " << state);
732  scheduler.setChoice(localChoice, state);
733  choiceFound = true;
734  }
735  }
736  STORM_LOG_THROW(choiceFound, storm::exceptions::UnexpectedException, "No choice selected at state " << state);
737  }
738  }
739  auto inducedModel = model.applyScheduler(scheduler)->template as<ModelType>();
740  Point inducedPoint;
741  for (uint64_t objIndex = 0; objIndex < objectiveHelper.size(); ++objIndex) {
742  ValueType inducedValue = objectiveHelper[objIndex].evaluateOnModel(env, *inducedModel);
743  if (objectiveHelper[objIndex].minimizing()) {
744  inducedValue = -inducedValue;
745  }
746  inducedPoint.push_back(storm::utility::convertNumber<GeometryValueType>(inducedValue));
747  // If this objective has weight zero, the lp solution is not necessarily correct
748  if (!storm::utility::isZero(currentWeightVector[objIndex])) {
749  ValueType lpValue = lpModel->getContinuousValue(currentObjectiveVariables[objIndex]);
750  double diff = storm::utility::convertNumber<double>(storm::utility::abs<ValueType>(inducedValue - lpValue));
751  STORM_LOG_WARN_COND(diff <= 1e-4 * std::abs(storm::utility::convertNumber<double>(inducedValue)), "Imprecise value for objective " << objIndex << ": LP says " << lpValue << " but scheduler induces " << inducedValue << " (difference is " << diff << ")");
752  }
753  }
754  return inducedPoint;
755  }
