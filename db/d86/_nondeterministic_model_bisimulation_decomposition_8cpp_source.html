97  template<typename ModelType>
98  void NondeterministicModelBisimulationDecomposition<ModelType>::buildQuotient() {
99  // In order to create the quotient model, we need to construct
100  // (a) the new transition matrix,
101  // (b) the new labeling,
102  // (c) the new reward structures.
103 
104  // Prepare a matrix builder for (a).
105  storm::storage::SparseMatrixBuilder<ValueType> builder(0, this->size(), 0, false, true, this->size());
106 
107  // Prepare the new state labeling for (b).
108  storm::models::sparse::StateLabeling newLabeling(this->size());
109  std::set<std::string> atomicPropositionsSet = this->options.respectedAtomicPropositions.get();
110  atomicPropositionsSet.insert("init");
111  std::vector<std::string> atomicPropositions = std::vector<std::string>(atomicPropositionsSet.begin(), atomicPropositionsSet.end());
112  for (auto const& ap : atomicPropositions) {
113  newLabeling.addLabel(ap);
114  }
115 
116  // If the model had state (action) rewards, we need to build the state rewards for the quotient as well.
117  boost::optional<std::vector<ValueType>> stateRewards;
118  boost::optional<std::vector<ValueType>> stateActionRewards;
119  boost::optional<storm::models::sparse::StandardRewardModel<ValueType> const&> rewardModel;
120  if (this->options.getKeepRewards() && this->model.hasRewardModel()) {
121  rewardModel = this->model.getUniqueRewardModel();
122  if (rewardModel.get().hasStateRewards()) {
123  stateRewards = std::vector<ValueType>(this->blocks.size());
124  }
125  if (rewardModel.get().hasStateActionRewards()) {
126  stateActionRewards = std::vector<ValueType>();
127  }
128  }
129 
130  // Now build (a) and (b) by traversing all blocks.
131  uint_fast64_t currentRow = 0;
132  std::vector<uint_fast64_t> nondeterministicChoiceIndices = this->model.getTransitionMatrix().getRowGroupIndices();
133  for (uint_fast64_t blockIndex = 0; blockIndex < this->blocks.size(); ++blockIndex) {
134  auto const& block = this->blocks[blockIndex];
135 
136  // Open new row group for the new meta state.
137  builder.newRowGroup(currentRow);
138 
139  // Pick one representative state. For strong bisimulation it doesn't matter which state it is, because
140  // they all behave equally.
141  storm::storage::sparse::state_type representativeState = *block.begin();
142  Block<BlockDataType> const& oldBlock = this->partition.getBlock(representativeState);
143 
144  // If the block is absorbing, we simply add a self-loop.
145  if (oldBlock.data().absorbing()) {
146  builder.addNextValue(currentRow, blockIndex, storm::utility::one<ValueType>());
147  ++currentRow;
148 
149  // If the block has a special representative state, we retrieve it now.
150  if (oldBlock.data().hasRepresentativeState()) {
151  representativeState = oldBlock.data().representativeState();
152  }
153 
154  // Give the choice a reward of zero as we artificially introduced that the block is absorbing.
155  if (this->options.getKeepRewards() && rewardModel && rewardModel.get().hasStateActionRewards()) {
156  stateActionRewards.get().push_back(storm::utility::zero<ValueType>());
157  }
158 
159  // Add all of the selected atomic propositions that hold in the representative state to the state
160  // representing the block.
161  for (auto const& ap : atomicPropositions) {
162  if (this->model.getStateLabeling().getStateHasLabel(ap, representativeState)) {
163  newLabeling.addLabelToState(ap, blockIndex);
164  }
165  }
166  } else {
167  // Add the outgoing choices of the block.
168  for (uint_fast64_t choice = nondeterministicChoiceIndices[representativeState]; choice < nondeterministicChoiceIndices[representativeState + 1]; ++choice) {
169  // If the choice is the same as the last one, we do not need to add it.
170  if (choice > nondeterministicChoiceIndices[representativeState] && quotientDistributions[choice - 1].equals(quotientDistributions[choice], this->comparator)) {
171  continue;
172  }
173 
174  for (auto entry : quotientDistributions[choice]) {
175  builder.addNextValue(currentRow, entry.first, entry.second);
176  }
177  if (this->options.getKeepRewards() && rewardModel && rewardModel.get().hasStateActionRewards()) {
178  stateActionRewards.get().push_back(quotientDistributions[choice].getReward());
179  }
180  ++currentRow;
181  }
182 
183  // Otherwise add all atomic propositions to the equivalence class that the representative state
184  // satisfies.
185  for (auto const& ap : atomicPropositions) {
186  if (this->model.getStateLabeling().getStateHasLabel(ap, representativeState)) {
187  newLabeling.addLabelToState(ap, blockIndex);
188  }
189  }
190  }
191 
192  // If the model has state rewards, we simply copy the state reward of the representative state, because
193  // all states in a block are guaranteed to have the same state reward.
194  if (this->options.getKeepRewards() && rewardModel && rewardModel.get().hasStateRewards()) {
195  stateRewards.get()[blockIndex] = rewardModel.get().getStateRewardVector()[representativeState];
196  }
197  }
198 
199  // Now check which of the blocks of the partition contain at least one initial state.
200  for (auto initialState : this->model.getInitialStates()) {
201  Block<BlockDataType> const& initialBlock = this->partition.getBlock(initialState);
202  newLabeling.addLabelToState("init", initialBlock.getId());
203  }
204 
205  // Construct the reward model mapping.
206  std::unordered_map<std::string, typename ModelType::RewardModelType> rewardModels;
207  if (this->options.getKeepRewards() && this->model.hasRewardModel()) {
208  STORM_LOG_THROW(this->model.hasUniqueRewardModel(), storm::exceptions::IllegalFunctionCallException, "Cannot preserve more than one reward model.");
209  typename std::unordered_map<std::string, typename ModelType::RewardModelType>::const_iterator nameRewardModelPair = this->model.getRewardModels().begin();
210  rewardModels.insert(std::make_pair(nameRewardModelPair->first, typename ModelType::RewardModelType(stateRewards, stateActionRewards)));
211  }
212 
213  // Finally construct the quotient model.
214  this->quotient = std::make_shared<ModelType>(builder.build(0,this->size(), this->size()), std::move(newLabeling), std::move(rewardModels));
215  }

222  template<typename ModelType>
223  void NondeterministicModelBisimulationDecomposition<ModelType>::updateQuotientDistributionsOfPredecessors(Block<BlockDataType> const& newBlock, Block<BlockDataType> const& oldBlock, std::vector<Block<BlockDataType>*>& splitterQueue) {
224  uint_fast64_t lastState = 0;
225  bool lastStateInitialized = false;
226 
227  for (auto stateIt = this->partition.begin(newBlock), stateIte = this->partition.end(newBlock); stateIt != stateIte; ++stateIt) {
228  for (auto predecessorEntry : this->backwardTransitions.getRow(*stateIt)) {
229  if (this->comparator.isZero(predecessorEntry.getValue())) {
230  continue;
231  }
232 
233  storm::storage::sparse::state_type predecessorChoice = predecessorEntry.getColumn();
234  storm::storage::sparse::state_type predecessorState = choiceToStateMapping[predecessorChoice];
235  Block<BlockDataType>& predecessorBlock = this->partition.getBlock(predecessorState);
236 
237  // If the predecessor block is marked as absorbing, we do not need to update anything.
238  if (predecessorBlock.data().absorbing()) {
239  continue;
240  }
241 
242  // If the predecessor block is not marked as to-be-refined, we do so now.
243  if (!predecessorBlock.data().splitter()) {
244  predecessorBlock.data().setSplitter();
245  splitterQueue.push_back(&predecessorBlock);
246  }
247 
248  if (lastStateInitialized) {
249  // If we have skipped to the choices of the next state, we need to repair the order of the
250  // distributions for the last state.
251  if (lastState != predecessorState) {
252  updateOrderedQuotientDistributions(lastState);
253  lastState = predecessorState;
254  }
255  } else {
256  lastStateInitialized = true;
257  lastState = choiceToStateMapping[predecessorChoice];
258  }
259 
260  // Now shift the probability from this transition from the old block to the new one.
261  this->quotientDistributions[predecessorChoice].shiftProbability(oldBlock.getId(), newBlock.getId(), predecessorEntry.getValue());
262  }
263  }
264 
265  if (lastStateInitialized) {
266  updateOrderedQuotientDistributions(lastState);
267  }
268  }

270  template<typename ModelType>
271  bool NondeterministicModelBisimulationDecomposition<ModelType>::checkQuotientDistributions() const {
272  std::vector<uint_fast64_t> nondeterministicChoiceIndices = this->model.getTransitionMatrix().getRowGroupIndices();
273  for (decltype(this->model.getNumberOfStates()) state = 0; state < this->model.getNumberOfStates(); ++state) {
274  for (auto choice = nondeterministicChoiceIndices[state]; choice < nondeterministicChoiceIndices[state + 1]; ++choice) {
275  storm::storage::DistributionWithReward<ValueType> distribution;
276  if (this->options.getKeepRewards() && this->model.hasRewardModel()) {
277  auto const& rewardModel = this->model.getUniqueRewardModel();
278  if (rewardModel.hasStateActionRewards()) {
279  distribution.setReward(rewardModel.getStateActionReward(choice));
280  }
281  }
282  for (auto const& element : this->model.getTransitionMatrix().getRow(choice)) {
283  distribution.addProbability(this->partition.getBlock(element.getColumn()).getId(), element.getValue());
284  }
285 
286  if (!distribution.equals(quotientDistributions[choice])) {
287  std::cout << "the distributions for choice " << choice << " of state " << state << " do not match." << std::endl;
288  std::cout << "is: " << quotientDistributions[choice] << " but should be " << distribution << std::endl;
289  exit(-1);
290  }
291 
292  bool less1 = distribution.less(quotientDistributions[choice], this->comparator);
293  bool less2 = quotientDistributions[choice].less(distribution, this->comparator);
294 
295  if (distribution.equals(quotientDistributions[choice]) && (less1 || less2)) {
296  std::cout << "mismatch of equality and less for " << std::endl;
297  std::cout << quotientDistributions[choice] << " vs " << distribution << std::endl;
298  exit(-1);
299  }
300  }
301 
302  for (auto choice = nondeterministicChoiceIndices[state]; choice < nondeterministicChoiceIndices[state + 1] - 1; ++choice) {
303  if (orderedQuotientDistributions[choice + 1]->less(*orderedQuotientDistributions[choice], this->comparator)) {
304  std::cout << "choice " << (choice+1) << " is less than predecessor" << std::endl;
305  std::cout << *orderedQuotientDistributions[choice] << " should be less than " << *orderedQuotientDistributions[choice + 1] << std::endl;
306  exit(-1);
307  }
308  }
309  }
310  return true;
311  }

383  template<typename ModelType>
384  bool NondeterministicModelBisimulationDecomposition<ModelType>::quotientDistributionsLess(storm::storage::sparse::state_type state1, storm::storage::sparse::state_type state2) const {
385  STORM_LOG_TRACE("Comparing the quotient distributions of state " << state1 << " and " << state2 << ".");
386  std::vector<uint_fast64_t> nondeterministicChoiceIndices = this->model.getTransitionMatrix().getRowGroupIndices();
387 
388  auto firstIt = orderedQuotientDistributions.begin() + nondeterministicChoiceIndices[state1];
389  auto firstIte = orderedQuotientDistributions.begin() + nondeterministicChoiceIndices[state1 + 1];
390  auto secondIt = orderedQuotientDistributions.begin() + nondeterministicChoiceIndices[state2];
391  auto secondIte = orderedQuotientDistributions.begin() + nondeterministicChoiceIndices[state2 + 1];
392 
393  for (; firstIt != firstIte && secondIt != secondIte; ++firstIt, ++secondIt) {
394  // If the current distributions are in a less-than relationship, we can return a result.
395  if ((*firstIt)->less(**secondIt, this->comparator)) {
396  return true;
397  } else if ((*secondIt)->less(**firstIt, this->comparator)) {
398  return false;
399  }
400 
401  // If the distributions matched, we need to advance both distribution iterators to the next distribution
402  // that is larger.
403  while (firstIt != firstIte && std::next(firstIt) != firstIte && !(*firstIt)->less(**std::next(firstIt), this->comparator)) {
404  ++firstIt;
405  }
406  while (secondIt != secondIte && std::next(secondIt) != secondIte && !(*secondIt)->less(**std::next(secondIt), this->comparator)) {
407  ++secondIt;
408  }
409  }
410 
411  if (firstIt == firstIte && secondIt != secondIte) {
412  return true;
413  }
414  return false;
415  }
