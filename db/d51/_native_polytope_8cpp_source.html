249  std::shared_ptr<Polytope<ValueType>> NativePolytope<ValueType>::minkowskiSum(std::shared_ptr<Polytope<ValueType>> const& rhs) const {
250  STORM_LOG_THROW(rhs->isNativePolytope(), storm::exceptions::InvalidArgumentException, "Invoked operation between a NativePolytope and a different polytope implementation. This is not supported");
251  NativePolytope<ValueType> const& nativeRhs = dynamic_cast<NativePolytope<ValueType> const&>(*rhs);
252 
253  if (this->isEmpty() || nativeRhs.isEmpty()) {
254  return std::make_shared<NativePolytope<ValueType>>(std::vector<Point>());
255  }
256 
257  std::vector<std::pair<EigenVector, ValueType>> resultConstraints;
258  resultConstraints.reserve(A.rows() + nativeRhs.A.rows());
259 
260  // evaluation of rhs in directions of lhs
261  for (Eigen::Index i = 0; i < A.rows(); ++i) {
262  auto optimizationRes = nativeRhs.optimize(A.row(i));
263  if ( optimizationRes.second ) {
264  resultConstraints.emplace_back(A.row(i), b(i) + (A.row(i) * optimizationRes.first)(0));
265  }
266  // If optimizationRes.second is false, it means that rhs is unbounded in this direction, i.e., the current constraint is not inserted
267  }
268 
269  // evaluation of lhs in directions of rhs
270  for (Eigen::Index i = 0; i < nativeRhs.A.rows(); ++i) {
271  auto optimizationRes = optimize(nativeRhs.A.row(i));
272  if ( optimizationRes.second ) {
273  resultConstraints.emplace_back(nativeRhs.A.row(i), nativeRhs.b(i) + (nativeRhs.A.row(i) * optimizationRes.first)(0));
274  }
275  // If optimizationRes.second is false, it means that rhs is unbounded in this direction, i.e., the current constraint is not inserted
276  }
277 
278  if (resultConstraints.empty()) {
279  return std::make_shared<NativePolytope<ValueType>>(std::vector<Halfspace<ValueType>>());
280  } else {
281  EigenMatrix newA(resultConstraints.size(), resultConstraints.front().first.rows());
282  EigenVector newb(resultConstraints.size());
283  for (Eigen::Index i = 0; i < newA.rows(); ++i) {
284  newA.row(i) = resultConstraints[i].first;
285  newb(i) = resultConstraints[i].second;
286  }
287  return std::make_shared<NativePolytope<ValueType>>(EmptyStatus::Nonempty, std::move(newA), std::move(newb));

434  break;
435  }
436  solver->pop();
437  }

439  newHalfspaces.reserve(keptConstraints.getNumberOfSetBits());
