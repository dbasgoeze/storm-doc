132  virtual boost::any visit(storm::expressions::BinaryNumericalFunctionExpression const& expression, boost::any const& data) override {
133  msat_term leftResult = boost::any_cast<msat_term>(expression.getFirstOperand()->accept(*this, data));
134  msat_term rightResult = boost::any_cast<msat_term>(expression.getSecondOperand()->accept(*this, data));
135 
136  msat_term result = leftResult;
137  int_fast64_t exponent;
138  int_fast64_t modulus;
139  storm::expressions::Variable freshAuxiliaryVariable;
140  msat_term modVariable;
141  msat_term lower;
142  msat_term upper;
143  typename storm::NumberTraits<storm::GmpRationalNumber>::IntegerType gmpModulus;
144  switch (expression.getOperatorType()) {
145  case storm::expressions::BinaryNumericalFunctionExpression::OperatorType::Plus:
146  return msat_make_plus(env, leftResult, rightResult);
147  case storm::expressions::BinaryNumericalFunctionExpression::OperatorType::Minus:
148  return msat_make_plus(env, leftResult, msat_make_times(env, msat_make_number(env, "-1"), rightResult));
149  case storm::expressions::BinaryNumericalFunctionExpression::OperatorType::Times:
150  return msat_make_times(env, leftResult, rightResult);
151  case storm::expressions::BinaryNumericalFunctionExpression::OperatorType::Divide:
152  return msat_make_divide(env, leftResult, rightResult);
153  case storm::expressions::BinaryNumericalFunctionExpression::OperatorType::Min:
154  return msat_make_term_ite(env, msat_make_leq(env, leftResult, rightResult), leftResult, rightResult);
155  case storm::expressions::BinaryNumericalFunctionExpression::OperatorType::Max:
156  return msat_make_term_ite(env, msat_make_leq(env, leftResult, rightResult), rightResult, leftResult);
157  case storm::expressions::BinaryNumericalFunctionExpression::OperatorType::Power:
158  exponent = expression.getSecondOperand()->evaluateAsInt();
159  STORM_LOG_THROW(exponent >= 0, storm::exceptions::ExpressionEvaluationException, "Cannot evaluate expression with negative exponent.");
160  --exponent;
161  if (exponent > 0) {
162  for (; exponent > 0; --exponent) {
163  result = msat_make_times(env, result, leftResult);
164  }
165  }
166  return result;
167  case storm::expressions::BinaryNumericalFunctionExpression::OperatorType::Modulo:
168  modulus = expression.getSecondOperand()->evaluateAsInt();
169  STORM_LOG_THROW(modulus > 0, storm::exceptions::ExpressionEvaluationException, "Cannot evaluate expression with negative modulus.");
170 
171  freshAuxiliaryVariable = manager.declareFreshVariable(manager.getIntegerType(), true);
172  modVariable = msat_make_constant(env, createVariable(freshAuxiliaryVariable));
173 
174  gmpModulus = typename storm::NumberTraits<storm::GmpRationalNumber>::IntegerType(static_cast<unsigned>(modulus));
175 
176  // Create the constraint that fixes the value of the fresh variable.
177  additionalConstraints.push_back(msat_make_int_modular_congruence(env, gmpModulus.get_mpz_t(), modVariable, leftResult));
178 
179  // Create the constraint that limits the value of the modulo operation to 0 <= val <= modulus-1.
180  lower = msat_make_number(env, "-1");
181  upper = msat_make_number(env, std::to_string(modulus - 1).c_str());
182  additionalConstraints.push_back(msat_make_and(env, msat_make_not(env, msat_make_leq(env, modVariable, lower)), msat_make_leq(env, modVariable, upper)));
183  return modVariable;
184  default:
185  STORM_LOG_THROW(false, storm::exceptions::ExpressionEvaluationException, "Cannot evaluate expression: unknown numerical binary operator '" << static_cast<uint_fast64_t>(expression.getOperatorType()) << "' in expression " << expression << ".");
186  }

195  if (expression.getFirstOperand()->getType().isBooleanType() && expression.getSecondOperand()->getType().isBooleanType()) {
196  return msat_make_iff(env, leftResult, rightResult);
197  } else {

201  if (expression.getFirstOperand()->getType().isBooleanType() && expression.getSecondOperand()->getType().isBooleanType()) {
202  return msat_make_not(env, msat_make_iff(env, leftResult, rightResult));
203  } else {

205  }

278  storm::expressions::Expression translateExpression(msat_term const& term) {
279  if (msat_term_is_and(env, term)) {
280  return translateExpression(msat_term_get_arg(term, 0)) && translateExpression(msat_term_get_arg(term, 1));
281  } else if (msat_term_is_or(env, term)) {
282  return translateExpression(msat_term_get_arg(term, 0)) || translateExpression(msat_term_get_arg(term, 1));
283  } else if (msat_term_is_iff(env, term)) {
284  return storm::expressions::iff(translateExpression(msat_term_get_arg(term, 0)), translateExpression(msat_term_get_arg(term, 1)));
285  } else if (msat_term_is_not(env, term)) {
286  return !translateExpression(msat_term_get_arg(term, 0));
287  } else if (msat_term_is_plus(env, term)) {
288  return translateExpression(msat_term_get_arg(term, 0)) + translateExpression(msat_term_get_arg(term, 1));
289  } else if (msat_term_is_times(env, term)) {
290  return translateExpression(msat_term_get_arg(term, 0)) * translateExpression(msat_term_get_arg(term, 1));
291  } else if (msat_term_is_equal(env, term)) {
292  return translateExpression(msat_term_get_arg(term, 0)) == translateExpression(msat_term_get_arg(term, 1));
293  } else if (msat_term_is_leq(env, term)) {
294  return translateExpression(msat_term_get_arg(term, 0)) <= translateExpression(msat_term_get_arg(term, 1));
295  } else if (msat_term_is_true(env, term)) {
296  return manager.boolean(true);
297  } else if (msat_term_is_false(env, term)) {
298  return manager.boolean(false);
299  } else if (msat_term_is_constant(env, term)) {
300  char* name = msat_decl_get_name(msat_term_get_decl(term));
301  std::string nameString(name);
302  storm::expressions::Expression result = manager.getVariableExpression(nameString.substr(0, nameString.find('/')));
303  msat_free(name);
304  return result;
305  } else if (msat_term_is_number(env, term)) {
306  char* termAsCString = msat_term_repr(term);
307  std::string termString(termAsCString);
308  msat_free(termAsCString);
309  if (msat_is_integer_type(env, msat_term_get_type(term))) {
310  return manager.integer(std::stoll(msat_term_repr(term)));
311  } else if (msat_is_rational_type(env, msat_term_get_type(term))) {
312  return manager.rational(storm::utility::convertNumber<storm::RationalNumber>(termString));
313  }
314  } else if (msat_term_is_term_ite(env, term)) {
315  return storm::expressions::ite(translateExpression(msat_term_get_arg(term, 0)), translateExpression(msat_term_get_arg(term, 1)), translateExpression(msat_term_get_arg(term, 2)));
316  }
317 
318  // If all other cases did not apply, we cannot represent the term in our expression framework.
319  char* termAsCString = msat_term_repr(term);
320  std::string termString(termAsCString);
321  msat_free(termAsCString);
322  STORM_LOG_THROW(false, storm::exceptions::ExpressionEvaluationException, "Cannot translate expression: unknown term: '" << termString << "'.");
323  }
