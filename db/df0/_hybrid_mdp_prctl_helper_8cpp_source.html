435  void eliminateEndComponentsAndTargetStatesReachabilityRewards(std::pair<storm::storage::SparseMatrix<ValueType>, std::vector<ValueType>>& explicitRepresentation, SolverRequirementsData<ValueType>& solverRequirementsData, storm::storage::BitVector const& targetStates, bool computeOneStepTargetProbabilities) {
436 
437  // Get easier handles to the data.
438  auto& transitionMatrix = explicitRepresentation.first;
439  auto& rewardVector = explicitRepresentation.second;
440 
441  // Start by computing the choices with reward 0, as we only want ECs within this fragment.
442  storm::storage::BitVector zeroRewardChoices(transitionMatrix.getRowCount());
443 
444  uint64_t index = 0;
445  for (auto const& e : rewardVector) {
446  if (storm::utility::isZero(e)) {
447  zeroRewardChoices.set(index);
448  }
449  ++index;
450  }
451 
452  // Compute the states that have some zero reward choice.
453  storm::storage::BitVector candidateStates(solverRequirementsData.properMaybeStates);
454  for (auto state : solverRequirementsData.properMaybeStates) {
455  bool keepState = false;
456 
457  for (auto row = transitionMatrix.getRowGroupIndices()[state], rowEnd = transitionMatrix.getRowGroupIndices()[state + 1]; row < rowEnd; ++row) {
458  if (zeroRewardChoices.get(row)) {
459  keepState = true;
460  break;
461  }
462  }
463 
464  if (!keepState) {
465  candidateStates.set(state, false);
466  }
467  }
468 
469  bool doDecomposition = !candidateStates.empty();
470 
471  storm::storage::MaximalEndComponentDecomposition<ValueType> endComponentDecomposition;
472  if (doDecomposition) {
473  auto backwardTransitions = transitionMatrix.transpose(true);
474 
475  // Only keep the candidate states that (under some scheduler) can stay in the set of candidates forever
476  candidateStates = storm::utility::graph::performProb0E(transitionMatrix, transitionMatrix.getRowGroupIndices(), backwardTransitions, candidateStates, ~candidateStates);
477 
478  doDecomposition = !candidateStates.empty();
479 
480  if (doDecomposition) {
481  // If there are candidates, compute the states that are in MECs with zero reward.
482  endComponentDecomposition = storm::storage::MaximalEndComponentDecomposition<ValueType>(transitionMatrix, backwardTransitions, candidateStates, zeroRewardChoices);
483  }
484  }
485 
486  // Only do more work if there are actually end-components.
487  if (doDecomposition && !endComponentDecomposition.empty()) {
488  STORM_LOG_DEBUG("Eliminating " << endComponentDecomposition.size() << " EC(s).");
489  if (computeOneStepTargetProbabilities) {
490  solverRequirementsData.oneStepTargetProbabilities = std::vector<ValueType>(solverRequirementsData.properMaybeStates.getNumberOfSetBits(), storm::utility::zero<ValueType>());
491  }
492 
493  std::vector<ValueType> subvector(solverRequirementsData.properMaybeStates.getNumberOfSetBits(), storm::utility::zero<ValueType>());
494  SparseMdpEndComponentInformation<ValueType>::eliminateEndComponents(endComponentDecomposition, transitionMatrix, solverRequirementsData.properMaybeStates, computeOneStepTargetProbabilities ? &targetStates : nullptr, nullptr, &rewardVector, transitionMatrix, computeOneStepTargetProbabilities ? &solverRequirementsData.oneStepTargetProbabilities.get() : nullptr, &subvector);
495  rewardVector = std::move(subvector);
496  } else {
497  STORM_LOG_DEBUG("Not eliminating ECs as there are none.");
498  if (computeOneStepTargetProbabilities) {
499  solverRequirementsData.oneStepTargetProbabilities = computeOneStepTargetProbabilitiesFromExtendedExplicitRepresentation(explicitRepresentation.first, solverRequirementsData.properMaybeStates, targetStates);
500  }
501  eliminateExtendedStatesFromExplicitRepresentation(explicitRepresentation, solverRequirementsData.initialScheduler, solverRequirementsData.properMaybeStates);
502  }

517  template<storm::dd::DdType DdType, typename ValueType>
518  std::unique_ptr<CheckResult> HybridMdpPrctlHelper<DdType, ValueType>::computeReachabilityRewards(Environment const& env, OptimizationDirection dir, storm::models::symbolic::NondeterministicModel<DdType, ValueType> const& model, storm::dd::Add<DdType, ValueType> const& transitionMatrix, RewardModelType const& rewardModel, storm::dd::Bdd<DdType> const& targetStates, bool qualitative) {
519 
520  // Only compute the result if there is at least one reward model.
521  STORM_LOG_THROW(!rewardModel.empty(), storm::exceptions::InvalidPropertyException, "Missing reward model for formula. Skipping formula.");
522 
523  // Determine which states have a reward of infinity by definition.
524  storm::dd::Bdd<DdType> infinityStates;
525  storm::dd::Bdd<DdType> transitionMatrixBdd = transitionMatrix.notZero();
526  if (dir == OptimizationDirection::Minimize) {
527  infinityStates = storm::utility::graph::performProb1E(model, transitionMatrixBdd, model.getReachableStates(), targetStates, storm::utility::graph::performProbGreater0E(model, transitionMatrixBdd, model.getReachableStates(), targetStates));
528  } else {
529  infinityStates = storm::utility::graph::performProb1A(model, transitionMatrixBdd, targetStates, storm::utility::graph::performProbGreater0A(model, transitionMatrixBdd, model.getReachableStates(), targetStates));
530  }
531  infinityStates = !infinityStates && model.getReachableStates();
532  storm::dd::Bdd<DdType> maybeStatesWithTargetStates = !infinityStates && model.getReachableStates();
533  storm::dd::Bdd<DdType> maybeStates = !targetStates && maybeStatesWithTargetStates;
534 
535  STORM_LOG_INFO("Preprocessing: " << infinityStates.getNonZeroCount() << " states with reward infinity, " << targetStates.getNonZeroCount() << " target states (" << maybeStates.getNonZeroCount() << " states remaining).");
536 
537  // Check whether we need to compute exact rewards for some states.
538  if (qualitative) {
539  // Set the values for all maybe-states to 1 to indicate that their reward values
540  // are neither 0 nor infinity.
541  return std::unique_ptr<CheckResult>(new SymbolicQuantitativeCheckResult<DdType, ValueType>(model.getReachableStates(), infinityStates.ite(model.getManager().getConstant(storm::utility::infinity<ValueType>()), model.getManager().template getAddZero<ValueType>()) + maybeStates.template toAdd<ValueType>() * model.getManager().getConstant(storm::utility::one<ValueType>())));
542  } else {
543  // If there are maybe states, we need to solve an equation system.
544  if (!maybeStates.isZero()) {
545  // If we maximize, we know that the solution to the equation system is unique.
546  bool hasNoEndComponents = dir == storm::solver::OptimizationDirection::Maximize;
547  bool hasUniqueSolution = hasNoEndComponents;
548  // Check for requirements of the solver this early so we can adapt the maybe states accordingly.
549  storm::solver::GeneralMinMaxLinearEquationSolverFactory<ValueType> linearEquationSolverFactory;
550  storm::solver::MinMaxLinearEquationSolverRequirements requirements = linearEquationSolverFactory.getRequirements(env, hasUniqueSolution, hasNoEndComponents, dir);
551  storm::solver::MinMaxLinearEquationSolverRequirements clearedRequirements = requirements;
552  bool extendMaybeStates = false;
553  if (clearedRequirements.hasEnabledRequirement()) {
554  if (clearedRequirements.uniqueSolution()) {
555  STORM_LOG_DEBUG("Scheduling EC elimination, because the solver requires it.");
556  extendMaybeStates = true;
557  clearedRequirements.clearUniqueSolution();
558  hasUniqueSolution = true;
559  // There might still be end components in which reward is collected.
560  }
561  if (clearedRequirements.validInitialScheduler()) {
562  STORM_LOG_DEBUG("Computing valid scheduler, because the solver requires it.");
563  extendMaybeStates = true;
564  clearedRequirements.clearValidInitialScheduler();
565  }
566  clearedRequirements.clearLowerBounds();
567  if (clearedRequirements.upperBounds()) {
568  STORM_LOG_DEBUG("Computing upper bounds, because the solver requires it.");
569  extendMaybeStates = true;
570  clearedRequirements.clearUpperBounds();
571  }
572  STORM_LOG_THROW(!clearedRequirements.hasEnabledCriticalRequirement(), storm::exceptions::UncheckedRequirementException, "Solver requirements " + clearedRequirements.getEnabledRequirementsAsString() + " not checked.");
573  }
574 
575  // Compute the set of maybe states that we are required to keep in the translation to explicit.
576  storm::dd::Bdd<DdType> requiredMaybeStates = extendMaybeStates ? maybeStatesWithTargetStates : maybeStates;
577 
578  storm::utility::Stopwatch conversionWatch;
579 
580  // Create the ODD for the translation between symbolic and explicit storage.
581  conversionWatch.start();
582  storm::dd::Odd odd = requiredMaybeStates.createOdd();
583  conversionWatch.stop();
584 
585  // Create the matrix and the vector for the equation system.
586  storm::dd::Add<DdType, ValueType> maybeStatesAdd = maybeStates.template toAdd<ValueType>();
587 
588  // Start by getting rid of
589  // (a) transitions from non-maybe states, and
590  // (b) the choices in the transition matrix that lead to a state that is neither a maybe state
591  // nor a target state ('infinity choices').
592  storm::dd::Add<DdType, ValueType> choiceFilterAdd = maybeStatesAdd * (transitionMatrixBdd && maybeStatesWithTargetStates.renameVariables(model.getRowVariables(), model.getColumnVariables())).existsAbstract(model.getColumnVariables()).template toAdd<ValueType>();
593  storm::dd::Add<DdType, ValueType> submatrix = transitionMatrix * choiceFilterAdd;
594 
595  // Then compute the reward vector to use in the computation.
596  storm::dd::Add<DdType, ValueType> subvector = rewardModel.getTotalRewardVector(maybeStatesAdd, choiceFilterAdd, submatrix, model.getColumnVariables());
597 
598  conversionWatch.start();
599  std::vector<uint_fast64_t> rowGroupSizes = (submatrix.notZero().existsAbstract(model.getColumnVariables()) || subvector.notZero()).template toAdd<uint_fast64_t>().sumAbstract(model.getNondeterminismVariables()).toVector(odd);
600  conversionWatch.stop();
601 
602  // Finally cut away all columns targeting non-maybe states (or non-(maybe or target) states, respectively).
603  submatrix *= extendMaybeStates ? maybeStatesWithTargetStates.swapVariables(model.getRowColumnMetaVariablePairs()).template toAdd<ValueType>() : maybeStatesAdd.swapVariables(model.getRowColumnMetaVariablePairs());
604 
605  // Translate the symbolic matrix/vector to their explicit representations.
606  conversionWatch.start();
607  std::pair<storm::storage::SparseMatrix<ValueType>, std::vector<ValueType>> explicitRepresentation = submatrix.toMatrixVector(std::move(rowGroupSizes), subvector, model.getRowVariables(), model.getColumnVariables(), model.getNondeterminismVariables(), odd, odd);
608  conversionWatch.stop();
609  STORM_LOG_INFO("Converting symbolic matrix/vector to explicit representation done in " << conversionWatch.getTimeInMilliseconds() << "ms.");
610 
611  // Fulfill the solver's requirements.
612  SolverRequirementsData<ValueType> solverRequirementsData;
613  if (extendMaybeStates) {
614  storm::storage::BitVector targetStates = computeTargetStatesForReachabilityRewardsFromExplicitRepresentation(explicitRepresentation.first);
615  solverRequirementsData.properMaybeStates = ~targetStates;
616 
617  if (requirements.uniqueSolution()) {
618  STORM_LOG_THROW(!requirements.validInitialScheduler(), storm::exceptions::UncheckedRequirementException, "The underlying solver requires a unique solution and an initial valid scheduler. This is currently not supported for expected reward properties.");
619  // eliminate the end components with reward 0.
620  // Note that this may also compute the oneStepTargetProbabilities if upper bounds are required.
621  eliminateEndComponentsAndTargetStatesReachabilityRewards(explicitRepresentation, solverRequirementsData, targetStates, requirements.upperBounds());
622  // The solution becomes unique after end components have been eliminated.
623  hasUniqueSolution = true;
624  }
625  else {
626  if (requirements.validInitialScheduler()) {
627  // Compute a valid initial scheduler.
628  solverRequirementsData.initialScheduler = computeValidInitialSchedulerForReachabilityRewards<ValueType>(explicitRepresentation.first, solverRequirementsData.properMaybeStates, targetStates);
629  }
630 
631  if (requirements.upperBounds()) {
632  solverRequirementsData.oneStepTargetProbabilities = computeOneStepTargetProbabilitiesFromExtendedExplicitRepresentation(explicitRepresentation.first, solverRequirementsData.properMaybeStates, targetStates);
633  }
634 
635  // Since we needed the transitions to target states to be translated as well for the computation
636  // of the scheduler, we have to get rid of them now.
637  eliminateExtendedStatesFromExplicitRepresentation(explicitRepresentation, solverRequirementsData.initialScheduler, solverRequirementsData.properMaybeStates);
638  }
639  }
640 
641  // Create the solution vector.
642  std::vector<ValueType> x(explicitRepresentation.first.getRowGroupCount(), storm::utility::zero<ValueType>());
643 
644  // Now solve the resulting equation system.
645  std::unique_ptr<storm::solver::MinMaxLinearEquationSolver<ValueType>> solver = linearEquationSolverFactory.create(env);
646 
647  // Set whether the equation system will have a unique solution / no end components
648  solver->setHasUniqueSolution(hasUniqueSolution);
649  solver->setHasNoEndComponents(hasNoEndComponents);
650 
651  // If the solver requires upper bounds, compute them now.
652  if (requirements.upperBounds()) {
653  setUpperRewardBounds(*solver, dir, explicitRepresentation.first, explicitRepresentation.second, solverRequirementsData.oneStepTargetProbabilities.get());
654  }
655 
656  solver->setMatrix(std::move(explicitRepresentation.first));
657 
658  // Move the scheduler to the solver.
659  if (solverRequirementsData.initialScheduler) {
660  solver->setInitialScheduler(std::move(solverRequirementsData.initialScheduler.get()));
661  }
662 
663  solver->setLowerBound(storm::utility::zero<ValueType>());
664  solver->setRequirementsChecked();
665  solver->solveEquations(env, dir, x, explicitRepresentation.second);
666 
667  // If we eliminated end components, we need to extend the solution vector.
668  if (requirements.uniqueSolution() && solverRequirementsData.ecInformation) {
669  std::vector<ValueType> extendedVector(solverRequirementsData.properMaybeStates.getNumberOfSetBits());
670  solverRequirementsData.ecInformation.get().setValues(extendedVector, solverRequirementsData.properMaybeStates, x);
671  x = std::move(extendedVector);
672  }
673 
674  // If we extended the maybe states, we create a new ODD that only contains proper maybe states.
675  if (extendMaybeStates) {
676  odd = maybeStates.createOdd();
677  }
678 
679  // Return a hybrid check result that stores the numerical values explicitly.
680  return std::unique_ptr<CheckResult>(new storm::modelchecker::HybridQuantitativeCheckResult<DdType, ValueType>(model.getReachableStates(), model.getReachableStates() && !maybeStates, infinityStates.ite(model.getManager().getConstant(storm::utility::infinity<ValueType>()), model.getManager().template getAddZero<ValueType>()), maybeStates, odd, x));
681  } else {
682  return std::unique_ptr<CheckResult>(new storm::modelchecker::SymbolicQuantitativeCheckResult<DdType, ValueType>(model.getReachableStates(), infinityStates.ite(model.getManager().getConstant(storm::utility::infinity<ValueType>()), model.getManager().template getAddZero<ValueType>())));
683  }
684  }
