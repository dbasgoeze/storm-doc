233  DdNode* InternalBdd<DdType::CUDD>::fromVectorRec(::DdManager* manager, uint_fast64_t& currentOffset, uint_fast64_t currentLevel, uint_fast64_t maxLevel, Odd const& odd, std::vector<uint_fast64_t> const& ddVariableIndices, std::function<bool (uint64_t)> const& filter) {
234  if (currentLevel == maxLevel) {
235  // If we are in a terminal node of the ODD, we need to check whether the then-offset of the ODD is one
236  // (meaning the encoding is a valid one) or zero (meaning the encoding is not valid). Consequently, we
237  // need to copy the next value of the vector iff the then-offset is greater than zero.
238  if (odd.getThenOffset() > 0) {
239  if (filter(currentOffset++)) {
240  return Cudd_ReadOne(manager);
241  } else {
242  return Cudd_ReadLogicZero(manager);
243  }
244  } else {
245  return Cudd_ReadZero(manager);
246  }
247  } else {
248  // If the total offset is zero, we can just return the constant zero DD.
249  if (odd.getThenOffset() + odd.getElseOffset() == 0) {
250  return Cudd_ReadZero(manager);
251  }
252 
253  // Determine the new else-successor.
254  DdNode* elseSuccessor = nullptr;
255  if (odd.getElseOffset() > 0) {
256  elseSuccessor = fromVectorRec(manager, currentOffset, currentLevel + 1, maxLevel, odd.getElseSuccessor(), ddVariableIndices, filter);
257  } else {
258  elseSuccessor = Cudd_ReadLogicZero(manager);
259  }
260  Cudd_Ref(elseSuccessor);
261 
262  // Determine the new then-successor.
263  DdNode* thenSuccessor = nullptr;
264  if (odd.getThenOffset() > 0) {
265  thenSuccessor = fromVectorRec(manager, currentOffset, currentLevel + 1, maxLevel, odd.getThenSuccessor(), ddVariableIndices, filter);
266  } else {
267  thenSuccessor = Cudd_ReadLogicZero(manager);
268  }
269  Cudd_Ref(thenSuccessor);
270 
271  // Create a node representing ITE(currentVar, thenSuccessor, elseSuccessor);
272  DdNode* currentVar = Cudd_bddIthVar(manager, static_cast<int>(ddVariableIndices[currentLevel]));
273  Cudd_Ref(currentVar);
274  DdNode* result = Cudd_bddIte(manager, currentVar, thenSuccessor, elseSuccessor);
275  Cudd_Ref(result);
276 
277  // Dispose of the intermediate results
278  Cudd_RecursiveDeref(manager, currentVar);
279  Cudd_RecursiveDeref(manager, thenSuccessor);
280  Cudd_RecursiveDeref(manager, elseSuccessor);
281 
282  // Before returning, we remove the protection imposed by the previous call to Cudd_Ref.
283  Cudd_Deref(result);
284 
285  return result;
286  }

341  std::shared_ptr<Odd> InternalBdd<DdType::CUDD>::createOddRec(DdNode const* dd, cudd::Cudd const& manager, uint_fast64_t currentLevel, uint_fast64_t maxLevel, std::vector<uint_fast64_t> const& ddVariableIndices, std::vector<std::unordered_map<DdNode const*, std::shared_ptr<Odd>>>& uniqueTableForLevels) {
342  // Check whether the ODD for this node has already been computed (for this level) and if so, return this instead.
343  auto it = uniqueTableForLevels[currentLevel].find(dd);
344  if (it != uniqueTableForLevels[currentLevel].end()) {
345  return it->second;
346  } else {
347  // Otherwise, we need to recursively compute the ODD.
348 
349  // If we are already at the maximal level that is to be considered, we can simply create an Odd without
350  // successors
351  if (currentLevel == maxLevel) {
352  auto oddNode = std::make_shared<Odd>(nullptr, 0, nullptr, dd != Cudd_ReadLogicZero(manager.getManager()) ? 1 : 0);
353  uniqueTableForLevels[currentLevel].emplace(dd, oddNode);
354  return oddNode;
355  } else if (ddVariableIndices[currentLevel] < Cudd_NodeReadIndex(dd)) {
356  // If we skipped the level in the DD, we compute the ODD just for the else-successor and use the same
357  // node for the then-successor as well.
358  std::shared_ptr<Odd> elseNode = createOddRec(dd, manager, currentLevel + 1, maxLevel, ddVariableIndices, uniqueTableForLevels);
359  std::shared_ptr<Odd> thenNode = elseNode;
360 
361  auto oddNode = std::make_shared<Odd>(elseNode, elseNode->getTotalOffset(), thenNode, elseNode->getTotalOffset());
362  uniqueTableForLevels[currentLevel].emplace(dd, oddNode);
363  return oddNode;
364  } else {
365  // Otherwise, we compute the ODDs for both the then- and else successors.
366  DdNode const* thenDdNode = Cudd_T_const(dd);
367  DdNode const* elseDdNode = Cudd_E_const(dd);
368 
369  if (Cudd_IsComplement(dd)) {
370  thenDdNode = Cudd_Not(thenDdNode);
371  elseDdNode = Cudd_Not(elseDdNode);
372  }
373 
374  std::shared_ptr<Odd> elseNode = createOddRec(elseDdNode, manager, currentLevel + 1, maxLevel, ddVariableIndices, uniqueTableForLevels);
375  std::shared_ptr<Odd> thenNode = createOddRec(thenDdNode, manager, currentLevel + 1, maxLevel, ddVariableIndices, uniqueTableForLevels);
376 
377  auto oddNode = std::make_shared<Odd>(elseNode, elseNode->getTotalOffset(), thenNode, thenNode->getTotalOffset());
378  uniqueTableForLevels[currentLevel].emplace(dd, oddNode);
379  return oddNode;
380  }

426  void InternalBdd<DdType::CUDD>::splitIntoGroupsRec(DdNode* dd, bool negated, std::vector<InternalBdd<DdType::CUDD>>& groups, std::vector<uint_fast64_t> const& ddGroupVariableIndices, uint_fast64_t currentLevel, uint_fast64_t maxLevel) const {
427  // For the empty DD, we do not need to create a group.

433  groups.push_back(InternalBdd<DdType::CUDD>(ddManager, cudd::BDD(ddManager->getCuddManager(), negated ? Cudd_Complement(dd) : dd)));

436  splitIntoGroupsRec(dd, negated, groups, ddGroupVariableIndices, currentLevel + 1, maxLevel);

442  splitIntoGroupsRec(thenNode, negated ^ Cudd_IsComplement(thenNode), groups, ddGroupVariableIndices, currentLevel + 1, maxLevel);

451  void InternalBdd<DdType::CUDD>::filterExplicitVectorRec(DdNode const* dd, cudd::Cudd const& manager, uint_fast64_t currentLevel, bool complement, uint_fast64_t maxLevel, std::vector<uint_fast64_t> const& ddVariableIndices, uint_fast64_t currentOffset, storm::dd::Odd const& odd, storm::storage::BitVector& result, uint_fast64_t& currentIndex, storm::storage::BitVector const& values) {
452  // If there are no more values to select, we can directly return.
453  if (dd == Cudd_ReadLogicZero(manager.getManager()) && !complement) {
454  return;
455  } else if (dd == Cudd_ReadOne(manager.getManager()) && complement) {
456  return;
457  }
458 
459  if (currentLevel == maxLevel) {
460  result.set(currentIndex++, values.get(currentOffset));
461  } else if (ddVariableIndices[currentLevel] < Cudd_NodeReadIndex(dd)) {
462  // If we skipped a level, we need to enumerate the explicit entries for the case in which the bit is set
463  // and for the one in which it is not set.
464  filterExplicitVectorRec(dd, manager, currentLevel + 1, complement, maxLevel, ddVariableIndices, currentOffset, odd.getElseSuccessor(), result, currentIndex, values);
465  filterExplicitVectorRec(dd, manager, currentLevel + 1, complement, maxLevel, ddVariableIndices, currentOffset + odd.getElseOffset(), odd.getThenSuccessor(), result, currentIndex, values);
466  } else {
467  // Otherwise, we compute the ODDs for both the then- and else successors.
468  DdNode const* thenDdNode = Cudd_T_const(dd);
469  DdNode const* elseDdNode = Cudd_E_const(dd);
470 
471  // Determine whether we have to evaluate the successors as if they were complemented.
472  bool elseComplemented = Cudd_IsComplement(elseDdNode) ^ complement;
473  bool thenComplemented = Cudd_IsComplement(thenDdNode) ^ complement;
474 
475  filterExplicitVectorRec(Cudd_Regular(elseDdNode), manager, currentLevel + 1, elseComplemented, maxLevel, ddVariableIndices, currentOffset, odd.getElseSuccessor(), result, currentIndex, values);
476  filterExplicitVectorRec(Cudd_Regular(thenDdNode), manager, currentLevel + 1, thenComplemented, maxLevel, ddVariableIndices, currentOffset + odd.getElseOffset(), odd.getThenSuccessor(), result, currentIndex, values);
477  }
478  }

480  std::pair<std::vector<storm::expressions::Expression>, std::unordered_map<uint_fast64_t, storm::expressions::Variable>> InternalBdd<DdType::CUDD>::toExpression(storm::expressions::ExpressionManager& manager) const {
481  std::pair<std::vector<storm::expressions::Expression>, std::unordered_map<uint_fast64_t, storm::expressions::Variable>> result;
482 
483  // Create (and maintain) a mapping from the DD nodes to a counter that says the how-many-th node (within the
484  // nodes of equal index) the node was.
485  std::unordered_map<DdNode const*, uint_fast64_t> nodeToCounterMap;
486  std::vector<uint_fast64_t> nextCounterForIndex(ddManager->getNumberOfDdVariables(), 0);
487  std::unordered_map<std::pair<uint_fast64_t, uint_fast64_t>, storm::expressions::Variable> countIndexToVariablePair;
488 
489  bool negated = Cudd_Regular(this->getCuddDdNode()) != this->getCuddDdNode();
490 
491  // Translate from the top node downwards.
492  storm::expressions::Variable topVariable = this->toExpressionRec(Cudd_Regular(this->getCuddDdNode()), ddManager->getCuddManager(), manager, result.first, result.second, countIndexToVariablePair, nodeToCounterMap, nextCounterForIndex);
493 
494  // Create the final expression.
495  if (negated) {
496  result.first.push_back(!topVariable);
497  } else {
498  result.first.push_back(topVariable);
499  }
500 
501  return result;
502  }

504  storm::expressions::Variable InternalBdd<DdType::CUDD>::toExpressionRec(DdNode const* dd, cudd::Cudd const& ddManager, storm::expressions::ExpressionManager& manager, std::vector<storm::expressions::Expression>& expressions, std::unordered_map<uint_fast64_t, storm::expressions::Variable>& indexToVariableMap, std::unordered_map<std::pair<uint_fast64_t, uint_fast64_t>, storm::expressions::Variable>& countIndexToVariablePair, std::unordered_map<DdNode const*, uint_fast64_t>& nodeToCounterMap, std::vector<uint_fast64_t>& nextCounterForIndex) {
505  STORM_LOG_ASSERT(dd == Cudd_Regular(dd), "Expected non-negated BDD node.");
506 
507  // First, try to look up the current node if it's not a terminal node.
508  auto nodeCounterIt = nodeToCounterMap.find(dd);
509  if (nodeCounterIt != nodeToCounterMap.end()) {
510  // If we have found the node, this means we can look up the counter-index pair and get the corresponding variable.
511  auto variableIt = countIndexToVariablePair.find(std::make_pair(nodeCounterIt->second, Cudd_NodeReadIndex(dd)));
512  STORM_LOG_ASSERT(variableIt != countIndexToVariablePair.end(), "Unable to find node.");
513  return variableIt->second;
514  }
515 
516  // If the node was not yet encountered, we create a variable and associate it with the appropriate expression.
517  storm::expressions::Variable newNodeVariable = manager.declareFreshBooleanVariable();
518 
519  // Since we want to reuse the variable whenever possible, we insert the appropriate entries in the hash table.
520  if (!Cudd_IsConstant_const(dd)) {
521  // If we are dealing with a non-terminal node, we count it as a new node with this index.
522  nodeToCounterMap[dd] = nextCounterForIndex[Cudd_NodeReadIndex(dd)];
523  countIndexToVariablePair[std::make_pair(nextCounterForIndex[Cudd_NodeReadIndex(dd)], Cudd_NodeReadIndex(dd))] = newNodeVariable;
524  ++nextCounterForIndex[Cudd_NodeReadIndex(dd)];
525  } else {
526  // If it's a terminal node, it is the one leaf and there's no need to keep track of a counter for this level.
527  nodeToCounterMap[dd] = 0;
528  countIndexToVariablePair[std::make_pair(0, Cudd_NodeReadIndex(dd))] = newNodeVariable;
529  }
530 
531  // In the terminal case, we can only have a one since we are considering non-negated nodes only.
532  if (dd == Cudd_ReadOne(ddManager.getManager())) {
533  // Push the expression that enforces that the new variable is true.
534  expressions.push_back(storm::expressions::iff(manager.boolean(true), newNodeVariable));
535  } else {
536  // In the non-terminal case, we recursively translate the children nodes and then construct and appropriate ite-expression.
537  DdNode const* t = Cudd_T_const(dd);
538  DdNode const* e = Cudd_E_const(dd);
539  DdNode const* T = Cudd_Regular(t);
540  DdNode const* E = Cudd_Regular(e);
541  storm::expressions::Variable thenVariable = toExpressionRec(T, ddManager, manager, expressions, indexToVariableMap, countIndexToVariablePair, nodeToCounterMap, nextCounterForIndex);
542  storm::expressions::Variable elseVariable = toExpressionRec(E, ddManager, manager, expressions, indexToVariableMap, countIndexToVariablePair, nodeToCounterMap, nextCounterForIndex);
543 
544  // Create the appropriate expression.
545  auto indexVariable = indexToVariableMap.find(Cudd_NodeReadIndex(dd));
546  storm::expressions::Variable levelVariable;
547  if (indexVariable == indexToVariableMap.end()) {
548  levelVariable = manager.declareFreshBooleanVariable();
549  indexToVariableMap[Cudd_NodeReadIndex(dd)] = levelVariable;
550  } else {
551  levelVariable = indexVariable->second;
552  }
553  expressions.push_back(storm::expressions::iff(newNodeVariable, storm::expressions::ite(levelVariable, t == T ? thenVariable : !thenVariable, e == E ? elseVariable : !elseVariable)));
554  }
555 
556  // Return the variable for this node.
557  return newNodeVariable;
558  }
