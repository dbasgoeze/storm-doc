52  bool EliminationLinearEquationSolver<ValueType>::internalSolveEquations(Environment const& env, std::vector<ValueType>& x, std::vector<ValueType> const& b) const {
53  // FIXME: This solver will not work for all input systems. More concretely, the current implementation will
54  // not work for systems that have a 1 on the diagonal. This is not a restriction of this technique in general
55  // but arbitrary matrices require pivoting, which is not currently implemented.
56 
57  STORM_LOG_INFO("Solving linear equation system (" << x.size() << " rows) with elimination");
58 
59  storm::storage::SparseMatrix<ValueType> const& transitionMatrix = localA ? *localA : *A;
60  storm::storage::SparseMatrix<ValueType> backwardTransitions = transitionMatrix.transpose();
61 
62  // Initialize the solution to the right-hand side of the equation system.
63  x = b;
64 
65  // Translate the matrix and its transpose into the flexible format.
66  storm::storage::FlexibleSparseMatrix<ValueType> flexibleMatrix(transitionMatrix, false);
67  storm::storage::FlexibleSparseMatrix<ValueType> flexibleBackwardTransitions(backwardTransitions, true);
68 
69  boost::optional<std::vector<uint_fast64_t>> distanceBasedPriorities;
70 
71  // TODO: get the order from the environment
72  storm::settings::modules::EliminationSettings::EliminationOrder order = storm::settings::getModule<storm::settings::modules::EliminationSettings>().getEliminationOrder();
73 
74  if (eliminationOrderNeedsDistances(order)) {
75  // Since we have no initial states at this point, we determine a representative of every BSCC regarding
76  // the backward transitions, because this means that every row is reachable from this set of rows, which
77  // we require to make sure we cover every row.
78  storm::storage::BitVector initialRows = storm::utility::graph::getBsccCover(backwardTransitions);
79  distanceBasedPriorities = getDistanceBasedPriorities(transitionMatrix, backwardTransitions, initialRows, b, eliminationOrderNeedsForwardDistances(order), eliminationOrderNeedsReversedDistances(order));
80  }
81 
82  std::shared_ptr<StatePriorityQueue> priorityQueue = createStatePriorityQueue<ValueType>(distanceBasedPriorities, flexibleMatrix, flexibleBackwardTransitions, b, storm::storage::BitVector(x.size(), true));
83 
84  // Create a state eliminator to perform the actual elimination.
85  PrioritizedStateEliminator<ValueType> eliminator(flexibleMatrix, flexibleBackwardTransitions, priorityQueue, x);
86 
87  // Eliminate all states.
88  while (priorityQueue->hasNext()) {
89  auto state = priorityQueue->pop();
90  eliminator.eliminateState(state, false);
91  }
92 
93  return true;
94  }
