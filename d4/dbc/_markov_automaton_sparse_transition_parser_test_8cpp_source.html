31 TEST(MarkovAutomatonSparseTransitionParserTest, BasicParsing) {
32 
33  // The file that will be used for the test.
34  std::string filename = STORM_TEST_RESOURCES_DIR "/tra/ma_general.tra";
35 
36  // Execute the parser.
37  typename storm::parser::MarkovAutomatonSparseTransitionParser<>::Result result = storm::parser::MarkovAutomatonSparseTransitionParser<>::parseMarkovAutomatonTransitions(filename);
38 
39  // Build the actual transition matrix.
40  storm::storage::SparseMatrix<double> transitionMatrix(result.transitionMatrixBuilder.build(0, 0));
41 
42  // Test all sizes and counts.
43  ASSERT_EQ(STATE_COUNT, transitionMatrix.getColumnCount());
44  ASSERT_EQ(CHOICE_COUNT, transitionMatrix.getRowCount());
45  ASSERT_EQ(12ul, transitionMatrix.getEntryCount());
46  ASSERT_EQ(6ul, transitionMatrix.getRowGroupCount());
47  ASSERT_EQ(7ul, transitionMatrix.getRowGroupIndices().size());
48  ASSERT_EQ(CHOICE_COUNT, result.markovianChoices.size());
49  ASSERT_EQ(STATE_COUNT, result.markovianStates.size());
50  ASSERT_EQ(2ul, result.markovianStates.getNumberOfSetBits());
51  ASSERT_EQ(STATE_COUNT, result.exitRates.size());
52 
53  // Test the general structure of the transition system (that will be an Markov automaton).
54 
55  // Test the mapping between states and transition matrix rows.
56  ASSERT_EQ(0ul, transitionMatrix.getRowGroupIndices()[0]);
57  ASSERT_EQ(1ul, transitionMatrix.getRowGroupIndices()[1]);
58  ASSERT_EQ(2ul, transitionMatrix.getRowGroupIndices()[2]);
59  ASSERT_EQ(3ul, transitionMatrix.getRowGroupIndices()[3]);
60  ASSERT_EQ(4ul, transitionMatrix.getRowGroupIndices()[4]);
61  ASSERT_EQ(6ul, transitionMatrix.getRowGroupIndices()[5]);
62  ASSERT_EQ(7ul, transitionMatrix.getRowGroupIndices()[6]);
63 
64  // Test the Markovian states.
65  ASSERT_TRUE(result.markovianStates.get(0));
66  ASSERT_FALSE(result.markovianStates.get(1));
67  ASSERT_TRUE(result.markovianStates.get(2));
68  ASSERT_FALSE(result.markovianStates.get(3));
69  ASSERT_FALSE(result.markovianStates.get(4));
70  ASSERT_FALSE(result.markovianStates.get(5));
71 
72  // Test the exit rates. These have to be 0 for all non-Markovian states.
73  ASSERT_EQ(2, result.exitRates[0]);
74  ASSERT_EQ(0, result.exitRates[1]);
75  ASSERT_EQ(15, result.exitRates[2]);
76  ASSERT_EQ(0, result.exitRates[3]);
77  ASSERT_EQ(0, result.exitRates[4]);
78  ASSERT_EQ(0, result.exitRates[5]);
79 
80  // Finally, test the transition matrix itself.
81  storm::storage::SparseMatrix<double>::const_iterator cIter = transitionMatrix.begin(0);
82 
83  ASSERT_EQ(2, cIter->getValue());
84  cIter++;
85  ASSERT_EQ(1, cIter->getValue());
86  cIter++;
87  ASSERT_EQ(1, cIter->getValue());
88  cIter++;
89  ASSERT_EQ(2, cIter->getValue());
90  cIter++;
91  ASSERT_EQ(4, cIter->getValue());
92  cIter++;
93  ASSERT_EQ(8, cIter->getValue());
94  cIter++;
95  ASSERT_EQ(0.5, cIter->getValue());
96  cIter++;
97  ASSERT_EQ(0.5, cIter->getValue());
98  cIter++;
99  ASSERT_EQ(1, cIter->getValue());
100  cIter++;
101  ASSERT_EQ(0.5, cIter->getValue());
102  cIter++;
103  ASSERT_EQ(0.5, cIter->getValue());
104  cIter++;
105  ASSERT_EQ(1, cIter->getValue());
106  cIter++;
107  ASSERT_EQ(transitionMatrix.end(), cIter);
108 }
