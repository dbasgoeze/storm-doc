252  void DFTASFChecker::generateVotConstraint(size_t i, std::vector<uint64_t> childVarIndices,
253  std::shared_ptr<storm::storage::DFTElement<ValueType> const> element) {
254  auto vot = std::static_pointer_cast<storm::storage::DFTVot<double> const>(element);
255  // VOTs are implemented via OR over ANDs with all possible combinations
256  std::vector<uint64_t> tmpVars;
257  size_t k = 0;
258  // Generate all permutations of k out of n
259  uint64_t combination = smallestIntWithNBitsSet(static_cast<uint64_t>(vot->threshold()));
260  do {
261  // Construct selected children from combination
262  std::vector<uint64_t> combinationChildren;
263  STORM_LOG_ASSERT(vot->nrChildren() < 64, "Too many children of a VOT Gate.");
264  for (uint8_t j = 0; j < static_cast<uint8_t>(vot->nrChildren()); ++j) {
265  if (combination & (1ul << j)) {
266  combinationChildren.push_back(childVarIndices.at(j));
267  }
268  }
269  // Introduce temporary variable for this AND
270  varNames.push_back("v_" + vot->name() + "_" + std::to_string(k));
271  size_t index = varNames.size() - 1;
272  tmpVars.push_back(index);
273  tmpTimePointVariables.push_back(index);
274  // AND over the selected children
275  constraints.push_back(std::make_shared<IsMaximum>(index, combinationChildren));
276  constraints.back()->setDescription("VOT gate " + element->name() + ": AND no. " + std::to_string(k));
277  // Generate next permutation
278  combination = nextBitPermutation(combination);
279  ++k;
280  } while (combination < (1ul << vot->nrChildren()) && combination != 0);
281 
282  // Constraint is OR over all possible combinations
283  constraints.push_back(std::make_shared<IsMinimum>(timePointVariables.at(i), tmpVars));
284  constraints.back()->setDescription("VOT gate " + element->name() + ": OR");
285  }

330  void DFTASFChecker::generateSpareConstraint(size_t i, std::vector<uint64_t> childVarIndices,
331  std::shared_ptr<storm::storage::DFTElement<ValueType> const> element) {
332  auto spare = std::static_pointer_cast<storm::storage::DFTSpare<double> const>(element);
333  auto const &children = spare->children();
334  uint64_t firstChild = children.front()->id();
335  uint64_t lastChild = children.back()->id();
336 
337  // First child of each spare is claimed in the beginning
338  constraints.push_back(std::make_shared<IsConstantValue>(getClaimVariableIndex(spare->id(), firstChild), 0));
339  constraints.back()->setDescription("SPARE gate " + spare->name() + " claims first child");
340 
341  // If last child is claimed before failure, then the spare fails when the last child fails (constraint 5)
342  std::shared_ptr<SmtConstraint> leftC = std::make_shared<IsLess>(
343  getClaimVariableIndex(spare->id(), lastChild), childVarIndices.back());
344  constraints.push_back(std::make_shared<Implies>(leftC, std::make_shared<IsEqual>(timePointVariables.at(i),
345  childVarIndices.back())));
346  constraints.back()->setDescription("Last child & claimed -> SPARE fails");
347 
348  // Construct constraint for trying to claim next child
349  STORM_LOG_ASSERT(children.size() >= 2, "Spare has only one child");
350  for (uint64_t currChild = 0; currChild < children.size() - 1; ++currChild) {
351  uint64_t timeCurrChild = childVarIndices.at(currChild); // Moment when current child fails
352  // If i-th child fails after being claimed, then try to claim next child (constraint 6)
353  std::shared_ptr<SmtConstraint> tryClaimC = generateTryToClaimConstraint(spare, currChild + 1,
354  timeCurrChild);
355  constraints.push_back(std::make_shared<Iff>(
356  std::make_shared<IsLess>(getClaimVariableIndex(spare->id(), children.at(currChild)->id()),
357  timeCurrChild), tryClaimC));
358  constraints.back()->setDescription("Try to claim " + std::to_string(currChild + 2) + "th child");
359  }
360  }

362  std::shared_ptr<SmtConstraint>
363  DFTASFChecker::generateTryToClaimConstraint(std::shared_ptr<storm::storage::DFTSpare<ValueType> const> spare,
364  uint64_t childIndex, uint64_t timepoint) const {
365  auto child = spare->children().at(childIndex);
366  uint64_t timeChild = timePointVariables.at(child->id()); // Moment when the child fails
367  uint64_t claimChild = getClaimVariableIndex(spare->id(), child->id()); // Moment the spare claims the child
368 
369  std::vector<std::shared_ptr<SmtConstraint>> noClaimingPossible;
370  // Child cannot be claimed.
371  if (childIndex + 1 < spare->children().size()) {
372  // Consider next child for claiming (second case in constraint 7)
373  noClaimingPossible.push_back(generateTryToClaimConstraint(spare, childIndex + 1, timepoint));
374  } else {
375  // Last child: spare fails at same point as this child (third case in constraint 7)
376  noClaimingPossible.push_back(std::make_shared<IsEqual>(timePointVariables.at(spare->id()), timepoint));
377  }
378  std::shared_ptr<SmtConstraint> elseCaseC = std::make_shared<And>(noClaimingPossible);
379 
380  // Check if next child is available (first case in constraint 7)
381  std::vector<std::shared_ptr<SmtConstraint>> claimingPossibleC;
382  // Next child is not yet failed
383  claimingPossibleC.push_back(std::make_shared<IsLess>(timepoint, timeChild));
384  // Child is not yet claimed by a different spare
385  for (auto const &otherSpare : child->parents()) {
386  if (otherSpare->id() == spare->id()) {
387  // not a different spare.
388  continue;
389  }
390  claimingPossibleC.push_back(std::make_shared<IsLess>(timepoint,
391  getClaimVariableIndex(otherSpare->id(),
392  child->id())));
393  }
394 
395  // Claim child if available
396  std::shared_ptr<SmtConstraint> firstCaseC = std::make_shared<IfThenElse>(
397  std::make_shared<And>(claimingPossibleC), std::make_shared<IsEqual>(claimChild, timepoint),
398  elseCaseC);
399  return firstCaseC;
400  }

402  void DFTASFChecker::addClaimingConstraints() {
403  // Only one spare can claim a child (constraint 8)
404  // and only not failed children can be claimed (addition to constrain 8)
405  for (size_t i = 0; i < dft.nrElements(); ++i) {
406  std::shared_ptr<storm::storage::DFTElement<ValueType> const> element = dft.getElement(i);
407  if (element->isSpareGate()) {
408  auto spare = std::static_pointer_cast<storm::storage::DFTSpare<double> const>(element);
409  for (auto const &child : spare->children()) {
410  std::vector<std::shared_ptr<SmtConstraint>> additionalC;
411  uint64_t timeClaiming = getClaimVariableIndex(spare->id(), child->id());
412  std::shared_ptr<SmtConstraint> leftC = std::make_shared<IsLessConstant>(timeClaiming,
413  notFailed);
414  // Child must be operational at time of claiming
415  additionalC.push_back(
416  std::make_shared<IsLess>(timeClaiming, timePointVariables.at(child->id())));
417  // No other spare claims this child
418  for (auto const &parent : child->parents()) {
419  if (parent->isSpareGate() && parent->id() != spare->id()) {
420  // Different spare
421  additionalC.push_back(std::make_shared<IsConstantValue>(
422  getClaimVariableIndex(parent->id(), child->id()), notFailed));
423  }
424  }
425  constraints.push_back(std::make_shared<Implies>(leftC, std::make_shared<And>(additionalC)));
426  constraints.back()->setDescription(
427  "Child " + child->name() + " must be operational at time of claiming by spare " +
428  spare->name() + " and can only be claimed by one spare.");
429  }
430  }
431  }

451  void DFTASFChecker::addMarkovianConstraints() {
452  uint64_t nrMarkovian = dft.nrBasicElements();
453  std::set<size_t> depElements;
454  // Vector containing (non-)Markovian constraints for each timepoint
455  std::vector<std::vector<std::shared_ptr<SmtConstraint>>> markovianC(nrMarkovian);
456  std::vector<std::vector<std::shared_ptr<SmtConstraint>>> nonMarkovianC(nrMarkovian);
457  std::vector<std::vector<std::shared_ptr<SmtConstraint>>> notColdC(nrMarkovian);
458 
459  // All dependent events of a failed trigger have failed as well (constraint 9)
460  for (size_t j = 0; j < dft.nrElements(); ++j) {
461  std::shared_ptr<storm::storage::DFTElement<ValueType> const> element = dft.getElement(j);
462  if (element->hasOutgoingDependencies()) {
463  for (uint64_t i = 0; i < nrMarkovian; ++i) {
464  std::shared_ptr<SmtConstraint> triggerFailed = std::make_shared<IsLessEqualConstant>(
465  timePointVariables.at(j), i);
466  std::vector<std::shared_ptr<SmtConstraint>> depFailed;
467  for (auto const &dependency : element->outgoingDependencies()) {
468  for (auto const &depElement : dependency->dependentEvents()) {
469  depFailed.push_back(
470  std::make_shared<IsLessEqualConstant>(timePointVariables.at(depElement->id()),
471  i));
472  }
473  }
474  markovianC[i].push_back(
475  std::make_shared<Implies>(triggerFailed, std::make_shared<And>(depFailed)));
476  }
477  }
478  }
479  for (uint64_t i = 0; i < nrMarkovian; ++i) {
480  constraints.push_back(
481  std::make_shared<Iff>(std::make_shared<IsBoolValue>(markovianVariables.at(i), true),
482  std::make_shared<And>(markovianC[i])));
483  constraints.back()->setDescription("Markovian (" + std::to_string(i) +
484  ") iff all dependent events which trigger failed also failed.");
485  }
486 
487  // In non-Markovian steps the next failed element is a dependent BE (constraint 10) + additions to specification in paper
488  for (size_t j = 0; j < dft.nrElements(); ++j) {
489  std::shared_ptr<storm::storage::DFTElement<ValueType> const> element = dft.getElement(j);
490  if (element->isBasicElement()) {
491  auto be = std::static_pointer_cast<storm::storage::DFTBE<double> const>(element);
492 
493  if (be->hasIngoingDependencies()) {
494  depElements.emplace(j);
495  for (uint64_t i = 0; i < nrMarkovian - 1; ++i) {
496  std::shared_ptr<SmtConstraint> nextFailure = std::make_shared<IsConstantValue>(
497  timePointVariables.at(j), i + 1);
498  std::vector<std::shared_ptr<SmtConstraint>> triggerFailed;
499  for (auto const &dependency : be->ingoingDependencies()) {
500  triggerFailed.push_back(std::make_shared<IsLessEqualConstant>(
501  timePointVariables.at(dependency->triggerEvent()->id()), i));
502  }
503  nonMarkovianC[i].push_back(
504  std::make_shared<Implies>(nextFailure, std::make_shared<Or>(triggerFailed)));
505  }
506  }
507  }
508  }
509  for (uint64_t i = 0; i < nrMarkovian; ++i) {
510  std::vector<std::shared_ptr<SmtConstraint>> dependentConstr;
511  for (auto dependentEvent: depElements) {
512  std::shared_ptr<SmtConstraint> nextFailure = std::make_shared<IsConstantValue>(
513  timePointVariables.at(dependentEvent), i + 1);
514  dependentConstr.push_back(nextFailure);
515  }
516  // Add Constraint that any DEPENDENT event has to fail next
517  nonMarkovianC[i].push_back(std::make_shared<Or>(dependentConstr));
518  constraints.push_back(
519  std::make_shared<Implies>(std::make_shared<IsBoolValue>(markovianVariables.at(i), false),
520  std::make_shared<And>(nonMarkovianC[i])));
521  constraints.back()->setDescription(
522  "Non-Markovian (" + std::to_string(i) + ") -> next failure is dependent BE.");
523  }
524 
525  // In Markovian steps the failure rate is positive (constraint 11)
526  for (size_t j = 0; j < dft.nrElements(); ++j) {
527  std::shared_ptr<storm::storage::DFTElement<ValueType> const> element = dft.getElement(j);
528  if (element->isBasicElement()) {
529  auto be = std::static_pointer_cast<storm::storage::DFTBE<double> const>(element);
530  for (uint64_t i = 0; i < nrMarkovian; ++i) {
531  std::shared_ptr<SmtConstraint> nextFailure = std::make_shared<IsConstantValue>(
532  timePointVariables.at(j), i + 1);
533  // BE is not cold
534  // TODO: implement use of activation variables here
535  notColdC[i].push_back(std::make_shared<Implies>(nextFailure, std::make_shared<IsTrue>(be->canFail())));
536  }
537  }
538  }
539  for (uint64_t i = 0; i < nrMarkovian; ++i) {
540  constraints.push_back(
541  std::make_shared<Implies>(std::make_shared<IsBoolValue>(markovianVariables.at(i), true),
542  std::make_shared<And>(notColdC[i])));
543  constraints.back()->setDescription("Markovian (" + std::to_string(i) + ") -> positive failure rate.");
544  }

710  storm::solver::SmtSolver::CheckResult
711  DFTASFChecker::checkDependencyConflict(uint64_t dep1Index, uint64_t dep2Index, uint64_t timeout) {
712  std::vector<std::shared_ptr<SmtConstraint>> andConstr;
713  std::vector<std::shared_ptr<SmtConstraint>> orConstr;
714  STORM_LOG_DEBUG(
715  "Check " << dft.getElement(dep1Index)->name() << " and " << dft.getElement(dep2Index)->name());
716  andConstr.clear();
717  // AND FDEP1 is triggered before FDEP2 is resolved
718  andConstr.push_back(std::make_shared<IsGreaterEqual>(
719  timePointVariables.at(dep1Index), timePointVariables.at(dep2Index)));
720  andConstr.push_back(std::make_shared<IsLess>(
721  timePointVariables.at(dep1Index), dependencyVariables.at(dep2Index)));
722  std::shared_ptr<SmtConstraint> betweenConstr1 = std::make_shared<And>(andConstr);
723 
724  andConstr.clear();
725  // AND FDEP2 is triggered before FDEP1 is resolved
726  andConstr.push_back(std::make_shared<IsGreaterEqual>(
727  timePointVariables.at(dep2Index), timePointVariables.at(dep1Index)));
728  andConstr.push_back(std::make_shared<IsLess>(
729  timePointVariables.at(dep2Index), dependencyVariables.at(dep1Index)));
730  std::shared_ptr<SmtConstraint> betweenConstr2 = std::make_shared<And>(andConstr);
731 
732  orConstr.clear();
733  // Either one of the above constraints holds
734  orConstr.push_back(betweenConstr1);
735  orConstr.push_back(betweenConstr2);
736 
737  // Both FDEPs were triggered before dependent elements have failed
738  andConstr.clear();
739  andConstr.push_back(std::make_shared<IsLess>(
740  timePointVariables.at(dep1Index), dependencyVariables.at(dep1Index)));
741  andConstr.push_back(std::make_shared<IsLess>(
742  timePointVariables.at(dep2Index), dependencyVariables.at(dep2Index)));
743  andConstr.push_back(std::make_shared<Or>(orConstr));
744 
745  std::shared_ptr<SmtConstraint> checkConstr = std::make_shared<And>(andConstr);
746 
747  std::shared_ptr<storm::expressions::ExpressionManager> manager = solver->getManager().getSharedPointer();
748  solver->push();
749  solver->add(checkConstr->toExpression(varNames, manager));
750  setSolverTimeout(timeout * 1000);
751  storm::solver::SmtSolver::CheckResult res = solver->check();
752  unsetSolverTimeout();
753  solver->pop();
754  return res;
755  }
