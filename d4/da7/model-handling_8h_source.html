218  ModelProcessingInformation getModelProcessingInformation(SymbolicInput const& input, std::shared_ptr<SymbolicInput> const& transformedJaniInput = nullptr) {
219  ModelProcessingInformation mpi;
220  auto ioSettings = storm::settings::getModule<storm::settings::modules::IOSettings>();
221  auto coreSettings = storm::settings::getModule<storm::settings::modules::CoreSettings>();
222  auto generalSettings = storm::settings::getModule<storm::settings::modules::GeneralSettings>();
223  auto bisimulationSettings = storm::settings::getModule<storm::settings::modules::BisimulationSettings>();
224 
225  // Set the engine.
226  mpi.engine = coreSettings.getEngine();
227 
228  // Set whether bisimulation is to be used.
229  mpi.applyBisimulation = generalSettings.isBisimulationSet();
230 
231  // Set the value type used for numeric values
232  if (generalSettings.isParametricSet()) {
233  mpi.verificationValueType = ModelProcessingInformation::ValueType::Parametric;
234  } else if (generalSettings.isExactSet()) {
235  mpi.verificationValueType = ModelProcessingInformation::ValueType::Exact;
236  } else {
237  mpi.verificationValueType = ModelProcessingInformation::ValueType::FinitePrecision;
238  }
239  auto originalVerificationValueType = mpi.verificationValueType;
240 
241  // Since the remaining settings could depend on the ones above, we need apply the automatic engine now.
242  bool useAutomatic = input.model.is_initialized() && mpi.engine == storm::utility::Engine::Automatic;
243  if (useAutomatic) {
244  if (input.model->isJaniModel()) {
245  // This can potentially overwrite the settings above, but will not overwrite settings that were explicitly set by the user (e.g. we will not disable bisimulation or disable exact arithmetic)
246  getModelProcessingInformationAutomatic(input, mpi);
247  } else {
248  // Transform Prism to jani first
249  STORM_LOG_ASSERT(input.model->isPrismProgram(), "Unexpected type of input.");
250  SymbolicInput janiInput;
251  janiInput.properties = input.properties;
252  storm::prism::Program const& prog = input.model.get().asPrismProgram();
253  auto modelAndProperties = prog.toJani(input.preprocessedProperties.is_initialized() ? input.preprocessedProperties.get() : input.properties);
254  janiInput.model = modelAndProperties.first;
255  if (!modelAndProperties.second.empty()) {
256  janiInput.preprocessedProperties = std::move(modelAndProperties.second);
257  }
258  // This can potentially overwrite the settings above, but will not overwrite settings that were explicitly set by the user (e.g. we will not disable bisimulation or disable exact arithmetic)
259  getModelProcessingInformationAutomatic(janiInput, mpi);
260  if (transformedJaniInput) {
261  // We cache the transformation result.
262  *transformedJaniInput = std::move(janiInput);
263  }
264  }
265  }
266 
267  // Check whether these settings are compatible with the provided input.
268  if (input.model) {
269  auto checkCompatibleSettings = [&mpi, &input] {
270  switch (mpi.verificationValueType) {
271  case ModelProcessingInformation::ValueType::Parametric:
272  return storm::utility::canHandle<storm::RationalFunction>(mpi.engine, input.preprocessedProperties.is_initialized() ? input.preprocessedProperties.get() : input.properties, input.model.get());
273  case ModelProcessingInformation::ValueType::Exact:
274  return storm::utility::canHandle<storm::RationalNumber>(mpi.engine, input.preprocessedProperties.is_initialized() ? input.preprocessedProperties.get() : input.properties, input.model.get());
275  break;
276  case ModelProcessingInformation::ValueType::FinitePrecision:
277  return storm::utility::canHandle<double>(mpi.engine, input.preprocessedProperties.is_initialized() ? input.preprocessedProperties.get() : input.properties, input.model.get());
278  }
279  return false;
280  };
281  mpi.isCompatible = checkCompatibleSettings();
282  if (!mpi.isCompatible) {
283  if (useAutomatic) {
284  bool useExact = mpi.verificationValueType != ModelProcessingInformation::ValueType::FinitePrecision;
285  STORM_LOG_WARN("The settings picked by the automatic engine (engine=" << mpi.engine << ", bisim=" << mpi.applyBisimulation << ", exact=" << useExact << ") are incompatible with this model. Falling back to default settings.");
286  mpi.engine = storm::utility::Engine::Sparse;
287  mpi.applyBisimulation = false;
288  mpi.verificationValueType = originalVerificationValueType;
289  // Retry check with new settings
290  mpi.isCompatible = checkCompatibleSettings();
291  }
292  }
293  } else {
294  // If there is no input model, nothing has to be done, actually
295  mpi.isCompatible = true;
296  }
297 
298  // Set whether a transformation to jani is required or necessary
299  mpi.transformToJani = ioSettings.isPrismToJaniSet();
300  if (input.model) {
301  auto builderType = storm::utility::getBuilderType(mpi.engine);
302  bool transformToJaniForJit = builderType == storm::builder::BuilderType::Jit;
303  STORM_LOG_WARN_COND(mpi.transformToJani || !transformToJaniForJit, "The JIT-based model builder is only available for JANI models, automatically converting the PRISM input model.");
304  bool transformToJaniForDdMA = (builderType == storm::builder::BuilderType::Dd) && (input.model->getModelType() == storm::storage::SymbolicModelDescription::ModelType::MA) && (!input.model->isJaniModel());
305  STORM_LOG_WARN_COND(mpi.transformToJani || !transformToJaniForDdMA, "Dd-based model builder for Markov Automata is only available for JANI models, automatically converting the input model.");
306  mpi.transformToJani |= (transformToJaniForJit || transformToJaniForDdMA);
307  }
308 
309  // Set the Valuetype used during model building
310  mpi.buildValueType = mpi.verificationValueType;
311  if (bisimulationSettings.useExactArithmeticInDdBisimulation()) {
312  if (storm::utility::getBuilderType(mpi.engine) == storm::builder::BuilderType::Dd && mpi.applyBisimulation) {
313  if (mpi.buildValueType == ModelProcessingInformation::ValueType::FinitePrecision) {
314  mpi.buildValueType = ModelProcessingInformation::ValueType::Exact;
315  }
316  } else {
317  STORM_LOG_WARN("Requested using exact arithmetic in Dd bisimulation but no dd bisimulation is applied.");
318  }
319  }
320 
321  // Set the Dd library
322  mpi.ddType = coreSettings.getDdLibraryType();
323  if (mpi.ddType == storm::dd::DdType::CUDD && coreSettings.isDdLibraryTypeSetFromDefaultValue()) {
324  if (!(mpi.buildValueType == ModelProcessingInformation::ValueType::FinitePrecision && mpi.verificationValueType == ModelProcessingInformation::ValueType::FinitePrecision)) {
325  STORM_LOG_INFO("Switching to DD library sylvan to allow for rational arithmetic.");
326  mpi.ddType = storm::dd::DdType::Sylvan;
327  }
328  }
329  return mpi;
330  }

370 
371 
372  // Check whether conversion for PRISM to JANI is requested or necessary.
373  if (output.model && output.model.get().isPrismProgram()) {
374  if (mpi.transformToJani) {
375  if (transformedJani->model) {
376  // Use the cached transformation if possible
377  output = std::move(*transformedJani);
378  } else {
379  storm::prism::Program const& model = output.model.get().asPrismProgram();
380  auto modelAndProperties = model.toJani(output.properties);
381 
382  output.model = modelAndProperties.first;
383

396  return {output, mpi};

928  std::vector<std::vector<storm::expressions::Expression>> parseInjectedRefinementPredicates(storm::expressions::ExpressionManager const& expressionManager, std::string const& refinementPredicatesString) {
929  std::vector<std::vector<storm::expressions::Expression>> injectedRefinementPredicates;
930 
931  storm::parser::ExpressionParser expressionParser(expressionManager);
932  std::unordered_map<std::string, storm::expressions::Expression> variableMapping;
933  for (auto const& variableTypePair : expressionManager) {
934  variableMapping[variableTypePair.first.getName()] = variableTypePair.first;
935  }
936  expressionParser.setIdentifierMapping(variableMapping);
937 
938  std::vector<std::string> predicateGroupsAsStrings;
939  boost::split(predicateGroupsAsStrings, refinementPredicatesString, boost::is_any_of(";"));
940 
941  if (!predicateGroupsAsStrings.empty()) {
942  for (auto const& predicateGroupString : predicateGroupsAsStrings) {
943  if (predicateGroupString.empty()) {
944  continue;
945  }
946 
947  std::vector<std::string> predicatesAsStrings;
948  boost::split(predicatesAsStrings, predicateGroupString, boost::is_any_of(":"));
949 
950  if (!predicatesAsStrings.empty()) {
951  injectedRefinementPredicates.emplace_back();
952  for (auto const& predicateString : predicatesAsStrings) {
953  storm::expressions::Expression predicate = expressionParser.parseFromString(predicateString);
954  STORM_LOG_TRACE("Adding special (user-provided) refinement predicate " << predicateString << ".");
955  injectedRefinementPredicates.back().emplace_back(predicate);
956  }
957
