101  void TemplateEdge::liftTransientDestinationAssignments(int64_t maxLevel) {
102  if (!destinations.empty()) {
103  auto const& destination = *destinations.begin();
104 
105  std::vector<std::shared_ptr<Assignment>> assignmentsToLift;
106 
107  for (auto const& assignment : destination.getOrderedAssignments().getTransientAssignments()) {
108  // Check if we can lift the assignment to the edge.
109  bool canBeLifted = assignment.getLevel() <= maxLevel;
110  if (canBeLifted) {
111  for (auto const& destination : destinations) {
112  if (!destination.hasAssignment(assignment)) {
113  canBeLifted = false;
114  break;
115  }
116  }
117  }
118 
119  if (canBeLifted) {
120  // Do not remove the assignment now, as we currently iterate over them.
121  // Also we need to make a copy of the assignment since we are about to delete it
122  assignmentsToLift.push_back(std::make_shared<Assignment>(assignment));
123  }
124  }
125 
126  // now actually lift the assignments
127  for (auto const& assignment : assignmentsToLift) {
128  this->addTransientAssignment(*assignment);
129  for (auto& destination : destinations) {
130  destination.removeAssignment(*assignment);
131  }
132  }
133  }
134  }
