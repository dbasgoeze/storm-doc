140  bool TopologicalMinMaxLinearEquationSolver<ValueType>::solveTrivialScc(uint64_t const& sccState, OptimizationDirection dir, std::vector<ValueType>& globalX, std::vector<ValueType> const& globalB) const {
141  ValueType& xi = globalX[sccState];
142  bool firstRow = true;
143  uint64_t bestRow;
144 
145  for (uint64_t row = this->A->getRowGroupIndices()[sccState]; row < this->A->getRowGroupIndices()[sccState + 1]; ++row) {
146  ValueType rowValue = globalB[row];
147  bool hasDiagonalEntry = false;
148  ValueType denominator;
149  for (auto const& entry : this->A->getRow(row)) {
150  if (entry.getColumn() == sccState) {
151  hasDiagonalEntry = true;
152  denominator = storm::utility::one<ValueType>() - entry.getValue();
153  } else {
154  rowValue += entry.getValue() * globalX[entry.getColumn()];
155  }
156  }
157  if (hasDiagonalEntry) {
158  STORM_LOG_WARN_COND_DEBUG(storm::NumberTraits<ValueType>::IsExact || !storm::utility::isAlmostZero(denominator) || storm::utility::isZero(denominator), "State " << sccState << " has a selfloop with probability '1-(" << denominator << ")'. This could be an indication for numerical issues.");
159  if (storm::utility::isZero(denominator)) {
160  // In this case we have a selfloop on this state. This can never an optimal choice:
161  // When minimizing, we are looking for the largest fixpoint (which will never be attained by this action)
162  // When maximizing, this choice reflects probability zero (non-optimal) or reward infinity (should already be handled during preprocessing).
163  continue;
164  } else {
165  rowValue /= denominator;
166  }
167  }
168  if (firstRow) {
169  xi = std::move(rowValue);
170  bestRow = row;
171  firstRow = false;
172  } else {
173  if (minimize(dir)) {
174  if (rowValue < xi) {
175  xi = std::move(rowValue);
176  bestRow = row;
177  }
178  } else {
179  if (rowValue > xi) {
180  xi = std::move(rowValue);
181  bestRow = row;
182  }
183  }
184  }
185  }
186  if (this->isTrackSchedulerSet()) {
187  this->schedulerChoices.get()[sccState] = bestRow - this->A->getRowGroupIndices()[sccState];
188  }
189  STORM_LOG_THROW(!firstRow, storm::exceptions::UnexpectedException, "Empty row group in MinMax equation system.");
190  //std::cout << "Solved trivial scc " << sccState << " with result " << globalX[sccState] << std::endl;
191  return true;
192  }
