28  if(!propositionalChecker.canHandle(formula.getSubformula().asUntilFormula().getLeftSubformula()) || !propositionalChecker.canHandle(formula.getSubformula().asUntilFormula().getRightSubformula())) {
29  STORM_LOG_DEBUG("Can not simplify when Until-formula has non-propositional subformula(s). Formula: " << formula);
30  return false;
31  }

34  std::pair<storm::storage::BitVector, storm::storage::BitVector> statesWithProbability01 = storm::utility::graph::performProb01(this->originalModel, phiStates, psiStates);
35  // Only consider the maybestates that are reachable from one initial state without hopping over a target (i.e., prob1) state
36  storm::storage::BitVector reachableGreater0States = storm::utility::graph::getReachableStates(this->originalModel.getTransitionMatrix(), this->originalModel.getInitialStates() & ~statesWithProbability01.first, ~statesWithProbability01.first, statesWithProbability01.second);
37  storm::storage::BitVector maybeStates = reachableGreater0States & ~statesWithProbability01.second;
38

41  typename storm::transformer::GoalStateMerger<SparseModelType>::ReturnType mergerResult = goalStateMerger.mergeTargetAndSinkStates(maybeStates, statesWithProbability01.second, statesWithProbability01.first);
42  this->simplifiedModel = mergerResult.model;
43  statesWithProbability01.second = storm::storage::BitVector(this->simplifiedModel->getNumberOfStates(), false);
44  if (mergerResult.targetState) {
45  statesWithProbability01.second.set(mergerResult.targetState.get(), true);
46  }
47  std::string targetLabel = "target";

49  targetLabel = "_" + targetLabel;

56  this->simplifiedFormula = std::make_shared<storm::logic::ProbabilityOperatorFormula const>(eventuallyFormula, formula.getOperatorInformation());
57

62  }
63  if (mergerResult.sinkState) {
64  considerForElimination.set(*mergerResult.sinkState, false);
65  }
