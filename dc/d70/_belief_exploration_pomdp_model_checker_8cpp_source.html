97  auto initialPomdpValueBounds = TrivialPomdpValueBoundsModelChecker<storm::models::sparse::Pomdp<ValueType>>(pomdp()).getValueBounds(formula, formulaInfo);
98  uint64_t initialPomdpState = pomdp().getInitialStates().getNextSetIndex(0);
99  Result result(initialPomdpValueBounds.getHighestLowerBound(initialPomdpState), initialPomdpValueBounds.getSmallestUpperBound(initialPomdpState));
100  STORM_LOG_INFO("Initial value bounds are [" << result.lowerBound << ", " << result.upperBound << "]");

102  boost::optional<std::string> rewardModelName;
103  std::set<uint32_t> targetObservations;

107  } else {

109  std::tie(preprocessedPomdp, targetObservations) = obsCloser.transform(formulaInfo.getTargetStates().states);
110  }
111  // FIXME: Instead of giving up, make sink states absorbing.
112  if (formulaInfo.isNonNestedReachabilityProbability()) {
113  if (!formulaInfo.getSinkStates().empty()) {

115  reachableFromSinkStates &= ~formulaInfo.getSinkStates().states;
116  STORM_LOG_THROW(reachableFromSinkStates.empty(), storm::exceptions::NotSupportedException, "There are sink states that can reach non-sink states. This is currently not supported");
117  }
118  } else {
119  // Expected reward formula!
120  rewardModelName = formulaInfo.getRewardModelName();
121  }
122  } else {
123  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Unsupported formula '" << formula << "'.");
124  }
125  if (storm::pomdp::detectFiniteBeliefMdp(pomdp(), formulaInfo.getTargetStates().states)) {
126  STORM_LOG_INFO("Detected that the belief MDP is finite.");
127  statistics.beliefMdpDetectedToBeFinite = true;
128  }
129 
130  if (options.refine) {
131  refineReachability(targetObservations, formulaInfo.minimize(), rewardModelName, initialPomdpValueBounds, result);
132  } else {
133  computeReachabilityOTF(targetObservations, formulaInfo.minimize(), rewardModelName, initialPomdpValueBounds, result);
134  }
135  // "clear" results in case they were actually not requested (this will make the output a bit more clear)
136  if ((formulaInfo.minimize() && !options.discretize) || (formulaInfo.maximize() && !options.unfold)) {
137  result.lowerBound = -storm::utility::infinity<ValueType>();
138  }
139  if ((formulaInfo.maximize() && !options.discretize) || (formulaInfo.minimize() && !options.unfold)) {
140  result.upperBound = storm::utility::infinity<ValueType>();
141  }
142 
143  if (storm::utility::resources::isTerminate()) {
144  statistics.aborted = true;
145  }
146  statistics.totalTime.stop();
147  return result;
148  }

437  BeliefValueType BeliefExplorationPomdpModelChecker<PomdpModelType, BeliefValueType>::rateObservation(typename ExplorerType::SuccessorObservationInformation const& info, BeliefValueType const& observationResolution, BeliefValueType const& maxResolution) {

439  auto one = storm::utility::one<BeliefValueType>();
440  if (storm::utility::isOne(n)) {
441  // If the belief is Dirac, it has to be approximated precisely.
442  // In this case, we return the best possible rating
443  return one;
444  } else {
445  // Create the rating for this observation at this choice from the given info
446  BeliefValueType obsChoiceRating = storm::utility::convertNumber<BeliefValueType, ValueType>(info.maxProbabilityToSuccessorWithObs / info.observationProbability);
447  // At this point, obsRating is the largest triangulation weight (which ranges from 1/n to 1
448  // Normalize the rating so that it ranges from 0 to 1, where
449  // 0 means that the actual belief lies in the middle of the triangulating simplex (i.e. a "bad" approximation) and 1 means that the belief is precisely approximated.
450  obsChoiceRating = (obsChoiceRating * n - one) / (n - one);
451  // Scale the ratings with the resolutions, so that low resolutions get a lower rating (and are thus more likely to be refined)
452  obsChoiceRating *= observationResolution / maxResolution;
453  return obsChoiceRating;
454  }

458  std::vector<BeliefValueType> BeliefExplorationPomdpModelChecker<PomdpModelType, BeliefValueType>::getObservationRatings(std::shared_ptr<ExplorerType> const& overApproximation, std::vector<BeliefValueType> const& observationResolutionVector) {
459  uint64_t numMdpStates = overApproximation->getExploredMdp()->getNumberOfStates();
460  auto const& choiceIndices = overApproximation->getExploredMdp()->getNondeterministicChoiceIndices();
461  BeliefValueType maxResolution = *std::max_element(observationResolutionVector.begin(), observationResolutionVector.end());
462 
463  std::vector<BeliefValueType> resultingRatings(pomdp().getNrObservations(), storm::utility::one<BeliefValueType>());
464 
465  std::map<uint32_t, typename ExplorerType::SuccessorObservationInformation> gatheredSuccessorObservations; // Declare here to avoid reallocations
466  for (uint64_t mdpState = 0; mdpState < numMdpStates; ++mdpState) {
467  // Check whether this state is reached under an optimal scheduler.
468  // The heuristic assumes that the remaining states are not relevant for the observation score.
469  if (overApproximation->stateIsOptimalSchedulerReachable(mdpState)) {
470  for (uint64_t mdpChoice = choiceIndices[mdpState]; mdpChoice < choiceIndices[mdpState + 1]; ++mdpChoice) {
471  // Similarly, only optimal actions are relevant
472  if (overApproximation->actionIsOptimal(mdpChoice)) {
473  // score the observations for this choice
474  gatheredSuccessorObservations.clear();
475  overApproximation->gatherSuccessorObservationInformationAtMdpChoice(mdpChoice, gatheredSuccessorObservations);
476  for (auto const& obsInfo : gatheredSuccessorObservations) {
477  auto const& obs = obsInfo.first;
478  BeliefValueType obsChoiceRating = rateObservation(obsInfo.second, observationResolutionVector[obs], maxResolution);
479 
480  // The rating of the observation will be the minimum over all choice-based observation ratings
481  resultingRatings[obs] = std::min(resultingRatings[obs], obsChoiceRating);
482  }
483  }
484  }
485  }
486  }

490  template<typename ValueType>
491  ValueType getGap(ValueType const& l, ValueType const& u) {
492  STORM_LOG_ASSERT(l >= storm::utility::zero<ValueType>() && u >= storm::utility::zero<ValueType>(), "Gap computation currently does not handle negative values.");
493  if (storm::utility::isInfinity(u)) {
494  if (storm::utility::isInfinity(l)) {
495  return storm::utility::zero<ValueType>();
496  } else {
497  return u;
498  }
499  } else if (storm::utility::isZero(u)) {
500  STORM_LOG_ASSERT(storm::utility::isZero(l), "Upper bound is zero but lower bound is " << l << ".");
501  return u;
502  } else {
503  STORM_LOG_ASSERT(!storm::utility::isInfinity(l), "Lower bound is infinity, but upper bound is " << u << ".");
504  // get the relative gap
505  return storm::utility::abs<ValueType>(u-l) * storm::utility::convertNumber<ValueType, uint64_t>(2) / (l+u);
506  }

509  template<typename PomdpModelType, typename BeliefValueType>
510  bool BeliefExplorationPomdpModelChecker<PomdpModelType, BeliefValueType>::buildOverApproximation(std::set<uint32_t> const &targetObservations, bool min, bool computeRewards, bool refine, HeuristicParameters const& heuristicParameters, std::vector<BeliefValueType>& observationResolutionVector, std::shared_ptr<BeliefManagerType>& beliefManager, std::shared_ptr<ExplorerType>& overApproximation) {

521  } else {

531  BeliefValueType numericPresicion = storm::utility::convertNumber<BeliefValueType>(options.numericPrecision);
532  if (std::any_of(obsRatings.begin(), obsRatings.end(), [&numericPresicion](ValueType const& value){return value + numericPresicion < storm::utility::one<ValueType>();})) {

535  }

537  STORM_LOG_DEBUG("Refining the resolution of " << refinedObservations.getNumberOfSetBits() << "/" << refinedObservations.size() << " observations.");
538  for (auto const& obs : refinedObservations) {
539  // Increment the resolution at the refined observations.
540  // Use storm's rational number to detect overflows properly.
541  storm::RationalNumber newObsResolutionAsRational = storm::utility::convertNumber<storm::RationalNumber>(observationResolutionVector[obs]) * storm::utility::convertNumber<storm::RationalNumber>(options.resolutionFactor);
542  static_assert(storm::NumberTraits<BeliefValueType>::IsExact || std::is_same<BeliefValueType, double>::value, "Unhandled belief value type");

544  observationResolutionVector[obs] = storm::utility::convertNumber<BeliefValueType>(std::numeric_limits<double>::max());
545  } else {
546  observationResolutionVector[obs] = storm::utility::convertNumber<BeliefValueType>(newObsResolutionAsRational);
547  }
548  }
549  overApproximation->restartExploration();
550  }
551  statistics.overApproximationMaxResolution = storm::utility::ceil(*std::max_element(observationResolutionVector.begin(), observationResolutionVector.end()));
552 
553  // Start exploration
554  storm::utility::Stopwatch explorationTime;
555  if (options.explorationTimeLimit) {
556  explorationTime.start();
557  }
558  bool timeLimitExceeded = false;
559  std::map<uint32_t, typename ExplorerType::SuccessorObservationInformation> gatheredSuccessorObservations; // Declare here to avoid reallocations
560  uint64_t numRewiredOrExploredStates = 0;
561  while (overApproximation->hasUnexploredState()) {
562  if (!timeLimitExceeded && options.explorationTimeLimit && static_cast<uint64_t>(explorationTime.getTimeInSeconds()) > options.explorationTimeLimit.get()) {
563  STORM_LOG_INFO("Exploration time limit exceeded.");
564  timeLimitExceeded = true;
565  STORM_LOG_INFO_COND(!fixPoint, "Not reaching a refinement fixpoint because the exploration time limit is exceeded.");
566  fixPoint = false;
567  }
568 
569  uint64_t currId = overApproximation->exploreNextState();
570  bool hasOldBehavior = refine && overApproximation->currentStateHasOldBehavior();
571  if (!hasOldBehavior) {
572  STORM_LOG_INFO_COND(!fixPoint, "Not reaching a refinement fixpoint because a new state is explored");
573  fixPoint = false; // Exploring a new state!
574  }
575  uint32_t currObservation = beliefManager->getBeliefObservation(currId);
576  if (targetObservations.count(currObservation) != 0) {
577  overApproximation->setCurrentStateIsTarget();
578  overApproximation->addSelfloopTransition();
579  } else {
580  // We need to decide how to treat this state (and each individual enabled action). There are the following cases:
581  // 1 The state has no old behavior and
582  // 1.1 we explore all actions or
583  // 1.2 we truncate all actions
584  // 2 The state has old behavior and was truncated in the last iteration and
585  // 2.1 we explore all actions or
586  // 2.2 we truncate all actions (essentially restoring old behavior, but we do the truncation step again to benefit from updated bounds)
587  // 3 The state has old behavior and was not truncated in the last iteration and the current action
588  // 3.1 should be rewired or
589  // 3.2 should get the old behavior but either
590  // 3.2.1 none of the successor observation has been refined since the last rewiring or exploration of this action
591  // 3.2.2 rewiring is only delayed as it could still have an effect in a later refinement step
592 
593  // Find out in which case we are
594  bool exploreAllActions = false;
595  bool truncateAllActions = false;
596  bool restoreAllActions = false;
597  bool checkRewireForAllActions = false;
598  // Get the relative gap
599  ValueType gap = getGap(overApproximation->getLowerValueBoundAtCurrentState(), overApproximation->getUpperValueBoundAtCurrentState());
600  if (!hasOldBehavior) {
601  // Case 1
602  // If we explore this state and if it has no old behavior, it is clear that an "old" optimal scheduler can be extended to a scheduler that reaches this state
603  if (!timeLimitExceeded && gap > heuristicParameters.gapThreshold && numRewiredOrExploredStates < heuristicParameters.sizeThreshold) {
604  exploreAllActions = true; // Case 1.1
605  } else {
606  truncateAllActions = true; // Case 1.2
607  overApproximation->setCurrentStateIsTruncated();
608  }
609  } else {
610  if (overApproximation->getCurrentStateWasTruncated()) {
611  // Case 2
612  if (!timeLimitExceeded && overApproximation->currentStateIsOptimalSchedulerReachable() && gap > heuristicParameters.gapThreshold && numRewiredOrExploredStates < heuristicParameters.sizeThreshold) {
613  exploreAllActions = true; // Case 2.1
614  STORM_LOG_INFO_COND(!fixPoint, "Not reaching a refinement fixpoint because a previously truncated state is now explored.");
615  fixPoint = false;
616  } else {
617  truncateAllActions = true; // Case 2.2
618  overApproximation->setCurrentStateIsTruncated();
619  if (fixPoint) {
620  // Properly check whether this can still be a fixpoint
621  if (overApproximation->currentStateIsOptimalSchedulerReachable() && !storm::utility::isZero(gap)) {
622  STORM_LOG_INFO_COND(!fixPoint, "Not reaching a refinement fixpoint because we truncate a state with non-zero gap " << gap << " that is reachable via an optimal sched.");
623  fixPoint = false;
624  }
625  //} else {
626  // In this case we truncated a state that is not reachable under optimal schedulers.
627  // If no other state is explored (i.e. fixPoint remaints true), these states should still not be reachable in subsequent iterations
628  }
629  }
630  } else {
631  // Case 3
632  // The decision for rewiring also depends on the corresponding action, but we have some criteria that lead to case 3.2 (independent of the action)
633  if (!timeLimitExceeded && overApproximation->currentStateIsOptimalSchedulerReachable() && gap > heuristicParameters.gapThreshold && numRewiredOrExploredStates < heuristicParameters.sizeThreshold) {
634  checkRewireForAllActions = true; // Case 3.1 or Case 3.2
635  } else {
636  restoreAllActions = true; // Definitely Case 3.2
637  // We still need to check for each action whether rewiring makes sense later
638  checkRewireForAllActions = true;
639  }
640  }
641  }
642  bool expandedAtLeastOneAction = false;
643  for (uint64 action = 0, numActions = beliefManager->getBeliefNumberOfChoices(currId); action < numActions; ++action) {
644  bool expandCurrentAction = exploreAllActions || truncateAllActions;
645  if (checkRewireForAllActions) {
646  assert(refine);
647  // In this case, we still need to check whether this action needs to be expanded
648  assert(!expandCurrentAction);
649  // Check the action dependent conditions for rewiring
650  // First, check whether this action has been rewired since the last refinement of one of the successor observations (i.e. whether rewiring would actually change the successor states)
651  assert(overApproximation->currentStateHasOldBehavior());
652  if (overApproximation->getCurrentStateActionExplorationWasDelayed(action) || overApproximation->currentStateHasSuccessorObservationInObservationSet(action, refinedObservations)) {
653  // Then, check whether the other criteria for rewiring are satisfied
654  if (!restoreAllActions && overApproximation->actionAtCurrentStateWasOptimal(action)) {
655  // Do the rewiring now! (Case 3.1)
656  expandCurrentAction = true;
657  STORM_LOG_INFO_COND(!fixPoint, "Not reaching a refinement fixpoint because we rewire a state.");
658  fixPoint = false;
659  } else {
660  // Delay the rewiring (Case 3.2.2)
661  overApproximation->setCurrentChoiceIsDelayed(action);
662  if (fixPoint) {
663  // Check whether this delay means that a fixpoint has not been reached
664  if (!overApproximation->getCurrentStateActionExplorationWasDelayed(action) || (overApproximation->currentStateIsOptimalSchedulerReachable() && overApproximation->actionAtCurrentStateWasOptimal(action) && !storm::utility::isZero(gap))) {
665  STORM_LOG_INFO_COND(!fixPoint, "Not reaching a refinement fixpoint because we delay a rewiring of a state with non-zero gap " << gap << " that is reachable via an optimal scheduler.");
666  fixPoint = false;
667  }
668  }
669  }
670  } // else { Case 3.2.1 }
671  }
672 
673  if (expandCurrentAction) {
674  expandedAtLeastOneAction = true;
675  if (!truncateAllActions) {
676  // Cases 1.1, 2.1, or 3.1
677  auto successorGridPoints = beliefManager->expandAndTriangulate(currId, action, observationResolutionVector);
678  for (auto const& successor : successorGridPoints) {
679  overApproximation->addTransitionToBelief(action, successor.first, successor.second, false);
680  }
681  if (computeRewards) {
682  overApproximation->computeRewardAtCurrentState(action);
683  }
684  } else {
685  // Cases 1.2 or 2.2
686  ValueType truncationProbability = storm::utility::zero<ValueType>();
687  ValueType truncationValueBound = storm::utility::zero<ValueType>();
688  auto successorGridPoints = beliefManager->expandAndTriangulate(currId, action, observationResolutionVector);
689  for (auto const& successor : successorGridPoints) {
690  bool added = overApproximation->addTransitionToBelief(action, successor.first, successor.second, true);
691  if (!added) {
692  // We did not explore this successor state. Get a bound on the "missing" value
693  truncationProbability += successor.second;
694  truncationValueBound += successor.second * (min ? overApproximation->computeLowerValueBoundAtBelief(successor.first) : overApproximation->computeUpperValueBoundAtBelief(successor.first));
695  }
696  }
697  if (computeRewards) {
698  // The truncationValueBound will be added on top of the reward introduced by the current belief state.
699  overApproximation->addTransitionsToExtraStates(action, truncationProbability);
700  overApproximation->computeRewardAtCurrentState(action, truncationValueBound);
701  } else {
702  overApproximation->addTransitionsToExtraStates(action, truncationValueBound, truncationProbability - truncationValueBound);
703  }
704  }
705  } else {
706  // Case 3.2
707  overApproximation->restoreOldBehaviorAtCurrentState(action);
708  }
709  }
710  if (expandedAtLeastOneAction) {
711  ++numRewiredOrExploredStates;
712  }
713  }
714 
715  if (storm::utility::resources::isTerminate()) {
716  break;
717  }
718  }
719 
720  if (storm::utility::resources::isTerminate()) {
721  // don't overwrite statistics of a previous, successful computation
722  if (!statistics.overApproximationStates) {
723  statistics.overApproximationBuildAborted = true;
724  statistics.overApproximationStates = overApproximation->getCurrentNumberOfMdpStates();
725  }
726  statistics.overApproximationBuildTime.stop();
727  return false;
728  }
729 
730  overApproximation->finishExploration();
731  statistics.overApproximationBuildTime.stop();
732 
733  statistics.overApproximationCheckTime.start();
734  overApproximation->computeValuesOfExploredMdp(min ? storm::solver::OptimizationDirection::Minimize : storm::solver::OptimizationDirection::Maximize);
735  statistics.overApproximationCheckTime.stop();
736 
737  // don't overwrite statistics of a previous, successful computation
738  if (!storm::utility::resources::isTerminate() || !statistics.overApproximationStates) {
739  statistics.overApproximationStates = overApproximation->getExploredMdp()->getNumberOfStates();
740  }
741  return fixPoint;
742  }
