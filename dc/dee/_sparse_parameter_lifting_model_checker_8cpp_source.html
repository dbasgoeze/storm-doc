58

60 
61  // Check if we need to check the formula on one point to decide whether to show AllSat or AllViolated
62  if (hypothesis == RegionResultHypothesis::Unknown && result == RegionResult::Unknown) {
63  result = getInstantiationChecker().check(env, region.getCenterPoint())->asExplicitQualitativeCheckResult()[*this->parametricModel->getInitialStates().begin()] ? RegionResult::CenterSat : RegionResult::CenterViolated;
64  }

75  } else if (hypothesis == RegionResultHypothesis::AllViolated || result == RegionResult::ExistsViolated || result == RegionResult::CenterViolated) {

78  if (!this->check(env, region, parameterOptimizationDirection)->asExplicitQualitativeCheckResult()[*this->parametricModel->getInitialStates().begin()]) {

81  result = sampleVertices(env, region, result);
82  }

99

101  auto vertices = region.getVerticesOfRegion(region.getVariables());
102  auto vertexIt = vertices.begin();
103  while (vertexIt != vertices.end() && !(hasSatPoint && hasViolatedPoint)) {
104  if (getInstantiationChecker().check(env, *vertexIt)->asExplicitQualitativeCheckResult()[*this->parametricModel->getInitialStates().begin()]) {
105  hasSatPoint = true;
106  } else {
107  hasViolatedPoint = true;
108  }
109  ++vertexIt;
110  }

113  if (hasViolatedPoint) {
114  result = RegionResult::ExistsBoth;
115  } else if (result != RegionResult::CenterSat) {
116  result = RegionResult::ExistsSat;
117  }
118  } else if (hasViolatedPoint && result != RegionResult::CenterViolated) {

157  std::pair<typename SparseModelType::ValueType, typename storm::storage::ParameterRegion<typename SparseModelType::ValueType>::Valuation> SparseParameterLiftingModelChecker<SparseModelType, ConstantType>::computeExtremalValue(Environment const& env, storm::storage::ParameterRegion<typename SparseModelType::ValueType> const& region, storm::solver::OptimizationDirection const& dir, typename SparseModelType::ValueType const& precision) {
158  STORM_LOG_THROW(this->parametricModel->getInitialStates().getNumberOfSetBits() == 1, storm::exceptions::NotSupportedException, "Getting extremal values at the initial state requires a model with a single initial state.");
159  boost::optional<ConstantType> value;
160  typename storm::storage::ParameterRegion<typename SparseModelType::ValueType>::Valuation valuation;
161 
162  for (auto const& v : region.getVerticesOfRegion(region.getVariables())) {
163  auto currValue = getInstantiationChecker().check(env, v)->template asExplicitQuantitativeCheckResult<ConstantType>()[*this->parametricModel->getInitialStates().begin()];
164  if (!value.is_initialized() || (storm::solver::minimize(dir) ? currValue < value.get() : currValue > value.get())) {
165  value = currValue;
166  valuation = v;
167  STORM_LOG_INFO("Current value for extremum: " << value.get() << ".");
168  }
169  }
170 
171  auto cmp = storm::solver::minimize(dir) ?
172  [](RegionBound<SparseModelType, ConstantType> const& lhs, RegionBound<SparseModelType, ConstantType> const& rhs) { return lhs.bound > rhs.bound; } :
173  [](RegionBound<SparseModelType, ConstantType> const& lhs, RegionBound<SparseModelType, ConstantType> const& rhs) { return lhs.bound < rhs.bound; };
174  std::priority_queue<RegionBound<SparseModelType, ConstantType>, std::vector<RegionBound<SparseModelType, ConstantType>>, decltype(cmp)> regionQueue(cmp);
175  regionQueue.emplace(region, storm::utility::zero<ConstantType>());
176  auto totalArea = storm::utility::convertNumber<ConstantType>(region.area());
177  auto coveredArea = storm::utility::zero<ConstantType>();
178  while (!regionQueue.empty()) {
179  auto const& currRegion = regionQueue.top().region;
180 
181  // Check whether this region contains a new 'good' value
182  auto currValue = getInstantiationChecker().check(env, currRegion.getCenterPoint())->template asExplicitQuantitativeCheckResult<ConstantType>()[*this->parametricModel->getInitialStates().begin()];
183  if (storm::solver::minimize(dir) ? currValue < value.get() : currValue > value.get()) {
184  value = currValue;
185  valuation = currRegion.getCenterPoint();
186  }
187 
188  // Check whether this region needs further investigation (i.e. splitting)
189  auto currBound = getBound(env, currRegion, dir)->template asExplicitQuantitativeCheckResult<ConstantType>()[*this->parametricModel->getInitialStates().begin()];
190  std::vector<storm::storage::ParameterRegion<typename SparseModelType::ValueType>> newRegions;
191  if (storm::solver::minimize(dir)) {
192  if (currBound < value.get() - storm::utility::convertNumber<ConstantType>(precision)) {
193  currRegion.split(currRegion.getCenterPoint(), newRegions);
194  }
195  } else {
196  if (currBound > value.get() + storm::utility::convertNumber<ConstantType>(precision)) {
197  currRegion.split(currRegion.getCenterPoint(), newRegions);
198  }
199  }
200 
201  if (newRegions.empty()) {
202  coveredArea += storm::utility::convertNumber<ConstantType>(currRegion.area());
203  }
204  regionQueue.pop();
205  for (auto const& r : newRegions) {
206  regionQueue.emplace(r, currBound);
207  }
208  STORM_LOG_INFO("Current value : " << value.get() << ", current bound: " << regionQueue.top().bound << ".");
209  STORM_LOG_INFO("Covered " << (coveredArea * storm::utility::convertNumber<ConstantType>(100.0) / totalArea) << "% of the region.");
210  }
211 
212  return std::make_pair(storm::utility::convertNumber<typename SparseModelType::ValueType>(value.get()), valuation);
213  }
