235  template<typename ModelType, typename BlockDataType>
236  void BisimulationDecomposition<ModelType, BlockDataType>::performPartitionRefinement() {
237  // Insert all blocks into the splitter queue as a (potential) splitter.
238  std::vector<Block<BlockDataType>*> splitterQueue;
239  std::for_each(partition.getBlocks().begin(), partition.getBlocks().end(), [&] (std::unique_ptr<Block<BlockDataType>> const& block) { block->data().setSplitter(); splitterQueue.push_back(block.get()); } );
240 
241  // Then perform the actual splitting until there are no more splitters.
242  uint_fast64_t iterations = 0;
243  while (!splitterQueue.empty()) {
244  ++iterations;
245 
246  // Get and prepare the next splitter.
247  // Sort the splitters according to their sizes to prefer small splitters. That is just a heuristic, but
248  // tends to work well.
249  std::sort(splitterQueue.begin(), splitterQueue.end(), [] (Block<BlockDataType> const* b1, Block<BlockDataType> const* b2) { return b1->getNumberOfStates() > b2->getNumberOfStates(); } );
250  Block<BlockDataType>* splitter = splitterQueue.back();
251  splitterQueue.pop_back();
252  splitter->data().setSplitter(false);
253 
254  // Now refine the partition using the current splitter.
255  refinePartitionBasedOnSplitter(*splitter, splitterQueue);
256 
257  if (storm::utility::resources::isTerminate()) {
258  std::cout << "Performed " << iterations << " iterations of partition refinement before abort." << std::endl;
259  STORM_LOG_THROW(false, storm::exceptions::AbortException, "Aborted in bisimulation computation.");
260  break;
261  }
262  }
