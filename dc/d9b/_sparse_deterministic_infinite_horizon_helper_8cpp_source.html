124  std::pair<ValueType, std::vector<ValueType>> SparseDeterministicInfiniteHorizonHelper<ValueType>::computeLraForBsccGainBias(Environment const& env, ValueGetter const& stateValuesGetter, ValueGetter const& actionValuesGetter, storm::storage::StronglyConnectedComponent const& bscc) {
125  // we want that the returned vector is sorted as the bscc. So let's assert that the bscc is sorted ascendingly.
126  STORM_LOG_ASSERT(std::is_sorted(bscc.begin(), bscc.end()), "Expected that bsccs are sorted.");
127 
128  // We build the equation system as in Line 3 of Algorithm 3 from
129  // Kretinsky, Meggendorfer: Efficient Strategy Iteration for Mean Payoff in Markov Decision Processes (ATVA 2017)
130  // https://doi.org/10.1007/978-3-319-68167-2_25
131  // The first variable corresponds to the gain of the bscc whereas the subsequent variables yield the bias for each state s_1, s_2, ....
132  // No bias variable for s_0 is needed since it is always set to zero, yielding an nxn equation system matrix
133  // To make this work for CTMC, we could uniformize the model. This preserves LRA and ensures that we can compute the
134  // LRA as for a DTMC (the soujourn time in each state is the same). If we then multiply the equations with the uniformization rate,
135  // the uniformization rate cancels out. Hence, we obtain the equation system below.
136 
137  // Get a mapping from global state indices to local ones.
138  std::unordered_map<uint64_t, uint64_t> toLocalIndexMap;
139  uint64_t localIndex = 0;
140  for (auto const& globalIndex : bscc) {
141  toLocalIndexMap[globalIndex] = localIndex;
142  ++localIndex;
143  }
144 
145  // Prepare an environment for the underlying equation solver
146  auto subEnv = env;
147  if (subEnv.solver().getLinearEquationSolverType() == storm::solver::EquationSolverType::Topological) {
148  // Topological solver does not make any sense since the BSCC is connected.
149  subEnv.solver().setLinearEquationSolverType(subEnv.solver().topological().getUnderlyingEquationSolverType(), subEnv.solver().topological().isUnderlyingEquationSolverTypeSetFromDefault());
150  }
151  subEnv.solver().setLinearEquationSolverPrecision(env.solver().lra().getPrecision(), env.solver().lra().getRelativeTerminationCriterion());
152 
153  // Build the equation system matrix and vector.
154  storm::solver::GeneralLinearEquationSolverFactory<ValueType> linearEquationSolverFactory;
155  bool isEquationSystemFormat = linearEquationSolverFactory.getEquationProblemFormat(subEnv) == storm::solver::LinearEquationSolverProblemFormat::EquationSystem;
156  storm::storage::SparseMatrixBuilder<ValueType> builder(bscc.size(), bscc.size());
157  std::vector<ValueType> eqSysVector;
158  eqSysVector.reserve(bscc.size());
159  // The first row asserts that the weighted bias variables and the reward at s_0 sum up to the gain
160  uint64_t row = 0;
161  ValueType entryValue;
162  for (auto const& globalState : bscc) {
163  ValueType rateAtState = this->_exitRates ? (*this->_exitRates)[globalState] : storm::utility::one<ValueType>();
164  // Coefficient for the gain variable
165  if (isEquationSystemFormat) {
166  // '1-0' in row 0 and -(-1) in other rows
167  builder.addNextValue(row, 0, storm::utility::one<ValueType>());
168  } else if (row > 0) {
169  // No coeficient in row 0, othwerise substract the gain
170  builder.addNextValue(row, 0, -storm::utility::one<ValueType>());
171  }
172  // Compute weighted sum over successor state. As this is a BSCC, each successor state will again be in the BSCC.
173  if (row > 0) {
174  if (isEquationSystemFormat) {
175  builder.addDiagonalEntry(row, rateAtState);
176  } else if (!storm::utility::isOne(rateAtState)) {
177  builder.addDiagonalEntry(row, storm::utility::one<ValueType>() - rateAtState);
178  }
179  }
180  for (auto const& entry : this->_transitionMatrix.getRow(globalState)) {
181  uint64_t col = toLocalIndexMap[entry.getColumn()];
182  if (col == 0) {
183  //Skip transition to state_0. This corresponds to setting the bias of state_0 to zero
184  continue;
185  }
186  entryValue = entry.getValue() * rateAtState;
187  if (isEquationSystemFormat) {
188  entryValue = -entryValue;
189  }
190  builder.addNextValue(row, col, entryValue);
191  }
192  eqSysVector.push_back(stateValuesGetter(globalState) + rateAtState * actionValuesGetter(globalState));
193  ++row;
194  }
195 
196  // Create a linear equation solver
197  auto solver = linearEquationSolverFactory.create(subEnv, builder.build());
198  // Check solver requirements.
199  auto requirements = solver->getRequirements(subEnv);
200  STORM_LOG_THROW(!requirements.hasEnabledCriticalRequirement(), storm::exceptions::UnmetRequirementException, "Solver requirements " + requirements.getEnabledRequirementsAsString() + " not checked.");
201  // Todo: Find bounds on the bias variables. Just inserting the maximal value from the vector probably does not work.
202 
203  std::vector<ValueType> eqSysSol(bscc.size(), storm::utility::zero<ValueType>());
204  // Take the mean of the rewards as an initial guess for the gain
205  //eqSysSol.front() = std::accumulate(eqSysVector.begin(), eqSysVector.end(), storm::utility::zero<ValueType>()) / storm::utility::convertNumber<ValueType, uint64_t>(bscc.size());
206  solver->solveEquations(subEnv, eqSysSol, eqSysVector);
207 
208  ValueType gain = eqSysSol.front();
209  // insert bias value for state 0
210  eqSysSol.front() = storm::utility::zero<ValueType>();
211  // Return the gain and the bias values
212  return std::pair<ValueType, std::vector<ValueType>>(std::move(gain), std::move(eqSysSol));
213  }

215  template <typename ValueType>
216  std::vector<ValueType> SparseDeterministicInfiniteHorizonHelper<ValueType>::computeSteadyStateDistrForBscc(Environment const& env, storm::storage::StronglyConnectedComponent const& bscc) {
217  // We want that the returned values are sorted properly. Let's assert that strongly connected components use a sorted container type
218  STORM_LOG_ASSERT(std::is_sorted(bscc.begin(), bscc.end()), "Expected that bsccs are sorted.");
219 
220  // Let A be ab auxiliary Matrix with A[s,s] = R(s,s) - r(s) & A[s,s'] = R(s,s') for s,s' in BSCC and s!=s'.
221  // We build and solve the equation system for
222  // x*A=0 & x_0+...+x_n=1 <=> A^t*x=0=x-x & x_0+...+x_n=1 <=> (1+A^t)*x = x & 1-x_0-...-x_n-1=x_n
223  // Then, x[i] will be the fraction of the time we are in state i.
224 
225  // This method assumes that this BSCC consist of more than one state.
226  // We therefore catch the (easy) case where the BSCC is a singleton.
227  if (bscc.size() == 1) {
228  return { storm::utility::one<ValueType>() };
229  }
230 
231  // Prepare an environment for the underlying linear equation solver
232  auto subEnv = env;
233  if (subEnv.solver().getLinearEquationSolverType() == storm::solver::EquationSolverType::Topological) {
234  // Topological solver does not make any sense since the BSCC is connected.
235  subEnv.solver().setLinearEquationSolverType(subEnv.solver().topological().getUnderlyingEquationSolverType(), subEnv.solver().topological().isUnderlyingEquationSolverTypeSetFromDefault());
236  }
237  subEnv.solver().setLinearEquationSolverPrecision(env.solver().lra().getPrecision(), env.solver().lra().getRelativeTerminationCriterion());
238  STORM_LOG_WARN_COND(!subEnv.solver().isForceSoundness(), "Sound computations are not properly implemented for this computation. You might get incorrect results.");
239 
240  // Get a mapping from global state indices to local ones as well as a bitvector containing states within the BSCC.
241  std::unordered_map<uint64_t, uint64_t> toLocalIndexMap;
242  storm::storage::BitVector bsccStates(this->_transitionMatrix.getRowCount(), false);
243  uint64_t localIndex = 0;
244  for (auto const& globalIndex : bscc) {
245  bsccStates.set(globalIndex, true);
246  toLocalIndexMap[globalIndex] = localIndex;
247  ++localIndex;
248  }
249 
250  // Build the auxiliary Matrix A.
251  auto auxMatrix = this->_transitionMatrix.getSubmatrix(false, bsccStates, bsccStates, true); // add diagonal entries!
252  uint64_t row = 0;
253  for (auto const& globalIndex : bscc) {
254  ValueType rateAtState = this->_exitRates ? (*this->_exitRates)[globalIndex] : storm::utility::one<ValueType>();
255  for (auto& entry : auxMatrix.getRow(row)) {
256  if (entry.getColumn() == row) {
257  // This value is non-zero since we have a BSCC with more than one state
258  entry.setValue(rateAtState * (entry.getValue() - storm::utility::one<ValueType>()));
259  } else if (this->isContinuousTime()) {
260  entry.setValue(entry.getValue() * rateAtState);
261  }
262  }
263  ++row;
264  }
265  assert(row == auxMatrix.getRowCount());
266 
267  // We need to consider A^t. This will not delete diagonal entries since they are non-zero.
268  auxMatrix = auxMatrix.transpose();
269 
270  // Check whether we need the fixpoint characterization
271  storm::solver::GeneralLinearEquationSolverFactory<ValueType> linearEquationSolverFactory;
272  bool isFixpointFormat = linearEquationSolverFactory.getEquationProblemFormat(subEnv) == storm::solver::LinearEquationSolverProblemFormat::FixedPointSystem;
273  if (isFixpointFormat) {
274  // Add a 1 on the diagonal
275  for (row = 0; row < auxMatrix.getRowCount(); ++row) {
276  for (auto& entry : auxMatrix.getRow(row)) {
277  if (entry.getColumn() == row) {
278  entry.setValue(storm::utility::one<ValueType>() + entry.getValue());
279  }
280  }
281  }
282  }
283 
284  // We now build the equation system matrix.
285  // We can drop the last row of A and add ones in this row instead to assert that the variables sum up to one
286  // Phase 1: replace the existing entries of the last row with ones
287  uint64_t col = 0;
288  uint64_t lastRow = auxMatrix.getRowCount() - 1;
289  for (auto& entry : auxMatrix.getRow(lastRow)) {
290  entry.setColumn(col);
291  if (isFixpointFormat) {
292  if (col == lastRow) {
293  entry.setValue(storm::utility::zero<ValueType>());
294  } else {
295  entry.setValue(-storm::utility::one<ValueType>());
296  }
297  } else {
298  entry.setValue(storm::utility::one<ValueType>());
299  }
300  ++col;
301  }
302  storm::storage::SparseMatrixBuilder<ValueType> builder(std::move(auxMatrix));
303  for (; col <= lastRow; ++col) {
304  if (isFixpointFormat) {
305  if (col != lastRow) {
306  builder.addNextValue(lastRow, col, -storm::utility::one<ValueType>());
307  }
308  } else {
309  builder.addNextValue(lastRow, col, storm::utility::one<ValueType>());
310  }
311  }
312 
313  std::vector<ValueType> bsccEquationSystemRightSide(bscc.size(), storm::utility::zero<ValueType>());
314  bsccEquationSystemRightSide.back() = storm::utility::one<ValueType>();
315 
316  // Create a linear equation solver
317  auto solver = linearEquationSolverFactory.create(subEnv, builder.build());
318  solver->setBounds(storm::utility::zero<ValueType>(), storm::utility::one<ValueType>());
319  // Check solver requirements.
320  auto requirements = solver->getRequirements(subEnv);
321  requirements.clearLowerBounds();
322  requirements.clearUpperBounds();
323  STORM_LOG_THROW(!requirements.hasEnabledCriticalRequirement(), storm::exceptions::UnmetRequirementException, "Solver requirements " + requirements.getEnabledRequirementsAsString() + " not checked.");
324 
325  std::vector<ValueType> steadyStateDistr(bscc.size(), storm::utility::one<ValueType>() / storm::utility::convertNumber<ValueType, uint64_t>(bscc.size()));
326  solver->solveEquations(subEnv, steadyStateDistr, bsccEquationSystemRightSide);
327 
328  // As a last step, we normalize these values to counter numerical inaccuracies a bit.
329  // This is only reasonable in non-exact mode.
330  if (!subEnv.solver().isForceExact()) {
331  ValueType sum = std::accumulate(steadyStateDistr.begin(), steadyStateDistr.end(), storm::utility::zero<ValueType>());
332  storm::utility::vector::scaleVectorInPlace<ValueType, ValueType>(steadyStateDistr, storm::utility::one<ValueType>() / sum);
333  }
334 
335  return steadyStateDistr;
336  }

360  template <typename ValueType>
361  std::pair<storm::storage::SparseMatrix<ValueType>, std::vector<ValueType>> SparseDeterministicInfiniteHorizonHelper<ValueType>::buildSspMatrixVector(std::vector<ValueType> const& bsccLraValues, std::vector<uint64_t> const& inputStateToBsccIndexMap, storm::storage::BitVector const& statesNotInComponent, bool asEquationSystem) {
362 
363  // Create SSP Matrix.
364  // In contrast to the version for nondeterministic models, we eliminate the auxiliary states representing each BSCC on the fly
365 
366  // Probability mass that would lead to a BSCC will be considered in the rhs of the equation system
367  auto sspMatrix = this->_transitionMatrix.getSubmatrix(false, statesNotInComponent, statesNotInComponent, asEquationSystem);
368  if (asEquationSystem) {
369  sspMatrix.convertToEquationSystem();
370  }
371 
372  // Create the SSP right-hand-side
373  std::vector<ValueType> rhs;
374  rhs.reserve(sspMatrix.getRowCount());
375  for (auto const& state : statesNotInComponent) {
376  ValueType stateValue = storm::utility::zero<ValueType>();
377  for (auto const& transition : this->_transitionMatrix.getRow(state)) {
378  if (!statesNotInComponent.get(transition.getColumn())) {
379  // This transition leads to a BSCC!
380  stateValue += transition.getValue() * bsccLraValues[inputStateToBsccIndexMap[transition.getColumn()]];
381  }
382  }
383  rhs.push_back(std::move(stateValue));
384  }
385 
386  return std::make_pair(std::move(sspMatrix), std::move(rhs));

389  template <typename ValueType>
390  std::vector<ValueType> SparseDeterministicInfiniteHorizonHelper<ValueType>::buildAndSolveSsp(Environment const& env, std::vector<ValueType> const& componentLraValues) {
391  STORM_LOG_ASSERT(this->_longRunComponentDecomposition != nullptr, "Decomposition not computed, yet.");
392 
393  // For fast transition rewriting, we build a mapping from the input state indices to the state indices of a new transition matrix
394  // which redirects all transitions leading to a former BSCC state to a new (imaginary) auxiliary state.
395  // Each auxiliary state gets assigned the value of that BSCC and we compute expected rewards (aka stochastic shortest path, SSP) on that new system.
396  // For efficiency reasons, we actually build the system where the auxiliary states are already eliminated.
397 
398  // First gather the states that are part of a component
399  // and create a mapping from states that lie in a component to the corresponding component index.
400  storm::storage::BitVector statesInComponents(this->_transitionMatrix.getRowGroupCount());
401  std::vector<uint64_t> stateIndexMap(this->_transitionMatrix.getRowGroupCount(), std::numeric_limits<uint64_t>::max());
402  for (uint64_t currentComponentIndex = 0; currentComponentIndex < this->_longRunComponentDecomposition->size(); ++currentComponentIndex) {
403  for (auto const& element : (*this->_longRunComponentDecomposition)[currentComponentIndex]) {
404  uint64_t state = internal::getComponentElementState(element);
405  statesInComponents.set(state);
406  stateIndexMap[state] = currentComponentIndex;
407  }
408  }
409  // Map the non-component states to their index in the SSP. Note that the order of these states will be preserved.
410  uint64_t numberOfNonComponentStates = 0;
411  storm::storage::BitVector statesNotInComponent = ~statesInComponents;
412  for (auto const& nonComponentState : statesNotInComponent) {
413  stateIndexMap[nonComponentState] = numberOfNonComponentStates;
414  ++numberOfNonComponentStates;
415  }
416 
417  // The next step is to create the equation system solving the SSP (unless the whole system consists of BSCCs)
418  std::vector<ValueType> sspValues;
419  if (numberOfNonComponentStates > 0) {
420  storm::solver::GeneralLinearEquationSolverFactory<ValueType> linearEquationSolverFactory;
421  bool isEqSysFormat = linearEquationSolverFactory.getEquationProblemFormat(env) == storm::solver::LinearEquationSolverProblemFormat::EquationSystem;
422  auto sspMatrixVector = buildSspMatrixVector(componentLraValues, stateIndexMap, statesNotInComponent, isEqSysFormat);
423  std::unique_ptr<storm::solver::LinearEquationSolver<ValueType>> solver = linearEquationSolverFactory.create(env, sspMatrixVector.first);
424  auto lowerUpperBounds = std::minmax_element(componentLraValues.begin(), componentLraValues.end());
425  solver->setBounds(*lowerUpperBounds.first, *lowerUpperBounds.second);
426  // Check solver requirements
427  auto requirements = solver->getRequirements(env);
428  requirements.clearUpperBounds();
429  requirements.clearLowerBounds();
430  STORM_LOG_THROW(!requirements.hasEnabledCriticalRequirement(), storm::exceptions::UnmetRequirementException, "Solver requirements " + requirements.getEnabledRequirementsAsString() + " not checked.");

465  auto bsccReachProbs = computeBsccReachabilityProbabilities(env, initialDistributionGetter);
466  // We are now ready to compute the resulting lra distribution

479  for (auto const& element : component) {
480  uint64_t state = internal::getComponentElementState(element);
481  steadyStateDistr[state] = *bsccDistrIt;
482  ++bsccDistrIt;
483  }
484  STORM_LOG_ASSERT(bsccDistrIt == bsccDistr.end(), "Unexpected number of entries in bscc distribution");
485  }

499  template <typename ValueType>
500  std::vector<ValueType> SparseDeterministicInfiniteHorizonHelper<ValueType>::computeBsccReachabilityProbabilities(Environment const& env, ValueGetter const& initialDistributionGetter) {
501  STORM_LOG_ASSERT(this->_longRunComponentDecomposition != nullptr, "Decomposition not computed, yet.");
502  uint64_t numBSCCs = this->_longRunComponentDecomposition->size();
503  STORM_LOG_ASSERT(numBSCCs > 0, "Found 0 BSCCs in a Markov chain. This should not be possible.");
504 
505  // Compute for each BSCC get the probability with which we reach that BSCC
506  std::vector<ValueType> bsccReachProbs(numBSCCs, storm::utility::zero<ValueType>());
507  if (numBSCCs == 1) {
508  bsccReachProbs.front() = storm::utility::one<ValueType>();
509  } else {
510  // Compute mapping from state indices to their BSCC and
511  // the set of states that do not lie in a BSCC
512  std::vector<uint64_t> stateToBsccMap(this->_transitionMatrix.getRowGroupCount(), std::numeric_limits<uint64_t>::max());
513  storm::storage::BitVector statesNotInComponent(this->_transitionMatrix.getRowGroupCount(), true);
514  for (uint64_t currentComponentIndex = 0; currentComponentIndex < this->_longRunComponentDecomposition->size(); ++currentComponentIndex) {
515  for (auto const& element : (*this->_longRunComponentDecomposition)[currentComponentIndex]) {
516  uint64_t state = internal::getComponentElementState(element);
517  statesNotInComponent.set(state, false);
518  stateToBsccMap[state] = currentComponentIndex;
519  }
520  }
521  // For each non-BSCC state, compute the expected number of times we visit that state by solving an equation system
522  // The equation system basically considers an equation `init(s) + sum_s' P(s',s) * x_s' = x_s` for each state s.
523  storm::solver::GeneralLinearEquationSolverFactory<ValueType> linearEquationSolverFactory;
524  bool isEqSysFormat = linearEquationSolverFactory.getEquationProblemFormat(env) == storm::solver::LinearEquationSolverProblemFormat::EquationSystem;
525  auto eqSysMatrix = this->_transitionMatrix.getSubmatrix(false, statesNotInComponent, statesNotInComponent, isEqSysFormat);
526  boost::optional<std::vector<ValueType>> upperBounds;
527  // Check solver requirements
528  auto requirements = linearEquationSolverFactory.getRequirements(env);
529  requirements.clearLowerBounds();
530  if (requirements.upperBounds()) {
531  auto toBsccProbabilities = this->_transitionMatrix.getConstrainedRowSumVector(statesNotInComponent, ~statesNotInComponent);
532  upperBounds = computeUpperBoundsForExpectedVisitingTimes(eqSysMatrix, toBsccProbabilities);
533  requirements.clearUpperBounds();
534  }
535  STORM_LOG_THROW(!requirements.hasEnabledCriticalRequirement(), storm::exceptions::UnmetRequirementException, "Solver requirements " + requirements.getEnabledRequirementsAsString() + " not checked.");
536  eqSysMatrix = eqSysMatrix.transpose(false, true); // Transpose so that each row considers the predecessors
537  if (isEqSysFormat) {
538  eqSysMatrix.convertToEquationSystem();
539  }
540  std::unique_ptr<storm::solver::LinearEquationSolver<ValueType>> solver = linearEquationSolverFactory.create(env, eqSysMatrix);
541  solver->setLowerBound(storm::utility::zero<ValueType>());
542  if (upperBounds) {
543  solver->setUpperBounds(std::move(upperBounds.get()));
544  }
545  std::vector<ValueType> eqSysRhs;
546  eqSysRhs.reserve(eqSysMatrix.getRowCount());
547  for (auto state : statesNotInComponent) {
548  eqSysRhs.push_back(initialDistributionGetter(state));
549  }
550  std::vector<ValueType> eqSysValues(eqSysMatrix.getRowCount());
551  solver->solveEquations(env, eqSysValues, eqSysRhs);
552 
553  // Now that we have the expected number of times we visit each non-BSCC state, we can compute the probabilities to reach each bscc.
554  // Run over all transitions that enter a BSCC
555  uint64_t eqSysStateIndex = 0;
556  for (auto state : statesNotInComponent) {
557  for (auto const& entry : this->_transitionMatrix.getRow(state)) {
558  if (!statesNotInComponent.get(entry.getColumn())) {
559  // Add the probability that the BSCC is reached from this state.
560  bsccReachProbs[stateToBsccMap[entry.getColumn()]] += entry.getValue() * eqSysValues[eqSysStateIndex];
561  }
562  }
563  ++eqSysStateIndex;
564  }
565  }
566  // As a last step, we normalize these values to counter numerical inaccuracies a bit.
567  // This is only reasonable in non-exact mode.
568  if (!env.solver().isForceExact()) {
569  ValueType sum = std::accumulate(bsccReachProbs.begin(), bsccReachProbs.end(), storm::utility::zero<ValueType>());
570  storm::utility::vector::scaleVectorInPlace<ValueType, ValueType>(bsccReachProbs, storm::utility::one<ValueType>() / sum);
571  }
572  return bsccReachProbs;
573  }
