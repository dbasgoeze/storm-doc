235  bool EpochManager::epochClassZigZagOrder(Epoch const& epoch1, Epoch const& epoch2) const {
236  STORM_LOG_ASSERT(dimensionCount > 0, "Invoked EpochManager with zero dimension count.");
237 
238  // Return true iff epoch 1 has to be computed before epoch 2
239 
240  if (!compareEpochClass(epoch1, epoch2)) {
241  return epochClassOrder(getEpochClass(epoch1), getEpochClass(epoch2));
242  }
243 
244  // check whether the sum of dimensions is the same
245  uint64_t e1Sum = 0;
246  uint64_t e2Sum = 0;
247  for (uint64_t dim = 0; dim < dimensionCount; ++dim) {
248  if (!isBottomDimension(epoch1, dim)) {
249  assert(!isBottomDimension(epoch2, dim));
250  e1Sum += getDimensionOfEpoch(epoch1, dim);
251  e2Sum += getDimensionOfEpoch(epoch2, dim);
252  }
253  }
254  if (e1Sum < e2Sum) {
255  return true;
256  } else if (e1Sum > e2Sum) {
257  return false;
258  }
259 
260  // find the first dimension where the epochs do not match.
261  // if the sum is even, we search from left to right, otherwise from right to left
262  bool sumEven = (e1Sum % 2) == 0;
263  if (sumEven) {
264  for (uint64_t dim = 0; dim < dimensionCount; ++dim) {
265  uint64_t e1Value = getDimensionOfEpoch(epoch1, dim);
266  uint64_t e2Value = getDimensionOfEpoch(epoch2, dim);
267  if (e1Value < e2Value) {
268  return true;
269  } else if (e1Value > e2Value) {
270  return false;
271  }
272  }
273  } else {
274  uint64_t dim = dimensionCount;
275  while (dim > 0) {
276  --dim;
277  uint64_t e1Value = getDimensionOfEpoch(epoch1, dim);
278  uint64_t e2Value = getDimensionOfEpoch(epoch2, dim);
279  if (e1Value < e2Value) {

285  }
