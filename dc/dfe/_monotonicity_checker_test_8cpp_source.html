111 TEST(MonotonicityCheckerTest, Brp_with_bisimulation_no_samples) {
112  std::string programFile = STORM_TEST_RESOURCES_DIR "/pdtmc/brp16_2.pm";
113  std::string formulaAsString = "P=? [true U s=4 & i=N ]";
114  std::string constantsAsString = ""; //e.g. pL=0.9,TOACK=0.5
115 
116  // Program and formula
117  storm::prism::Program program = storm::api::parseProgram(programFile);
118  program = storm::utility::prism::preprocess(program, constantsAsString);
119  std::vector<std::shared_ptr<const storm::logic::Formula>> formulas = storm::api::extractFormulasFromProperties(storm::api::parsePropertiesForPrismProgram(formulaAsString, program));
120  std::shared_ptr<storm::models::sparse::Dtmc<storm::RationalFunction>> model = storm::api::buildSparseModel<storm::RationalFunction>(program, formulas)->as<storm::models::sparse::Dtmc<storm::RationalFunction>>();
121  std::shared_ptr<storm::models::sparse::Dtmc<storm::RationalFunction>> dtmc = model->as<storm::models::sparse::Dtmc<storm::RationalFunction>>();
122  auto simplifier = storm::transformer::SparseParametricDtmcSimplifier<storm::models::sparse::Dtmc<storm::RationalFunction>>(*dtmc);
123  ASSERT_TRUE(simplifier.simplify(*(formulas[0])));
124  model = simplifier.getSimplifiedModel();
125 
126  // Apply bisimulation
127  storm::storage::BisimulationType bisimType = storm::storage::BisimulationType::Strong;
128  if (storm::settings::getModule<storm::settings::modules::BisimulationSettings>().isWeakBisimulationSet()) {
129  bisimType = storm::storage::BisimulationType::Weak;
130  }
131 
132  dtmc = storm::api::performBisimulationMinimization<storm::RationalFunction>(model, formulas, bisimType)->as<storm::models::sparse::Dtmc<storm::RationalFunction>>();
133 
134 
135  // Create the region
136  typename storm::storage::ParameterRegion<storm::RationalFunction>::Valuation lowerBoundaries;
137  typename storm::storage::ParameterRegion<storm::RationalFunction>::Valuation upperBoundaries;
138  std::set<typename storm::storage::ParameterRegion<storm::RationalFunction>::VariableType> vars = storm::models::sparse::getProbabilityParameters(*dtmc);
139  for (auto var : vars) {
140  typename storm::storage::ParameterRegion<storm::RationalFunction>::CoefficientType lb = storm::utility::convertNumber<typename storm::storage::ParameterRegion<storm::RationalFunction>::CoefficientType>(0 + 0.000001);
141  typename storm::storage::ParameterRegion<storm::RationalFunction>::CoefficientType ub = storm::utility::convertNumber<typename storm::storage::ParameterRegion<storm::RationalFunction>::CoefficientType>(1 - 0.000001);
142  lowerBoundaries.emplace(std::make_pair(var, lb));
143  upperBoundaries.emplace(std::make_pair(var, ub));
144  }
145  auto region = storm::storage::ParameterRegion<storm::RationalFunction>(std::move(lowerBoundaries), std::move(upperBoundaries));
146 std::vector<storm::storage::ParameterRegion<storm::RationalFunction>> regions = {region};
147 
148  ASSERT_EQ(dtmc->getNumberOfStates(), 99ull);
149  ASSERT_EQ(dtmc->getNumberOfTransitions(), 195ull);
150 
151  storm::analysis::MonotonicityChecker<storm::RationalFunction> monotonicityChecker = storm::analysis::MonotonicityChecker<storm::RationalFunction>(dtmc, formulas, regions, true);
152  auto result = monotonicityChecker.checkMonotonicity(std::cout);
153  EXPECT_EQ(1ul, result.size());
154  EXPECT_EQ(2ul, result.begin()->second.size());
155  auto monotone = result.begin()->second.begin();
156  EXPECT_EQ(true, monotone->second.first);
157  EXPECT_EQ(false, monotone->second.second);
158 }
