60  std::shared_ptr<Polytope<ValueType>> Polytope<ValueType>::createSelectiveDownwardClosure(std::vector<Point> const& points, storm::storage::BitVector const& selectedDimensions) {
61  if (points.empty()) {
62  // In this case, the downwardclosure is empty
63  return createEmptyPolytope();
64  }
65  if (selectedDimensions.empty()) {
66  return create(points);
67  }
68  assert(points.front().size() == selectedDimensions.size());
69 
70  std::vector<Halfspace<ValueType>> halfspaces;
71  // We build the convex hull of the given points.
72  // However, auxiliary points (that will always be in the selective downward closure) are added.
73  // Then, the halfspaces of the resulting polytope are a superset of the halfspaces of the downward closure.
74  std::vector<Point> auxiliaryPoints = points;
75  auxiliaryPoints.reserve(auxiliaryPoints.size() * (1 + selectedDimensions.getNumberOfSetBits()));
76  for (auto const& point : points) {
77  for (auto const& dim : selectedDimensions) {
78  auxiliaryPoints.push_back(point);
79  auxiliaryPoints.back()[dim] -= storm::utility::one<ValueType>();
80  }
81  }
82  std::vector<Halfspace<ValueType>> auxiliaryHalfspaces = create(auxiliaryPoints)->getHalfspaces();
83  // The downward closure is obtained by erasing the halfspaces for which the normal vector is negative for one of the selected dimensions.
84  for (auto& h : auxiliaryHalfspaces) {
85  bool allGreaterEqZero = true;
86  for (auto const& dim : selectedDimensions) {
87  allGreaterEqZero &= (h.normalVector()[dim] >= storm::utility::zero<ValueType>());
88  }
89  if (allGreaterEqZero){
90  halfspaces.push_back(std::move(h));
91  }
92  }
93  return create(halfspaces);
94  }

108  std::vector<typename Polytope<storm::RationalNumber>::Point> Polytope<storm::RationalNumber>::getVerticesInClockwiseOrder() const{
109  std::vector<Point> vertices = getVertices();
110  if(vertices.size()<=2) {
111  // In this case, every ordering is clockwise
112  return vertices;
113  }
114  STORM_LOG_THROW(vertices.front().size()==2, storm::exceptions::IllegalFunctionCallException, "Getting Vertices in clockwise order is only possible for a 2D-polytope.");
115 
116  std::vector<storm::storage::BitVector> neighborsOfVertices(vertices.size(), storm::storage::BitVector(vertices.size(), false));
117  std::vector<Halfspace<storm::RationalNumber>> halfspaces = getHalfspaces();
118  for(auto const& h : halfspaces) {
119  storm::storage::BitVector verticesOnHalfspace(vertices.size(), false);
120  for(uint_fast64_t v = 0; v<vertices.size(); ++v) {
121  if(h.isPointOnBoundary(vertices[v])) {
122  verticesOnHalfspace.set(v);
123  }
124  }
125  for(auto const& v : verticesOnHalfspace) {
126  neighborsOfVertices[v] |= verticesOnHalfspace;
127  neighborsOfVertices[v].set(v, false);
128  }
129  }
130 
131  std::vector<Point> result;
132  result.reserve(vertices.size());
133  storm::storage::BitVector unprocessedVertices(vertices.size(), true);
134 
135  uint_fast64_t currentVertex = 0;
136  for(uint_fast64_t v = 1; v<vertices.size(); ++v) {
137  if(vertices[v].front() < vertices[currentVertex].front()) {
138  currentVertex = v;
139  }
140  }
141  STORM_LOG_ASSERT(neighborsOfVertices[currentVertex].getNumberOfSetBits()==2, "For 2D Polytopes with at least 3 vertices, each vertex should have exactly 2 neighbors");
142  uint_fast64_t firstNeighbor = neighborsOfVertices[currentVertex].getNextSetIndex(0);
143  uint_fast64_t secondNeighbor = neighborsOfVertices[currentVertex].getNextSetIndex(firstNeighbor+1);
144  uint_fast64_t previousVertex = vertices[firstNeighbor].back() <= vertices[secondNeighbor].back() ? firstNeighbor : secondNeighbor;
145  do {
146  unprocessedVertices.set(currentVertex, false);
147  result.push_back(std::move(vertices[currentVertex]));
148 
149  STORM_LOG_ASSERT(neighborsOfVertices[currentVertex].getNumberOfSetBits()==2, "For 2D Polytopes with at least 3 vertices, each vertex should have exactly 2 neighbors");
150  firstNeighbor = neighborsOfVertices[currentVertex].getNextSetIndex(0);
151  secondNeighbor = neighborsOfVertices[currentVertex].getNextSetIndex(firstNeighbor+1);
152  uint_fast64_t nextVertex = firstNeighbor != previousVertex ? firstNeighbor : secondNeighbor;
153  previousVertex = currentVertex;
154  currentVertex = nextVertex;
155  } while(!unprocessedVertices.empty());
156 
157  return result;
158  }
