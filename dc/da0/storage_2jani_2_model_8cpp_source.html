201  result.templateEdge->addDestination(templateDestination);
202

315  if (modelReference.getBooleanValue(edgeVariables[outerIndex][innerIndex])) {
316  chosenEdges.emplace_back(possibleEdges[outerIndex][innerIndex]);
317  break;
318  }
319  }
320  }
321

323  // Note that there is still information missing, which we need to add (like the action index etc.).
324  ConditionalMetaEdge conditionalMetaEdge = createSynchronizedMetaEdge(newAutomaton, chosenEdges);
325

327  conditionalMetaEdge.components = components;

330  conditionalMetaEdge.actionIndex = resultingActionIndex;
331

333

338  }

549  newAutomaton.registerTemplateEdge(conditionalMetaEdge.templateEdge);
550  conditionalMetaEdge.actionIndex = edge.getActionIndex();
551  conditionalMetaEdge.components.emplace_back(static_cast<uint64_t>(i));
552  conditionalMetaEdge.condition.emplace_back(edge.getSourceLocationIndex());
553  conditionalMetaEdge.rate = edge.getOptionalRate();
554  for (auto const& destination : edge.getDestinations()) {
555  conditionalMetaEdge.templateEdge->addDestination(destination.getOrderedAssignments());
556  conditionalMetaEdge.effects.emplace_back();
557 
558  conditionalMetaEdge.effects.back().emplace_back(destination.getLocationIndex());
559  conditionalMetaEdge.probabilities.emplace_back(destination.getProbability());
560  }
561  }
562  }
563  }
564 
565  // Now that all meta edges have been built, we can explore the location space and add all edges based
566  // on the templates.
567  addEdgesToReachableLocations(composedAutomata, newAutomaton, conditionalMetaEdges);
568 
569  // Fix all variables mentioned in assignments by applying the constructed remapping.
570  newAutomaton.changeAssignmentVariables(variableRemapping);
571 
572  // Finalize the flattened model.
573  storm::expressions::Expression initialStatesRestriction = getManager().boolean(true);
574  for (auto const& automaton : composedAutomata) {
575  if (automaton.get().hasInitialStatesRestriction()) {
576  initialStatesRestriction = initialStatesRestriction && automaton.get().getInitialStatesRestriction();
577  }
578  for (auto const& funDef : automaton.get().getFunctionDefinitions()) {
579  newAutomaton.addFunctionDefinition(funDef.second);
580  }
581  }
582 
583  newAutomaton.setInitialStatesRestriction(this->getInitialStatesExpression(composedAutomata));
584  if (this->hasInitialStatesRestriction()) {
585  flattenedModel.setInitialStatesRestriction(this->getInitialStatesRestriction());
586  }
587  flattenedModel.addAutomaton(newAutomaton);
588  flattenedModel.setStandardSystemComposition();
589  flattenedModel.finalize();
590 
591  return flattenedModel;
592  }

994  void Model::simplifyComposition() {
995  CompositionInformationVisitor visitor(*this, this->getSystemComposition());
996  CompositionInformation info = visitor.getInformation();
997  if (info.containsNestedParallelComposition()) {
998  STORM_LOG_WARN("Unable to simplify non-standard compliant system composition.");
999  }
1000 
1001  // Check whether we need to copy certain automata
1002  std::unordered_map<std::string, std::vector<std::string>> automatonToCopiesMap;
1003  for (auto const& automatonMultiplicity : info.getAutomatonToMultiplicityMap()) {
1004  if (automatonMultiplicity.second > 1) {
1005  std::vector<std::string> copies = {automatonMultiplicity.first};
1006  // We need to copy this automaton n-1 times.
1007  for (uint64_t copyIndex = 1; copyIndex < automatonMultiplicity.second; ++copyIndex) {
1008  std::string copyPrefix = "Copy__" + std::to_string(copyIndex) + "_Of";
1009  std::string copyAutName = copyPrefix + automatonMultiplicity.first;
1010  this->addAutomaton(this->getAutomaton(automatonMultiplicity.first).clone(getManager(), copyAutName, copyPrefix));
1011  copies.push_back(copyAutName);
1012  }
1013  // For esthetic reasons we reverse the list of copies so that the ones with the lowest index will be pop_back'ed first
1014  std::reverse(copies.begin(), copies.end());
1015  // We insert the copies in reversed order as they will be popped in reversed order, as well.
1016  automatonToCopiesMap[automatonMultiplicity.first] = std::move(copies);
1017  }
1018  }
1019 
1020  // Traverse the system composition and exchange automata by their copy
1021  auto newComposition = CompositionSimplificationVisitor(automatonToCopiesMap).simplify(getSystemComposition());
1022  this->setSystemComposition(newComposition);
1023 
1024  }
