414  }

420  xCopy = xCopy.minAbstract(this->choiceVariables);
421  } else {
422  xCopy = xCopy.maxAbstract(this->choiceVariables);
423  }
424  }

445  MinMaxLinearEquationSolverRequirements SymbolicMinMaxLinearEquationSolver<DdType, ValueType>::getRequirements(Environment const& env, boost::optional<storm::solver::OptimizationDirection> const& direction) const {
446  MinMaxLinearEquationSolverRequirements requirements;
447 
448  auto method = getMethod(env, std::is_same<ValueType, storm::RationalNumber>::value);
449  if (method == MinMaxMethod::PolicyIteration) {
450  if (!this->hasUniqueSolution()) {
451  requirements.requireValidInitialScheduler();
452  }
453  } else if (method == MinMaxMethod::ValueIteration) {
454  if (!this->hasUniqueSolution()) {
455  if (!direction || direction.get() == storm::solver::OptimizationDirection::Maximize) {
456  requirements.requireLowerBounds();
457  }
458  if (!direction || direction.get() == storm::solver::OptimizationDirection::Minimize) {
459  requirements.requireValidInitialScheduler();
460  }
461  }
462  } else if (method == MinMaxMethod::RationalSearch) {
463  requirements.requireLowerBounds();
464  if (!this->hasUniqueSolution() && (!direction || direction.get() == storm::solver::OptimizationDirection::Minimize)) {
465  requirements.requireUniqueSolution();
466  }
467  } else {
468  STORM_LOG_THROW(false, storm::exceptions::InvalidEnvironmentException, "The selected min max technique is not supported by this solver.");
469  }
470 
471  return requirements;
472  }
