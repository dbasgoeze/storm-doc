487  if (useGaussSeidelMultiplication) {

504  }

506  }

510  if (useDiffs) {

512  }

515  } else {

529  }

533  // precision here. Doing so, we need to take the means of the lower and upper values later to guarantee
534  // the original precision.
535  if (this->hasRelevantValues()) {

548  status = SolverStatus::TerminatedEarly;
549  }
550  }

553  this->showProgressIterative(iterations);
554

560  }

568  } else if (&x == this->cachedRowVector.get()) {
569  std::swap(x, *this->cachedRowVector);
570  }
571

573  clearCache();
574  }
575  this->reportStatus(status, iterations);
576

713  template<typename ValueType>
714  template<typename RationalType, typename ImpreciseType>
715  bool NativeLinearEquationSolver<ValueType>::solveEquationsRationalSearchHelper(Environment const& env, NativeLinearEquationSolver<ImpreciseType> const& impreciseSolver, storm::storage::SparseMatrix<RationalType> const& rationalA, std::vector<RationalType>& rationalX, std::vector<RationalType> const& rationalB, storm::storage::SparseMatrix<ImpreciseType> const& A, std::vector<ImpreciseType>& x, std::vector<ImpreciseType> const& b, std::vector<ImpreciseType>& tmpX) const {
716 
717  ValueType precision = storm::utility::convertNumber<ValueType>(env.solver().native().getPrecision());
718  uint64_t maxIter = env.solver().native().getMaximalNumberOfIterations();
719  bool relative = env.solver().native().getRelativeTerminationCriterion();
720  auto multiplicationStyle = env.solver().native().getPowerMethodMultiplicationStyle();
721 
722  std::vector<ImpreciseType> const* originalX = &x;
723 
724  std::vector<ImpreciseType>* currentX = &x;
725  std::vector<ImpreciseType>* newX = &tmpX;
726 
727  SolverStatus status = SolverStatus::InProgress;
728  uint64_t overallIterations = 0;
729  uint64_t valueIterationInvocations = 0;
730  impreciseSolver.startMeasureProgress();
731  while (status == SolverStatus::InProgress && overallIterations < maxIter) {
732  // Perform value iteration with the current precision.
733  typename NativeLinearEquationSolver<ImpreciseType>::PowerIterationResult result = impreciseSolver.performPowerIteration(env, currentX, newX, b, storm::utility::convertNumber<ImpreciseType, ValueType>(precision), relative, SolverGuarantee::LessOrEqual, overallIterations, maxIter, multiplicationStyle);
734 
735  // At this point, the result of the imprecise value iteration is stored in the (imprecise) current x.
736 
737  ++valueIterationInvocations;
738  STORM_LOG_TRACE("Completed " << valueIterationInvocations << " power iteration invocations, the last one with precision " << precision << " completed in " << result.iterations << " iterations.");
739 
740  // Count the iterations.
741  overallIterations += result.iterations;
742 
743  // Compute maximal precision until which to sharpen.
744  uint64_t p = storm::utility::convertNumber<uint64_t>(storm::utility::ceil(storm::utility::log10<ValueType>(storm::utility::one<ValueType>() / precision)));
745 
746  // Make sure that currentX and rationalX are not aliased.
747  std::vector<RationalType>* temporaryRational = TemporaryHelper<RationalType, ImpreciseType>::getTemporary(rationalX, currentX, newX);
748 
749  // Sharpen solution and place it in the temporary rational.
750  bool foundSolution = sharpen(p, rationalA, *currentX, rationalB, *temporaryRational);
751 
752  // After sharpen, if a solution was found, it is contained in the free rational.
753 
754  if (foundSolution) {
755  status = SolverStatus::Converged;
756 
757  TemporaryHelper<RationalType, ImpreciseType>::swapSolutions(rationalX, temporaryRational, x, currentX, newX);
758  } else {
759  // Increase the precision.
760  precision = precision / 10;
761  }
762  if (storm::utility::resources::isTerminate()) {
763  status = SolverStatus::Aborted;
764  }
765  }
766 
767  // Swap the two vectors if the current result is not in the original x.
768  if (currentX != originalX) {
769  std::swap(x, tmpX);
770  }
771 
772  if (status == SolverStatus::InProgress && overallIterations == maxIter) {
773  status = SolverStatus::MaximalIterationsExceeded;
774  }
775 
776  this->logIterations(status == SolverStatus::Converged, status == SolverStatus::TerminatedEarly, overallIterations);
777 
778  return status == SolverStatus::Converged || status == SolverStatus::TerminatedEarly;
779  }
