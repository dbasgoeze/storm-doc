160  template <storm::dd::DdType DdType, typename ValueType>
161  std::unique_ptr<MenuGame<DdType, ValueType>> JaniMenuGameAbstractor<DdType, ValueType>::buildGame() {
162  // As long as there is only one automaton, we only build its game representation.
163  GameBddResult<DdType> game = automata.front().abstract();
164 
165  // Add the locations to the transitions.
166  game.bdd &= edgeDecoratorAdd.notZero();
167 
168  // Construct a set of all unnecessary variables, so we can abstract from it.
169  std::set<storm::expressions::Variable> variablesToAbstract(abstractionInformation.getPlayer1VariableSet(abstractionInformation.getPlayer1VariableCount()));
170  std::set<storm::expressions::Variable> successorAndAuxVariables(abstractionInformation.getSuccessorVariables());
171  auto player2Variables = abstractionInformation.getPlayer2VariableSet(game.numberOfPlayer2Variables);
172  variablesToAbstract.insert(player2Variables.begin(), player2Variables.end());
173  auto auxVariables = abstractionInformation.getAuxVariableSet(0, abstractionInformation.getAuxVariableCount());
174  variablesToAbstract.insert(auxVariables.begin(), auxVariables.end());
175  successorAndAuxVariables.insert(auxVariables.begin(), auxVariables.end());
176 
177  storm::utility::Stopwatch relevantStatesWatch(true);
178  storm::dd::Bdd<DdType> nonTerminalStates = this->abstractionInformation.getDdManager().getBddOne();
179  if (this->isRestrictToRelevantStatesSet()) {
180  // Compute which states are non-terminal.
181  for (auto const& expression : this->terminalStateExpressions) {
182  nonTerminalStates &= !this->getStates(expression);
183  }
184  if (this->hasTargetStateExpression()) {
185  nonTerminalStates &= !this->getStates(this->getTargetStateExpression());
186  }
187  }
188  relevantStatesWatch.stop();
189 
190  storm::dd::Bdd<DdType> extendedTransitionRelation = nonTerminalStates && game.bdd;
191  storm::dd::Bdd<DdType> initialStates = initialLocationsBdd && initialStateAbstractor.getAbstractStates();
192  if (this->restrictToValidBlocks) {
193  STORM_LOG_DEBUG("Restricting to valid blocks.");
194  storm::dd::Bdd<DdType> validBlocks = validBlockAbstractor.getValidBlocks();
195 
196  // Compute the choices with only valid successors so we can restrict the game to these.
197  auto choicesWithOnlyValidSuccessors = !game.bdd.andExists(!validBlocks.swapVariables(abstractionInformation.getSourceSuccessorVariablePairs()), successorAndAuxVariables) && game.bdd.existsAbstract(successorAndAuxVariables);
198 
199  // Restrict the proper parts.
200  extendedTransitionRelation &= validBlocks && choicesWithOnlyValidSuccessors;
201  initialStates &= validBlocks;
202  }
203 
204  // Do a reachability analysis on the raw transition relation.
205  storm::dd::Bdd<DdType> transitionRelation = extendedTransitionRelation.existsAbstract(variablesToAbstract);
206  initialStates.addMetaVariables(abstractionInformation.getSourcePredicateVariables());
207  storm::dd::Bdd<DdType> reachableStates = storm::utility::dd::computeReachableStates(initialStates, transitionRelation, abstractionInformation.getSourceVariables(), abstractionInformation.getSuccessorVariables()).first;
208 
209  relevantStatesWatch.start();
210  if (this->isRestrictToRelevantStatesSet() && this->hasTargetStateExpression()) {
211  // Get the target state BDD.
212  storm::dd::Bdd<DdType> targetStates = reachableStates && this->getStates(this->getTargetStateExpression());
213 
214  // In the presence of target states, we keep only states that can reach the target states.
215  reachableStates = storm::utility::dd::computeBackwardsReachableStates(targetStates, reachableStates, transitionRelation, abstractionInformation.getSourceVariables(), abstractionInformation.getSuccessorVariables());
216 
217  // Include all successors of reachable states, because the backward search otherwise potentially
218  // cuts probability 0 choices of these states.
219  reachableStates |= (reachableStates && !targetStates).relationalProduct(transitionRelation, abstractionInformation.getSourceVariables(), abstractionInformation.getSuccessorVariables());
220 
221  // Restrict transition relation to relevant fragment for computation of deadlock states.
222  transitionRelation &= reachableStates && reachableStates.swapVariables(abstractionInformation.getExtendedSourceSuccessorVariablePairs());
223 
224  relevantStatesWatch.stop();
225  STORM_LOG_TRACE("Restricting to relevant states took " << relevantStatesWatch.getTimeInMilliseconds() << "ms.");
226  }
227 
228  // Find the deadlock states in the model. Note that this does not find the 'deadlocks' in bottom states,
229  // as the bottom states are not contained in the reachable states.
230  storm::dd::Bdd<DdType> deadlockStates = transitionRelation.existsAbstract(abstractionInformation.getSuccessorVariables());
231  deadlockStates = reachableStates && !deadlockStates;
232 
233  // If there are deadlock states, we fix them now.
234  storm::dd::Add<DdType, ValueType> deadlockTransitions = abstractionInformation.getDdManager().template getAddZero<ValueType>();
235  if (!deadlockStates.isZero()) {
236  deadlockTransitions = (deadlockStates && abstractionInformation.getAllPredicateIdentities() && abstractionInformation.getAllLocationIdentities() && abstractionInformation.encodePlayer1Choice(0, abstractionInformation.getPlayer1VariableCount()) && abstractionInformation.encodePlayer2Choice(0, 0, game.numberOfPlayer2Variables) && abstractionInformation.encodeAux(0, 0, abstractionInformation.getAuxVariableCount())).template toAdd<ValueType>();
237  }
238 
239  // Compute bottom states and the appropriate transitions if necessary.
240  BottomStateResult<DdType> bottomStateResult(abstractionInformation.getDdManager().getBddZero(), abstractionInformation.getDdManager().getBddZero());
241  bottomStateResult = automata.front().getBottomStateTransitions(reachableStates, game.numberOfPlayer2Variables);
242  bool hasBottomStates = !bottomStateResult.states.isZero();
243 
244  // Construct the transition matrix by cutting away the transitions of unreachable states.
245  // Note that we also restrict the successor states of transitions, because there might be successors
246  // that are not in the set of relevant states we restrict to.
247  storm::dd::Add<DdType, ValueType> transitionMatrix = (extendedTransitionRelation && reachableStates && reachableStates.swapVariables(abstractionInformation.getExtendedSourceSuccessorVariablePairs())).template toAdd<ValueType>();
248  transitionMatrix *= edgeDecoratorAdd;
249  transitionMatrix += deadlockTransitions;
250 
251  // Extend the current game information with the 'non-bottom' tag before potentially adding bottom state transitions.
252  transitionMatrix *= (abstractionInformation.getBottomStateBdd(true, true) && abstractionInformation.getBottomStateBdd(false, true)).template toAdd<ValueType>();
253  reachableStates &= abstractionInformation.getBottomStateBdd(true, true);
254  initialStates &= abstractionInformation.getBottomStateBdd(true, true);
255 
256  // If there are bottom transitions, exnted the transition matrix and reachable states now.
257  if (hasBottomStates) {
258  transitionMatrix += bottomStateResult.transitions.template toAdd<ValueType>();
259  reachableStates |= bottomStateResult.states;
260  }
261 
262  std::set<storm::expressions::Variable> allNondeterminismVariables = player2Variables;
263  allNondeterminismVariables.insert(abstractionInformation.getPlayer1Variables().begin(), abstractionInformation.getPlayer1Variables().end());
264 
265  std::set<storm::expressions::Variable> allSourceVariables(abstractionInformation.getSourceVariables());
266  allSourceVariables.insert(abstractionInformation.getBottomStateVariable(true));
267  std::set<storm::expressions::Variable> allSuccessorVariables(abstractionInformation.getSuccessorVariables());
268  allSuccessorVariables.insert(abstractionInformation.getBottomStateVariable(false));
269 
270  return std::make_unique<MenuGame<DdType, ValueType>>(abstractionInformation.getDdManagerAsSharedPointer(), reachableStates, initialStates, abstractionInformation.getDdManager().getBddZero(), transitionMatrix, bottomStateResult.states, allSourceVariables, allSuccessorVariables, abstractionInformation.getExtendedSourceSuccessorVariablePairs(), std::set<storm::expressions::Variable>(abstractionInformation.getPlayer1Variables().begin(), abstractionInformation.getPlayer1Variables().end()), player2Variables, allNondeterminismVariables, auxVariables, abstractionInformation.getPredicateToBddMap());
271  }
