98  StateBehavior<ValueType, StateType> DftNextStateGenerator<ValueType, StateType>::exploreState(StateToIdCallback const& stateToIdCallback, bool exploreDependencies, bool takeFirstDependency) {
99  // Prepare the result, in case we return early.
100  StateBehavior<ValueType, StateType> result;
101 
102  //size_t failableCount = hasDependencies ? state->nrFailableDependencies() : state->nrFailableBEs();
103  //size_t currentFailable = 0;
104  state->getFailableElements().init(exploreDependencies);
105 
106  // Check for absorbing state:
107  // - either no relevant event remains (i.e., all relevant events have failed already), or
108  // - no BE can fail
109  if (!state->hasOperationalRelevantEvent() || state->getFailableElements().isEnd()) {
110  Choice<ValueType, StateType> choice(0, true);
111  // Add self loop
112  choice.addProbability(state->getId(), storm::utility::one<ValueType>());
113  STORM_LOG_TRACE("Added self loop for " << state->getId());
114  // No further exploration required
115  result.addChoice(std::move(choice));
116  result.setExpanded();
117  return result;
118  }
119 
120  Choice<ValueType, StateType> choice(0, !exploreDependencies);
121 
122  // Let BE fail
123  bool isFirst = true;
124  while (!state->getFailableElements().isEnd()) {
125  //TODO outside
126  if (takeFirstDependency && exploreDependencies && !isFirst) {
127  // We discard further exploration as we already chose one dependent event
128  break;
129  }
130  isFirst = false;
131 
132  // Construct new state as copy from original one
133  DFTStatePointer newState = state->copy();
134  std::pair<std::shared_ptr<storm::storage::DFTBE<ValueType> const>, bool> nextBEPair = newState->letNextBEFail(state->getFailableElements().get(), exploreDependencies);
135  std::shared_ptr<storm::storage::DFTBE<ValueType> const>& nextBE = nextBEPair.first;
136  STORM_LOG_ASSERT(nextBE, "NextBE is null.");
137  STORM_LOG_ASSERT(nextBEPair.second == exploreDependencies, "Failure due to dependencies does not match.");
138  STORM_LOG_TRACE("With the failure of: " << nextBE->name() << " [" << nextBE->id() << "] in " << mDft.getStateString(state));
139 
140  // Propagate
141  storm::storage::DFTStateSpaceGenerationQueues<ValueType> queues;
142 
143  propagateFailure(newState, nextBE, queues);
144 
145  bool transient = false;
146  // TODO handle for all types of BEs.
147  if (nextBE->beType() == storm::storage::BEType::EXPONENTIAL) {
148  auto beExp = std::static_pointer_cast<storm::storage::BEExponential<ValueType> const>(nextBE);
149  transient = beExp->isTransient();
150  }
151 
152  if(newState->isInvalid() || (transient && !newState->hasFailed(mDft.getTopLevelIndex()))) {
153  // Continue with next possible state
154  state->getFailableElements().next();
155  STORM_LOG_TRACE("State is ignored because " << (newState->isInvalid() ? "it is invalid" : "the transient fault is ignored"));
156  continue;
157  }
158 
159  // Get the id of the successor state
160  StateType newStateId;
161  if (newState->hasFailed(mDft.getTopLevelIndex()) && uniqueFailedState) {
162  // Use unique failed state
163  newStateId = 0;
164  } else {
165  propagateFailsafe(newState, nextBE, queues);
166 
167  // Update failable dependencies
168  newState->updateFailableDependencies(nextBE->id());
169  newState->updateDontCareDependencies(nextBE->id());
170  newState->updateFailableInRestrictions(nextBE->id());
171 
172  // Add new state
173  newStateId = stateToIdCallback(newState);
174  }
175 
176  // Set transitions
177  if (exploreDependencies) {
178  // Failure is due to dependency -> add non-deterministic choice if necessary
179  ValueType probability = mDft.getDependency(state->getFailableElements().get())->probability();
180  choice.addProbability(newStateId, probability);
181  STORM_LOG_TRACE("Added transition to " << newStateId << " with probability " << probability);
182 
183  if (!storm::utility::isOne(probability)) {
184  // Add transition to state where dependency was unsuccessful
185  DFTStatePointer unsuccessfulState = state->copy();
186  unsuccessfulState->letDependencyBeUnsuccessful(state->getFailableElements().get());
187  // Add state
188  StateType unsuccessfulStateId = stateToIdCallback(unsuccessfulState);
189  ValueType remainingProbability = storm::utility::one<ValueType>() - probability;
190  choice.addProbability(unsuccessfulStateId, remainingProbability);
191  STORM_LOG_TRACE("Added transition to " << unsuccessfulStateId << " with remaining probability " << remainingProbability);
192  STORM_LOG_ASSERT(unsuccessfulStateId != state->getId(), "Self loop was added (through PDEP) for " << unsuccessfulStateId << " and failure of " << nextBE->name());
193  }
194  result.addChoice(std::move(choice));
195  } else {
196  // Failure is due to "normal" BE failure
197  // Set failure rate according to activation
198  STORM_LOG_THROW(nextBE->beType() == storm::storage::BEType::EXPONENTIAL, storm::exceptions::NotSupportedException, "BE of type '" << nextBE->type() << "' is not supported.");
199  auto beExp = std::static_pointer_cast<storm::storage::BEExponential<ValueType> const>(nextBE);
200  bool isActive = true;
201  if (mDft.hasRepresentant(beExp->id())) {
202  // Active must be checked for the state we are coming from as this state is responsible for the rate
203  isActive = state->isActive(mDft.getRepresentant(beExp->id()));
204  }
205  ValueType rate = isActive ? beExp->activeFailureRate() : beExp->passiveFailureRate();
206  STORM_LOG_ASSERT(!storm::utility::isZero(rate), "Rate is 0.");
207  choice.addProbability(newStateId, rate);
208  STORM_LOG_TRACE("Added transition to " << newStateId << " with " << (isActive ? "active" : "passive") << " failure rate " << rate);
209  }
210  STORM_LOG_ASSERT(newStateId != state->getId(), "Self loop was added for " << newStateId << " and failure of " << nextBE->name());
211 
212  state->getFailableElements().next();
213  } // end while failing BE
214 
215  if (exploreDependencies) {
216  if (result.empty()) {
217  // Dependencies might have been prevented from sequence enforcer
218  // -> explore BEs now
219  return exploreState(stateToIdCallback, false, takeFirstDependency);
220  }
221  } else {
222  if (choice.size() == 0) {
223  // No transition was generated
224  STORM_LOG_TRACE("No transitions were generated.");
225  // Add self loop
226  choice.addProbability(state->getId(), storm::utility::one<ValueType>());
227  STORM_LOG_TRACE("Added self loop for " << state->getId());
228  }
229  STORM_LOG_ASSERT(choice.size() > 0, "At least one choice should have been generated.");
230  // Add all rates as one choice
231  result.addChoice(std::move(choice));
232  }
233 
234  STORM_LOG_TRACE("Finished exploring state: " << mDft.getStateString(state));
235  result.setExpanded();
236  return result;
237  }
