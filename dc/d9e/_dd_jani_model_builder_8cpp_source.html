62  auto features = model.getModelFeatures();
63  features.remove(storm::jani::ModelFeature::Arrays); // can be substituted
64  features.remove(storm::jani::ModelFeature::DerivedOperators);

66  features.remove(storm::jani::ModelFeature::StateExitRewards);
67  if (!features.empty()) {

79  for (auto const& p : properties.get()) {
80  p.gatherReferencedRewardModels(rewardModels);
81  }

84  STORM_LOG_INFO("Symbolic engine can not build Jani model due to non-trivial reward expressions.");

92  }

96  // As this method is supposed to be a quick check, we just return true at this point.

294  CompositionVariables<Type, ValueType> create() {
295  // First, check whether every automaton appears exactly once in the system composition. Simultaneously,
296  // we determine the set of non-silent actions used by the composition.
297  automata.clear();
298  this->model.getSystemComposition().accept(*this, boost::none);
299  STORM_LOG_THROW(automata.size() == this->model.getNumberOfAutomata(), storm::exceptions::InvalidArgumentException, "Cannot build symbolic model from JANI model whose system composition refers to a subset of automata.");
300 
301  STORM_LOG_THROW(!this->model.hasTransientEdgeDestinationAssignments(), storm::exceptions::InvalidArgumentException, "The symbolic JANI model builder currently does not support transient edge destination assignments.");
302 
303  // Then, check that the model does not contain non-transient unbounded integer or non-transient real variables.
304  STORM_LOG_THROW(!this->model.getGlobalVariables().containsNonTransientUnboundedIntegerVariables(), storm::exceptions::InvalidArgumentException, "Cannot build symbolic model from JANI model that contains non-transient global unbounded integer variables.");
305  STORM_LOG_THROW(!this->model.getGlobalVariables().containsNonTransientRealVariables(), storm::exceptions::InvalidArgumentException, "Cannot build symbolic model from JANI model that contains global non-transient real variables.");
306  for (auto const& automaton : this->model.getAutomata()) {
307  STORM_LOG_THROW(!automaton.getVariables().containsNonTransientUnboundedIntegerVariables(), storm::exceptions::InvalidArgumentException, "Cannot build symbolic model from JANI model that contains non-transient unbounded integer variables in automaton '" << automaton.getName() << "'.");
308  STORM_LOG_THROW(!automaton.getVariables().containsNonTransientRealVariables(), storm::exceptions::InvalidArgumentException, "Cannot build symbolic model from JANI model that contains non-transient real variables in automaton '" << automaton.getName() << "'.");
309  }
310 
311  // Based on this assumption, we create the variables.
312  return createVariables();
313  }

335  result.actionVariablesMap[nonSilentActionIndex] = variablePair.first;
336  result.allNondeterminismVariables.insert(variablePair.first);
337  }
338 
339  // FIXME: check how many nondeterminism variables we should actually allocate.
340  uint64_t numberOfNondeterminismVariables = this->model.getNumberOfAutomata();
341  for (auto const& automaton : this->model.getAutomata()) {
342  numberOfNondeterminismVariables += automaton.getNumberOfEdges();
343  }
344  for (uint_fast64_t i = 0; i < numberOfNondeterminismVariables; ++i) {
345  std::pair<storm::expressions::Variable, storm::expressions::Variable> variablePair = result.manager->addMetaVariable("nondet" + std::to_string(i));
346  result.localNondeterminismVariables.push_back(variablePair.first);
347  result.allNondeterminismVariables.insert(variablePair.first);

352  result.probabilisticMarker = result.manager->getEncoding(result.probabilisticNondeterminismVariable, 1);
353  result.allNondeterminismVariables.insert(result.probabilisticNondeterminismVariable);
354  }

357  storm::jani::Automaton const& automaton = this->model.getAutomaton(automatonName);

370  result.columnMetaVariables.insert(variablePair.second);
371

398  identity &= variableIdentity;
399  range &= result.manager->getRange(locationVariables.first);
400 
401  // Then create variables for the variables of the automaton.
402  for (auto const& variable : automaton.getVariables()) {
403  // Only create the variable if it's non-transient.
404  if (variable.isTransient()) {
405  continue;
406  }
407 
408  createVariable(variable, result);
409  identity &= result.variableToIdentityMap.at(variable.getExpressionVariable()).toBdd();
410  range &= result.manager->getRange(result.variableToRowMetaVariableMap->at(variable.getExpressionVariable()));
411  }
412 
413  result.automatonToIdentityMap[automaton.getName()] = identity.template toAdd<ValueType>();
414  result.automatonToRangeMap[automaton.getName()] = (range && globalVariableRanges).template toAdd<ValueType>();
415  }
416 
417  ParameterCreator<Type, ValueType> parameterCreator;
418  parameterCreator.create(model, *result.rowExpressionAdapter);
419  if (std::is_same<ValueType, storm::RationalFunction>::value) {
420  result.parameters = parameterCreator.getParameters();
421  }
422 
423  return result;

529  template <storm::dd::DdType Type, typename ValueType>
530  EdgeDestinationDd<Type, ValueType> buildEdgeDestinationDd(storm::jani::Automaton const& automaton, storm::jani::EdgeDestination const& destination, storm::dd::Bdd<Type> const& guard, CompositionVariables<Type, ValueType> const& variables) {
531  storm::dd::Add<Type, ValueType> transitions = variables.rowExpressionAdapter->translateExpression(destination.getProbability());
532 
533  STORM_LOG_TRACE("Translating edge destination.");
534 
535  // Iterate over all assignments (boolean and integer) and build the DD for it.
536  std::set<storm::expressions::Variable> assignedVariables;
537  for (auto const& assignment : destination.getOrderedAssignments().getNonTransientAssignments()) {
538  // Record the variable as being written.
539  STORM_LOG_TRACE("Assigning to variable " << variables.variableToRowMetaVariableMap->at(assignment.getExpressionVariable()).getName());
540  assignedVariables.insert(assignment.getExpressionVariable());
541 
542  // Translate the written variable.
543  auto const& primedMetaVariable = variables.variableToColumnMetaVariableMap->at(assignment.getExpressionVariable());
544  storm::dd::Add<Type, ValueType> writtenVariable = variables.manager->template getIdentity<ValueType>(primedMetaVariable);
545 
546  // Translate the expression that is being assigned.
547  storm::dd::Add<Type, ValueType> assignedExpression = variables.rowExpressionAdapter->translateExpression(assignment.getAssignedExpression());
548 
549  // Combine the assigned expression with the guard.
550  storm::dd::Add<Type, ValueType> result = assignedExpression * guard.template toAdd<ValueType>();
551 
552  // Combine the variable and the assigned expression.
553  result = result.equals(writtenVariable).template toAdd<ValueType>();
554  result *= guard.template toAdd<ValueType>();
555 
556  // Restrict the transitions to the range of the written variable.
557  result = result * variables.variableToRangeMap.at(primedMetaVariable).template toAdd<ValueType>();
558 
559  // Combine the assignment DDs.
560  transitions *= result;
561  }
562 
563  // Compute the set of assigned global variables.
564  std::set<storm::expressions::Variable> assignedGlobalVariables;
565  std::set_intersection(assignedVariables.begin(), assignedVariables.end(), variables.allGlobalVariables.begin(), variables.allGlobalVariables.end(), std::inserter(assignedGlobalVariables, assignedGlobalVariables.begin()));
566 
567  // All unassigned boolean variables need to keep their value.
568  for (storm::jani::BooleanVariable const& variable : automaton.getVariables().getBooleanVariables()) {
569  if (assignedVariables.find(variable.getExpressionVariable()) == assignedVariables.end()) {
570  STORM_LOG_TRACE("Multiplying identity of variable " << variable.getName());
571  transitions *= variables.variableToIdentityMap.at(variable.getExpressionVariable());
572  }
573  }
574 
575  // All unassigned integer variables need to keep their value.
576  for (storm::jani::BoundedIntegerVariable const& variable : automaton.getVariables().getBoundedIntegerVariables()) {
577  if (assignedVariables.find(variable.getExpressionVariable()) == assignedVariables.end()) {
578  STORM_LOG_TRACE("Multiplying identity of variable " << variable.getName());
579  transitions *= variables.variableToIdentityMap.at(variable.getExpressionVariable());
580  }
581  }
582 
583  transitions *= variables.manager->getEncoding(variables.automatonToLocationDdVariableMap.at(automaton.getName()).second, destination.getLocationIndex()).template toAdd<ValueType>();
584 
585  return EdgeDestinationDd<Type, ValueType>(transitions, assignedGlobalVariables);
586  }

945  boost::any visit(storm::jani::ParallelComposition const& composition, boost::any const& data) override {

949 
950  // Prepare storage for the subautomata of the composition.
951  std::vector<AutomatonDd> subautomata;
952 
953  // The outer loop iterates over the indices of the subcomposition, because the first subcomposition needs
954  // to be built before the second and so on.
955  uint64_t silentActionIndex = actionInformation.getActionIndex(storm::jani::Model::SILENT_ACTION_NAME);
956  for (uint64_t subcompositionIndex = 0; subcompositionIndex < composition.getNumberOfSubcompositions(); ++subcompositionIndex) {
957  // Now build a new set of action instantiations for the current subcomposition index.
958  ActionInstantiations actionInstantiations;
959  actionInstantiations[silentActionIndex].emplace_back(silentActionIndex, 0, isCtmc);
960  if (this->model.getModelType() == storm::jani::ModelType::MA) {
961  actionInstantiations[storm::jani::Model::SILENT_ACTION_INDEX].emplace_back(silentActionIndex, 0, true);
962  }
963 
964  for (uint64_t synchronizationVectorIndex = 0; synchronizationVectorIndex < composition.getNumberOfSynchronizationVectors(); ++synchronizationVectorIndex) {
965  auto const& synchVector = composition.getSynchronizationVector(synchronizationVectorIndex);
966 
967  // Determine the first participating subcomposition, because we need to build the corresponding action
968  // from all local nondeterminism variable offsets that the output action of the synchronization vector
969  // is required to have.
970  if (subcompositionIndex == synchVector.getPositionOfFirstParticipatingAction()) {
971  uint64_t actionIndex = actionInformation.getActionIndex(synchVector.getInput(subcompositionIndex));
972  actionInstantiations[actionIndex].emplace_back(actionIndex, synchronizationVectorIndex, 0, isCtmc);
973  } else if (synchVector.getInput(subcompositionIndex) != storm::jani::SynchronizationVector::NO_ACTION_INPUT) {
974  uint64_t actionIndex = actionInformation.getActionIndex(synchVector.getInput(subcompositionIndex));
975 
976  // If this subcomposition is participating in the synchronization vector, but it's not the first
977  // such subcomposition, then we have to retrieve the offset we need for the participating action
978  // by looking at the maximal offset used by the preceding participating action.
979  boost::optional<uint64_t> previousActionPosition = synchVector.getPositionOfPrecedingParticipatingAction(subcompositionIndex);
980  STORM_LOG_ASSERT(previousActionPosition, "Inconsistent information about synchronization vector.");
981  AutomatonDd const& previousAutomatonDd = subautomata[previousActionPosition.get()];
982  auto precedingActionIndex = actionInformation.getActionIndex(synchVector.getInput(previousActionPosition.get()));
983  auto precedingActionIt = previousAutomatonDd.actions.find(ActionIdentification(precedingActionIndex, synchronizationVectorIndex, isCtmc));
984 
985  uint64_t highestLocalNondeterminismVariable = 0;
986  if (precedingActionIt != previousAutomatonDd.actions.end()) {
987  highestLocalNondeterminismVariable = precedingActionIt->second.getHighestLocalNondeterminismVariable();
988  } else {

1002  AutomatonDd composeInParallel(std::vector<AutomatonDd> const& subautomata, std::vector<storm::jani::SynchronizationVector> const& synchronizationVectors) {

1006  storm::dd::Bdd<Type> nonMarkovianActionGuards = this->variables.manager->getBddZero();

1009  std::unordered_map<ActionIdentification, std::vector<ActionDd>, ActionIdentificationHash> actions;
1010  for (uint64_t synchronizationVectorIndex = 0; synchronizationVectorIndex < synchronizationVectors.size(); ++synchronizationVectorIndex) {
1011  auto const& synchVector = synchronizationVectors[synchronizationVectorIndex];
1012

1014  if (synchronizingAction) {
1015  if (applyMaximumProgress) {

1017  // By the JANI standard, we can assume that synchronizing actions of MAs are always non-Markovian.
1018  nonMarkovianActionGuards |= synchronizingAction->guard;

1022  }
1023

1025  ActionIdentification silentActionIdentification(storm::jani::Model::SILENT_ACTION_INDEX);
1026  ActionIdentification silentMarkovianActionIdentification(storm::jani::Model::SILENT_ACTION_INDEX, true);
1027

1031  for (auto const& automaton : subautomata) {
1032  for (auto& actionDd : silentActionDds) {
1033  STORM_LOG_TRACE("Extending previous (non-Markovian) silent action by identity of current automaton.");

1035  }
1036  for (auto& actionDd : silentMarkovianActionDds) {
1037  STORM_LOG_TRACE("Extending previous (Markovian) silent action by identity of current automaton.");
1038  actionDd = actionDd.multiplyTransitions(automaton.identity);
1039  }

1042  if (silentActionIt != automaton.actions.end()) {
1043  STORM_LOG_TRACE("Extending (non-Markovian) silent action by running identity.");

1045  }
1046 
1047  silentActionIt = automaton.actions.find(silentMarkovianActionIdentification);
1048  if (silentActionIt != automaton.actions.end()) {
1049  STORM_LOG_TRACE("Extending (Markovian) silent action by running identity.");
1050  silentMarkovianActionDds.emplace_back(silentActionIt->second.multiplyTransitions(result.identity));
1051  }

1054 
1055  // Add the transient location assignments of the automata.
1056  addToTransientAssignmentMap(result.transientLocationAssignments, automaton.transientLocationAssignments);
1057  }
1058

1060  auto& allSilentActionDds = actions[silentActionIdentification];
1061  allSilentActionDds.insert(allSilentActionDds.end(), silentActionDds.begin(), silentActionDds.end());
1062  }
1063 
1064  // Add guards of non-markovian actions
1065  if (applyMaximumProgress) {
1066  auto allSilentActionDdsIt = actions.find(silentActionIdentification);
1067  if (allSilentActionDdsIt != actions.end()) {
1068  for (ActionDd const& silentActionDd : allSilentActionDdsIt->second) {
1069  nonMarkovianActionGuards |= silentActionDd.guard;
1070  }
1071  }
1072  }
1073

1075  auto& allMarkovianSilentActionDds = actions[silentMarkovianActionIdentification];
1076  allMarkovianSilentActionDds.insert(allMarkovianSilentActionDds.end(), silentMarkovianActionDds.begin(), silentMarkovianActionDds.end());

1078  auto invertedNonMarkovianGuards = !nonMarkovianActionGuards;
1079  for (ActionDd& markovianActionDd : allMarkovianSilentActionDds) {
1080  markovianActionDd.conjunctGuardWith(invertedNonMarkovianGuards);
1081  }
1082  }
1083  }
1084

1086  for (auto const& actionDds : actions) {

1088  if (actionDds.first == silentMarkovianActionIdentification) {
1089  // For the Markovian transitions, we can simply add the actions.
1090  combinedAction = actionDds.second.front();
1091  for (uint64_t i = 1; i < actionDds.second.size(); ++i) {
1092  combinedAction = combinedAction.add(actionDds.second[i]);
1093  }
1094  } else {
1095  combinedAction = actionDds.second.size() > 1 ? combineUnsynchronizedActions(actionDds.second) : actionDds.second.front();
1096  }
1097  result.actions[actionDds.first] = combinedAction;
1098  result.extendLocalNondeterminismVariables(combinedAction.getLocalNondeterminismVariables());
1099  }
1100 
1101  // Construct combined identity.
1102  for (auto const& subautomaton : subautomata) {
1103  result.identity *= subautomaton.identity;
1104  }
1105 
1106  return result;
1107  }

1247  inputEnabledGuard &= action.guard;
1248  }

1377  EdgeDd buildEdgeDd(storm::jani::Automaton const& automaton, storm::jani::Edge const& edge) {
1378  STORM_LOG_TRACE("Translating guard " << edge.getGuard());
1379

1381  // slower when the full range is applied.
1382  storm::dd::Bdd<Type> guard = this->variables.rowExpressionAdapter->translateBooleanExpression(edge.getGuard());
1383  storm::dd::Bdd<Type> rangedGuard = guard && this->variables.automatonToRangeMap.at(automaton.getName()).toBdd();
1384  STORM_LOG_WARN_COND(!rangedGuard.isZero(), "The guard '" << edge.getGuard() << "' is unsatisfiable.");
1385 
1386  if (!rangedGuard.isZero()) {
1387  // Create the DDs representing the individual updates.
1388  std::vector<EdgeDestinationDd<Type, ValueType>> destinationDds;
1389  for (storm::jani::EdgeDestination const& destination : edge.getDestinations()) {
1390  destinationDds.push_back(buildEdgeDestinationDd(automaton, destination, guard, this->variables));
1391 
1392  STORM_LOG_WARN_COND(!destinationDds.back().transitions.isZero(), "Destination does not have any effect.");
1393  }
1394 
1395  // Now that we have built the destinations, we always take the full guard.
1396  storm::dd::Bdd<Type> sourceLocationBdd = this->variables.manager->getEncoding(this->variables.automatonToLocationDdVariableMap.at(automaton.getName()).first, edge.getSourceLocationIndex());
1397  guard = sourceLocationBdd && rangedGuard;
1398 
1399  // Start by gathering all variables that were written in at least one destination.
1400  std::set<storm::expressions::Variable> globalVariablesInSomeDestination;
1401 
1402  // If the edge is not labeled with the silent action, we have to analyze which portion of the global
1403  // variables was written by any of the updates and make all update results equal w.r.t. this set. If
1404  // the edge is labeled with the silent action, we can already multiply the identities of all global variables.
1405  if (edge.getActionIndex() != storm::jani::Model::SILENT_ACTION_INDEX) {
1406  for (auto const& edgeDestinationDd : destinationDds) {
1407  globalVariablesInSomeDestination.insert(edgeDestinationDd.writtenGlobalVariables.begin(), edgeDestinationDd.writtenGlobalVariables.end());
1408  }
1409  } else {
1410  globalVariablesInSomeDestination = this->variables.allGlobalVariables;
1411  }
1412 
1413  // Then, multiply the missing identities.
1414  for (auto& destinationDd : destinationDds) {
1415  std::set<storm::expressions::Variable> missingIdentities;
1416  std::set_difference(globalVariablesInSomeDestination.begin(), globalVariablesInSomeDestination.end(), destinationDd.writtenGlobalVariables.begin(), destinationDd.writtenGlobalVariables.end(), std::inserter(missingIdentities, missingIdentities.begin()));
1417 
1418  for (auto const& variable : missingIdentities) {
1419  STORM_LOG_TRACE("Multiplying identity for variable " << variable.getName() << " to destination DD.");
1420  destinationDd.transitions *= this->variables.variableToIdentityMap.at(variable);
1421  }
1422  }
1423 
1424  // Now combine the destination DDs to the edge DD.
1425  storm::dd::Add<Type, ValueType> transitions = this->variables.manager->template getAddZero<ValueType>();
1426  for (auto const& destinationDd : destinationDds) {
1427  transitions += destinationDd.transitions;
1428  }
1429 
1430  // Add the source location and the guard.
1431  storm::dd::Add<Type, ValueType> guardAdd = guard.template toAdd<ValueType>();
1432  transitions *= guardAdd;
1433 
1434  // If we multiply the ranges of global variables, make sure everything stays within its bounds.
1435  if (!globalVariablesInSomeDestination.empty()) {
1436  transitions *= this->variables.globalVariableRanges;
1437  }
1438 
1439  // If the edge has a rate, we multiply it to the DD.

1656  ActionDd combineEdgesToActionNondeterministic(std::vector<EdgeDd> const& edges, uint64_t localNondeterminismVariableOffset) {
1657  // Sum all guards, so we can read off the maximal number of nondeterministic choices in any given state.
1658  storm::dd::Bdd<Type> allGuards = this->variables.manager->getBddZero();
1659  storm::dd::Add<Type, uint_fast64_t> sumOfGuards = this->variables.manager->template getAddZero<uint_fast64_t>();
1660  for (auto const& edge : edges) {
1661  STORM_LOG_ASSERT(!edge.isMarkovian, "Unexpected Markovian edge.");
1662  sumOfGuards += edge.guard.template toAdd<uint_fast64_t>();
1663  allGuards |= edge.guard;
1664  }
1665  uint_fast64_t maxChoices = sumOfGuards.getMax();
1666  STORM_LOG_TRACE("Found " << maxChoices << " non-Markovian local choices.");
1667 
1668  // Depending on the maximal number of nondeterminstic choices, we need to use some variables to encode the nondeterminism.
1669  if (maxChoices <= 1) {
1670  return combineEdgesBySummation(allGuards, edges);
1671  } else {
1672  // Calculate number of required variables to encode the nondeterminism.
1673  uint_fast64_t numberOfBinaryVariables = static_cast<uint_fast64_t>(std::ceil(storm::utility::math::log2(maxChoices)));
1674 
1675  storm::dd::Add<Type, ValueType> allEdges = this->variables.manager->template getAddZero<ValueType>();
1676  std::map<storm::expressions::Variable, storm::dd::Bdd<Type>> globalVariableToWritingFragment;
1677  std::map<storm::expressions::Variable, storm::dd::Add<Type, ValueType>> transientAssignments;
1678 
1679  storm::dd::Bdd<Type> equalsNumberOfChoicesDd;
1680  std::vector<storm::dd::Add<Type, ValueType>> choiceDds(maxChoices, this->variables.manager->template getAddZero<ValueType>());
1681  std::vector<storm::dd::Bdd<Type>> remainingDds(maxChoices, this->variables.manager->getBddZero());
1682  std::vector<std::pair<storm::dd::Bdd<Type>, storm::dd::Add<Type, ValueType>>> indicesEncodedWithLocalNondeterminismVariables;
1683  for (uint64_t j = 0; j < maxChoices; ++j) {
1684  storm::dd::Add<Type, ValueType> indexEncoding = encodeIndex(j, localNondeterminismVariableOffset, numberOfBinaryVariables, this->variables);
1685  indicesEncodedWithLocalNondeterminismVariables.push_back(std::make_pair(indexEncoding.toBdd(), indexEncoding));
1686  }
1687 
1688  for (uint_fast64_t currentChoices = 1; currentChoices <= maxChoices; ++currentChoices) {
1689  // Determine the set of states with exactly currentChoices choices.
1690  equalsNumberOfChoicesDd = sumOfGuards.equals(this->variables.manager->getConstant(currentChoices));
1691 
1692  // If there is no such state, continue with the next possible number of choices.
1693  if (equalsNumberOfChoicesDd.isZero()) {
1694  continue;
1695  }
1696 
1697  // Reset the previously used intermediate storage.
1698  for (uint_fast64_t j = 0; j < currentChoices; ++j) {
1699  choiceDds[j] = this->variables.manager->template getAddZero<ValueType>();
1700  remainingDds[j] = equalsNumberOfChoicesDd;
1701  }
1702 
1703  for (std::size_t j = 0; j < edges.size(); ++j) {
1704  EdgeDd const& currentEdge = edges[j];
1705 
1706  // Check if edge guard overlaps with equalsNumberOfChoicesDd. That is, there are states with exactly currentChoices
1707  // choices such that one outgoing choice is given by the j-th edge.
1708  storm::dd::Bdd<Type> guardChoicesIntersection = currentEdge.guard && equalsNumberOfChoicesDd;
1709 
1710  // If there is no such state, continue with the next command.
1711  if (guardChoicesIntersection.isZero()) {
1712  continue;
1713  }
1714 
1715  // Split the currentChoices nondeterministic choices.
1716  for (uint_fast64_t k = 0; k < currentChoices; ++k) {
1717  // Calculate the overlapping part of command guard and the remaining DD.
1718  storm::dd::Bdd<Type> remainingGuardChoicesIntersection = guardChoicesIntersection && remainingDds[k];
1719 
1720  // Check if we can add some overlapping parts to the current index.
1721  if (!remainingGuardChoicesIntersection.isZero()) {
1722  // Remove overlapping parts from the remaining DD.
1723  remainingDds[k] = remainingDds[k] && !remainingGuardChoicesIntersection;
1724 
1725  // Combine the overlapping part of the guard with command updates and add it to the resulting DD.
1726  choiceDds[k] += remainingGuardChoicesIntersection.template toAdd<ValueType>() * currentEdge.transitions;
1727 
1728  // Keep track of the fragment of transient assignments.
1729  for (auto const& transientAssignment : currentEdge.transientEdgeAssignments) {
1730  addToTransientAssignmentMap(transientAssignments, transientAssignment.first, remainingGuardChoicesIntersection.template toAdd<ValueType>() * transientAssignment.second * indicesEncodedWithLocalNondeterminismVariables[k].second);
1731  }
1732 
1733  // Keep track of the written global variables of the fragment.
1734  for (auto const& variableFragment : currentEdge.variableToWritingFragment) {
1735  addToVariableWritingFragmentMap(globalVariableToWritingFragment, variableFragment.first, remainingGuardChoicesIntersection && variableFragment.second && indicesEncodedWithLocalNondeterminismVariables[k].first);
1736  }
1737  }
1738 
1739  // Remove overlapping parts from the command guard DD
1740  guardChoicesIntersection = guardChoicesIntersection && !remainingGuardChoicesIntersection;
1741 
1742  // If the guard DD has become equivalent to false, we can stop here.
1743  if (guardChoicesIntersection.isZero()) {
1744  break;
1745  }
1746  }
1747  }
1748 
1749  // Add the meta variables that encode the nondeterminisim to the different choices.
1750  for (uint_fast64_t j = 0; j < currentChoices; ++j) {
1751  allEdges += indicesEncodedWithLocalNondeterminismVariables[j].second * choiceDds[j];
1752  }
1753 
1754  // Delete currentChoices out of overlapping DD
1755  sumOfGuards = sumOfGuards * (!equalsNumberOfChoicesDd).template toAdd<uint_fast64_t>();
1756  }
1757 
1758  return ActionDd(allGuards, allEdges, transientAssignments, std::make_pair(localNondeterminismVariableOffset, localNondeterminismVariableOffset + numberOfBinaryVariables), globalVariableToWritingFragment, this->variables.manager->getBddZero());
1759  }
1760  }
