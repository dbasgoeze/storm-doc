46  storm::prism::Program PrismParser::parseFromString(std::string const& input, std::string const& filename, bool prismCompatibility) {

51  PositionIteratorType last(input.end());

53

75  std::size_t lineNumber = boost::spirit::get_line(e.first);
76 
77  // Now propagate exception.
78  STORM_LOG_THROW(false, storm::exceptions::WrongFormatException, "Parsing error in line " << lineNumber << " of file " << filename << ".");
79  }
80 
81  STORM_LOG_TRACE("Parsed PRISM input: " << result);

335  (qi::lit("<") > -identifier > qi::lit(">")[qi::_a = true]))
336  > *expressionParser
337  > qi::lit("->")
338  > updateListDefinition(qi::_r1)
339  > qi::lit(";"))[qi::_val = phoenix::bind(&PrismParser::createCommand, phoenix::ref(*this), qi::_a, qi::_1, qi::_2, qi::_3, qi::_r1)];
340 
341  auto setLocationInfoFunction = this->annotate(qi::_val, qi::_1, qi::_3);
342  qi::on_success(commandDefinition, setLocationInfoFunction);
343 
344  formulaDefinition = (qi::lit("formula") > identifier > qi::lit("=") > *expressionParser > qi::lit(";"))[qi::_val = phoenix::bind(&PrismParser::createFormulaSecondRun, phoenix::ref(*this), qi::_1, qi::_2)];
345  formulaDefinition.name("formula definition");
346  this->secondRun = true;
347  this->expressionParser->setIdentifierMapping(&this->identifiers_);
348 
349  // We need to parse the formula rhs between the first run and the second run, because
350  // * in the first run, the type of the formula (int, bool, clock) is not known
351  // * in the second run, formulas might be used before they are declared
352  createFormulaIdentifiers(this->globalProgramInformation.formulas);
353 
354  this->globalProgramInformation.moveToSecondRun();
355  }

357  void PrismParser::createFormulaIdentifiers(std::vector<storm::prism::Formula> const& formulas) {
358  STORM_LOG_THROW(formulas.size() == this->formulaExpressions.size(), storm::exceptions::UnexpectedException, "Unexpected number of formulas and formula expressions");
359  this->formulaOrder.clear();
360  storm::storage::BitVector unprocessed(formulas.size(), true);
361  // It might be that formulas are declared in a weird order.
362  // We follow a trial-and-error approach: If we can not parse the expression for one formula,
363  // we assume a subsequent formula has to be evaluated first.
364  // We cycle through the formulas until no further progress is made
365  bool progress = true;
366  while (progress) {
367  progress = false;
368  for (uint64_t formulaIndex = unprocessed.getNextSetIndex(0); formulaIndex < formulas.size(); formulaIndex = unprocessed.getNextSetIndex(formulaIndex + 1)) {
369  storm::expressions::Expression expression = this->expressionParser->parseFromString(formulaExpressions[formulaIndex], true);
370  if (expression.isInitialized()) {
371  progress = true;
372  unprocessed.set(formulaIndex, false);
373  formulaOrder.push_back(formulaIndex);
374  storm::expressions::Variable variable;
375  try {
376  if (expression.hasIntegerType()) {
377  variable = manager->declareIntegerVariable(formulas[formulaIndex].getName());
378  } else if (expression.hasBooleanType()) {
379  variable = manager->declareBooleanVariable(formulas[formulaIndex].getName());
380  } else {
381  STORM_LOG_ASSERT(expression.hasNumericalType(), "Unexpected type for formula expression of formula " << formulas[formulaIndex].getName());
382  variable = manager->declareRationalVariable(formulas[formulaIndex].getName());
383  }
384  this->identifiers_.add(formulas[formulaIndex].getName(), variable.getExpression());
385  } catch (storm::exceptions::InvalidArgumentException const& e) {
386  STORM_LOG_THROW(false, storm::exceptions::WrongFormatException, "Parsing error in " << this->getFilename() << ": illegal identifier '" << formulas[formulaIndex].getName() << "' at line '" << formulas[formulaIndex].getLineNumber());
387  }
388  this->expressionParser->setIdentifierMapping(&this->identifiers_);
389  }
390  }
391  }
392  if (!unprocessed.empty()) {
393  for (auto const& formulaIndex : unprocessed) {
394  STORM_LOG_ERROR("Parsing error in " << this->getFilename() << ": Invalid expression for formula '" << formulas[formulaIndex].getName() << "' at line '" << formulas[formulaIndex].getLineNumber() << "':\n\t" << formulaExpressions[formulaIndex]);
395  }
396  STORM_LOG_THROW(unprocessed.getNumberOfSetBits() == 1, storm::exceptions::WrongFormatException, "Unable to parse expressions for " << unprocessed.getNumberOfSetBits() << " formulas. This could be due to circular dependencies");
397  STORM_LOG_THROW(false, storm::exceptions::WrongFormatException, "Unable to parse expression for formula '" << formulas[unprocessed.getNextSetIndex(0)].getName() << "'.");
398  }
399  }

764  bool PrismParser::isValidModuleRenaming(std::string const& oldModuleName, storm::prism::ModuleRenaming const& moduleRenaming, GlobalProgramInformation const& globalProgramInformation) const {
765  if (!this->secondRun) {
766  auto const& renaming = moduleRenaming.getRenaming();
767  auto const& moduleIndexPair = globalProgramInformation.moduleToIndexMap.find(oldModuleName);
768  if (moduleIndexPair == globalProgramInformation.moduleToIndexMap.end()) {
769  STORM_LOG_ERROR("Parsing error in " << this->getFilename() << ": No module named '" << oldModuleName << "' to rename.");
770  return false;
771  }
772  storm::prism::Module const& moduleToRename = globalProgramInformation.modules[moduleIndexPair->second];
773  // Check whether all varialbes are renamed.
774  for (auto const& variable : moduleToRename.getBooleanVariables()) {
775  auto const& renamingPair = renaming.find(variable.getName());
776  if (renamingPair == renaming.end()) {
777  STORM_LOG_ERROR("Parsing error in renaiming of module '" << oldModuleName << "': Boolean variable '" << variable.getName() << "' was not renamed.");
778  return false;
779  }
780  }
781  for (auto const& variable : moduleToRename.getIntegerVariables()) {
782  auto const& renamingPair = renaming.find(variable.getName());
783  if (renamingPair == renaming.end()) {
784  STORM_LOG_ERROR("Parsing error in renaiming of module '" << oldModuleName << "': Integer variable '" << variable.getName() << "' was not renamed.");
785  return false;
786  }
787  }
788  for (auto const& variable : moduleToRename.getClockVariables()) {

803  storm::prism::Module PrismParser::createRenamedModule(std::string const& newModuleName, std::string const& oldModuleName, storm::prism::ModuleRenaming const& moduleRenaming, GlobalProgramInformation& globalProgramInformation) const {
804  // Check whether the module to rename actually exists.
805  auto const& moduleIndexPair = globalProgramInformation.moduleToIndexMap.find(oldModuleName);
806  STORM_LOG_THROW(moduleIndexPair != globalProgramInformation.moduleToIndexMap.end(), storm::exceptions::WrongFormatException, "Parsing error in " << this->getFilename() << ": No module named '" << oldModuleName << "' to rename.");
807  storm::prism::Module const& moduleToRename = globalProgramInformation.modules[moduleIndexPair->second];
808  auto const& renaming = moduleRenaming.getRenaming();
809 
810  if (!this->secondRun) {
811  // Register all (renamed) variables for later use.
812  // We already checked before, whether the renaiming is valid.
813  for (auto const& variable : moduleToRename.getBooleanVariables()) {
814  auto const& renamingPair = renaming.find(variable.getName());
815  STORM_LOG_THROW(renamingPair != renaming.end(), storm::exceptions::WrongFormatException, "Parsing error in " << this->getFilename() << ": Boolean variable '" << variable.getName() << " was not renamed.");
816  storm::expressions::Variable renamedVariable = manager->declareBooleanVariable(renamingPair->second);
817  this->identifiers_.add(renamingPair->second, renamedVariable.getExpression());
818  if(this->observables.count(renamingPair->second) > 0) {
819  this->observables.erase(renamingPair->second);
820  }
821  }
822  for (auto const& variable : moduleToRename.getIntegerVariables()) {
823  auto const& renamingPair = renaming.find(variable.getName());
824  STORM_LOG_THROW(renamingPair != renaming.end(), storm::exceptions::WrongFormatException, "Parsing error in " << this->getFilename() << ": Integer variable '" << variable.getName() << " was not renamed.");
825  storm::expressions::Variable renamedVariable = manager->declareIntegerVariable(renamingPair->second);
826  this->identifiers_.add(renamingPair->second, renamedVariable.getExpression());
827  if(this->observables.count(renamingPair->second) > 0) {
828  this->observables.erase(renamingPair->second);
829  }
830  }
831  for (auto const& variable : moduleToRename.getClockVariables()) {
832  auto const& renamingPair = renaming.find(variable.getName());
833  STORM_LOG_THROW(renamingPair != renaming.end(), storm::exceptions::WrongFormatException, "Parsing error in " << this->getFilename() << ": Clock variable '" << variable.getName() << " was not renamed.");
834  storm::expressions::Variable renamedVariable = manager->declareRationalVariable(renamingPair->second);
835  this->identifiers_.add(renamingPair->second, renamedVariable.getExpression());
836  if(this->observables.count(renamingPair->second) > 0) {
837  this->observables.erase(renamingPair->second);
838  }
839  }
840 
841  for (auto const& command : moduleToRename.getCommands()) {
842  std::string newActionName = command.getActionName();
843  auto const& renamingPair = renaming.find(command.getActionName());
844  if (renamingPair != renaming.end()) {
845  newActionName = renamingPair->second;
846  }
847 
848  // Record any newly occurring action names/indices.
849  auto nameIndexPair = globalProgramInformation.actionIndices.find(newActionName);
850  if (nameIndexPair == globalProgramInformation.actionIndices.end()) {
851  std::size_t nextIndex = globalProgramInformation.actionIndices.size();
852  globalProgramInformation.actionIndices.emplace(newActionName, nextIndex);
853  }
854  }
855 
856  // Return a dummy module in the first pass.
857  return storm::prism::Module();
858  } else {
859  // Add a mapping from the new module name to its (future) index.
860  globalProgramInformation.moduleToIndexMap[newModuleName] = globalProgramInformation.modules.size();
861 
862  // Create a mapping from identifiers to the expressions they need to be replaced with.
863  std::map<storm::expressions::Variable, storm::expressions::Expression> expressionRenaming;
864  for (auto const& namePair : renaming) {
865  storm::expressions::Expression const* substitutedExpression = this->identifiers_.find(namePair.second);
866  // If the mapped-to-value is an expression, we need to replace it.
867  if (substitutedExpression != nullptr) {
868  expressionRenaming.emplace(manager->getVariable(namePair.first), *substitutedExpression);
869  }
870  }
871 
872  // Rename the boolean variables.
873  std::vector<storm::prism::BooleanVariable> booleanVariables;
874  for (auto const& variable : moduleToRename.getBooleanVariables()) {
875  auto const& renamingPair = renaming.find(variable.getName());
876  STORM_LOG_THROW(renamingPair != renaming.end(), storm::exceptions::WrongFormatException, "Parsing error in " << this->getFilename() << ": Boolean variable '" << variable.getName() << " was not renamed.");
877  bool observable = this->observables.count(renamingPair->second) > 0;
878  if(observable) {
879  this->observables.erase(renamingPair->second);
880  }
881  booleanVariables.push_back(storm::prism::BooleanVariable(manager->getVariable(renamingPair->second), variable.hasInitialValue() ? variable.getInitialValueExpression().substitute(expressionRenaming) : variable.getInitialValueExpression(), observable, this->getFilename(), moduleRenaming.getLineNumber()));
882  }
883 
884  // Rename the integer variables.
885  std::vector<storm::prism::IntegerVariable> integerVariables;
886  for (auto const& variable : moduleToRename.getIntegerVariables()) {
887  auto const& renamingPair = renaming.find(variable.getName());
888  STORM_LOG_THROW(renamingPair != renaming.end(), storm::exceptions::WrongFormatException, "Parsing error in " << this->getFilename() << ": Integer variable '" << variable.getName() << " was not renamed.");
889  bool observable = this->observables.count(renamingPair->second) > 0;
890  if(observable) {
891  this->observables.erase(renamingPair->second);
892  }
893  integerVariables.push_back(storm::prism::IntegerVariable(manager->getVariable(renamingPair->second), variable.getLowerBoundExpression().substitute(expressionRenaming), variable.getUpperBoundExpression().substitute(expressionRenaming), variable.hasInitialValue() ? variable.getInitialValueExpression().substitute(expressionRenaming) : variable.getInitialValueExpression(), observable, this->getFilename(), moduleRenaming.getLineNumber()));
894  }
895 
896  // Rename the clock variables.
897  std::vector<storm::prism::ClockVariable> clockVariables;
898  for (auto const& variable : moduleToRename.getClockVariables()) {
899  auto const& renamingPair = renaming.find(variable.getName());
900  STORM_LOG_THROW(renamingPair != renaming.end(), storm::exceptions::WrongFormatException, "Parsing error in " << this->getFilename() << ": Clock variable '" << variable.getName() << " was not renamed.");
901  bool observable = this->observables.count(renamingPair->second) > 0;
902  if (observable) {
903  this->observables.erase(renamingPair->second);
904  }
905  clockVariables.push_back(storm::prism::ClockVariable(manager->getVariable(renamingPair->second), observable, this->getFilename(), moduleRenaming.getLineNumber()));
906  }
907 
908  // Rename invariant (if present)
909  storm::expressions::Expression invariant;
910  if (moduleToRename.hasInvariant()) {
911  invariant = moduleToRename.getInvariant().substitute(expressionRenaming);
912  }
913 
914  // Rename commands.
915  std::vector<storm::prism::Command> commands;
916  for (auto const& command : moduleToRename.getCommands()) {
917  std::vector<storm::prism::Update> updates;
918  for (auto const& update : command.getUpdates()) {
919  std::vector<storm::prism::Assignment> assignments;
920  for (auto const& assignment : update.getAssignments()) {
921  auto const& renamingPair = renaming.find(assignment.getVariableName());
922  if (renamingPair != renaming.end()) {
923  assignments.emplace_back(manager->getVariable(renamingPair->second), assignment.getExpression().substitute(expressionRenaming), this->getFilename(), moduleRenaming.getLineNumber());
924  } else {
925  assignments.emplace_back(assignment.getVariable(), assignment.getExpression().substitute(expressionRenaming), this->getFilename(), moduleRenaming.getLineNumber());
926  }
927  }
928  updates.emplace_back(globalProgramInformation.currentUpdateIndex, update.getLikelihoodExpression().substitute(expressionRenaming), assignments, this->getFilename(), moduleRenaming.getLineNumber());
929  ++globalProgramInformation.currentUpdateIndex;
930  }
931 
932  std::string newActionName = command.getActionName();
933  auto const& renamingPair = renaming.find(command.getActionName());
934  if (renamingPair != renaming.end()) {
935  newActionName = renamingPair->second;
936  }
937 
938  uint_fast64_t actionIndex = 0;
939  auto nameIndexPair = globalProgramInformation.actionIndices.find(newActionName);
940  if (nameIndexPair == globalProgramInformation.actionIndices.end()) {
941  std::size_t nextIndex = globalProgramInformation.actionIndices.size();
942  globalProgramInformation.actionIndices.emplace(newActionName, nextIndex);
943  actionIndex = nextIndex;
944  } else {
945  actionIndex = nameIndexPair->second;
946  }
947 
948  commands.emplace_back(globalProgramInformation.currentCommandIndex, command.isMarkovian(), actionIndex, newActionName, command.getGuardExpression().substitute(expressionRenaming), updates, this->getFilename(), moduleRenaming.getLineNumber());
949  ++globalProgramInformation.currentCommandIndex;
950  }
951 
952  return storm::prism::Module(newModuleName, booleanVariables, integerVariables, clockVariables, invariant, commands, oldModuleName, renaming);
953  }
954  }
