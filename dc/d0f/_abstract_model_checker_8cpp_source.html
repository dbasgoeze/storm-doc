220  std::unique_ptr<CheckResult> AbstractModelChecker<ModelType>::checkBinaryBooleanStateFormula(Environment const& env, CheckTask<storm::logic::BinaryBooleanStateFormula, ValueType> const& checkTask) {
221  storm::logic::BinaryBooleanStateFormula const& stateFormula = checkTask.getFormula();
222  STORM_LOG_THROW(stateFormula.getLeftSubformula().isStateFormula() && stateFormula.getRightSubformula().isStateFormula(), storm::exceptions::InvalidArgumentException, "The given formula is invalid.");
223 
224  std::unique_ptr<CheckResult> leftResult = this->check(env, checkTask.template substituteFormula<storm::logic::Formula>(stateFormula.getLeftSubformula().asStateFormula()));
225  std::unique_ptr<CheckResult> rightResult = this->check(env, checkTask.template substituteFormula<storm::logic::Formula>(stateFormula.getRightSubformula().asStateFormula()));
226 
227  STORM_LOG_THROW(leftResult->isQualitative() && rightResult->isQualitative(), storm::exceptions::InternalTypeErrorException, "Expected qualitative results.");
228 
229  if (stateFormula.isAnd()) {
230  leftResult->asQualitativeCheckResult() &= rightResult->asQualitativeCheckResult();
231  } else if (stateFormula.isOr()) {
232  leftResult->asQualitativeCheckResult() |= rightResult->asQualitativeCheckResult();
233  } else {
234  STORM_LOG_THROW(false, storm::exceptions::InvalidArgumentException, "The given formula '" << stateFormula << "' is invalid.");
235  }
236 
237  return leftResult;
238  }
