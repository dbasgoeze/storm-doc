155  PomdpMemory PomdpMemoryBuilder::buildSettableBitsMemory(uint64_t numStates) const {
156  // compute the number of bits, i.e., floor(log(numStates))
157  uint64_t numBits = 0;
158  uint64_t actualNumStates = 1;
159  while (actualNumStates * 2 <= numStates) {
160  actualNumStates *= 2;
161  ++numBits;
162  }
163 
164  STORM_LOG_WARN_COND(actualNumStates == numStates, "The number of memory states for the settable bits pattern has to be a power of 2. Shrinking the number of memory states to " << actualNumStates << ".");
165 
166  std::vector<storm::storage::BitVector> transitions(actualNumStates, storm::storage::BitVector(actualNumStates, false));
167  for (uint64_t state = 0; state < actualNumStates; ++state) {
168  transitions[state].set(state);
169  for (uint64_t bit = 0; bit < numBits; ++bit) {
170  uint64_t bitMask = 1u << bit;
171  transitions[state].set(state | bitMask);
