187 TEST(NondeterministicSparseTransitionParserTest, Whitespaces) {
188  // Test the resilience of the parser against whitespaces.
189  // Do so by comparing the hashes of the transition matices and the rowMapping vectors element by element.
190  storm::storage::SparseMatrix<double> correctResult(storm::parser::NondeterministicSparseTransitionParser<>::parseNondeterministicTransitions(STORM_TEST_RESOURCES_DIR "/tra/mdp_general.tra"));
191  storm::storage::SparseMatrix<double> whitespaceResult = storm::parser::NondeterministicSparseTransitionParser<>::parseNondeterministicTransitions(STORM_TEST_RESOURCES_DIR "/tra/mdp_whitespaces.tra");
192  ASSERT_EQ(correctResult.hash(), whitespaceResult.hash());
193  ASSERT_EQ(correctResult.getRowGroupIndices().size(), whitespaceResult.getRowGroupIndices().size());
194  for(uint_fast64_t i = 0; i < correctResult.getRowGroupIndices().size(); i++) {
195  ASSERT_EQ(correctResult.getRowGroupIndices()[i], whitespaceResult.getRowGroupIndices()[i]);
196  }
197 
198  // Do the same (minus the unused rowMapping) for the corresponding transition rewards file (with and without whitespaces)
199  uint_fast64_t correctHash = storm::parser::NondeterministicSparseTransitionParser<>::parseNondeterministicTransitionRewards(STORM_TEST_RESOURCES_DIR "/rew/mdp_general.trans.rew", correctResult).hash();
200  ASSERT_EQ(correctHash, storm::parser::NondeterministicSparseTransitionParser<>::parseNondeterministicTransitionRewards(STORM_TEST_RESOURCES_DIR "/rew/mdp_whitespaces.trans.rew", whitespaceResult).hash());
201 }
