352  bool IOSettings::check() const {
353  // Ensure that at most one symbolic input model is given.
354  uint64_t numSymbolicInputs = isJaniInputSet() ? 1 : 0;
355  numSymbolicInputs += isPrismInputSet() ? 1 : 0;
356  numSymbolicInputs += isQvbsInputSet() ? 1 : 0;
357  STORM_LOG_THROW(numSymbolicInputs <= 1, storm::exceptions::InvalidSettingsException, "Multiple symbolic input models.");
358  STORM_LOG_THROW(!isExportJaniDotSet() || isJaniInputSet() || isQvbsInputSet(), storm::exceptions::InvalidSettingsException, "Jani-to-dot export is only available for jani models" );
359 
360  // Ensure that not two explicit input models were given.
361  uint64_t numExplicitInputs = isExplicitSet() ? 1 : 0;
362  numExplicitInputs += isExplicitDRNSet() ? 1 : 0;
363  numExplicitInputs += isExplicitIMCASet() ? 1 : 0;
364  STORM_LOG_THROW(numExplicitInputs <= 1, storm::exceptions::InvalidSettingsException, "Multiple explicit input models");
365 
366  // Ensure that the model was given either symbolically or explicitly.
367  STORM_LOG_THROW(numSymbolicInputs + numExplicitInputs <= 1, storm::exceptions::InvalidSettingsException, "The model may be either given in an explicit or a symbolic format (PRISM or JANI), but not both.");
368 
369  // Make sure PRISM-to-JANI conversion is only set if the actual input is in PRISM format.
370  STORM_LOG_THROW(!isPrismToJaniSet() || isPrismInputSet(), storm::exceptions::InvalidSettingsException, "For the transformation from PRISM to JANI, the input model must be given in the prism format.");
371 
372  return true;
373  }
