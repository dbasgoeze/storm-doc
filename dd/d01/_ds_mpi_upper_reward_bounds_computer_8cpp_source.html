65  ValueType DsMpiDtmcUpperRewardBoundsComputer<ValueType>::computeLambdaForChoice(uint64_t choice) const {
66  ValueType localLambda = storm::utility::zero<ValueType>();
67  uint64_t state = this->getStateForChoice(choice);
68 
69  // Check whether condition (I) or (II) applies.
70  ValueType probSum = originalOneStepTargetProbabilities[choice];
71  for (auto const& e : transitionMatrix.getRow(choice)) {
72  probSum += e.getValue() * p[e.getColumn()];
73  }
74 
75  if (p[state] < probSum) {
76  STORM_LOG_TRACE("Condition (I) does apply for state " << state << " as " << p[state] << " < " << probSum << ".");
77  // Condition (I) applies.
78  localLambda = probSum - p[state];
79  ValueType nominator = originalRewards[choice];
80  for (auto const& e : transitionMatrix.getRow(choice)) {
81  nominator += e.getValue() * w[e.getColumn()];
82  }
83  nominator -= w[state];
84  localLambda = nominator / localLambda;
85  } else {
86  STORM_LOG_TRACE("Condition (I) does not apply for state " << state << std::setprecision(30) << " as " << probSum << " <= " << p[state] << ".");
87  // Here, condition (II) automatically applies and as the resulting local lambda is 0, we
88  // don't need to consider it.
89 
90 #ifndef NDEBUG
91  // Actually check condition (II).
92  ValueType rewardSum = originalRewards[choice];
93  for (auto const& e : transitionMatrix.getRow(choice)) {
94  rewardSum += e.getValue() * w[e.getColumn()];
95  }
96  STORM_LOG_WARN_COND(w[state] >= rewardSum || storm::utility::ConstantsComparator<ValueType>().isEqual(w[state], rewardSum), "Expected condition (II) to hold in state " << state << ", but " << w[state] << " < " << rewardSum << ".");

234  void DsMpiMdpUpperRewardBoundsComputer<ValueType>::sweep() {
235  // Create a priority queue that allows for easy retrieval of the currently best state.
236  storm::storage::ConsecutiveUint64DynamicPriorityQueue<DsMpiMdpPriorityLess<ValueType>> queue(this->transitionMatrix.getRowGroupCount(), DsMpiMdpPriorityLess<ValueType>(*this));
237 
238  // Keep track of visited states.
239  storm::storage::BitVector visited(this->transitionMatrix.getRowGroupCount());
240 
241  while (!queue.empty()) {
242  // Get first entry in queue.
243  storm::storage::sparse::state_type currentState = queue.popTop();
244 
245  // Mark state as visited.
246  visited.set(currentState);
247 
248  // Set weight and probability for the state.
249  uint64_t choiceInCurrentState = this->getChoiceInState(currentState);
250  this->w[currentState] = this->rewards[choiceInCurrentState];
251  this->p[currentState] = this->targetProbabilities[choiceInCurrentState];
252 
253  for (auto const& choiceEntry : this->backwardTransitions.getRow(currentState)) {
254  uint64_t predecessor = this->getStateForChoice(choiceEntry.getColumn());
255  if (visited.get(predecessor)) {
256  continue;
257  }
258 
259  // Update reward/probability values.
260  this->rewards[choiceEntry.getColumn()] += choiceEntry.getValue() * this->w[currentState];
261  this->targetProbabilities[choiceEntry.getColumn()] += choiceEntry.getValue() * this->p[currentState];
262 
263  // If the choice is not the one that is currently taken in the predecessor state, we might need
264  // to update it.
265  uint64_t currentChoiceInPredecessor = this->getChoiceInState(predecessor);
266  if (currentChoiceInPredecessor != choiceEntry.getColumn()) {
267  // Check whether the updates choice now becomes a better choice in the predecessor state.
268  ValueType const& newTargetProbability = this->targetProbabilities[choiceEntry.getColumn()];
269  ValueType const& newReward = this->rewards[choiceEntry.getColumn()];
270  ValueType const& currentTargetProbability = this->targetProbabilities[this->getChoiceInState(predecessor)];
271  ValueType const& currentReward = this->rewards[this->getChoiceInState(predecessor)];
272 
273  if (newTargetProbability > currentTargetProbability || (newTargetProbability == currentTargetProbability && newReward < currentReward)) {
274  setChoiceInState(predecessor, choiceEntry.getColumn());
275  }
276  }
277 
278  // Notify the priority of a potential increase of the priority of the element.
279  queue.increase(predecessor);
280  }
281  }
