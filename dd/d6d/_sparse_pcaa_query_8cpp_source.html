42  typename SparsePcaaQuery<SparseModelType, GeometryValueType>::WeightVector SparsePcaaQuery<SparseModelType, GeometryValueType>::findSeparatingVector(Point const& pointToBeSeparated) {
43  STORM_LOG_DEBUG("Searching a weight vector to seperate the point given by " << storm::utility::vector::toString(storm::utility::vector::convertNumericVector<double>(pointToBeSeparated)) << ".");
44 
45  if(underApproximation->isEmpty()) {
46  // In this case, every weight vector is separating
47  uint_fast64_t objIndex = diracWeightVectorsToBeChecked.getNextSetIndex(0) % pointToBeSeparated.size();
48  WeightVector result(pointToBeSeparated.size(), storm::utility::zero<GeometryValueType>());
49  result[objIndex] = storm::utility::one<GeometryValueType>();
50  diracWeightVectorsToBeChecked.set(objIndex, false);
51  return result;
52  }
53 
54  // Reaching this point means that the underApproximation contains halfspaces. The seperating vector has to be the normal vector of one of these halfspaces.
55  // We pick one with maximal distance to the given point. However, Dirac weight vectors that only assign a non-zero weight to a single objective take precedence.
56  STORM_LOG_ASSERT(!underApproximation->contains(pointToBeSeparated), "Tried to find a separating point but the point is already contained in the underApproximation");
57  std::vector<storm::storage::geometry::Halfspace<GeometryValueType>> halfspaces = underApproximation->getHalfspaces();
58  uint_fast64_t farestHalfspaceIndex = halfspaces.size();
59  GeometryValueType farestDistance = -storm::utility::one<GeometryValueType>();
60  bool foundSeparatingDiracVector = false;
61  for(uint_fast64_t halfspaceIndex = 0; halfspaceIndex < halfspaces.size(); ++halfspaceIndex) {
62  GeometryValueType distance = halfspaces[halfspaceIndex].euclideanDistance(pointToBeSeparated);
63  if(!storm::utility::isZero(distance)) {
64  storm::storage::BitVector nonZeroVectorEntries = ~storm::utility::vector::filterZero<GeometryValueType>(halfspaces[halfspaceIndex].normalVector());
65  bool isSingleObjectiveVector = nonZeroVectorEntries.getNumberOfSetBits() == 1 && diracWeightVectorsToBeChecked.get(nonZeroVectorEntries.getNextSetIndex(0));
66  // Check if this halfspace is a better candidate than the current one
67  if((!foundSeparatingDiracVector && isSingleObjectiveVector ) || (foundSeparatingDiracVector==isSingleObjectiveVector && distance>farestDistance)) {
68  foundSeparatingDiracVector = foundSeparatingDiracVector || isSingleObjectiveVector;
69  farestHalfspaceIndex = halfspaceIndex;
70  farestDistance = distance;
71  }
72  }
73  }
74  if(foundSeparatingDiracVector) {
75  diracWeightVectorsToBeChecked &= storm::utility::vector::filterZero<GeometryValueType>(halfspaces[farestHalfspaceIndex].normalVector());
76  }
77 
78  STORM_LOG_THROW(farestHalfspaceIndex<halfspaces.size(), storm::exceptions::UnexpectedException, "There is no seperating vector.");
79  STORM_LOG_DEBUG("Found separating weight vector: " << storm::utility::vector::toString(storm::utility::vector::convertNumericVector<double>(halfspaces[farestHalfspaceIndex].normalVector())) << ".");
80  return halfspaces[farestHalfspaceIndex].normalVector();
81  }

83  template <class SparseModelType, typename GeometryValueType>
84  void SparsePcaaQuery<SparseModelType, GeometryValueType>::performRefinementStep(Environment const& env, WeightVector&& direction) {
85  // Normalize the direction vector so that the entries sum up to one
86  storm::utility::vector::scaleVectorInPlace(direction, storm::utility::one<GeometryValueType>() / std::accumulate(direction.begin(), direction.end(), storm::utility::zero<GeometryValueType>()));
87  weightVectorChecker->check(env, storm::utility::vector::convertNumericVector<typename SparseModelType::ValueType>(direction));
88  STORM_LOG_DEBUG("weighted objectives checker result (under approximation) is " << storm::utility::vector::toString(storm::utility::vector::convertNumericVector<double>(weightVectorChecker->getUnderApproximationOfInitialStateResults())));
89  RefinementStep step;
90  step.weightVector = direction;
91  step.lowerBoundPoint = storm::utility::vector::convertNumericVector<GeometryValueType>(weightVectorChecker->getUnderApproximationOfInitialStateResults());
92  step.upperBoundPoint = storm::utility::vector::convertNumericVector<GeometryValueType>(weightVectorChecker->getOverApproximationOfInitialStateResults());
93  // For the minimizing objectives, we need to scale the corresponding entries with -1 as we want to consider the downward closure
94  for (uint_fast64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
95  if (storm::solver::minimize(this->objectives[objIndex].formula->getOptimalityType())) {
96  step.lowerBoundPoint[objIndex] *= -storm::utility::one<GeometryValueType>();
97  step.upperBoundPoint[objIndex] *= -storm::utility::one<GeometryValueType>();
98  }
99  }
100  refinementSteps.push_back(std::move(step));
101 
102  updateOverApproximation();
103  updateUnderApproximation();
104  }

106  template <class SparseModelType, typename GeometryValueType>
107  void SparsePcaaQuery<SparseModelType, GeometryValueType>::updateOverApproximation() {
108  storm::storage::geometry::Halfspace<GeometryValueType> h(refinementSteps.back().weightVector, storm::utility::vector::dotProduct(refinementSteps.back().weightVector, refinementSteps.back().upperBoundPoint));
109 
110  // Due to numerical issues, it might be the case that the updated overapproximation does not contain the underapproximation,
111  // e.g., when the new point is strictly contained in the underapproximation. Check if this is the case.
112  GeometryValueType maximumOffset = h.offset();
113  for(auto const& step : refinementSteps){
114  maximumOffset = std::max(maximumOffset, storm::utility::vector::dotProduct(h.normalVector(), step.lowerBoundPoint));
115  }
116  if(maximumOffset > h.offset()){
117  // We correct the issue by shifting the halfspace such that it contains the underapproximation
118  h.offset() = maximumOffset;
119  STORM_LOG_WARN("Numerical issues: The overapproximation would not contain the underapproximation. Hence, a halfspace is shifted by " << storm::utility::convertNumber<double>(h.invert().euclideanDistance(refinementSteps.back().upperBoundPoint)) << ".");
120  }
121  overApproximation = overApproximation->intersection(h);
122  STORM_LOG_DEBUG("Updated OverApproximation to " << overApproximation->toString(true));
123  }
