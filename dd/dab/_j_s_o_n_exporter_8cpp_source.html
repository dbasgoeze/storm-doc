787  void getBoundsFromConstraints(ExportJsonType& typeDesc, storm::expressions::Variable const& var, storm::expressions::Expression const& constraint, std::vector<storm::jani::Constant> const& constants) {
788  if (constraint.getBaseExpression().isBinaryBooleanFunctionExpression() && constraint.getBaseExpression().getOperator() == storm::expressions::OperatorType::And) {
789  getBoundsFromConstraints(typeDesc, var, constraint.getBaseExpression().getOperand(0), constants);
790  getBoundsFromConstraints(typeDesc, var, constraint.getBaseExpression().getOperand(1), constants);
791  } else if (constraint.getBaseExpression().isBinaryRelationExpression()) {
792  bool leq = constraint.getBaseExpression().getOperator() == storm::expressions::OperatorType::LessOrEqual;
793  bool geq = constraint.getBaseExpression().getOperator() == storm::expressions::OperatorType::GreaterOrEqual;
794  std::vector<bool> varOps(2, false);
795  for (int i : {0,1}) {
796  varOps[i] = constraint.getOperand(i).isVariable() && constraint.getOperand(i).getBaseExpression().asVariableExpression().getVariable() == var;
797  }
798  storm::expressions::Expression boundExpr = varOps[0] ? constraint.getOperand(1) : constraint.getOperand(0);
799  if ((leq && varOps[0]) || (geq && varOps[1])) {
800  typeDesc["upper-bound"] = buildExpression(boundExpr, constants);
801  } else if ((leq && varOps[1]) || (geq && varOps[0])) {
802  typeDesc["lower-bound"] = buildExpression(boundExpr, constants);
803  } else {
804  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Jani Export for constant constraint " << constraint << " is not supported.");
805  }
806  } else {
807  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Jani Export for constant constraint " << constraint << " is not supported.");
808  }
809  }
