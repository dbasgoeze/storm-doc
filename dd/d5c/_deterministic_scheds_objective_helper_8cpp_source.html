120  auto const& subformula = formula.getSubformula();
121  std::map<uint64_t, ValueType> result;
122  if (formula.isProbabilityOperatorFormula() && subformula.isUntilFormula()) {
123  // In this case, there is nothing to be done.
124  } else if (formula.isRewardOperatorFormula() && (subformula.isTotalRewardFormula() || subformula.isEventuallyFormula())) {

128

134  if (stateValues.find(state) == stateValues.end()) {
135  for (uint64_t choice = rowGroupIndices[state]; choice < rowGroupIndices[state + 1]; ++choice) {
136  if (!storm::utility::isZero(choiceBasedRewards[choice])) {
137  result[choice] = choiceBasedRewards[choice];
138  }
139  }
140  }
141  }
142  } else if (formula.isTimeOperatorFormula() && subformula.isEventuallyFormula()) {

149  rates = &ma->getExitRates();
150  ms = &ma->getMarkovianStates();
151  }
152  // Set all choice offsets to one, except for the ones at states in scheduerIndependentStateValues.
153  for (uint64_t state = 0; state < model.getNumberOfStates(); ++state) {

155  ValueType value = storm::utility::one<ValueType>();
156  if (rates) {
157  if (ms->get(state)) {
158  value /= (*rates)[state];
159  } else {
160  // Nothing to be done for probabilistic states
161  continue;
162  }
163  }
164  for (uint64_t choice = rowGroupIndices[state]; choice < rowGroupIndices[state + 1]; ++choice) {
165  result[choice] = value;
166  }
167  }
168  }
169  } else {
170  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "The given formula " << formula << " is not supported.");
171  }

173  }
174  return choiceValueOffsets.get();

195  template <typename ModelType>
196  std::vector<typename ModelType::ValueType> computeValueBounds(Environment const& env, bool lowerValueBounds, ModelType const& model, storm::logic::Formula const& formula) {
197  // Change the optimization direction in the formula.
198  auto newFormula = storm::logic::CloneVisitor().clone(formula);
199  newFormula->asOperatorFormula().setOptimalityType(lowerValueBounds ? storm::solver::OptimizationDirection::Minimize : storm::solver::OptimizationDirection::Maximize);
200 
201  if (std::is_same<typename ModelType::ValueType, storm::RationalNumber>::value) {
202  // don't have to worry about precision in exact mode.
203  return evaluateOperatorFormula(env, model, *newFormula);
204  } else {
205  // Create an environment where sound results are enforced
206  storm::Environment soundEnv(env);
207  soundEnv.solver().setForceSoundness(true);
208  auto result = evaluateOperatorFormula(soundEnv, model, *newFormula);
209 
210  auto eps = storm::utility::convertNumber<typename ModelType::ValueType>(soundEnv.solver().minMax().getPrecision());
211  // Add/substract eps to all entries to make up for precision errors
212  if (lowerValueBounds) {
213  eps = -eps;
214  }
215  for (auto& v : result) {
216  v += eps;
217  }
218  return result;
219  }

496  template <typename ModelType>
497  void DeterministicSchedsObjectiveHelper<ModelType>::computeUpperBounds(Environment const& env) const {
498  if (!upperResultBounds) {
499  upperResultBounds = computeValueBounds(env, false, model, *objective.formula);
500  auto upperResultBound = objective.upperResultBound;
501  if (!upperResultBound.is_initialized() && storm::utility::vector::hasInfinityEntry(upperResultBounds.get())) {
502  STORM_LOG_THROW(objective.formula->isRewardOperatorFormula(), storm::exceptions::NotSupportedException, "The upper bound for objective " << *objective.originalFormula << " is infinity at some state. This is only supported for reachability rewards / total rewards.");

504  auto rewards = getTotalRewardVector(model, *objective.formula);
505  auto zeroValuedStates = storm::utility::vector::filterZero(upperResultBounds.get());
506  auto expVisits = computeUpperBoundOnExpectedVisitingTimes(model.getTransitionMatrix(), zeroValuedStates, ~zeroValuedStates, true);
507  ValueType upperBound = storm::utility::zero<ValueType>();
508  for (uint64_t state = 0; state < expVisits.size(); ++state) {
509  ValueType maxReward = storm::utility::zero<ValueType>();
510  for (auto row = model.getTransitionMatrix().getRowGroupIndices()[state], endRow = model.getTransitionMatrix().getRowGroupIndices()[state + 1]; row < endRow; ++row) {
511  maxReward = std::max(maxReward, rewards[row]);
512  }
513  upperBound += expVisits[state] * maxReward;
514  }
515  upperResultBound = upperBound;
516  }
517  storm::utility::vector::clip(upperResultBounds.get(), objective.lowerResultBound, upperResultBound);
518  }
