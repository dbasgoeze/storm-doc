214  template <typename ValueType>
215  void computeSchedulerProb1(storm::storage::SparseMatrix<ValueType> const& transitionMatrix, storm::storage::SparseMatrix<ValueType> const& backwardTransitions, storm::storage::BitVector const& consideredStates, storm::storage::BitVector const& statesToReach, std::vector<uint64_t>& choices, storm::storage::BitVector const* allowedChoices = nullptr) {
216 
217  std::vector<uint64_t> stack;
218  storm::storage::BitVector processedStates = statesToReach;
219  stack.insert(stack.end(), processedStates.begin(), processedStates.end());
220  uint64_t currentState = 0;
221 
222  while (!stack.empty()) {
223  currentState = stack.back();
224  stack.pop_back();
225 
226  for (auto const& predecessorEntry : backwardTransitions.getRow(currentState)) {
227  auto predecessor = predecessorEntry.getColumn();
228  if (consideredStates.get(predecessor) & !processedStates.get(predecessor)) {
229  // Find a choice leading to an already processed state (such a choice has to exist since this is a predecessor of the currentState)
230  auto const& groupStart = transitionMatrix.getRowGroupIndices()[predecessor];
231  auto const& groupEnd = transitionMatrix.getRowGroupIndices()[predecessor + 1];
232  uint64_t row = allowedChoices ? allowedChoices->getNextSetIndex(groupStart) : groupStart;
233  for (; row < groupEnd; row = allowedChoices ? allowedChoices->getNextSetIndex(row + 1) : row + 1) {
234  bool hasSuccessorInProcessedStates = false;
235  for (auto const& successorOfPredecessor : transitionMatrix.getRow(row)) {
236  if (processedStates.get(successorOfPredecessor.getColumn())) {
237  hasSuccessorInProcessedStates = true;
238  break;
239  }
240  }
241  if (hasSuccessorInProcessedStates) {
242  choices[predecessor] = row - groupStart;
243  processedStates.set(predecessor, true);
244  stack.push_back(predecessor);
245  break;
246  }
247  }
248  STORM_LOG_ASSERT(allowedChoices || row < groupEnd, "Unable to find choice at a predecessor of a processed state that leads to a processed state.");
249  }
250  }
251  }
252  STORM_LOG_ASSERT(consideredStates.isSubsetOf(processedStates), "Not all states have been processed.");
253  }

415 
416  // Use the (0...0) vector as initial guess for the solution.
417  std::fill(ecQuotient->auxStateValues.begin(), ecQuotient->auxStateValues.end(), storm::utility::zero<ValueType>());
418 
419  solver->solveEquations(env, ecQuotient->auxStateValues, ecQuotient->auxChoiceValues);
420  this->weightedResult = std::vector<ValueType>(transitionMatrix.getRowGroupCount());

425  template <class SparseModelType>
426  void StandardPcaaWeightVectorChecker<SparseModelType>::unboundedIndividualPhase(Environment const& env,std::vector<ValueType> const& weightVector) {
427  if (objectivesWithNoUpperTimeBound.getNumberOfSetBits() == 1 && storm::utility::isOne(weightVector[*objectivesWithNoUpperTimeBound.begin()])) {
428  uint_fast64_t objIndex = *objectivesWithNoUpperTimeBound.begin();
429  objectiveResults[objIndex] = weightedResult;
430  if (storm::solver::minimize(this->objectives[objIndex].formula->getOptimalityType())) {
431  storm::utility::vector::scaleVectorInPlace(objectiveResults[objIndex], -storm::utility::one<ValueType>());
432  }
433  for (uint_fast64_t objIndex2 = 0; objIndex2 < this->objectives.size(); ++objIndex2) {
434  if (objIndex != objIndex2) {
435  objectiveResults[objIndex2] = std::vector<ValueType>(transitionMatrix.getRowGroupCount(), storm::utility::zero<ValueType>());
436  }
437  }
438  } else {
439  storm::storage::SparseMatrix<ValueType> deterministicMatrix = transitionMatrix.selectRowsFromRowGroups(this->optimalChoices, false);
440  storm::storage::SparseMatrix<ValueType> deterministicBackwardTransitions = deterministicMatrix.transpose();
441  std::vector<ValueType> deterministicStateRewards(deterministicMatrix.getRowCount()); // allocate here
442  storm::solver::GeneralLinearEquationSolverFactory<ValueType> linearEquationSolverFactory;
443 
444  auto infiniteHorizonHelper = createDetInfiniteHorizonHelper(deterministicMatrix);
445  infiniteHorizonHelper.provideBackwardTransitions(deterministicBackwardTransitions);
446 
447  // We compute an estimate for the results of the individual objectives which is obtained from the weighted result and the result of the objectives computed so far.
448  // Note that weightedResult = Sum_{i=1}^{n} w_i * objectiveResult_i.
449  std::vector<ValueType> weightedSumOfUncheckedObjectives = weightedResult;
450  ValueType sumOfWeightsOfUncheckedObjectives = storm::utility::vector::sum_if(weightVector, objectivesWithNoUpperTimeBound);
451 
452  for (uint_fast64_t const &objIndex : storm::utility::vector::getSortedIndices(weightVector)) {
453  auto const& obj = this->objectives[objIndex];
454  if (objectivesWithNoUpperTimeBound.get(objIndex)) {
455  offsetsToUnderApproximation[objIndex] = storm::utility::zero<ValueType>();
456  offsetsToOverApproximation[objIndex] = storm::utility::zero<ValueType>();
457  if (lraObjectives.get(objIndex)) {
458  auto actionValueGetter = [&] (uint64_t const& a) { return actionRewards[objIndex][transitionMatrix.getRowGroupIndices()[a] + this->optimalChoices[a]]; };
459  typename storm::modelchecker::helper::SparseNondeterministicInfiniteHorizonHelper<ValueType>::ValueGetter stateValueGetter;
460  if (stateRewards.empty() || stateRewards[objIndex].empty()) {
461  stateValueGetter = [] (uint64_t const&) { return storm::utility::zero<ValueType>(); };
462  } else {
463  stateValueGetter = [&] (uint64_t const& s) { return stateRewards[objIndex][s]; };
464  }
465  objectiveResults[objIndex] = infiniteHorizonHelper.computeLongRunAverageValues(env, stateValueGetter, actionValueGetter);
466  } else { // i.e. a total reward objective
467  storm::utility::vector::selectVectorValues(deterministicStateRewards, this->optimalChoices, transitionMatrix.getRowGroupIndices(), actionRewards[objIndex]);
468  storm::storage::BitVector statesWithRewards = ~storm::utility::vector::filterZero(deterministicStateRewards);
469  // As maybestates we pick the states from which a state with reward is reachable
470  storm::storage::BitVector maybeStates = storm::utility::graph::performProbGreater0(deterministicBackwardTransitions, storm::storage::BitVector(deterministicMatrix.getRowCount(), true), statesWithRewards);
471 
472  // Compute the estimate for this objective
473  if (!storm::utility::isZero(weightVector[objIndex])) {
474  objectiveResults[objIndex] = weightedSumOfUncheckedObjectives;
475  ValueType scalingFactor = storm::utility::one<ValueType>() / sumOfWeightsOfUncheckedObjectives;
476  if (storm::solver::minimize(obj.formula->getOptimalityType())) {
477  scalingFactor *= -storm::utility::one<ValueType>();
478  }
479  storm::utility::vector::scaleVectorInPlace(objectiveResults[objIndex], scalingFactor);
480  storm::utility::vector::clip(objectiveResults[objIndex], obj.lowerResultBound, obj.upperResultBound);
481  }
482  // Make sure that the objectiveResult is initialized correctly
483  objectiveResults[objIndex].resize(transitionMatrix.getRowGroupCount(), storm::utility::zero<ValueType>());
484 
485  if (!maybeStates.empty()) {
486  bool needEquationSystem = linearEquationSolverFactory.getEquationProblemFormat(env) == storm::solver::LinearEquationSolverProblemFormat::EquationSystem;
487  storm::storage::SparseMatrix<ValueType> submatrix = deterministicMatrix.getSubmatrix(
488  true, maybeStates, maybeStates, needEquationSystem);
489  if (needEquationSystem) {
490  // Converting the matrix from the fixpoint notation to the form needed for the equation
491  // system. That is, we go from x = A*x + b to (I-A)x = b.
492  submatrix.convertToEquationSystem();
493  }
494 
495  // Prepare solution vector and rhs of the equation system.
496  std::vector<ValueType> x = storm::utility::vector::filterVector(objectiveResults[objIndex], maybeStates);
497  std::vector<ValueType> b = storm::utility::vector::filterVector(deterministicStateRewards, maybeStates);
498 
499  // Now solve the resulting equation system.
500  std::unique_ptr<storm::solver::LinearEquationSolver<ValueType>> solver = linearEquationSolverFactory.create(env, submatrix);
501  auto req = solver->getRequirements(env);
502  solver->clearBounds();
503  storm::storage::BitVector submatrixRowsWithSumLessOne = deterministicMatrix.getRowFilter(maybeStates, maybeStates) % maybeStates;
504  submatrixRowsWithSumLessOne.complement();
505  this->setBoundsToSolver(*solver, req.lowerBounds(), req.upperBounds(), objIndex, submatrix, submatrixRowsWithSumLessOne, b);
506  if (solver->hasLowerBound()) {
507  req.clearLowerBounds();
508  }
509  if (solver->hasUpperBound()) {
510  req.clearUpperBounds();
511  }
512  STORM_LOG_THROW(!req.hasEnabledCriticalRequirement(), storm::exceptions::UncheckedRequirementException, "Solver requirements " + req.getEnabledRequirementsAsString() + " not checked.");
513  solver->solveEquations(env, x, b);
514  // Set the result for this objective accordingly
515  storm::utility::vector::setVectorValues<ValueType>(objectiveResults[objIndex], maybeStates, x);
516  }
517  storm::utility::vector::setVectorValues<ValueType>(objectiveResults[objIndex], ~maybeStates, storm::utility::zero<ValueType>());
518  }
519  // Update the estimate for the next objectives.
520  if (!storm::utility::isZero(weightVector[objIndex])) {
521  storm::utility::vector::addScaledVector(weightedSumOfUncheckedObjectives, objectiveResults[objIndex], -weightVector[objIndex]);
522  sumOfWeightsOfUncheckedObjectives -= weightVector[objIndex];
523  }
524  } else {
525  objectiveResults[objIndex] = std::vector<ValueType>(transitionMatrix.getRowGroupCount(), storm::utility::zero<ValueType>());
526  }
527  }
528  }
529  }

531  template <class SparseModelType>
532  void StandardPcaaWeightVectorChecker<SparseModelType>::updateEcQuotient(std::vector<ValueType> const& weightedRewardVector) {
533  // Check whether we need to update the currently cached ecElimResult
534  storm::storage::BitVector newTotalReward0Choices = storm::utility::vector::filterZero(weightedRewardVector);
535  storm::storage::BitVector zeroLraRewardChoices(weightedRewardVector.size(), true);
536  if (lraMecDecomposition) {
537  for (uint64_t mecIndex = 0; mecIndex < lraMecDecomposition->mecs.size(); ++mecIndex) {
538  if (!storm::utility::isZero(lraMecDecomposition->auxMecValues[mecIndex])) {
539  // The mec has a non-zero value, so flag all its choices as non-zero
540  auto const& mec = lraMecDecomposition->mecs[mecIndex];
541  for (auto const& stateChoices : mec) {
542  for (auto const& choice : stateChoices.second) {
543  zeroLraRewardChoices.set(choice, false);
544  }
545  }
546  }
547  }
548  }
549  storm::storage::BitVector newReward0Choices = newTotalReward0Choices & zeroLraRewardChoices;
550  if (!ecQuotient || ecQuotient->origReward0Choices != newReward0Choices) {
551 
552  // It is sufficient to consider the states from which a transition with non-zero reward is reachable. (The remaining states always have reward zero).
553  auto nonZeroRewardStates = transitionMatrix.getRowGroupFilter(newReward0Choices, true);
554  nonZeroRewardStates.complement();
555  storm::storage::BitVector subsystemStates = storm::utility::graph::performProbGreater0E(transitionMatrix.transpose(true), storm::storage::BitVector(transitionMatrix.getRowGroupCount(), true), nonZeroRewardStates);
556 
557  // Remove neutral end components, i.e., ECs in which no total reward is earned.
558  // Note that such ECs contain one (or maybe more) LRA ECs.
559  auto ecElimResult = storm::transformer::EndComponentEliminator<ValueType>::transform(transitionMatrix, subsystemStates, ecChoicesHint & newTotalReward0Choices, totalReward0EStates);
560 
561  storm::storage::BitVector rowsWithSumLessOne(ecElimResult.matrix.getRowCount(), false);
562  for (uint64_t row = 0; row < rowsWithSumLessOne.size(); ++row) {
563  if (ecElimResult.matrix.getRow(row).getNumberOfEntries() == 0) {
564  rowsWithSumLessOne.set(row, true);
565  } else {
566  for (auto const& entry : transitionMatrix.getRow(ecElimResult.newToOldRowMapping[row])) {
567  if (!subsystemStates.get(entry.getColumn())) {
568  rowsWithSumLessOne.set(row, true);
569  break;
570  }
571  }
572  }
573  }
574 
575  ecQuotient = EcQuotient();
576  ecQuotient->matrix = std::move(ecElimResult.matrix);
577  ecQuotient->ecqToOriginalChoiceMapping = std::move(ecElimResult.newToOldRowMapping);
578  ecQuotient->originalToEcqStateMapping = std::move(ecElimResult.oldToNewStateMapping);
579  ecQuotient->ecqToOriginalStateMapping.resize(ecQuotient->matrix.getRowGroupCount());
580  for (uint64_t state = 0; state < ecQuotient->originalToEcqStateMapping.size(); ++state) {
581  uint64_t ecqState = ecQuotient->originalToEcqStateMapping[state];
582  if (ecqState < ecQuotient->matrix.getRowGroupCount()) {
583  ecQuotient->ecqToOriginalStateMapping[ecqState].insert(state);
584  }
585  }
586  ecQuotient->ecqStayInEcChoices = std::move(ecElimResult.sinkRows);
587  ecQuotient->origReward0Choices = std::move(newReward0Choices);
588  ecQuotient->rowsWithSumLessOne = std::move(rowsWithSumLessOne);
589  ecQuotient->auxStateValues.resize(ecQuotient->matrix.getRowGroupCount());
590  ecQuotient->auxChoiceValues.resize(ecQuotient->matrix.getRowCount());
591 
592  }

641  template <class SparseModelType>
642  void StandardPcaaWeightVectorChecker<SparseModelType>::computeAndSetBoundsToSolver(storm::solver::AbstractEquationSolver<ValueType>& solver, bool requiresLower, bool requiresUpper, storm::storage::SparseMatrix<ValueType> const& transitions, storm::storage::BitVector const& rowsWithSumLessOne, std::vector<ValueType> const& rewards) const {
643 
644  // Compute the one step target probs
645  std::vector<ValueType> oneStepTargetProbs(transitions.getRowCount(), storm::utility::zero<ValueType>());
646  for (auto row : rowsWithSumLessOne) {
647  oneStepTargetProbs[row] = storm::utility::one<ValueType>() - transitions.getRowSum(row);
648  }
649 
650  if (requiresLower && !solver.hasLowerBound()) {
651  // Compute lower bounds
652  std::vector<ValueType> negativeRewards;
653  negativeRewards.reserve(transitions.getRowCount());
654  uint64_t row = 0;
655  for (auto const& rew : rewards) {
656  if (rew < storm::utility::zero<ValueType>()) {
657  negativeRewards.resize(row, storm::utility::zero<ValueType>());
658  negativeRewards.push_back(-rew);
659  }
660  ++row;
661  }
662  if (!negativeRewards.empty()) {
663  negativeRewards.resize(row, storm::utility::zero<ValueType>());

666  solver.setLowerBounds(std::move(lowerBounds));
667  } else {
668  solver.setLowerBound(storm::utility::zero<ValueType>());
669  }
670  }
671 
672  // Compute upper bounds
673  if (requiresUpper && !solver.hasUpperBound()) {
674  std::vector<ValueType> positiveRewards;
675  positiveRewards.reserve(transitions.getRowCount());
676  uint64_t row = 0;
677  for (auto const& rew : rewards) {
678  if (rew > storm::utility::zero<ValueType>()) {
679  positiveRewards.resize(row, storm::utility::zero<ValueType>());
680  positiveRewards.push_back(rew);
681  }
682  ++row;
683  }
684  if (!positiveRewards.empty()) {
685  positiveRewards.resize(row, storm::utility::zero<ValueType>());
686  solver.setUpperBound(storm::modelchecker::helper::BaierUpperRewardBoundsComputer<ValueType>(transitions, positiveRewards, oneStepTargetProbs).computeUpperBound());
687  } else {
688  solver.setUpperBound(storm::utility::zero<ValueType>());
689  }
690  }

693  template <class SparseModelType>
694  void StandardPcaaWeightVectorChecker<SparseModelType>::transformEcqSolutionToOriginalModel(std::vector<ValueType> const& ecqSolution,
695  std::vector<uint_fast64_t> const& ecqOptimalChoices,
696  std::map<uint64_t, uint64_t> const& ecqStateToOptimalMecMap,
697  std::vector<ValueType>& originalSolution,
698  std::vector<uint_fast64_t>& originalOptimalChoices) const {
699 
700  auto backwardsTransitions = transitionMatrix.transpose(true);
701 
702  // Keep track of states for which no choice has been set yet.
703  storm::storage::BitVector unprocessedStates(transitionMatrix.getRowGroupCount(), true);
704 
705  // For each eliminated ec, keep track of the states (within the ec) that we want to reach and the states for which a choice needs to be set
706  // (Declared already at this point to avoid expensive allocations in each loop iteration)
707  storm::storage::BitVector ecStatesToReach(transitionMatrix.getRowGroupCount(), false);
708  storm::storage::BitVector ecStatesToProcess(transitionMatrix.getRowGroupCount(), false);
709 
710  // Run through each state of the ec quotient as well as the associated state(s) of the original model
711  for (uint64_t ecqState = 0; ecqState < ecqSolution.size(); ++ecqState) {
712  uint64_t ecqChoice = ecQuotient->matrix.getRowGroupIndices()[ecqState] + ecqOptimalChoices[ecqState];
713  uint_fast64_t origChoice = ecQuotient->ecqToOriginalChoiceMapping[ecqChoice];
714  auto const& origStates = ecQuotient->ecqToOriginalStateMapping[ecqState];
715  STORM_LOG_ASSERT(!origStates.empty(), "Unexpected empty set of original states.");
716  if (ecQuotient->ecqStayInEcChoices.get(ecqChoice)) {
717  // We stay in the current state(s) forever (End component)
718  // We need to set choices in a way that (i) the optimal LRA Mec is reached (if there is any) and (ii) 0 total reward is collected.
719  if (!ecqStateToOptimalMecMap.empty()) {
720  // The current ecqState represents an elimnated EC and we need to stay in this EC and we need to make sure that optimal MEC decisions are performed within this EC.
721  STORM_LOG_ASSERT(ecqStateToOptimalMecMap.count(ecqState) > 0, "No Lra Mec associated to given eliminated EC");
722  auto const& lraMec = lraMecDecomposition->mecs[ecqStateToOptimalMecMap.at(ecqState)];
723  if (lraMec.size() == origStates.size()) {
724  // LRA mec and eliminated EC coincide
725  for (auto const& state : origStates) {
726  STORM_LOG_ASSERT(lraMec.containsState(state), "Expected state to be contained in the lra mec.");
727  // Note that the optimal choice for this state has already been set in the infinite horizon phase.
728  unprocessedStates.set(state, false);
729  originalSolution[state] = ecqSolution[ecqState];
730  }
731  } else {
732  // LRA mec is proper subset of eliminated ec. There are also other states for which we have to set choices leading to the LRA MEC inside.
733  STORM_LOG_ASSERT(lraMec.size() < origStates.size(), "Lra Mec (" << lraMec.size() << " states) should be a proper subset of the eliminated ec (" << origStates.size() << " states).");
734  for (auto const& state : origStates) {
735  if (lraMec.containsState(state)) {
736  ecStatesToReach.set(state, true);
737  // Note that the optimal choice for this state has already been set in the infinite horizon phase.
738  } else {
739  ecStatesToProcess.set(state, true);
740  }
741  unprocessedStates.set(state, false);
742  originalSolution[state] = ecqSolution[ecqState];
743  }
744  computeSchedulerProb1(transitionMatrix, backwardsTransitions, ecStatesToProcess, ecStatesToReach, originalOptimalChoices);
745  // Clear bitvectors for next ecqState.
746  ecStatesToProcess.clear();
747  ecStatesToReach.clear();
748  }
749  } else {
750  // If there is no LRA Mec to reach, we just need to make sure that finite total reward is collected for all objectives
751  // In this branch our BitVectors have a slightly different meaning, so we create more readable aliases
752  storm::storage::BitVector& ecStatesToAvoid = ecStatesToReach;
753  bool needSchedulerComputation = false;
754  STORM_LOG_ASSERT(storm::utility::isZero(ecqSolution[ecqState]), "Solution for state that stays inside EC must be zero. Got " << ecqSolution[ecqState] << " instead.");
755  for (auto const& state : origStates) {
756  originalSolution[state] = storm::utility::zero<ValueType>(); // i.e. ecqSolution[ecqState];
757  ecStatesToProcess.set(state, true);
758  }
759  auto validChoices = transitionMatrix.getRowFilter(ecStatesToProcess, ecStatesToProcess);
760  auto valid0RewardChoices = validChoices & actionsWithoutRewardInUnboundedPhase;
761  for (auto const& state : origStates) {
762  auto groupStart = transitionMatrix.getRowGroupIndices()[state];
763  auto groupEnd = transitionMatrix.getRowGroupIndices()[state + 1];
764  auto nextValidChoice = valid0RewardChoices.getNextSetIndex(groupStart);
765  if (nextValidChoice < groupEnd) {
766  originalOptimalChoices[state] = nextValidChoice - groupStart;
767  } else {
768  // this state should not be reached infinitely often
769  ecStatesToAvoid.set(state, true);
770  needSchedulerComputation = true;
771  }
772  }
773  if (needSchedulerComputation) {
774  // There are ec states which we should not visit infinitely often
775  auto ecStatesThatCanAvoid = storm::utility::graph::performProbGreater0A(transitionMatrix, transitionMatrix.getRowGroupIndices(), backwardsTransitions, ecStatesToProcess, ecStatesToAvoid, false, 0, valid0RewardChoices);
776  ecStatesThatCanAvoid.complement();
777  // Set the choice for all states that can achieve value 0
778  computeSchedulerProb0(transitionMatrix, backwardsTransitions, ecStatesThatCanAvoid, ecStatesToAvoid, valid0RewardChoices, originalOptimalChoices);
779  // Set the choice for all remaining states
780  computeSchedulerProb1(transitionMatrix, backwardsTransitions, ecStatesToProcess & ~ecStatesToAvoid, ecStatesToAvoid, originalOptimalChoices, &validChoices);
781  }
782  ecStatesToAvoid.clear();
783  ecStatesToProcess.clear();
784  }
785  } else {
786  // We eventually leave the current state(s)
787  // In this case, we can safely take the origChoice at the corresponding state (say 's').
788  // For all other origStates associated with ecqState (if there are any), we make sure that the state 's' is reached almost surely.
789  if (origStates.size() > 1) {
790  for (auto const& state : origStates) {
791  // Check if the orig choice originates from this state
792  auto groupStart = transitionMatrix.getRowGroupIndices()[state];
793  auto groupEnd = transitionMatrix.getRowGroupIndices()[state + 1];
794  if (origChoice >= groupStart && origChoice < groupEnd) {
795  originalOptimalChoices[state] = origChoice - groupStart;
796  ecStatesToReach.set(state, true);
797  } else {
798  STORM_LOG_ASSERT(origStates.size() > 1, "Multiple original states expected.");
799  ecStatesToProcess.set(state, true);
800  }
801  unprocessedStates.set(state, false);
802  originalSolution[state] = ecqSolution[ecqState];
803  }
804  computeSchedulerProb1(transitionMatrix, backwardsTransitions, ecStatesToProcess, ecStatesToReach, originalOptimalChoices);
805  // Clear bitvectors for next ecqState.
806  ecStatesToProcess.clear();
807  ecStatesToReach.clear();
808  } else {
809  // There is just one state so we take the associated choice.
810  auto state = *origStates.begin();
811  auto groupStart = transitionMatrix.getRowGroupIndices()[state];
812  STORM_LOG_ASSERT(origChoice >= groupStart && origChoice < transitionMatrix.getRowGroupIndices()[state + 1], "Invalid choice: " << originalOptimalChoices[state] << " at a state with " << transitionMatrix.getRowGroupSize(state) << " choices.");
813  originalOptimalChoices[state] = origChoice - groupStart;
814  originalSolution[state] = ecqSolution[ecqState];
815  unprocessedStates.set(state, false);
816  }
817  }
818  }
819 
820  // The states that still not have been processed, there is no associated state of the ec quotient.
821  // This is because the value for these states will be 0 under all (lra optimal-) schedulers.
822  storm::utility::vector::setVectorValues(originalSolution, unprocessedStates, storm::utility::zero<ValueType>());
823  // Get a set of states for which we know that no reward (for all objectives) will be collected
824  if (this->lraMecDecomposition) {
825  // In this case, all unprocessed non-lra mec states should reach an (unprocessed) lra mec
826  for (auto const& mec : this->lraMecDecomposition->mecs) {
827  for (auto const& sc : mec) {
828  if (unprocessedStates.get(sc.first)) {
829  ecStatesToReach.set(sc.first, true);
830  }
831  }
832  }
833  } else {
