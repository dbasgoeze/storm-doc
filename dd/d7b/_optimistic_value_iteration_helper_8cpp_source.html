113  template<bool HasRowGroups, storm::solver::OptimizationDirection Dir>
114  uint64_t IterationHelper<ValueType>::repeatedIterateInternal(std::vector<ValueType>& x, std::vector<ValueType> const& b, ValueType precision, bool relative) {
115  // Do a backwards gauss-seidel style iteration
116  bool convergence = true;
117  IndexType i = x.size();
118  while (i > 0) {
119  --i;
120  ValueType newXi = HasRowGroups ? multiplyRowGroup<Dir>(i, b, x) : multiplyRow(i, b[i], x);
121  ValueType& oldXi = x[i];
122  // Check if we converged
123  if (relative) {
124  if (storm::utility::isZero(oldXi)) {
125  if (!storm::utility::isZero(newXi)) {
126  convergence = false;
127  break;
128  }
129  } else if (storm::utility::abs<ValueType>((newXi - oldXi) / oldXi) > precision) {
130  convergence = false;
131  break;
132  }
133  } else {
134  if (storm::utility::abs<ValueType>((newXi - oldXi)) > precision) {
135  convergence = false;
136  break;
137  }
138  }
139  }
140  if (!convergence) {
141  // we now know that we did not converge. We still need to set the remaining values
142  while (i > 0) {
143  --i;
144  x[i] = HasRowGroups ? multiplyRowGroup<Dir>(i, b, x) : multiplyRow(i, b[i], x);
145  }
146  }
147  return convergence;
148  }

166  typename IterationHelper<ValueType>::IterateResult IterationHelper<ValueType>::iterateUpperInternal(std::vector<ValueType>& x, std::vector<ValueType> const& b, bool takeMinOfOldAndNew) {
167  // For each row compare the new upper bound candidate with the old one
168  bool newUpperBoundAlwaysHigherEqual = true;
169  bool newUpperBoundAlwaysLowerEqual = true;
170  // Do a backwards gauss-seidel style iteration
171  for (IndexType i = x.size(); i > 0;) {
172  --i;
173  ValueType newXi = HasRowGroups ? multiplyRowGroup<Dir>(i, b, x) : multiplyRow(i, b[i], x);
174  ValueType& oldXi = x[i];
175  if (newXi > oldXi) {
176  newUpperBoundAlwaysLowerEqual = false;
177  if (!takeMinOfOldAndNew) {
178  oldXi = newXi;
179  }
180  } else if (newXi != oldXi) {
181  assert(newXi < oldXi);
182  newUpperBoundAlwaysHigherEqual = false;
183  oldXi = newXi;
184  }
185  }
186  // Return appropriate result
187  if (newUpperBoundAlwaysLowerEqual) {
188  if (newUpperBoundAlwaysHigherEqual) {
189  return IterateResult::Equal;
190  } else {
191  return IterateResult::AlwaysLowerOrEqual;
192  }
193  } else {

222  STORM_LOG_ASSERT(row < groupEnd, "Empty row group not expected.");
