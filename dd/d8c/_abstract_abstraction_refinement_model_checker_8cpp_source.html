111  STORM_LOG_THROW(checkTask->isOnlyInitialStatesRelevantSet(), storm::exceptions::InvalidPropertyException, "The abstraction-refinement model checkers can only compute the result for the initial states.");
112 
113  // Notify the underlying implementation that the abstraction-refinement process is being started.
114  this->initializeAbstractionRefinement();
115 
116  uint64_t iterations = 0;
117  std::unique_ptr<CheckResult> result;
118  auto start = std::chrono::high_resolution_clock::now();
119  while (!result) {
120  ++iterations;
121 
122  // Obtain the abstract model.
123  auto abstractionStart = std::chrono::high_resolution_clock::now();
124  std::shared_ptr<storm::models::Model<ValueType>> abstractModel = this->getAbstractModel();
125  auto abstractionEnd = std::chrono::high_resolution_clock::now();
126  STORM_LOG_TRACE("Model in iteration " << iterations << " has " << abstractModel->getNumberOfStates() << " states and " << abstractModel->getNumberOfTransitions() << " transitions (retrieved in " << std::chrono::duration_cast<std::chrono::milliseconds>(abstractionEnd - abstractionStart).count() << "ms).");
127 
128  // Obtain lower and upper bounds from the abstract model.
129  std::pair<std::unique_ptr<CheckResult>, std::unique_ptr<CheckResult>> bounds = computeBounds(env, *abstractModel);
130 
131  // Try to derive the final result from the obtained bounds.
132  if (bounds.first || bounds.second) {
133  result = tryToObtainResultFromBounds(*abstractModel, bounds);
134  }
135  if (!result) {
136  if (bounds.first && bounds.second) {
137  printBoundsInformation(bounds);
138  }
139 
140  auto refinementStart = std::chrono::high_resolution_clock::now();
141  this->refineAbstractModel();
142  auto refinementEnd = std::chrono::high_resolution_clock::now();
143  STORM_LOG_TRACE("Refinement in iteration " << iterations << " completed in " << std::chrono::duration_cast<std::chrono::milliseconds>(refinementEnd - refinementStart).count() << "ms.");
144  }
145  }
146  auto end = std::chrono::high_resolution_clock::now();
147  STORM_LOG_TRACE("Completed abstraction-refinement (" << this->getName() << ") in " << std::chrono::duration_cast<std::chrono::milliseconds>(end - start).count() << "ms.");

153  std::pair<std::unique_ptr<CheckResult>, std::unique_ptr<CheckResult>> AbstractAbstractionRefinementModelChecker<ModelType>::computeBounds(Environment const& env, storm::models::Model<ValueType> const& abstractModel) {
154  // We go through two phases. In phase (1) we are solving the qualitative part and in phase (2) the quantitative part.
155 
156  std::pair<std::unique_ptr<CheckResult>, std::unique_ptr<CheckResult>> result;
157 
158  // Preparation: determine the constraint states and the target states of the reachability objective.
159  std::pair<std::unique_ptr<storm::abstraction::StateSet>, std::unique_ptr<storm::abstraction::StateSet>> constraintAndTargetStates = getConstraintAndTargetStates(abstractModel);
160 
161  // Phase (1): solve qualitatively.
162  lastQualitativeResults = computeQualitativeResult(env, abstractModel, *constraintAndTargetStates.first, *constraintAndTargetStates.second);
163 
164  // Check whether the answer can be given after the qualitative solution.
165  result.first = checkForResultAfterQualitativeCheck(abstractModel);
166  if (result.first) {
167  return result;
168  }
169 
170  // Check whether we should skip the quantitative solution (for example if there are initial states for which
171  // the value is already known to be different at this point.
172  bool doSkipQuantitativeSolution = skipQuantitativeSolution(abstractModel, *lastQualitativeResults);
173  STORM_LOG_TRACE("" << (doSkipQuantitativeSolution ? "Skipping" : "Not skipping") << " quantitative solution.");
174 
175  // Phase (2): solve quantitatively.
176  if (!doSkipQuantitativeSolution) {
177  lastBounds = computeQuantitativeResult(env, abstractModel, *constraintAndTargetStates.first, *constraintAndTargetStates.second, *lastQualitativeResults);
178  STORM_LOG_ASSERT(lastBounds.first && lastBounds.second, "Expected both bounds.");
179 
180  result = std::make_pair(lastBounds.first->clone(), lastBounds.second->clone());
181  filterInitialStates(abstractModel, result);
182 
183  // Check whether the answer can be given after the quantitative solution.
184  if (checkForResultAfterQuantitativeCheck(abstractModel, true, result.first->asQuantitativeCheckResult<ValueType>())) {
185  result.second = nullptr;
186  } else if (checkForResultAfterQuantitativeCheck(abstractModel, false, result.second->asQuantitativeCheckResult<ValueType>())) {
187  result.first = nullptr;
188  }
189  }
190 
191  return result;
192  }

704  template<typename ModelType>
705  std::unique_ptr<CheckResult> AbstractAbstractionRefinementModelChecker<ModelType>::checkForResultAfterQualitativeCheck(storm::models::symbolic::Model<DdType, ValueType> const& abstractModel) {
706  std::unique_ptr<CheckResult> result;
707 
708  auto const& symbolicQualitativeResultMinMax = lastQualitativeResults->asSymbolicQualitativeResultMinMax<DdType>();
709 
710  bool isRewardFormula = checkTask->getFormula().isEventuallyFormula() && checkTask->getFormula().asEventuallyFormula().getContext() == storm::logic::FormulaContext::Reward;
711  if (isRewardFormula) {
712  // In the reachability reward case, we can give an answer if all initial states of the system are infinity
713  // states in the min result.
714  if ((abstractModel.getInitialStates() && !symbolicQualitativeResultMinMax.getProb1Min().getStates()) == abstractModel.getInitialStates()) {
715  result = std::make_unique<storm::modelchecker::SymbolicQuantitativeCheckResult<DdType, ValueType>>(abstractModel.getReachableStates(), abstractModel.getInitialStates(), abstractModel.getInitialStates().ite(abstractModel.getManager().getConstant(storm::utility::infinity<ValueType>()), abstractModel.getManager().template getAddZero<ValueType>()));
716  }
717  } else {
718  // In the reachability probability case, we can give the answer if all initial states are prob1 states
719  // in the min result or if all initial states are prob0 in the max case.
720  // Furthermore, we can give the answer if there are initial states with probability > 0 in the min case
721  // and the probability bound was 0 or if there are initial states with probability < 1 in the max case
722  // and the probability bound was 1.
723  if ((abstractModel.getInitialStates() && symbolicQualitativeResultMinMax.getProb1Min().getStates()) == abstractModel.getInitialStates()) {
724  result = std::make_unique<storm::modelchecker::SymbolicQuantitativeCheckResult<DdType, ValueType>>(abstractModel.getReachableStates(), abstractModel.getInitialStates(), abstractModel.getManager().template getAddOne<ValueType>());
725  } else if ((abstractModel.getInitialStates() && symbolicQualitativeResultMinMax.getProb0Max().getStates()) == abstractModel.getInitialStates()) {
726  result = std::make_unique<storm::modelchecker::SymbolicQuantitativeCheckResult<DdType, ValueType>>(abstractModel.getReachableStates(), abstractModel.getInitialStates(), abstractModel.getManager().template getAddZero<ValueType>());
727  } else if (checkTask->isBoundSet() && checkTask->getBoundThreshold() == storm::utility::zero<ValueType>() && (abstractModel.getInitialStates() && symbolicQualitativeResultMinMax.getProb0Min().getStates()) != abstractModel.getInitialStates()) {
728  result = std::make_unique<storm::modelchecker::SymbolicQuantitativeCheckResult<DdType, ValueType>>(abstractModel.getReachableStates(), abstractModel.getInitialStates(), (abstractModel.getInitialStates() && symbolicQualitativeResultMinMax.getProb0Min().getStates()).ite(abstractModel.getManager().template getConstant<ValueType>(0.5), abstractModel.getManager().template getAddZero<ValueType>()));
729  } else if (checkTask->isBoundSet() && checkTask->getBoundThreshold() == storm::utility::one<ValueType>() && (abstractModel.getInitialStates() && symbolicQualitativeResultMinMax.getProb1Max().getStates()) != abstractModel.getInitialStates()) {
730  result = std::make_unique<storm::modelchecker::SymbolicQuantitativeCheckResult<DdType, ValueType>>(abstractModel.getReachableStates(), abstractModel.getInitialStates(), (abstractModel.getInitialStates() && symbolicQualitativeResultMinMax.getProb1Max().getStates()).ite(abstractModel.getManager().template getConstant<ValueType>(0.5), abstractModel.getManager().template getAddZero<ValueType>()) + symbolicQualitativeResultMinMax.getProb1Max().getStates().template toAdd<ValueType>());
731  }
732  }
733 
734  if (result) {
735  STORM_LOG_TRACE("Found result after qualitative check.");
736  } else {
737  STORM_LOG_TRACE("Did not find result after qualitative check.");
738  }
739 
740  return result;
741  }
