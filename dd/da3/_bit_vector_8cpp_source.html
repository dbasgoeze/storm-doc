490  uint_fast64_t BitVector::getAsInt(uint_fast64_t bitIndex, uint_fast64_t numberOfBits) const {
491  STORM_LOG_ASSERT(numberOfBits <= 64, "Number of bits must be <= 64.");
492  uint64_t bucket = bitIndex >> 6;
493  uint64_t bitIndexInBucket = bitIndex & mod64mask;
494 
495  uint64_t mask;
496  if (bitIndexInBucket == 0) {
497  mask = -1ull;
498  } else {
499  mask = (1ull << (64 - bitIndexInBucket)) - 1ull;
500  }
501 
502  if (bitIndexInBucket + numberOfBits < 64) {
503  // If the value stops before the end of the bucket, we need to erase some lower bits.
504  mask &= ~((1ull << (64 - (bitIndexInBucket + numberOfBits))) - 1ull);
505  return (buckets[bucket] & mask) >> (64 - (bitIndexInBucket + numberOfBits));
506  } else if (bitIndexInBucket + numberOfBits > 64) {
507  // In this case, the integer "crosses" the bucket line.
508  uint64_t result = (buckets[bucket] & mask);
509  ++bucket;
510 
511  // Compute the remaining number of bits.
512  numberOfBits -= (64 - bitIndexInBucket);
513 
514  // Shift the intermediate result to the right location.
515  result <<= numberOfBits;
516 
517  // Strip away everything from the second bucket that is beyond the final index and add it to the
518  // intermediate result.
519  mask = ~((1ull << (64 - numberOfBits)) - 1ull);
520  uint64_t lowerBits = buckets[bucket] & mask;
521  result |= (lowerBits >> (64 - numberOfBits));
522 
523  return result;
524  } else {
525  // In this case, it suffices to take the current mask.
526  return buckets[bucket] & mask;
527  }

566

715  uint_fast64_t BitVector::getNextIndexWithValue(bool value, uint64_t const* dataPtr, uint_fast64_t startingIndex, uint_fast64_t endIndex) {
716  uint_fast8_t currentBitInByte = startingIndex & mod64mask;
717  uint64_t const* bucketIt = dataPtr + (startingIndex >> 6);
718  startingIndex = (startingIndex >> 6 << 6);
719 
720  uint64_t mask;
721  if (currentBitInByte == 0) {
722  mask = -1ull;
723  } else {
724  mask = (1ull << (64 - currentBitInByte)) - 1ull;
725  }
726 
727  // For efficiency reasons, we branch on the desired value at this point
728  if (value) {
729  while (startingIndex < endIndex) {
730  // Compute the remaining bucket content.
731  uint64_t remainingInBucket = *bucketIt & mask;
732 
733  // Check if there is at least one bit in the remainder of the bucket that is set to true.
734  if (remainingInBucket != 0) {
735  // As long as the current bit is not set, move the current bit.
736  while ((remainingInBucket & (1ull << (63 - currentBitInByte))) == 0) {
737  ++currentBitInByte;
738  }
739 
740  // Only return the index of the set bit if we are still in the valid range.
741  if (startingIndex + currentBitInByte < endIndex) {
742  return startingIndex + currentBitInByte;
743  } else {
744  return endIndex;
745  }
746  }
747 
748  // Advance to the next bucket.
749  startingIndex += 64;
750  ++bucketIt;
751  mask = -1ull;
752  currentBitInByte = 0;
753  }
754  } else {
755  while (startingIndex < endIndex) {
756  // Compute the remaining bucket content.
757  uint64_t remainingInBucket = *bucketIt & mask;
758 
759  // Check if there is at least one bit in the remainder of the bucket that is set to false.
760  if (remainingInBucket != (-1ull & mask)) {
761  // As long as the current bit is not false, move the current bit.
762  while ((remainingInBucket & (1ull << (63 - currentBitInByte))) != 0) {
763  ++currentBitInByte;
764  }
765 
766  // Only return the index of the set bit if we are still in the valid range.
767  if (startingIndex + currentBitInByte < endIndex) {
768  return startingIndex + currentBitInByte;
769  } else {
770  return endIndex;
771  }
772  }
773 
774  // Advance to the next bucket.
775  startingIndex += 64;
776  ++bucketIt;
777  mask = -1ull;
778  currentBitInByte = 0;
779  }
780  }
781 
782  return endIndex;
783  }
